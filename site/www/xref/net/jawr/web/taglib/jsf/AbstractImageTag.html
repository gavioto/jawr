184     /***
185 	 * Prepare the image URL
186 	 * @param context the faces context
187 	 * @param results the result
188      * @throws IOException if an exception occurs
189 	 */
190 	protected void prepareImageUrl(FacesContext context, StringBuffer results) throws IOException {
191 		
192 		ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) context.getExternalContext().getApplicationMap().get(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
193 		if(imgRsHandler == null){
194 			throw new IllegalStateException("You are using a Jawr image tag while the Jawr Image servlet has not been initialized. Initialization of Jawr Image servlet either failed or never occurred.");
195 		}
196 	
197 		String imgSrc = (String) getAttribute("src");
198 		
199 		String newUrl = (String) imgRsHandler.getCacheUrl(imgSrc);
200 		
201         if(newUrl == null){
202         	try {
203 				newUrl = CheckSumUtils.getCacheBustedUrl(imgSrc, imgRsHandler.getRsHandler(), imgRsHandler.getJawrConfig());
204 				imgRsHandler.addMapping(imgSrc, newUrl);
205 	    	} catch (IOException e) {
206 	    		
207 	    		throw new IOException("An IOException occured while processing the image '"+imgSrc+"'."+ e.getMessage());
208 			}
209     	}
210         
211 		if(newUrl == null){
212         	newUrl = imgSrc;
213         	logger.debug("No mapping found for the image : "+imgSrc);
214         }
215         
216         String imageServletMapping = imgRsHandler.getJawrConfig().getServletMapping();
217 		if("".equals(imageServletMapping)){
218 			if(newUrl.startsWith("/")){
219 				newUrl = newUrl.substring(1);
220 			}
221 		}else{
222 			newUrl = PathNormalizer.joinDomainToPath(imageServletMapping, newUrl);
223 		}
224         
225 		HttpServletResponse response = ((HttpServletResponse)context.getExternalContext().getResponse());
226         prepareAttribute(results, "src", response.encodeURL(newUrl));
227 	}
