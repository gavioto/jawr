32  /***
33   * Instances of this class will find all the resources which don't belong to 
34   * any defined bundle. Will return a mapping for each of them. 
35   * 
36   * @author Jordi Hernández Sellés
37   * @author Ibrahim Chaehoi
38   *
39   */
40  public class OrphanResourceBundlesMapper {
41  	
42  	/*** The logger */
43  	private static final Logger LOGGER = Logger.getLogger(OrphanResourceBundlesMapper.class);
44  	
45  	/*** The base directory */
46  	protected String baseDir;
47  	
48  	/*** The resource handler */
49  	protected ResourceReaderHandler rsHandler;
50  	
51  	/*** The generator registry */
52  	protected GeneratorRegistry generatorRegistry;
53  	
54  	/*** The list of current bundles */
55  	protected List currentBundles;
56  	
57  	/*** The resourcefile extension */
58  	protected String resourceExtension;
59  	
60  	/*** The bundle mapping */
61  	private List bundleMapping;
62  	
63  	/***
64  	 * Constructor
65  	 * 
66  	 * @param baseDir the base directory of the resource mapper
67  	 * @param rsHandler the resource handler
68  	 * @param generatorRegistry the generator registry
69  	 * @param currentBundles the list of current bundles
70  	 * @param resourceExtension the resource file extension
71  	 */
72  	public OrphanResourceBundlesMapper(String baseDir,
73  			ResourceReaderHandler rsHandler, GeneratorRegistry generatorRegistry, List currentBundles,
74  			String resourceExtension) {
75  		if(!"".equals(baseDir) && !"/".equals(baseDir))
76  			this.baseDir = "/" + PathNormalizer.normalizePath(baseDir) + "/**";
77  		else this.baseDir = "/**";
78  		
79  		this.rsHandler = rsHandler;
80  		this.generatorRegistry = generatorRegistry;
81  		this.currentBundles = new ArrayList();
82  		if(null != currentBundles)
83  			this.currentBundles.addAll(currentBundles);
84  		this.resourceExtension = resourceExtension;
85  		this.bundleMapping = new ArrayList();
86  	}
87  
88  	/***
89  	 * Scan all dirs starting at baseDir, and add each orphan 
90  	 * resource to the resources map.  
91  	 * @return
92  	 */
93  	public List getOrphansList() throws DuplicateBundlePathException {
94  		
95  		// Create a mapping for every resource available
96  		JoinableResourceBundleImpl tempBundle = new JoinableResourceBundleImpl("orphansTemp", "orphansTemp",
97  																				this.resourceExtension,
98  																				new InclusionPattern(),
99  																				Collections.singletonList(this.baseDir),
100 																				rsHandler, generatorRegistry);
101 		
102 		// Add licenses
103 		Set licensesPathList = tempBundle.getLicensesPathList();
104 		for(Iterator it = licensesPathList.iterator(); it.hasNext();) {
105 			addFileIfNotMapped((String)it.next());
106 		}
107 		
108 		// Add resources
109 		List allPaths = tempBundle.getItemPathList();
110 		for(Iterator it = allPaths.iterator(); it.hasNext();) {
111 			addFileIfNotMapped((String)it.next());
112 		}
113 		return this.bundleMapping;
114 	}
115 
116 	
117 	
118 	/***
119 	 * Determine wether a resource is already added to some bundle, add it to the list if it is not. 
120 	 * @param filePath
121 	 * @param currentMappedResources
122 	 */
123 	private void addFileIfNotMapped(String filePath)  throws DuplicateBundlePathException{				
124 		
125 		for(Iterator it = currentBundles.iterator();it.hasNext(); ) {
126 			JoinableResourceBundle bundle = (JoinableResourceBundle) it.next();			
127 			List items = bundle.getItemPathList();
128 			Set licenses = bundle.getLicensesPathList();
129 			
130 			if(items.contains(filePath))
131 				return;
132 			else if (licenses.contains(filePath))
133 				return;
134 			else if(filePath.equals(bundle.getId())){
135 				LOGGER.fatal("Duplicate bundle id resulted from orphan mapping of:" + filePath);
136 				throw new DuplicateBundlePathException(filePath);
137 			}
138 		}
139 		if(LOGGER.isDebugEnabled())
140 			LOGGER.debug("Adding orphan resource: " + filePath);
141 		
142 		// If we got here, the resource belongs to no other bundle.  
143 		bundleMapping.add(filePath);		
144 	}
