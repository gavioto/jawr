14  package net.jawr.web.resource.bundle.generator.css.less;
15  
16  import java.io.InputStream;
17  import java.io.InputStreamReader;
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  import org.mozilla.javascript.ConsString;
22  import org.mozilla.javascript.Context;
23  import org.mozilla.javascript.Function;
24  import org.mozilla.javascript.JavaScriptException;
25  import org.mozilla.javascript.NativeArray;
26  import org.mozilla.javascript.Scriptable;
27  import org.mozilla.javascript.ScriptableObject;
28  import org.mozilla.javascript.tools.shell.Global;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;

32  /**
33   * This class defines the less engine
34   * 
35   * Original code from LessCssEngine
36   * 
37   * @author (Original) Rostislav Hristov
38   * @author (Original) Uriah Carpenter
39   * @author (Original) Noah Sloan
40   */
41  public class LessEngine {
42  
43  	/** The logger */
44  	private final Logger logger = LoggerFactory.getLogger(getClass());
45  
46  	/** The scope */
47  	private Scriptable scope;
48  	
49  	/** The compileString function */
50  	private Function cs;
51  	
52  	/**
53  	 * Constructor 
54  	 */
55  	public LessEngine() {
56  		this(LessEngine.class.getResourceAsStream("less.js"), LessEngine.class.getResourceAsStream("engine.js"), LessEngine.class.getResourceAsStream("browser.js"));
57  	}
58  
59  	/**
60  	 * Constructor
61  	 * @param less the URL to the less.js file
62  	 */
63  	public LessEngine(InputStream less, InputStream engine, InputStream browser) {
64  		
65  		try {
66  			logger.debug("Initializing LESS Engine.");
67  			Context cx = Context.enter();
68  			logger.warn("Using implementation version: "
69  					+ cx.getImplementationVersion());
70  			cx.setOptimizationLevel(9);
71  			Global global = new Global();
72  			global.init(cx);
73  			scope = cx.initStandardObjects(global);
74  			cx.evaluateReader(scope, new InputStreamReader(browser), "browser.js", 1,
75  					null);
76  			cx.evaluateReader(scope, new InputStreamReader(less), "less.js", 1,
77  					null);
78  			cx.evaluateReader(scope, new InputStreamReader(engine), "engine.js", 1,
79  					null);
80  			cs = (Function) scope.get("compileString", scope);
81  			Context.exit();
82  		} catch (Exception e) {
83  			logger.error("LESS Engine intialization failed.", e);
84  		}
85  	}
86  
87  	/**
88  	 * Compiles the input string
89  	 * @param input the input string
90  	 * @return the compiled string
91  	 * @throws LessException if an error occurs during the compilation 
92  	 */
93  	public String compile(String input) throws LessException {
94  		try {
95  			long time = System.currentTimeMillis();
96  			String result = call(cs, new Object[] { input });
97  			logger.debug("The compilation of '" + input + "' took "
98  					+ (System.currentTimeMillis() - time) + " ms.");
99  			return result;
100 		} catch (Exception e) {
101 			throw parseLessException(e);
102 		}
103 	}
104 
105 	/**
106 	 * Calls the function using the argument array
107 	 * @param fn the function
108 	 * @param args the arguments
109 	 * @return the result of the function call
110 	 */
111 	private synchronized String call(Function fn, Object[] args) {
112 		return (String) Context.call(null, fn, scope, scope, args);
113 	}
114 
115 	/**
116 	 * Parse the exception and return a LessException
117 	 * @param root the root exception
118 	 * @return a LessException 
119 	 * @throws LessException if an exception occurs 
120 	 */
121 	private LessException parseLessException(Exception root)
122 			throws LessException {
123 
124 		logger.debug("Parsing LESS Exception", root);
125 
126 		if (root instanceof JavaScriptException) {
127 
128 			Scriptable value = (Scriptable) ((JavaScriptException) root)
129 					.getValue();
130 
131 			boolean hasName = ScriptableObject.hasProperty(value, "name");
132 			boolean hasType = ScriptableObject.hasProperty(value, "type");
133 
134 			if (hasName || hasType) {
135 				String errorType = "Error";
136 
137 				if (hasName) {
138 					String type = (String) ScriptableObject.getProperty(value,
139 							"name");
140 					if ("ParseError".equals(type)) {
141 						errorType = "Parse Error";
142 					} else {
143 						errorType = type + " Error";
144 					}
145 				} else if (hasType) {
146 					Object prop = ScriptableObject.getProperty(value, "type");
147 					if (prop instanceof String) {
148 						errorType = (String) prop + " Error";
149 					}
150 				}
151 
152 				String message = ScriptableObject.getProperty(value,
153 						"message").toString();
154 
155 				String filename = "";
156 				
157 				if (ScriptableObject.getProperty(value, "filename") != null) { 
158 					filename = ScriptableObject.getProperty(value, "filename").toString(); 
159 				}
160 				
161 				int line = -1;
162 				if (ScriptableObject.hasProperty(value, "line")) {
163 					line = ((Double) ScriptableObject
164 							.getProperty(value, "line")).intValue();
165 				}
166 
167 				int column = -1;
168 				if (ScriptableObject.hasProperty(value, "column")) {
169 					column = ((Double) ScriptableObject.getProperty(value,
170 							"column")).intValue();
171 				}
172 
173 				List<String> extractList = new ArrayList<String>();
174 				if (ScriptableObject.hasProperty(value, "extract")) {
175 					NativeArray extract = (NativeArray) ScriptableObject
176 							.getProperty(value, "extract");
177 					for (int i = 0; i < extract.getLength(); i++) {
178 						if (extract.get(i, extract) instanceof String) {
179 							extractList.add(((String) extract.get(i, extract))
180 									.replace("\t", " "));
181 						}
182 					}
183 				}
184 
185 				throw new LessException(message.toString(), errorType, filename, line,
186 						column, extractList);
187 			}
188 		}
189 
190 		throw new LessException(root);
191 	}
