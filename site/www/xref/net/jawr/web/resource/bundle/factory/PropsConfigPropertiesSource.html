14  package net.jawr.web.resource.bundle.factory;
15  
16  import java.util.Properties;
17  
18  import net.jawr.web.resource.bundle.factory.util.ConfigPropertiesSource;
19  import net.jawr.web.resource.bundle.factory.util.PropsFilePropertiesSource;
20  
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  /**
25   * ConfigPropertiesSource implementation that reads its values from a Properties
26   * object.
27   * 
28   * @author Ibrahim Chaehoi
29   */
30  public class PropsConfigPropertiesSource implements ConfigPropertiesSource {
31  
32  	/** The logger */
33  	private static final Logger LOGGER = LoggerFactory
34  			.getLogger(PropsFilePropertiesSource.class.getName());
35  
36  	/** The configuration properties */
37  	private Properties configProps;
38  
39  	/** The properties hashcode */
40  	protected int propsHashCode;
41  
42  	/**
43  	 * Constructor
44  	 * 
45  	 * @param props
46  	 *            the properties
47  	 */
48  	public PropsConfigPropertiesSource(Properties props) {
49  		this.configProps = props;
50  	}
51  
52  	/*
53  	 * (non-Javadoc)
54  	 * 
55  	 * @see net.jawr.web.resource.bundle.factory.util.ConfigPropertiesSource#
56  	 * configChanged()
57  	 */
58  	public boolean configChanged() {
59  		int currentConfigHash = this.configProps.hashCode();
60  		boolean configChanged = this.propsHashCode != currentConfigHash;
61  
62  		if (configChanged && LOGGER.isDebugEnabled())
63  			LOGGER.debug("Changes in configuration properties file detected.");
64  
65  		this.propsHashCode = currentConfigHash;
66  
67  		return configChanged;
68  	}
69  
70  	/*
71  	 * (non-Javadoc)
72  	 * 
73  	 * @see net.jawr.web.resource.bundle.factory.util.ConfigPropertiesSource#
74  	 * getConfigProperties()
75  	 */
76  	public Properties getConfigProperties() {
77  		return configProps;
78  	}
79  
80  }
