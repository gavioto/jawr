50  	/** The properties configuration helper */
51  	private PropertiesConfigHelper props;
52  	
53  	/** The bundle handler factory */
54  	private BundlesHandlerFactory factory;

58  	 */
59  	protected PropertiesBasedBundlesHandlerFactory(){

61  	}

63  	/**
64  	 * Create a PropertiesBasedBundlesHandlerFactory using the specified
65  	 * properties.
66  	 * 
67  	 * @param properties the properties
68  	 * @param resourceType
69  	 *            js or css
70  	 * @param rsHandler
71  	 *            ResourceHandler to access files.
72  	 */
73  	public PropertiesBasedBundlesHandlerFactory(Properties properties,
74  			String resourceType, ResourceReaderHandler rsHandler, ResourceBundleHandler rsBundleHandler,
75  			JawrConfig jawrConfig) {
76  		this.props = new PropertiesConfigHelper(properties, resourceType);
77  		
78  		// Create the BundlesHandlerFactory
79  		factory = new BundlesHandlerFactory(jawrConfig);
80  		factory.setResourceReaderHandler(rsHandler);
81  		factory.setResourceBundleHandler(rsBundleHandler);
82  		factory.setBundlesType(resourceType);
83  
84  		// Root resources dir
85  		factory.setBaseDir(props.getProperty(PropertiesBundleConstant.RESOURCES_BASEDIR, "/"));
86  
87  		// Use cache by default
88  		factory.setUseCacheManager(Boolean.valueOf(
89  				props.getProperty(PropertiesBundleConstant.RESOURCES_USE_CACHE, "true")).booleanValue());
90  
91  		// Postprocessor definitions
92  		factory.setGlobalPostProcessorKeys(props
93  				.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_POSTPROCESSOR));
94  		factory.setUnitPostProcessorKeys(props
95  				.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_FILE_POSTPROCESSOR));
96  		factory.setGlobalCompositePostProcessorKeys(props
97  				.getProperty(PropertiesBundleConstant.COMPOSITE_BUNDLE_FACTORY_POSTPROCESSOR));
98  		factory.setUnitCompositePostProcessorKeys(props
99  				.getProperty(PropertiesBundleConstant.COMPOSITE_BUNDLE_FACTORY_FILE_POSTPROCESSOR));
100 		factory.setResourceTypePreprocessorKeys(props
101 				.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_GLOBAL_PREPROCESSORS));
102 		factory.setResourceTypePostprocessorKeys(props
103 				.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_GLOBAL_POSTPROCESSORS));
104 		
105 		// Single or multiple bundle for orphans settings.
106 		factory.setUseSingleResourceFactory(Boolean.valueOf(
107 				props.getProperty(PropertiesBundleConstant.FACTORY_USE_SINGLE_BUNDLE, "false"))
108 				.booleanValue());
109 		factory.setSingleFileBundleName(props
110 				.getProperty(PropertiesBundleConstant.FACTORY_SINGLE_FILE_NAME));
111 		
112 		// Use orphans resolution at all, on by default. FACTORY_PROCESS_ORPHANS
113 		factory.setScanForOrphans(Boolean.valueOf(
114 				props.getCommonProperty(PropertiesBundleConstant.FACTORY_PROCESS_ORPHANS, "true")).booleanValue());
115 
116 		// Use the automatic directory-as-bundle mapper.
117 		factory.setUseDirMapperFactory(Boolean.valueOf(
118 				props.getProperty(PropertiesBundleConstant.FACTORY_USE_DIR_MAPPER, "false"))
119 				.booleanValue());
120 		factory.setExludedDirMapperDirs(props
121 				.getPropertyAsSet(PropertiesBundleConstant.FACTORY_DIR_MAPPER_EXCLUSION));
122 
123 		// Initialize custom generators
124 		Iterator<String> generators = props.getCommonPropertyAsSet(PropertiesBundleConstant.CUSTOM_GENERATORS)
125 				.iterator();
126 		GeneratorRegistry generatorRegistry = jawrConfig.getGeneratorRegistry();
127 		while (generators.hasNext()) {
128 			String generatorClass = (String) generators.next();
129 			generatorRegistry.registerGenerator(generatorClass);
130 		}
131 		
132 		// Initialize variant resolvers
133 		Iterator<String> resolvers = props.getCommonPropertyAsSet(PropertiesBundleConstant.CUSTOM_RESOLVERS)
134 				.iterator();
135 		while (resolvers.hasNext()) {
136 			String resolverClass = (String) resolvers.next();
137 			generatorRegistry.registerVariantResolver(resolverClass);
138 		}
139 
140 		// Initialize custom bundles
141 		Set<ResourceBundleDefinition> customBundles = new HashSet<ResourceBundleDefinition>();
142 		// Check if we should use the bundle names property or
143 		// find the bundle name using the bundle id declaration :
144 		// jawr.<type>.bundle.<name>.id
145 		if(null != props.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_NAMES)) {
146 			StringTokenizer tk = new StringTokenizer(props
147 					.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_NAMES), JawrConstant.COMMA_SEPARATOR);
148 			while (tk.hasMoreTokens()) {
149 				customBundles.add(buildCustomBundleDefinition(tk.nextToken()
150 						.trim(), false, generatorRegistry));
151 			}
152 		} else {
153 			Iterator<String> bundleNames = props.getPropertyBundleNameSet().iterator();
154 			while (bundleNames.hasNext()) {
155 				customBundles.add(buildCustomBundleDefinition(
156 						(String) bundleNames.next(), false, generatorRegistry));
157 			}
158 		}
159 		
160 		factory.setBundleDefinitions(customBundles);
161 
162 		// Set global bundle preprocessor
163 		factory.setCustomGlobalPreprocessors(props.getCustomGlobalPreprocessorMap());
164 		
165 		// Set global bundle postprocessor
166 		factory.setCustomGlobalPostprocessors(props.getCustomGlobalPostprocessorMap());
167 		
168 		// Set custom postprocessor
169 		factory.setCustomPostprocessors(props.getCustomPostProcessorMap());
170 		
171 	}

176 	/**
177 	 * Build a resources handler based on the configuration.
178 	 * 
179 	 * @param jawrConfig the jawr config
180 	 * @return a resources handler based on the configuration.
181 	 * @throws DuplicateBundlePathException
182 	 * @throws BundleDependencyException  if an error exists in the dependency definition
183 	 */
184 	public ResourceBundlesHandler buildResourceBundlesHandler() throws DuplicateBundlePathException, BundleDependencyException {
185 		return factory.buildResourceBundlesHandler();
186 	}

188 	/**
189 	 * Create a BundleDefinition based on the properties file.
190 	 * 
191 	 * @param bundleName the bundle name
192 	 * @param generatorRegistry the generator registry
193 	 * @return the bundleDefinition
194 	 */
195 	private ResourceBundleDefinition buildCustomBundleDefinition(
196 			String bundleName, boolean isChildBundle, GeneratorRegistry generatorRegistry) {
197 
198 		// Id for the bundle
199 		String bundleId = props.getCustomBundleProperty(bundleName,
200 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_ID);
201 		if (null == bundleId && !isChildBundle)
202 			throw new IllegalArgumentException(
203 					"No id defined for the bundle with name:" + bundleName
204 							+ ". Please specify one in configuration. ");
205 
206 		// Wether it's a composite or not
207 		boolean isComposite = Boolean.valueOf(
208 				props.getCustomBundleProperty(bundleName,
209 						PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_COMPOSITE_FLAG, "false"))
210 				.booleanValue();
211 
212 		// Create definition and set its id
213 		ResourceBundleDefinition bundle = new ResourceBundleDefinition();
214 		bundle.setBundleId(bundleId);
215 		bundle.setBundleName(bundleName);
216 		bundle.setBundlePrefix(props.getCustomBundleProperty(bundleName,
217 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_BUNDLE_PREFIX));
218 		
219 		// Wether it's global or not
220 		Boolean isGlobal = Boolean.valueOf(props.getCustomBundleProperty(
221 				bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_GLOBAL_FLAG, "false"));
222 		bundle.setGlobal(isGlobal.booleanValue());
223 
224 		// Set order if its a global bundle
225 		if (isGlobal.booleanValue()) {
226 			Integer order = Integer.valueOf(props.getCustomBundleProperty(
227 					bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_ORDER, "0"));
228 			bundle.setInclusionOrder(order.intValue());
229 		}
230 
231 		// Override bundle postprocessor
232 		if (null != props.getCustomBundleProperty(bundleName,
233 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_POSTPROCESSOR))
234 			bundle.setBundlePostProcessorKeys(props.getCustomBundleProperty(
235 					bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_POSTPROCESSOR));
236 
237 		// Override unitary postprocessor
238 		if (null != props.getCustomBundleProperty(bundleName,
239 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_FILE_POSTPROCESSOR))
240 			bundle.setUnitaryPostProcessorKeys(props.getCustomBundleProperty(
241 					bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_FILE_POSTPROCESSOR));
242 
243 		// Use only with debug mode on
244 		Boolean isDebugOnly = Boolean.valueOf(props.getCustomBundleProperty(
245 				bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEBUGONLY, "false"));
246 		bundle.setDebugOnly(isDebugOnly.booleanValue());
247 
248 		// Use only with debug mode off
249 		Boolean isDebugNever = Boolean.valueOf(props.getCustomBundleProperty(
250 				bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEBUGNEVER, "false"));
251 		bundle.setDebugNever(isDebugNever.booleanValue());
252 
253 		// Set conditional comment for IE, in case one is specified
254 		if (null != props.getCustomBundleProperty(bundleName,
255 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_IE_CONDITIONAL_EXPRESSION))
256 			bundle.setIeConditionalExpression(props
257 					.getCustomBundleProperty(bundleName,
258 							PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_IE_CONDITIONAL_EXPRESSION));
259 
260 		// Sets the alternate URL for production mode. 
261 		if (null != props.getCustomBundleProperty(bundleName,
262 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_PRODUCTION_ALT_URL))
263 			bundle.setAlternateProductionURL(props.getCustomBundleProperty(bundleName,
264 					PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_PRODUCTION_ALT_URL));
265 			
266 		
267 		if (isComposite) {
268 			String childBundlesProperty = props.getCustomBundleProperty(
269 					bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_COMPOSITE_NAMES);
270 			if (null == childBundlesProperty)
271 				throw new IllegalArgumentException(
272 						"No child bundle names were defined for the composite bundle with name:"
273 								+ bundleName
274 								+ ". Please specify at least one in configuration. ");
275 
276 			bundle.setComposite(true);
277 
278 			// add children
279 			List<ResourceBundleDefinition> children = new ArrayList<ResourceBundleDefinition>();
280 			StringTokenizer tk = new StringTokenizer(childBundlesProperty, JawrConstant.COMMA_SEPARATOR);
281 			while (tk.hasMoreTokens()) {
282 				ResourceBundleDefinition childDef = buildCustomBundleDefinition(
283 						tk.nextToken().trim(), true, generatorRegistry);
284 				childDef.setBundleId(bundleId);
285 				children.add(childDef);
286 			}
287 			bundle.setChildren(children);
288 		} else {
289 			String mappingsProperty = props.getCustomBundleProperty(bundleName,
290 					PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_MAPPINGS);
291 			if (null == mappingsProperty)
292 				throw new IllegalArgumentException(
293 						"No mappings were defined for the bundle with name:"
294 								+ bundleName
295 								+ ". Please specify at least one in configuration. ");
296 
297 			// Add the mappings
298 			List<String> mappings = new ArrayList<String>();
299 			Map<String, VariantSet> variants = new TreeMap<String, VariantSet>();
300 			StringTokenizer tk = new StringTokenizer(mappingsProperty, JawrConstant.COMMA_SEPARATOR);
301 			while (tk.hasMoreTokens()){
302 				String mapping = tk.nextToken().trim();
303 				mappings.add(mapping);
304 				// Add local variants
305 				variants = VariantUtils.concatVariants(variants, generatorRegistry.getAvailableVariants(mapping));
306 			}
307 			bundle.setMappings(mappings);
308 			bundle.setVariants(variants);
309 		}
310 
311 		// dependencies
312 		List<String> dependencies = props.getCustomBundlePropertyAsList(bundleName,
313 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEPENDENCIES);
314 		bundle.setDependencies(dependencies);
315 		
316 		return bundle;
317 	}
