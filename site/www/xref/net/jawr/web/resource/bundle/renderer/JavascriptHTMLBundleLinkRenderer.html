14  package net.jawr.web.resource.bundle.renderer;
15  
16  import java.io.IOException;
17  import java.io.Writer;
18  
19  import net.jawr.web.JawrConstant;
20  import net.jawr.web.resource.bundle.handler.ResourceBundlesHandler;
21  
22  /**
23   * Renderer that creates javascript link tags. 
24   * 
25   * @author Jordi Hernández Sellés
26   * @author Ibrahim Chaehoi
27   */
28  public class JavascriptHTMLBundleLinkRenderer extends AbstractBundleLinkRenderer implements JsBundleLinkRenderer{
29      
30  	/** The serial version UID */
31  	private static final long serialVersionUID = -7753927288041046690L;
32  
33  	/** The start tag */
34      private static final String PRE_TAG = "<script type=\"text/javascript\" src=\"";
35      
36      /** The source attribute*/
37      private static final String SRC_END_ATTRIBUTE = "\"";
38      
39      /** The async attribute */
40      private static final String ASYNC_ATTRIBUTE = " async=\"async\"";
41      
42      /** The defer attribute */
43      private static final String DEFER_ATTRIBUTE = " defer=\"defer\"";
44      
45      /** The end tag */
46      private static final String POST_TAG = " ></script>\n";
47      
48      /** The defer attribute */
49      private boolean defer;
50      
51      /** The async attribute */
52      private boolean async;
53      
54      /** A flag indicating if we are rendering the global links */
55      private boolean renderGlobalLinks;
56      
57      /** Creates a new instance of JavascriptHTMLBundleLinkRenderer */
58      public JavascriptHTMLBundleLinkRenderer(){
59      	
60      }
61      
62      /* (non-Javadoc)
63       * @see net.jawr.web.resource.bundle.renderer.JsBundleLinkRenderer#init(net.jawr.web.resource.bundle.handler.ResourceBundlesHandler, java.lang.Boolean, java.lang.Boolean, java.lang.Boolean)
64       */
65      @Override
66      public void init(ResourceBundlesHandler bundler, Boolean useRandomParam,
67      		Boolean async, Boolean defer) {
68      	init(bundler, useRandomParam);
69          if(async != null){
70          	this.async = async.booleanValue(); 
71          }
72          if(defer != null){
73          	this.defer = defer.booleanValue(); 
74          }
75      }
76  
77      /* (non-Javadoc)
78  	 * @see net.jawr.web.resource.bundle.renderer.BundleRenderer#getResourceType()
79  	 */
80  	public String getResourceType() {
81  		return JawrConstant.JS_TYPE;
82  	}
83  	
84  	/**
85  	 * Performs the global bundle rendering
86  	 * @param ctx the context
87  	 * @param out the writer
88  	 * @param debugOn the flag indicating if we are in debug mode or not
89  	 * @throws IOException if an IO exception occurs
90  	 */
91  	protected void performGlobalBundleLinksRendering(BundleRendererContext ctx,
92  			Writer out, boolean debugOn) throws IOException {
93  		
94  		renderGlobalLinks = true;
95  		super.performGlobalBundleLinksRendering(ctx, out, debugOn);
96  		renderGlobalLinks = false;
97  	}
98  	
99      /* (non-Javadoc)
100      * @see net.jawr.web.resource.bundle.renderer.AbstractBundleLinkRenderer#createBundleLink(java.lang.String, java.lang.String)
101      */
102     protected String renderLink(String fullPath) {
103     	
104     	//if(bundler.get)
105     	StringBuffer sb = new StringBuffer(PRE_TAG);
106     	sb.append(fullPath);
107     	sb.append(SRC_END_ATTRIBUTE);
108     	if(async && !renderGlobalLinks){
109     		sb.append(ASYNC_ATTRIBUTE);
110     	}
111     	if(defer && !renderGlobalLinks){
112     		sb.append(DEFER_ATTRIBUTE);
113     	}
114     	sb.append(POST_TAG); 
115         return sb.toString();
116     }
117     
118 }
