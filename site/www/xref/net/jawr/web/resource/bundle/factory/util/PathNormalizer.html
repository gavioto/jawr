56  	/***
57  	 * Removes the URL prefix defined in the configuration from a path. If the prefix contains a variant information, it adds it to the name.
58  	 * 
59  	 * @param path the path
60  	 * @return the path without the prefix
61  	 */
62  	public static String removeVariantPrefixFromPath(String path) {
63  		
64  		// Remove first slash
65  		String resultPath = path.substring(1);
66  		// eval the existence of a suffix
67  		String prefix = resultPath.substring(0, resultPath.indexOf("/"));
68  
69  		// The prefix also contains variant information after a '.'
70  		if (prefix.indexOf('.') != -1) {
71  			String variantPrefix = prefix.substring(prefix.indexOf('.') + 1);
72  			String suffix = '@' + variantPrefix + resultPath.substring(resultPath.lastIndexOf('.'));
73  			resultPath = resultPath.substring(resultPath.indexOf("/"), resultPath.lastIndexOf('.')) + suffix;
74  		} else
75  			resultPath = resultPath.substring(resultPath.indexOf("/"));
76  		
77  		return resultPath;
78  	}

80  	/***
81  	 * Removes the URL prefix defined in the configuration from a path. If the prefix contains a variant information, it adds it to the name.
82  	 * 
83  	 * @param path the path
84  	 * @return the bundle info from the  
85  	 */
86  	public static String[] extractBundleInfoFromPath(String path) {
87  		
88  		String[] result = new String[3];
89  		
90  		String resultPath = null;
91  		if(path.startsWith(BundleRenderer.GZIP_PATH_PREFIX)){
92  			// Remove the gzip prefix
93  			resultPath = path.substring(BundleRenderer.GZIP_PATH_PREFIX.length());
94  		}else{
95  			// Remove first slash
96  			resultPath = path.substring(1);
97  		}
98  		
99  		// eval the existence of a suffix
100 		int realPathStartIdx = resultPath.indexOf("/");
101 		String prefix = "";
102 		if(realPathStartIdx != -1){
103 			prefix = resultPath.substring(0, realPathStartIdx);
104 		}
105 		
106 		String hashcode = null;
107 		String variantPrefix = null;
108 		// The prefix also contains variant information after a '.'
109 		int hashCodeVariantSeparatorIdx = prefix.indexOf('.');
110 		if (hashCodeVariantSeparatorIdx != -1) {
111 			hashcode = prefix.substring(0, hashCodeVariantSeparatorIdx);
112 			variantPrefix = prefix.substring(hashCodeVariantSeparatorIdx + 1);
113 			resultPath = resultPath.substring(realPathStartIdx);
114 		} else{
115 			hashcode = prefix.substring(0, realPathStartIdx);
116 			resultPath = resultPath.substring(realPathStartIdx);
117 		}
118 		
119 		result[0] = resultPath;
120 		result[1] = variantPrefix;
121 		result[2] = hashcode;
122 		
123 		return result;
124 	}

158 	 * Normalizes two paths and joins them as a single path. 
159 	 * @param prefix
160 	 * @param path
161 	 * @return the joined path
162 	 */
163 	public static String joinPaths(String prefix,String path) {
164 		
165 		String joinedPath = null;
166 		if(prefix.startsWith(JawrConstant.HTTP_URL_PREFIX) || prefix.startsWith(JawrConstant.HTTPS_URL_PREFIX) || prefix.startsWith("//")){
167 			joinedPath = joinDomainToPath(prefix, path);
168 		}else{
169 			
170 			String normalizedPrefix = PathNormalizer.normalizePath(prefix);
171 			
172 			StringBuffer sb = new StringBuffer(JawrConstant.URL_SEPARATOR);
173 			if(!"".equals(normalizedPrefix))
174 				sb.append(normalizedPrefix).append(JawrConstant.URL_SEPARATOR);
175 			sb.append(PathNormalizer.normalizePath(path));
176 			joinedPath = sb.toString();
177 		}
178 		
179 		return joinedPath; 
180 	}

202 	/***
203 	 * Removes leading and trailing separators from a path, and removes 
204 	 * double separators (// is replaced by /). 
205 	 * @param path the path to normalize
206 	 * @return the normalized path
207 	 */
208 	public static final String normalizePath(String path) {
209 		String normalizedPath = path.replaceAll("//", JawrConstant.URL_SEPARATOR);
210 		StringTokenizer tk = new StringTokenizer(normalizedPath,JawrConstant.URL_SEPARATOR);
211 		StringBuffer sb = new StringBuffer();
212 		while(tk.hasMoreTokens()) {
213 			sb.append(tk.nextToken());
214 			if(tk.hasMoreTokens())
215 				sb.append(JawrConstant.URL_SEPARATOR);
216 		}
217 		return sb.toString();
218 		
219 	}

227 		Set ret = new HashSet();
228 		for(Iterator it = paths.iterator();it.hasNext();) {
229 			String path = normalizePath((String)it.next());
230 			ret.add(path);
231 		}
232 		return ret;

289     /***
290      * Determines the parent path of a filename or a directory.
291      * 
292      * <pre>
293      * PathUtils.getParentPath( null )                           = ""
294      * PathUtils.getParentPath( "" )                             = ""
295      * PathUtils.getParentPath( "/" )                            = "/"
296      * PathUtils.getParentPath( "/usr/local/" )                  = "/usr/local/"
297      * PathUtils.getRelativePath( "/usr/local/bin/java.sh" )     = ""/usr/local/bin/"
298      * </pre>
299      * 
300      * @param path the path
301      * @return the parent path.
302      */
303     public static String getParentPath(String path){
304     
305     	String parentPath = null;
306     	if(StringUtils.isEmpty(path)){
307     		parentPath = "";
308     	}else{
309     		
310     		parentPath = path;
311 	    	if(parentPath.length() > 1 && parentPath.endsWith(JawrConstant.URL_SEPARATOR)){
312 	    		parentPath = parentPath.substring(0, parentPath.length()-2);
313 	    	}
314 	    	int index = parentPath.lastIndexOf(JawrConstant.URL_SEPARATOR);
315 	    	if(index > 0){
316 	    		return parentPath.substring(0, index+1);
317 	    	}else{
318 	    		parentPath = JawrConstant.URL_SEPARATOR;
319 	    	}
320     	}
321     	
322     	return parentPath; 
323     }

361     /***
362      * Determines the relative path of a filename from a base directory.
363      * This method is useful in building relative links within pages of
364      * a web site.  It provides similar functionality to Anakia's
365      * <code>$relativePath</code> context variable.  The arguments to
366      * this method may contain either forward or backward slashes as
367      * file separators.  The relative path returned is formed using
368      * forward slashes as it is expected this path is to be used as a
369      * link in a web page (again mimicking Anakia's behavior).
370      * <p/>
371      * This method is thread-safe.
372      * <br/>
373      * <pre>
374      * PathUtils.getRelativePath( null, null )                                   = ""
375      * PathUtils.getRelativePath( null, "/usr/local/java/bin" )                  = ""
376      * PathUtils.getRelativePath( "/usr/local/", null )                          = ""
377      * PathUtils.getRelativePath( "/usr/local/", "/usr/local/java/bin" )         = ".."
378      * PathUtils.getRelativePath( "/usr/local/", "/usr/local/java/bin/java.sh" ) = "../.."
379      * PathUtils.getRelativePath( "/usr/local/java/bin/java.sh", "/usr/local/" ) = ""
380      * </pre>
381      *
382      * @param basedir The base directory.
383      * @param filename The filename that is relative to the base
384      * directory.
385      * @return The relative path of the filename from the base
386      * directory.  This value is not terminated with a forward slash.
387      * A zero-length string is returned if: the filename is not relative to
388      * the base directory, <code>basedir</code> is null or zero-length,
389      * or <code>filename</code> is null or zero-length.
390      */
391     public static final String getRelativePath( String basedir, String filename )
392     {
393         String basedirPath = uppercaseDrive(basedir);
394         String filenamePath = uppercaseDrive(filename);
395 
396         /*
397          * Verify the arguments and make sure the filename is relative
398          * to the base directory.
399          */
400         if ( basedirPath == null || basedirPath.length() == 0 || filenamePath == null
401             || filenamePath.length() == 0 || !filenamePath.startsWith( basedirPath ) )
402         {
403             return "";
404         }
405 
406         /*
407          * Normalize the arguments.  First, determine the file separator
408          * that is being used, then strip that off the end of both the
409          * base directory and filename.
410          */
411         String separator = determineSeparator( filenamePath );
412         basedirPath = StringUtils.chompLast( basedirPath, separator );
413         filenamePath = StringUtils.chompLast( filenamePath, separator );
414 
415         /*
416          * Remove the base directory from the filename to end up with a
417          * relative filename (relative to the base directory).  This
418          * filename is then used to determine the relative path.
419          */
420         String relativeFilename = filenamePath.substring( basedirPath.length() );
421 
422         return determineRelativePath( relativeFilename, separator );
423     }

425     /***
426      * Concatenates a filename to a base web path.
427      * If the base path doesn't end with "/", it will consider as base path the parent folder of the base path passed as parameter.
428      * 
429      * <pre>
430      * PathUtils.concatWebPath("", null));								   			= null
431 	 * PathUtils.concatWebPath(null, null));								   		= null
432 	 * PathUtils.concatWebPath(null, ""));								   			= null
433 	 * PathUtils.concatWebPath(null, "a"));								   			= null			
434 	 * PathUtils.concatWebPath(null, "/a"));								   		= "/a" 
435  	 * PathUtils.concatWebPath( "/css/folder/subfolder/", "icons/img.png" )      	= "/css/folder/subfolder/icons/img.png"
436      * PathUtils.concatWebPath( "/css/folder/subfolder/style.css", "icons/img.png") = "/css/folder/subfolder/icons/img.png"
437      * PathUtils.concatWebPath( "/css/folder/", "../icons/img.png" )      			= "/css/icons/img.png"
438      * PathUtils.concatWebPath( "/css/folder/style.css", "../icons/img.png" ) 		= "/css/icons/img.png"
439      * </pre>
440      * 
441      * @param basePath the base path
442      * @param fullFilenameToAdd the file name to add
443      * @return the concatenated path, or null if invalid
444      */
445     public static String concatWebPath(String basePath, String fullFilenameToAdd){
446      	
447     	if(fullFilenameToAdd == null || basePath == null && (fullFilenameToAdd.length() == 0
448     			|| fullFilenameToAdd.charAt(0) != JawrConstant.URL_SEPARATOR_CHAR)){
449     		return null;
450     	}
451     	
452     	if(basePath == null){
453     		basePath = "";
454     	}
455     	// If the basePath is pointing to a file, set the base path to the parent directory
456     	if(basePath.length() > 1 && basePath.charAt(basePath.length()-1) != '/'){
457     		basePath = getParentPath(basePath);
458     	}
459     	
460     	int len = basePath.length();
461     	String fullPath = null;
462         if (len == 0) {
463         	return doNormalizeIgnoreOtherSeparator(fullFilenameToAdd, true);
464         }
465         
466     	char ch = basePath.charAt(len - 1);
467         if (ch == JawrConstant.URL_SEPARATOR_CHAR) {
468         	fullPath = basePath + fullFilenameToAdd;
469         } else {
470         	fullPath = basePath + '/' + fullFilenameToAdd;
471         }
472         
473         return doNormalizeIgnoreOtherSeparator(fullPath, true); 
474         
475     }

477     /***
478      * Internal method to perform the normalization.
479      *
480      * @param filename  the filename
481      * @param keepSeparator  true to keep the final separator
482      * @return the normalized filename
483      */
484     private static String doNormalizeIgnoreOtherSeparator(String filename, boolean keepSeparator) {
485         if (filename == null) {
486             return null;
487         }
488         int size = filename.length();
489         if (size == 0) {
490             return filename;
491         }
492         int prefix = 0;
493 //        int prefix = getPrefixLength(filename);
494 //        if (prefix < 0) {
495 //            return null;
496 //        }
497         
498         char[] array = new char[size + 2];  // +1 for possible extra slash, +2 for arraycopy
499         filename.getChars(0, filename.length(), array, 0);
500         
501         // add extra separator on the end to simplify code below
502         boolean lastIsDirectory = true;
503         if (array[size - 1] != JawrConstant.URL_SEPARATOR_CHAR) {
504             array[size++] = JawrConstant.URL_SEPARATOR_CHAR;
505             lastIsDirectory = false;
506         }
507         
508         // adjoining slashes
509         for (int i = prefix + 1; i < size; i++) {
510             if (array[i] == JawrConstant.URL_SEPARATOR_CHAR && array[i - 1] == JawrConstant.URL_SEPARATOR_CHAR) {
511                 System.arraycopy(array, i, array, i - 1, size - i);
512                 size--;
513                 i--;
514             }
515         }
516         
517         // dot slash
518         for (int i = prefix + 1; i < size; i++) {
519             if (array[i] == JawrConstant.URL_SEPARATOR_CHAR && array[i - 1] == '.' &&
520                     (i == prefix + 1 || array[i - 2] == JawrConstant.URL_SEPARATOR_CHAR)) {
521                 if (i == size - 1) {
522                     lastIsDirectory = true;
523                 }
524                 System.arraycopy(array, i + 1, array, i - 1, size - i);
525                 size -=2;
526                 i--;
527             }
528         }
529         
530         // double dot slash
531         outer:
532         for (int i = prefix + 2; i < size; i++) {
533             if (array[i] == JawrConstant.URL_SEPARATOR_CHAR && array[i - 1] == '.' && array[i - 2] == '.' &&
534                     (i == prefix + 2 || array[i - 3] == JawrConstant.URL_SEPARATOR_CHAR)) {
535                 if (i == prefix + 2) {
536                     return null;
537                 }
538                 if (i == size - 1) {
539                     lastIsDirectory = true;
540                 }
541                 int j;
542                 for (j = i - 4 ; j >= prefix; j--) {
543                     if (array[j] == JawrConstant.URL_SEPARATOR_CHAR) {
544                         // remove b/../ from a/b/../c
545                         System.arraycopy(array, i + 1, array, j + 1, size - i);
546                         size -= (i - j);
547                         i = j + 1;
548                         continue outer;
549                     }
550                 }
551                 // remove a/../ from a/../c
552                 System.arraycopy(array, i + 1, array, prefix, size - i);
553                 size -= (i + 1 - prefix);
554                 i = prefix + 1;
555             }
556         }
557         
558         if (size <= 0) {  // should never be less than 0
559             return "";
560         }
561         if (size <= prefix) {  // should never be less than prefix
562             return new String(array, 0, size);
563         }
564         if (lastIsDirectory && keepSeparator) {
565             return new String(array, 0, size);  // keep trailing separator
566         }
567         return new String(array, 0, size - 1);  // lose trailing separator
568     }

570     /***
571      * This method can calculate the relative path between two pathes on a web site.
572      * <br/>
573      * <pre>
574      * PathUtils.getRelativeWebPath( null, null )                                          = ""
575      * PathUtils.getRelativeWebPath( null, "http://plexus.codehaus.org/" )                 = ""
576      * PathUtils.getRelativeWebPath( "http://plexus.codehaus.org/", null )                 = ""
577      * PathUtils.getRelativeWebPath( "http://plexus.codehaus.org/",
578      *                      "http://plexus.codehaus.org/plexus-utils/index.html" )        = "plexus-utils/index.html"
579      * PathUtils.getRelativeWebPath( "http://plexus.codehaus.org/plexus-utils/index.html",
580      *                      "http://plexus.codehaus.org/"                                 = "../../"
581      * </pre>
582      *
583      * @param oldPath
584      * @param newPath
585      * @return a relative web path from <code>oldPath</code>.
586      */
587     public static final String getRelativeWebPath( final String oldPath, final String newPath )
588     {
589         if ( StringUtils.isEmpty( oldPath ) || StringUtils.isEmpty( newPath ) )
590         {
591             return "";
592         }
593 
594         String resultPath = buildRelativePath( newPath, oldPath, '/' );
595 
596         if ( newPath.endsWith( "/" ) && !resultPath.endsWith( "/" ) )
597         {
598             return resultPath + "/";
599         }
600 
601         return resultPath;
602     }

604     /***
605      * This method can calculate the relative path between two pathes on a file system.
606      * <br/>
607      * <pre>
608      * PathUtils.getRelativeFilePath( null, null )                                   = ""
609      * PathUtils.getRelativeFilePath( null, "/usr/local/java/bin" )                  = ""
610      * PathUtils.getRelativeFilePath( "/usr/local", null )                           = ""
611      * PathUtils.getRelativeFilePath( "/usr/local", "/usr/local/java/bin" )          = "java/bin"
612      * PathUtils.getRelativeFilePath( "/usr/local", "/usr/local/java/bin/" )         = "java/bin"
613      * PathUtils.getRelativeFilePath( "/usr/local/java/bin", "/usr/local/" )         = "../.."
614      * PathUtils.getRelativeFilePath( "/usr/local/", "/usr/local/java/bin/java.sh" ) = "java/bin/java.sh"
615      * PathUtils.getRelativeFilePath( "/usr/local/java/bin/java.sh", "/usr/local/" ) = "../../.."
616      * PathUtils.getRelativeFilePath( "/usr/local/", "/bin" )                        = "../../bin"
617      * PathUtils.getRelativeFilePath( "/bin", "/usr/local/" )                        = "../usr/local"
618      * </pre>
619      * Note: On Windows based system, the <code>/</code> character should be replaced by <code>\</code> character.
620      *
621      * @param oldPath
622      * @param newPath
623      * @return a relative file path from <code>oldPath</code>.
624      */
625     public static final String getRelativeFilePath( final String oldPath, final String newPath )
626     {
627         if ( StringUtils.isEmpty( oldPath ) || StringUtils.isEmpty( newPath ) )
628         {
629             return "";
630         }
631 
632         // normalise the path delimiters
633         String fromPath = new File( oldPath ).getPath();
634         String toPath = new File( newPath ).getPath();
635 
636         // strip any leading slashes if its a windows path
637         if ( toPath.matches( "^//[a-zA-Z]:" ) )
638         {
639             toPath = toPath.substring( 1 );
640         }
641         if ( fromPath.matches( "^//[a-zA-Z]:" ) )
642         {
643             fromPath = fromPath.substring( 1 );
644         }
645 
646         // lowercase windows drive letters.
647         if ( fromPath.startsWith( ":", 1 ) )
648         {
649             fromPath = Character.toLowerCase( fromPath.charAt( 0 ) ) + fromPath.substring( 1 );
650         }
651         if ( toPath.startsWith( ":", 1 ) )
652         {
653             toPath = Character.toLowerCase( toPath.charAt( 0 ) ) + toPath.substring( 1 );
654         }
655 
656         // check for the presence of windows drives. No relative way of
657         // traversing from one to the other.
658         if ( ( toPath.startsWith( ":", 1 ) && fromPath.startsWith( ":", 1 ) )
659                         && ( !toPath.substring( 0, 1 ).equals( fromPath.substring( 0, 1 ) ) ) )
660         {
661             // they both have drive path element but they dont match, no
662             // relative path
663             return null;
664         }
665 
666         if ( ( toPath.startsWith( ":", 1 ) && !fromPath.startsWith( ":", 1 ) )
667                         || ( !toPath.startsWith( ":", 1 ) && fromPath.startsWith( ":", 1 ) ) )
668         {
669             // one has a drive path element and the other doesnt, no relative
670             // path.
671             return null;
672         }
673 
674         String resultPath = buildRelativePath( toPath, fromPath, File.separatorChar );
675 
676         if ( newPath.endsWith( File.separator ) && !resultPath.endsWith( File.separator ) )
677         {
678             return resultPath + File.separator;
679         }
680 
681         return resultPath;
682     }

688     /***
689      * Determines the relative path of a filename.  For each separator
690      * within the filename (except the leading if present), append the
691      * "../" string to the return value.
692      *
693      * @param filename The filename to parse.
694      * @param separator The separator used within the filename.
695      * @return The relative path of the filename.  This value is not
696      * terminated with a forward slash.  A zero-length string is
697      * returned if: the filename is zero-length.
698      */
699     private static final String determineRelativePath( String filename,
700                                                        String separator )
701     {
702         if ( filename.length() == 0 )
703         {
704             return "";
705         }
706 
707 
708         /*
709          * Count the slashes in the relative filename, but exclude the
710          * leading slash.  If the path has no slashes, then the filename
711          * is relative to the current directory.
712          */
713         int slashCount = StringUtils.countMatches( filename, separator ) - 1;
714         if ( slashCount <= 0 )
715         {
716             return ".";
717         }
718 
719         /*
720          * The relative filename contains one or more slashes indicating
721          * that the file is within one or more directories.  Thus, each
722          * slash represents a "../" in the relative path.
723          */
724         StringBuffer sb = new StringBuffer();
725         for ( int i = 0; i < slashCount; i++ )
726         {
727             sb.append( "../" );
728         }
729 
730         /*
731          * Finally, return the relative path but strip the trailing
732          * slash to mimic Anakia's behavior.
733          */
734         return StringUtils.chop( sb.toString() );
735     }

737     /***
738      * Helper method to determine the file separator (forward or
739      * backward slash) used in a filename.  The slash that occurs more
740      * often is returned as the separator.
741      *
742      * @param filename The filename parsed to determine the file
743      * separator.
744      * @return The file separator used within <code>filename</code>.
745      * This value is either a forward or backward slash.
746      */
747     private static final String determineSeparator( String filename )
748     {
749         int forwardCount = StringUtils.countMatches( filename, "/" );
750         int backwardCount = StringUtils.countMatches( filename, "//" );
751 
752         return forwardCount >= backwardCount ? "/" : "//";
753     }

775     private static final String buildRelativePath( String toPath,  String fromPath, final char separatorChar )
776     {
777         // use tokeniser to traverse paths and for lazy checking
778         StringTokenizer toTokeniser = new StringTokenizer( toPath, String.valueOf( separatorChar ) );
779         StringTokenizer fromTokeniser = new StringTokenizer( fromPath, String.valueOf( separatorChar ) );
780 
781         int count = 0;
782 
783         // walk along the to path looking for divergence from the from path
784         while ( toTokeniser.hasMoreTokens() && fromTokeniser.hasMoreTokens() )
785         {
786             if ( separatorChar == '//' )
787             {
788                 if ( !fromTokeniser.nextToken().equalsIgnoreCase( toTokeniser.nextToken() ) )
789                 {
790                     break;
791                 }
792             }
793             else
794             {
795                 if ( !fromTokeniser.nextToken().equals( toTokeniser.nextToken() ) )
796                 {
797                     break;
798                 }
799             }
800 
801             count++;
802         }
803 
804         // reinitialise the tokenisers to count positions to retrieve the
805         // gobbled token
806 
807         toTokeniser = new StringTokenizer( toPath, String.valueOf( separatorChar ) );
808         fromTokeniser = new StringTokenizer( fromPath, String.valueOf( separatorChar ) );
809 
810         while ( count-- > 0 )
811         {
812             fromTokeniser.nextToken();
813             toTokeniser.nextToken();
814         }
815 
816         StringBuffer relativePath = new StringBuffer();
817 
818         // add back refs for the rest of from location.
819         while ( fromTokeniser.hasMoreTokens() )
820         {
821             fromTokeniser.nextToken();
822 
823             relativePath.append("..");
824 
825             if ( fromTokeniser.hasMoreTokens() )
826             {
827             	relativePath.append(separatorChar);
828             }
829         }
830 
831         if ( relativePath.length() != 0 && toTokeniser.hasMoreTokens() )
832         {
833         	relativePath.append(separatorChar);
834         }
835 
836         // add fwd fills for whatevers left of newPath.
837         while ( toTokeniser.hasMoreTokens() )
838         {
839         	relativePath.append(toTokeniser.nextToken());
840 
841             if ( toTokeniser.hasMoreTokens() )
842             {
843             	relativePath.append(separatorChar);
844             }
845         }
846         return relativePath.toString();
847     }
