55  /**
56   * Creates a script which holds the data from a message bundle(s). The script is
57   * such that properties can be accessed as functions (i.e.:
58   * alert(com.mycompany.mymessage()); ).
59   * 
60   * @author Jordi Hernández Sellés
61   * @author ibrahim Chaehoi
62   */
63  public class MessageBundleScriptCreator {
64  
65  	/** The logger */
66  	private static final Logger LOGGER = LoggerFactory
67  			.getLogger(MessageBundleScriptCreator.class.getName());
68  
69  	public static final String DEFAULT_NAMESPACE = "messages";
70  
71  	private static final String SCRIPT_TEMPLATE = "/net/jawr/web/resource/bundle/message/messages.js";
72  
73  	private static final String DEFAULT_RESOURCE_BUNDLE_CHARSET = "ISO-8859-1";
74  
75  	protected static StringBuffer template;
76  	protected String configParam;
77  	protected String namespace;
78  	private String filter;
79  	protected Locale locale;
80  	private List<String> filterList;
81  	protected ServletContext servletContext;
82  	protected boolean fallbackToSystemLocale = true;
83  	protected boolean addQuoteToMessageKey = false;
84  	protected Charset resourceBundleCharset;
85  
86  	public MessageBundleScriptCreator(GeneratorContext context) {
87  		super();
88  		this.servletContext = context.getServletContext();
89  		if (null == template)
90  			template = loadScriptTemplate();
91  
92  		this.locale = context.getLocale();
93  
94  		// Set the namespace
95  		namespace = context.getParenthesesParam();
96  		namespace = null == namespace ? DEFAULT_NAMESPACE : namespace;
97  
98  		// Set the filter
99  		filter = context.getBracketsParam();
100 		if (null != filter) {
101 			StringTokenizer tk = new StringTokenizer(filter, "\\|");
102 			filterList = new ArrayList<String>();
103 			while (tk.hasMoreTokens())
104 				filterList.add(tk.nextToken());
105 		}
106 
107 		this.configParam = context.getPath();
108 
109 		Properties configProperties = context.getConfig().getConfigProperties();
110 		this.fallbackToSystemLocale = PropertiesConfigHelper.getBooleanValue(
111 				configProperties,
112 				JawrConstant.JAWR_LOCALE_GENERATOR_FALLBACK_TO_SYSTEM_LOCALE,
113 				true);
114 
115 		this.addQuoteToMessageKey = PropertiesConfigHelper.getBooleanValue(
116 				configProperties,
117 				JawrConstant.JAWR_LOCALE_GENERATOR_ADD_QUOTE_TO_MSG_KEY,
118 				false);
119 
120 		String charsetName = context.getConfig().getProperty(
121 				JawrConstant.JAWR_LOCALE_GENERATOR_RESOURCE_BUNDLE_CHARSET,
122 				DEFAULT_RESOURCE_BUNDLE_CHARSET);
123 
124 		resourceBundleCharset = Charset.forName(charsetName);
125 	}
126 
127 	/**
128 	 * Loads a template containing the functions which convert properties into
129 	 * methods.
130 	 * 
131 	 * @return the template containing the functions which convert properties
132 	 *         into methods.
133 	 */
134 	private StringBuffer loadScriptTemplate() {
135 		StringWriter sw = new StringWriter();
136 		InputStream is = null;
137 		try {
138 			is = ClassLoaderResourceUtils.getResourceAsStream(SCRIPT_TEMPLATE,
139 					this);
140 			IOUtils.copy(is, sw);
141 		} catch (IOException e) {
142 			Marker fatal = MarkerFactory.getMarker("FATAL");
143 			LOGGER.error(fatal,
144 					"a serious error occurred when initializing MessageBundleScriptCreator");
145 			throw new BundlingProcessException(
146 					"Classloading issues prevent loading the message template to be loaded. ",
147 					e);
148 		} finally {
149 			IOUtils.close(is);
150 		}
151 
152 		return sw.getBuffer();
153 	}

155 	/**
156 	 * Create the message resource bundles specified and uses a
157 	 * BundleStringJsonifier to generate the properties.
158 	 * 
159 	 * @return the script
160 	 */
161 	public Reader createScript(Charset charset) {
162 
163 		String[] names = configParam.split("\\|");
164 		Properties props = new Properties();
165 
166 		Locale currentLocale = getLocaleToApply();
167 
168 		MessageBundleControl control = new MessageBundleControl(
169 				fallbackToSystemLocale, resourceBundleCharset);
170 		for (int x = 0; x < names.length; x++) {
171 
172 			ResourceBundle bundle;
173 
174 			try {
175 				bundle = ResourceBundle.getBundle(names[x], currentLocale,
176 						control);
177 			} catch (MissingResourceException ex) {
178 				// Fixes problems with some servers, e.g. WLS 10
179 				try {
180 					bundle = ResourceBundle.getBundle(names[x], currentLocale,
181 							getClass().getClassLoader(), control);
182 				} catch (Exception e) {
183 					bundle = ResourceBundle.getBundle(names[x], currentLocale,
184 							Thread.currentThread().getContextClassLoader(),
185 							control);
186 				}
187 			}
188 
189 			updateProperties(bundle, props, charset);
190 		}
191 		return doCreateScript(props);
192 	}
193 
194 	/**
195 	 * Returns the locale to use to retrieve the ResourceBundle
196 	 * 
197 	 * @return the locale to use to retrieve the ResourceBundle
198 	 */
199 	protected Locale getLocaleToApply() {
200 		Locale currentLocale = locale;
201 
202 		if (currentLocale == null) {
203 			if (fallbackToSystemLocale) {
204 				currentLocale = Locale.getDefault();
205 			} else {
206 				currentLocale = new Locale("", "");
207 			}
208 		}
209 		return currentLocale;
210 	}
211 
212 	/**
213 	 * Loads the message resource bundles specified and uses a
214 	 * BundleStringJasonifier to generate the properties.
215 	 * 
216 	 * @return
217 	 */
218 	public Reader createScript(Charset charset, ResourceBundle bundle) {
219 
220 		Properties props = new Properties();
221 		updateProperties(bundle, props, charset);
222 		return doCreateScript(props);
223 	}
224 
225 	/**
226 	 * Loads the message resource bundles specified and uses a
227 	 * BundleStringJasonifier to generate the properties.
228 	 * 
229 	 * @param charset
230 	 * @return
231 	 * @throws UnsupportedEncodingException
232 	 */
233 	public void updateProperties(ResourceBundle bundle, Properties props,
234 			Charset charset) {
235 
236 		Enumeration<String> keys = bundle.getKeys();
237 
238 		while (keys.hasMoreElements()) {
239 			String key = keys.nextElement();
240 
241 			if (matchesFilter(key)) {
242 				String value = bundle.getString(key);
243 				props.put(key, value);
244 			}
245 		}
246 	}
247 
248 	/**
249 	 * Returns the JS script from the message properties
250 	 * 
251 	 * @param props
252 	 *            the message properties
253 	 * @return the JS script from the message properties
254 	 */
255 	protected Reader doCreateScript(Properties props) {
256 		BundleStringJsonifier bsj = new BundleStringJsonifier(props, addQuoteToMessageKey);
257 		String script = template.toString();
258 		String messages = bsj.serializeBundles().toString();
259 		script = script.replaceFirst("@namespace",
260 				RegexUtil.adaptReplacementToMatcher(namespace));
261 		script = script.replaceFirst("@messages",
262 				RegexUtil.adaptReplacementToMatcher(messages));
263 
264 		return new StringReader(script);
265 	}
266 
267 	/**
268 	 * Determines wether a key matches any of the set filters.
269 	 * 
270 	 * @param key
271 	 *            the property key
272 	 * @return true if the key matches any of the set filters.
273 	 */
274 	protected boolean matchesFilter(String key) {
275 		boolean rets = (null == filterList);
276 		if (!rets) {
277 			for (Iterator<String> it = filterList.iterator(); it.hasNext()
278 					&& !rets;)
279 				rets = key.startsWith(it.next());
280 		}
281 		return rets;
282 
283 	}

285 	public class MessageBundleControl extends ResourceBundle.Control {
286 
287 		private boolean fallbackToSystemLocale = true;
288 		private Charset charset;
289 
290 		/**
291 		 * Constructor
292 		 * 
293 		 * @param fallbackToSystemLocale
294 		 *            the flag indicating if the fallback local is the System
295 		 *            locale or not
296 		 * @param charset
297 		 *            the ResourceBundle charset
298 		 */
299 		public MessageBundleControl(boolean fallbackToSystemLocale,
300 				Charset charset) {
301 
302 			this.fallbackToSystemLocale = fallbackToSystemLocale;
303 			this.charset = charset;
304 		}
305 
306 		@Override
307 		public Locale getFallbackLocale(String baseName, Locale locale) {
308 			if (baseName == null) {
309 				throw new NullPointerException();
310 			}
311 			Locale defaultLocale = Locale.getDefault();
312 
313 			if (fallbackToSystemLocale) {
314 				defaultLocale = Locale.getDefault();
315 			} else {
316 				defaultLocale = new Locale("", "");
317 			}
318 
319 			return locale.equals(defaultLocale) ? null : defaultLocale;
320 		}
321 
322 		@Override
323 		public ResourceBundle newBundle(String baseName, Locale locale,
324 				String format, ClassLoader loader, boolean reload)
325 				throws IllegalAccessException, InstantiationException,
326 				IOException {
327 
328 			String bundleName = toBundleName(baseName, locale);
329 			ResourceBundle bundle = null;
330 			if (format.equals("java.class")) {
331 				try {
332 					@SuppressWarnings("unchecked")
333 					Class<? extends ResourceBundle> bundleClass = (Class<? extends ResourceBundle>) loader
334 							.loadClass(bundleName);
335 
336 					// If the class isn't a ResourceBundle subclass, throw a
337 					// ClassCastException.
338 					if (ResourceBundle.class.isAssignableFrom(bundleClass)) {
339 						bundle = bundleClass.newInstance();
340 					} else {
341 						throw new ClassCastException(bundleClass.getName()
342 								+ " cannot be cast to ResourceBundle");
343 					}
344 				} catch (ClassNotFoundException e) {
345 				}
346 			} else if (format.equals("java.properties")) {
347 				final String resourceName = toResourceName(bundleName,
348 						"properties");
349 				final ClassLoader classLoader = loader;
350 				final boolean reloadFlag = reload;
351 				InputStream stream = null;
352 				Reader rd = null;
353 				try {
354 					stream = AccessController
355 							.doPrivileged(new PrivilegedExceptionAction<InputStream>() {
356 								public InputStream run() throws IOException {
357 									InputStream is = null;
358 									if (reloadFlag) {
359 										URL url = classLoader
360 												.getResource(resourceName);
361 										if (url != null) {
362 											URLConnection connection = url
363 													.openConnection();
364 											if (connection != null) {
365 												// Disable caches to get fresh
366 												// data for
367 												// reloading.
368 												connection.setUseCaches(false);
369 												is = connection
370 														.getInputStream();
371 											}
372 										}
373 									} else {
374 										is = classLoader
375 												.getResourceAsStream(resourceName);
376 									}
377 									return is;
378 								}
379 							});
380 				} catch (PrivilegedActionException e) {
381 					throw (IOException) e.getException();
382 				}
383 				if (stream != null) {
384 					try {
385 						rd = new InputStreamReader(stream, charset);
386 						bundle = new PropertyResourceBundle(rd);
387 					} finally {
388 						IOUtils.close(rd);
389 						IOUtils.close(stream);
390 					}
391 				}
392 			} else {
393 				throw new IllegalArgumentException("unknown format: " + format);
394 			}
395 			return bundle;
396 		}
