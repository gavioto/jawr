31  /**
32   * This class will manage the serialization of the joinable resource bundle.
33   * 
34   * @author Ibrahim Chaehoi
35   * 
36   */
37  public class JoinableResourceBundlePropertySerializer {
38  
39  	/**
40  	 * This method will serialize the properties of the bundle in the Properties object
41  	 * 
42  	 * This method will serialize all bundle except the childs of composite bundle,
43  	 * for which the mappings will be part of their parent bundles.  
44  	 * The properties associated to a bundle are the same as the one define in the jawr configuration file.
45  	 * Only the following properties are different from the standard configuration file :
46  	 *   - The mapping, which will contains path to each resources of the bundle ( no wildcard like : myfolder/** ) 
47  	 *   - The variants which will be explicitly specified.
48  	 *   - The bundle hash codes will be define as properties, so we will not have to compute them.
49  	 *   For a bundle with local variants, there will be an hash code for each variant + one, which is the hash 
50  	 *   code of the default bundle.
51  	 *   - The licence path list.
52  	 *   
53  	 * @param bundle the bundle to serialize
54  	 * @param props the properties to update
55  	 */
56  	public static void serializeInProperties(JoinableResourceBundle bundle,
57  			String type, Properties props) {
58  
59  		// If the bundle is a child of a composite bundle, 
60  		// no need to serialize it, it will be integrated with the composite bundle
61  		if (StringUtils.isEmpty(bundle.getId())) {
62  			return;
63  		}
64  
65  		String bundleName = bundle.getName();
66  		String prefix = PropertiesBundleConstant.PROPS_PREFIX + type + "."+ PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_PROPERTY + bundleName;
67  		InclusionPattern inclusion = bundle.getInclusionPattern();
68  
69  		// Set the ID
70  		props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_ID, bundle.getId());
71  		
72  		if(StringUtils.isNotEmpty(bundle.getBundlePrefix())){
73  			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_BUNDLE_PREFIX, bundle.getBundlePrefix());
74  		}
75  		props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_ID, bundle.getId());
76  		
77  		if (inclusion.isGlobal()) {
78  			props
79  					.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_GLOBAL_FLAG, Boolean.toString(inclusion
80  							.isGlobal()));
81  		}
82  		if (inclusion.getInclusionOrder() != 0) {
83  			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_ORDER, Integer.toString(inclusion
84  					.getInclusionOrder()));
85  		}
86  
87  		if (inclusion.isIncludeOnlyOnDebug()) {
88  			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEBUGONLY, Boolean.toString(inclusion
89  					.isIncludeOnlyOnDebug()));
90  		}
91  		if (inclusion.isExcludeOnDebug()) {
92  			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEBUGNEVER, Boolean.toString(inclusion
93  					.isExcludeOnDebug()));
94  		}
95  		if (StringUtils.isNotEmpty(bundle.getExplorerConditionalExpression())) {
96  			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_IE_CONDITIONAL_EXPRESSION, bundle
97  					.getExplorerConditionalExpression());
98  		}
99  		if (StringUtils.isNotEmpty(bundle.getAlternateProductionURL())) {
100 			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_PRODUCTION_ALT_URL, bundle
101 					.getAlternateProductionURL());
102 		}
103 		
104 		if (bundle.getBundlePostProcessor() != null) {
105 			props
106 					.put(prefix +PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_POSTPROCESSOR,
107 							getBundlePostProcessorsName((ChainedResourceBundlePostProcessor) bundle
108 									.getBundlePostProcessor()));
109 		}
110 
111 		if (bundle.getUnitaryPostProcessor() != null) {
112 			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_FILE_POSTPROCESSOR,
113 					getBundlePostProcessorsName((ChainedResourceBundlePostProcessor) bundle
114 							.getUnitaryPostProcessor()));
115 		}
116 		
117 		// Add variants and the bundle hashcode
118 		Map<String, VariantSet> variants = bundle.getVariants();
119 		if (variants != null && !variants.isEmpty()) {
120 			String serializedVariants = serializeVariantSets(variants);
121 			if (StringUtils.isNotEmpty(serializedVariants)) {
122 				props.put(prefix +  PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_VARIANTS, serializedVariants);
123 			}
124 
125 			List<String> variantKeys = bundle.getVariantKeys();
126 			for (Iterator<String> iterator = variantKeys.iterator(); iterator
127 					.hasNext();) {
128 				String variantKey = iterator.next();
129 				if (StringUtils.isNotEmpty(variantKey)) {
130 					props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_HASHCODE_VARIANT + variantKey,
131 							bundle.getBundleDataHashCode(variantKey));
132 				}
133 			}
134 		} 
135 			
136 		String bundleHashcode = bundle.getBundleDataHashCode(null);
137 		if(bundleHashcode != null){
138 			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_HASHCODE, bundleHashcode);
139 		}
140 		
141 
142 		// mapping
143 		List<BundlePath> itemPathList = null; 
144 		if(!bundle.getInclusionPattern().isIncludeOnDebug()){
145 			itemPathList = bundle.getItemPathList();
146 		}else if(!bundle.getInclusionPattern().isExcludeOnDebug()){
147 			itemPathList = bundle.getItemDebugPathList();
148 		}
149 		
150 		if (itemPathList != null && !itemPathList.isEmpty()) {
151 			props
152 					.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_MAPPINGS,
153 							getCommaSeparatedStringForBundlePath(itemPathList));
154 		}
155 		List<JoinableResourceBundle> dependencies = bundle.getDependencies();
156 		if (dependencies != null && !dependencies.isEmpty()) {
157 			List<String> dependenciesBundleName = getBundleNames(dependencies);
158 			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEPENDENCIES, getCommaSeparatedString(dependenciesBundleName));
159 		}
160 		
161 		Set<String> licensesPathList = bundle.getLicensesPathList();
162 		if (licensesPathList != null && !licensesPathList.isEmpty()) {
163 			props.put(prefix + PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_LICENCE_PATH_LIST,
164 					getCommaSeparatedString(licensesPathList));
165 		}
166 	}
167 
168 	/**
169 	 * Serialize the variant sets.
170 	 * 
171 	 * @param map the map to serialize
172 	 * @return the serialized variant sets
173 	 */
174 	private static String serializeVariantSets(Map<String, VariantSet> map) {
175 		StringBuffer result = new StringBuffer();
176 		
177 		for (Iterator<Entry<String, VariantSet>> iterator = map.entrySet().iterator(); iterator.hasNext();) {
178 			Entry<String, VariantSet> entry = iterator.next();
179 			result.append(entry.getKey()+":");
180 			VariantSet variantSet = (VariantSet) entry.getValue();
181 			result.append(variantSet.getDefaultVariant()+":");
182 			result.append(getCommaSeparatedString(variantSet));
183 			result.append(";");
184 		}
185 		
186 		return result.toString();
187 	}
188 
189 	/**
190 	 * Returns the list of bundle names
191 	 * @param bundles the bundles
192 	 * @return the list of bundle names
193 	 */
194 	private static List<String> getBundleNames(List<JoinableResourceBundle> bundles) {
195 		
196 		List<String> bundleNames = new ArrayList<String>();
197 		for (Iterator<JoinableResourceBundle> iterator = bundles.iterator(); iterator
198 				.hasNext();) {
199 			bundleNames.add(iterator.next().getName());
200 		}
201 		return bundleNames;
202 	}
203 
204 	/**
205 	 * Returns the mapping list
206 	 * 
207 	 * @param itemPathList the item path list
208 	 * @return the item path list
209 	 */
210 	private static String getCommaSeparatedString(Collection<String> coll) {
211 
212 		StringBuffer buffer = new StringBuffer();
213 		for (Iterator<String> eltIterator = coll.iterator(); eltIterator.hasNext();) {
214 			String elt = eltIterator.next();
215 			buffer.append(elt);
216 			if(eltIterator.hasNext()){
217 				buffer.append(",");
218 			}
219 		}
220 		return buffer.toString();
221 	}
222 	
223 	/**
224 	 * Returns the mapping list
225 	 * 
226 	 * @param itemPathList the item path list
227 	 * @return the item path list
228 	 */
229 	private static String getCommaSeparatedStringForBundlePath(Collection<BundlePath> coll) {
230 
231 		StringBuffer buffer = new StringBuffer();
232 		for (Iterator<BundlePath> eltIterator = coll.iterator(); eltIterator.hasNext();) {
233 			String elt = eltIterator.next().getPath();
234 			buffer.append(elt);
235 			if(eltIterator.hasNext()){
236 				buffer.append(",");
237 			}
238 		}
239 		return buffer.toString();
240 	}

242 	/**
243 	 * Returns the bundle post processor name separated by a comma character
244 	 * 
245 	 * @param processor the post processor
246 	 * @return the bundle post processor name separated by a comma character
247 	 */
248 	private static String getBundlePostProcessorsName(
249 			ChainedResourceBundlePostProcessor processor) {
250 
251 		String bundlePostProcessor = "";
252 		if (processor != null) {
253 			bundlePostProcessor = processor.getId();
254 		}
255 
256 		return bundlePostProcessor;
257 	}
