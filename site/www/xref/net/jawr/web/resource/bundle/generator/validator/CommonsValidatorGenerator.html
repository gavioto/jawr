78  	 * (non-Javadoc)
79  	 * 
80  	 * @see
81  	 * net.jawr.web.resource.bundle.generator.ResourceGenerator#createResource
82  	 * (java.lang.String, javax.servlet.ServletContext,
83  	 * java.nio.charset.Charset)
84  	 */
85  	public Reader createResource(GeneratorContext context) {
86  		Locale locale = null == context.getLocale() ? Locale.getDefault()
87  				: context.getLocale();
88  		String validators = context.getPath();
89  		String validatorParams = context.getParenthesesParam();
90  
91  		String messagesNS = MessageBundleScriptCreator.DEFAULT_NAMESPACE;
92  		boolean stopOnErrors = true;
93  
94  		StringTokenizer valparams = new StringTokenizer(validatorParams, "|");
95  		String validatorMappingKey = valparams.nextToken();
96  		if (valparams.hasMoreTokens())
97  			messagesNS = valparams.nextToken();
98  		if (valparams.hasMoreTokens())
99  			stopOnErrors = Boolean.valueOf(valparams.nextToken())
100 					.booleanValue();
101 
102 		// Build the ValidatorResources if it doesn't exist yet
103 		if (!validatorResourcesMap.containsKey(validatorMappingKey)) {
104 			createValidatorResources(validatorMappingKey, context.getConfig());
105 		}
106 
107 		ValidatorResources validatorResources = (ValidatorResources) validatorResourcesMap
108 				.get(validatorMappingKey);
109 		StringBuffer sb = new StringBuffer();
110 		StringTokenizer tk = new StringTokenizer(validators, "|");
111 		while (tk.hasMoreTokens()) {
112 			String validator = tk.nextToken();
113 			if (STATIC_JAVASCRIPT_KEY.equals(validator)) {
114 				sb.append(buildStaticJavascript(validatorResources));
115 			} else
116 				sb.append(buildFormJavascript(validatorResources, validator,
117 						locale, messagesNS, stopOnErrors));
118 
119 		}
120 
121 		return new StringReader(sb.toString());
122 	}

137 	 * Creates a validator for a specific form.
138 	 * 
139 	 * @param validatorResources
140 	 * @param formName
141 	 * @param locale
142 	 * @return
143 	 */
144 	private StringBuffer buildFormJavascript(
145 			ValidatorResources validatorResources, String formName,
146 			Locale locale, String messageNS, boolean stopOnErrors) {
147 		Form form = validatorResources.getForm(locale, formName);
148 		return createDynamicJavascript(validatorResources, form, messageNS,
149 				stopOnErrors);
150 	}

152 	@SuppressWarnings("unchecked")
153 	private StringBuffer createDynamicJavascript(ValidatorResources resources,
154 			Form form, String messageNS, boolean stopOnErrors) {
155 		StringBuffer results = new StringBuffer();
156 
157 		List<ValidatorAction> actions = this.createActionList(resources, form);
158 
159 		final String methods = this.createMethods(actions, stopOnErrors);
160 
161 		String jsFormName = form.getName();
162 
163 		results.append(this.getJavascriptBegin(jsFormName, methods));
164 
165 		for (Iterator<ValidatorAction> i = actions.iterator(); i.hasNext();) {
166 			ValidatorAction va = i.next();
167 			int jscriptVar = 0;
168 			String functionName = null;
169 
170 			if ((va.getJsFunctionName() != null)
171 					&& (va.getJsFunctionName().length() > 0)) {
172 				functionName = va.getJsFunctionName();
173 			} else {
174 				functionName = va.getName();
175 			}
176 
177 			results.append("    function " + jsFormName + "_" + functionName
178 					+ " () { \n");
179 
180 			for (Iterator<Field> x = form.getFields().iterator(); x.hasNext();) {
181 				Field field = x.next();
182 
183 				// Skip indexed fields for now until there is a good way to
184 				// handle error messages (and the length of the list (could
185 				// retrieve from scope?))
186 				if (field.isIndexed() || !field.isDependency(va.getName())) {
187 					continue;
188 				}
189 
190 				String message = null;
191 				Msg msg = field.getMessage(va.getName());
192 				if ((msg != null) && !msg.isResource()) {
193 					message = JavascriptStringUtil.quote(msg.toString());
194 				} else {
195 					if (msg == null) {
196 						message = va.getMsg();
197 					} else {
198 						message = msg.getKey();
199 					}
200 					Arg[] args = field.getArgs(va.getName());
201 
202 					message = messageNS + "." + message + "(";
203 					for (int a = 0; a < args.length; a++) {
204 						if (args[a] != null) {
205 							if (args[a].isResource()) {
206 								message += messageNS + "." + args[a].getKey()
207 										+ "()";
208 							} else {
209 								message += "\"" + args[a].getKey() + "\"";
210 							}
211 							message += ",";
212 						}
213 					}
214 					message += "null)";
215 				}
216 
217 				message = (message != null) ? message : "";
218 
219 				// prefix variable with 'a' to make it a legal identifier
220 				results.append("     this.a" + jscriptVar++ + " = [\""
221 						+ field.getKey() + "\", " + (message) + ", ");
222 
223 				results.append("function(varName){");
224 
225 				Map<String, Var> vars = field.getVars();
226 
227 				// Loop through the field's variables.
228 				Iterator<Entry<String, Var>> varsIterator = vars.entrySet()
229 						.iterator();
230 
231 				while (varsIterator.hasNext()) {
232 
233 					Entry<String, Var> varEntry = varsIterator.next();
234 					String varName = varEntry.getKey();
235 					Var var = varEntry.getValue();
236 					String varValue = var.getValue();
237 
238 					// Non-resource variable
239 					if (var.isResource()) {
240 						varValue = messageNS + "." + varValue + "()";
241 					} else
242 						varValue = "'" + varValue + "'";
243 					String jsType = var.getJsType();
244 
245 					// skip requiredif variables field, fieldIndexed, fieldTest,
246 					// fieldValue
247 					if (varName.startsWith("field")) {
248 						continue;
249 					}
250 
251 					String varValueEscaped = JavascriptStringUtil
252 							.escape(varValue);
253 
254 					if (Var.JSTYPE_INT.equalsIgnoreCase(jsType)) {
255 						results.append("this." + varName + "=+"
256 								+ varValueEscaped + "; ");
257 					} else if (Var.JSTYPE_REGEXP.equalsIgnoreCase(jsType)) {
258 						results.append("this." + varName + "=eval('/'+"
259 								+ varValueEscaped + "+'/'); ");
260 					} else if (Var.JSTYPE_STRING.equalsIgnoreCase(jsType)) {
261 						results.append("this." + varName + "="
262 								+ varValueEscaped + "; ");
263 
264 						// So everyone using the latest format doesn't need to
265 						// change their xml files immediately.
266 					} else if ("mask".equalsIgnoreCase(varName)) {
267 						results.append("this." + varName + "=eval('/'+"
268 								+ varValueEscaped + "+'/'); ");
269 					} else {
270 						results.append("this." + varName + "="
271 								+ varValueEscaped + "; ");
272 					}
273 				}
274 
275 				results.append(" return this[varName];}];\n");
276 			}
277 
278 			results.append("    } \n\n");
279 		}
280 		return results;
281 	}

283 	/**
284 	 * Returns the opening script element and some initial javascript.
285 	 */
286 	protected String getJavascriptBegin(String jsFormName, String methods) {
287 		StringBuffer sb = new StringBuffer();
288 		String name = jsFormName.replace('/', '_'); // remove any '/' characters
289 
290 		name = jsFormName.substring(0, 1).toUpperCase()
291 				+ jsFormName.substring(1, jsFormName.length());
292 
293 		sb.append("\n    var bCancel = false; \n\n");
294 		sb.append("    function validate" + name + "(form) { \n");
295 		sb.append("        if (bCancel) { \n");
296 		sb.append("            return true; \n");
297 		sb.append("        } else { \n");
298 
299 		// Always return true if there aren't any Javascript validation methods
300 		if ((methods == null) || (methods.length() == 0)) {
301 			sb.append("            return true; \n");
302 		} else {
303 			sb.append("            var formValidationResult; \n");
304 			sb.append("            formValidationResult = " + methods + "; \n");
305 			if (methods.indexOf("&&") >= 0) {
306 				sb.append("            return (formValidationResult); \n");
307 			} else {
308 				// Making Sure that Bitwise operator works:
309 				sb.append("            return (formValidationResult == 1); \n");
310 			}
311 		}
312 		sb.append("        } \n");
313 		sb.append("    } \n\n");
314 
315 		return sb.toString();
316 	}

318 	@SuppressWarnings("unchecked")
319 	private List<ValidatorAction> createActionList(
320 			ValidatorResources resources, Form form) {
321 		List<String> actionMethods = new ArrayList<String>();
322 
323 		Iterator<Field> itFields = form.getFields().iterator();
324 
325 		while (itFields.hasNext()) {
326 			Field field = itFields.next();
327 
328 			for (Iterator<String> x = field.getDependencyList().iterator(); x
329 					.hasNext();) {
330 				String o = x.next();
331 
332 				if ((o != null) && !actionMethods.contains(o)) {
333 					actionMethods.add(o);
334 				}
335 			}
336 		}
337 
338 		List<ValidatorAction> actions = new ArrayList<ValidatorAction>();
339 
340 		// Create list of ValidatorActions based on actionMethods
341 		Iterator<String> iterator = actionMethods.iterator();
342 
343 		while (iterator.hasNext()) {
344 			String depends = iterator.next();
345 			ValidatorAction va = resources.getValidatorAction(depends);
346 
347 			// throw nicer NPE for easier debugging
348 			if (va == null) {
349 				throw new NullPointerException("Depends string \"" + depends
350 						+ "\" was not found in validator-rules.xml.");
351 			}
352 
353 			if ((va.getJavascript() != null)
354 					&& (va.getJavascript().length() > 0)) {
355 				actions.add(va);
356 			} else {
357 				iterator.remove();
358 			}
359 		}
360 
361 		Collections.sort(actions, ACTION_COMPARATOR);
362 
363 		return actions;
364 	}

366 	private String createMethods(List<ValidatorAction> actions,
367 			boolean stopOnErrors) {
368 		StringBuffer methods = new StringBuffer();
369 		final String methodOperator = stopOnErrors ? " && " : " & ";
370 
371 		Iterator<ValidatorAction> iter = actions.iterator();
372 
373 		while (iter.hasNext()) {
374 			ValidatorAction va = iter.next();
375 
376 			if (methods.length() > 0) {
377 				methods.append(methodOperator);
378 			}
379 
380 			methods.append(va.getMethod()).append("(form)");
381 		}
382 
383 		return methods.toString();
384 	}

386 	@SuppressWarnings("unchecked")
387 	private StringBuffer buildStaticJavascript(
388 			ValidatorResources validatorResources) {
389 		StringBuffer sb = new StringBuffer();
390 
391 		Iterator<ValidatorAction> actions = validatorResources
392 				.getValidatorActions().values().iterator();
393 
394 		while (actions.hasNext()) {
395 			ValidatorAction va = actions.next();
396 
397 			if (va != null) {
398 				String javascript = va.getJavascript();
399 
400 				if ((javascript != null) && (javascript.length() > 0)) {
401 					sb.append(javascript).append("\n");
402 				}
403 			}
404 		}
405 
406 		return sb;
407 	}

409 	private void createValidatorResources(String path, JawrConfig config) {
410 
411 		ValidatorResources validatorResources = null;
412 		String configPaths = config.getConfigProperties().getProperty(path);
413 
414 		StringTokenizer st = new StringTokenizer(configPaths, ",");
415 		InputStream[] inputStreams = new InputStream[st.countTokens()];
416 		int pos = 0;
417 
418 		try {
419 			while (st.hasMoreTokens()) {
420 				String validatorRules = st.nextToken().trim();
421 
422 				if (LOGGER.isDebugEnabled()) {
423 					LOGGER.debug("Validation rules file from '"
424 							+ validatorRules + "'");
425 				}
426 				InputStream is = null;
427 				try {
428 					is = ClassLoaderResourceUtils.getResourceAsStream(
429 							validatorRules, this);
430 				} catch (FileNotFoundException fos) {
431 					is = config.getContext()
432 							.getResourceAsStream(validatorRules);
433 				}
434 				inputStreams[pos] = is;
435 				pos++;
436 			}
437 			validatorResources = new ValidatorResources(inputStreams);
438 		} catch (SAXException ex) {
439 			throw new BundlingProcessException(ex);
440 		} catch (IOException e) {
441 			throw new BundlingProcessException(e);
442 		}
443 		validatorResourcesMap.put(path, validatorResources);
444 	}

446 	private static final Comparator<ValidatorAction> ACTION_COMPARATOR = new Comparator<ValidatorAction>() {
447 		public int compare(ValidatorAction va1, ValidatorAction va2) {
448 
449 			if (((va1.getDepends() == null) || (va1.getDepends().length() == 0))
450 					&& ((va2.getDepends() == null) || (va2.getDepends()
451 							.length() == 0))) {
452 				return 0;
453 			} else if (((va1.getDepends() != null) && (va1.getDepends()
454 					.length() > 0))
455 					&& ((va2.getDepends() == null) || (va2.getDepends()
456 							.length() == 0))) {
457 				return 1;
458 			} else if (((va1.getDepends() == null) || (va1.getDepends()
459 					.length() == 0))
460 					&& ((va2.getDepends() != null) && (va2.getDepends()
461 							.length() > 0))) {
462 				return -1;
463 			} else {
464 				return va1.getDependencyList().size()
465 						- va2.getDependencyList().size();
466 			}
467 		}
468 	};
