14  package net.jawr.web.resource.bundle.generator.resolver;
15  
16  
17  /**
18   * This class defines the factory for prefix and suffix resolver
19   * 
20   * @author ibrahim Chaehoi
21   */
22  public class ResourceGeneratorResolverFactory {
23  
24  	/**
25  	 * Create the prefix generator resolver 
26  	 * @param prefix the prefix
27  	 * @return the prefix generator resolver
28  	 */
29  	public static ResourceGeneratorResolver createPrefixResolver(String prefix){
30  		return new PrefixedPathResolver(prefix);
31  	}
32  	
33  	/**
34  	 * Create the suffix generator resolver 
35  	 * @param suffix the prefix
36  	 * @return the suffix generator resolver
37  	 */
38  	public static ResourceGeneratorResolver createSuffixResolver(String suffix){
39  		return new SuffixedPathResolver(suffix);
40  	}
41  }
