41  /**
42   * This class defines the manager for resource reader.
43   * 
44   * @author Ibrahim Chaehoi
45   */
46  public class ServletContextResourceReaderHandler implements
47  		ResourceReaderHandler {
48  
49  	/** The logger */
50  	private static final Logger LOGGER = LoggerFactory
51  			.getLogger(ServletContextResourceReaderHandler.class);
52  
53  	/** The servlet context */
54  	private ServletContext servletContext;
55  
56  	/** The working directory */
57  	private String workingDirectory;
58  
59  	/** The generator registry */
60  	private GeneratorRegistry generatorRegistry;
61  
62  	/** The list of resource readers */
63  	private List<TextResourceReader> resourceReaders = new ArrayList<TextResourceReader>();
64  
65  	/** The list of stream resource readers */
66  	private List<StreamResourceReader> streamResourceReaders = new ArrayList<StreamResourceReader>();
67  
68  	/** The list of resource info providers */
69  	private List<ResourceBrowser> resourceInfoProviders = new ArrayList<ResourceBrowser>();
70  
71  	/** The allowed file extension */
72  	private List<String> allowedExtensions = new ArrayList<String>();
73  
74  	/**
75  	 * Constructor
76  	 * 
77  	 * @param servletContext
78  	 *            the servlet context
79  	 * @param jawrConfig
80  	 *            the Jawr config
81  	 * @param generatorRegistry
82  	 *            the generator registry
83  	 * @throws IOException
84  	 *             if an IOException occurs.
85  	 */
86  	public ServletContextResourceReaderHandler(ServletContext servletContext,
87  			JawrConfig jawrConfig, GeneratorRegistry generatorRegistry)
88  			throws IOException {
89  
90  		String tempWorkingDirectory = ((File) servletContext
91  				.getAttribute(JawrConstant.SERVLET_CONTEXT_TEMPDIR))
92  				.getCanonicalPath();
93  		if (jawrConfig.getUseBundleMapping()
94  				&& StringUtils.isNotEmpty(jawrConfig.getJawrWorkingDirectory())) {
95  			tempWorkingDirectory = jawrConfig.getJawrWorkingDirectory();
96  		}
97  
98  		this.servletContext = servletContext;
99  		this.generatorRegistry = generatorRegistry;
100 		this.generatorRegistry.setResourceReaderHandler(this);
101 		if (tempWorkingDirectory.startsWith(JawrConstant.FILE_URI_PREFIX)) {
102 			tempWorkingDirectory = tempWorkingDirectory
103 					.substring(JawrConstant.FILE_URI_PREFIX.length());
104 		}
105 
106 		// add the default extension
107 		allowedExtensions.addAll(JawrConstant.DEFAULT_RESOURCE_EXTENSIONS);
108 
109 		if (JawrConstant.IMG_TYPE.equals(jawrConfig.getResourceType())) {
110 			for (Object key : ImageMIMETypesSupport.getSupportedProperties(
111 					JawrConfig.class).keySet()) {
112 				if (!this.allowedExtensions.contains((String) key)) {
113 					this.allowedExtensions.add((String) key);
114 				}
115 			}
116 		} else {
117 			allowedExtensions.add(jawrConfig.getResourceType());
118 		}
119 
120 		this.workingDirectory = tempWorkingDirectory;
121 
122 		ServletContextResourceReader rd = (ServletContextResourceReader) ClassLoaderResourceUtils
123 				.buildObjectInstance(jawrConfig
124 						.getServletContextResourceReaderClass());
125 		rd.init(servletContext, jawrConfig);
126 		addResourceReaderToEnd(rd);
127 
128 		// Add FileSystemResourceReader if needed
129 		String baseContextDir = jawrConfig
130 				.getProperty(JawrConstant.JAWR_BASECONTEXT_DIRECTORY);
131 		if (StringUtils.isNotEmpty(baseContextDir)) {
132 			ResourceReader fileRd = new FileSystemResourceReader(jawrConfig);
133 			if (LOGGER.isDebugEnabled()) {
134 				LOGGER.debug("The base directory context is set to "
135 						+ baseContextDir);
136 			}
137 
138 			boolean baseContextDirHighPriority = Boolean
139 					.valueOf(jawrConfig
140 							.getProperty(JawrConstant.JAWR_BASECONTEXT_DIRECTORY_HIGH_PRIORITY));
141 			if (baseContextDirHighPriority) {
142 				addResourceReaderToStart(fileRd);
143 				if (LOGGER.isDebugEnabled()) {
144 					LOGGER.debug("Jawr will search in priority in the base directory context before searching in the war content.");
145 				}
146 			} else {
147 				addResourceReaderToEnd(fileRd);
148 				if (LOGGER.isDebugEnabled()) {
149 					LOGGER.debug("Jawr will search in priority in the war content before searching in the base directory context.");
150 				}
151 			}
152 		}
153 	}

175 	/**
176 	 * Initialize the reader
177 	 * 
178 	 * @param obj
179 	 *            the reader to initialize
180 	 */
181 	private void initReader(Object obj) {
182 		if (obj instanceof WorkingDirectoryLocationAware) {
183 			((WorkingDirectoryLocationAware) obj)
184 					.setWorkingDirectory(workingDirectory);
185 		}
186 		if (obj instanceof ServletContextAware) {
187 			((ServletContextAware) obj).setServletContext(servletContext);
188 		}
189 
190 		if (obj instanceof ResourceBrowser) {
191 			resourceInfoProviders.add(0, (ResourceBrowser) obj);
192 		}
193 	}
194 
195 	/**
196 	 * Adds the resource reader to the list of available resource readers.
197 	 * 
198 	 * @param rd
199 	 *            the resource reader
200 	 */
201 	public void addResourceReaderToEnd(ResourceReader rd) {
202 
203 		if (rd instanceof TextResourceReader) {
204 			resourceReaders.add((TextResourceReader) rd);
205 		}
206 
207 		if (rd instanceof StreamResourceReader) {
208 			streamResourceReaders.add((StreamResourceReader) rd);
209 		}
210 
211 		initReader(rd);
212 	}
213 
214 	/**
215 	 * Adds the resource reader to the list of available resource readers at the
216 	 * specified position.
217 	 * 
218 	 * @param rd
219 	 */
220 	public void addResourceReaderToStart(ResourceReader rd) {
221 		if (rd instanceof TextResourceReader) {
222 			resourceReaders.add(0, (TextResourceReader) rd);
223 		}
224 		if (rd instanceof StreamResourceReader) {
225 			streamResourceReaders.add(0, (StreamResourceReader) rd);
226 		}
227 
228 		initReader(rd);
229 	}

258 	/*
259 	 * (non-Javadoc)
260 	 * 
261 	 * @see
262 	 * net.jawr.web.resource.handler.ResourceReader#getResource(java.lang.String
263 	 * , boolean)
264 	 */
265 	public Reader getResource(String resourceName, boolean processingBundle,
266 			List<Class<?>> excludedReader) throws ResourceNotFoundException {
267 
268 		Reader rd = null;
269 
270 		String resourceExtension = FileNameUtils.getExtension(resourceName);
271 		boolean generatedPath = generatorRegistry.isPathGenerated(resourceName);
272 		if (generatedPath
273 				|| allowedExtensions.contains(resourceExtension.toLowerCase())) {
274 			for (Iterator<TextResourceReader> iterator = resourceReaders
275 					.iterator(); iterator.hasNext();) {
276 				TextResourceReader rsReader = iterator.next();
277 
278 				if (!isInstanceOf(rsReader, excludedReader)) {
279 					if (!(rsReader instanceof ResourceGenerator)
280 							|| ((ResourceGenerator) rsReader).getResolver()
281 									.matchPath(resourceName)) {
282 
283 						rd = rsReader.getResource(resourceName,
284 								processingBundle);
285 						if (rd != null) {
286 							break;
287 						}
288 					}
289 				}
290 
291 			}
292 		} else {
293 			LOGGER.warn("The resource '"
294 					+ resourceName
295 					+ "' will not be read as its extension is not an allowed one.");
296 		}
297 
298 		if (rd == null) {
299 			throw new ResourceNotFoundException(resourceName);
300 		}
301 
302 		return rd;
303 	}
304 
305 	private boolean isInstanceOf(Object rd, List<Class<?>> interfaces) {
306 
307 		boolean result = false;
308 
309 		for (Class<?> class1 : interfaces) {
310 			if (class1.isInstance(rd)) {
311 				result = true;
312 				break;
313 			}
314 		}
315 
316 		return result;
317 	}

332 	/*
333 	 * (non-Javadoc)
334 	 * 
335 	 * @see
336 	 * net.jawr.web.resource.handler.stream.StreamResourceReader#getResourceAsStream
337 	 * (java.lang.String, boolean)
338 	 */
339 	public InputStream getResourceAsStream(String resourceName,
340 			boolean processingBundle) throws ResourceNotFoundException {
341 
342 		generatorRegistry.loadGeneratorIfNeeded(resourceName);
343 		InputStream is = null;
344 
345 		String resourceExtension = FileNameUtils.getExtension(resourceName);
346 		boolean generatedPath = generatorRegistry.isPathGenerated(resourceName);
347 		if (generatedPath
348 				|| allowedExtensions.contains(resourceExtension.toLowerCase())) {
349 			for (Iterator<StreamResourceReader> iterator = streamResourceReaders
350 					.iterator(); iterator.hasNext();) {
351 
352 				StreamResourceReader rsReader = iterator.next();
353 				if (!(rsReader instanceof ResourceGenerator)
354 						|| ((ResourceGenerator) rsReader).getResolver()
355 								.matchPath(resourceName)) {
356 
357 					is = rsReader.getResourceAsStream(resourceName);
358 					if (is != null) {
359 						break;
360 					}
361 				}
362 			}
363 		} else {
364 			LOGGER.warn("The resource '"
365 					+ resourceName
366 					+ "' will not be read as its extension is not an allowed one.");
367 		}
368 		if (is == null) {
369 			throw new ResourceNotFoundException(resourceName);
370 		}
371 
372 		return is;
373 	}

375 	/*
376 	 * (non-Javadoc)
377 	 * 
378 	 * @see
379 	 * net.jawr.web.resource.handler.reader.ResourceBrowser#getResourceNames
380 	 * (java.lang.String)
381 	 */
382 	public Set<String> getResourceNames(String dirName) {
383 		Set<String> resourceNames = new TreeSet<String>();
384 		for (Iterator<ResourceBrowser> iterator = resourceInfoProviders
385 				.iterator(); iterator.hasNext();) {
386 			ResourceBrowser rsBrowser = iterator.next();
387 			if (generatorRegistry.isPathGenerated(dirName)) {
388 				if (rsBrowser instanceof ResourceGenerator) {
389 					ResourceGenerator rsGeneratorBrowser = (ResourceGenerator) rsBrowser;
390 					if (rsGeneratorBrowser.getResolver().matchPath(dirName)) {
391 						resourceNames.addAll(rsBrowser
392 								.getResourceNames(dirName));
393 						break;
394 					}
395 				}
396 			} else {
397 				if (!(rsBrowser instanceof ResourceGenerator)) {
398 					resourceNames.addAll(rsBrowser.getResourceNames(dirName));
399 					break;
400 				}
401 			}
402 		}
403 
404 		return resourceNames;
405 	}

407 	/*
408 	 * (non-Javadoc)
409 	 * 
410 	 * @see
411 	 * net.jawr.web.resource.handler.reader.ResourceBrowser#isDirectory(java
412 	 * .lang.String)
413 	 */
414 	public boolean isDirectory(String resourceName) {
415 		boolean result = false;
416 		for (Iterator<ResourceBrowser> iterator = resourceInfoProviders
417 				.iterator(); iterator.hasNext() && !result;) {
418 			ResourceBrowser rsBrowser = iterator.next();
419 			if (generatorRegistry.isPathGenerated(resourceName)) {
420 				if (rsBrowser instanceof ResourceGenerator) {
421 					ResourceGenerator rsGeneratorBrowser = (ResourceGenerator) rsBrowser;
422 					if (rsGeneratorBrowser.getResolver()
423 							.matchPath(resourceName)) {
424 						result = rsBrowser.isDirectory(resourceName);
425 					}
426 				}
427 			} else {
428 				if (!(rsBrowser instanceof ResourceGenerator)) {
429 					result = rsBrowser.isDirectory(resourceName);
430 				}
431 			}
432 		}
433 		return result;
434 	}
