32  	// The line separator string of the operating system.
33  	private static final String systemLineSeparator = System
34  			.getProperty("line.separator");

36  	// Mapping table from 6-bit nibbles to Base64 characters.
37  	private static char[] map1 = new char[64];
38  	static {
39  		int i = 0;
40  		for (char c = 'A'; c <= 'Z'; c++)
41  			map1[i++] = c;
42  		for (char c = 'a'; c <= 'z'; c++)
43  			map1[i++] = c;
44  		for (char c = '0'; c <= '9'; c++)
45  			map1[i++] = c;
46  		map1[i++] = '+';
47  		map1[i++] = '/';
48  	}
49  
50  	// Mapping table from Base64 characters to 6-bit nibbles.
51  	private static byte[] map2 = new byte[128];
52  	static {
53  		for (int i = 0; i < map2.length; i++)
54  			map2[i] = -1;
55  		for (int i = 0; i < 64; i++)
56  			map2[map1[i]] = (byte) i;
57  	}

84  	/***
85  	 * Encodes a byte array into Base 64 format and breaks the output into
86  	 * lines.
87  	 * 
88  	 * @param in
89  	 *            An array containing the data bytes to be encoded.
90  	 * @param iOff
91  	 *            Offset of the first byte in <code>in</code> to be processed.
92  	 * @param iLen
93  	 *            Number of bytes to be processed in <code>in</code>, starting
94  	 *            at <code>iOff</code>.
95  	 * @param lineLen
96  	 *            Line length for the output data. Should be a multiple of 4.
97  	 * @param lineSeparator
98  	 *            The line separator to be used to separate the output lines.
99  	 * @return A String containing the Base64 encoded data, broken into lines.
100 	 */
101 	public static String encodeLines(byte[] in, int iOff, int iLen,
102 			int lineLen, String lineSeparator) {
103 		int blockLen = (lineLen * 3) / 4;
104 		if (blockLen <= 0)
105 			throw new IllegalArgumentException();
106 		int lines = (iLen + blockLen - 1) / blockLen;
107 		int bufLen = ((iLen + 2) / 3) * 4 + lines * lineSeparator.length();
108 		StringBuffer buf = new StringBuffer(bufLen);
109 		int ip = 0;
110 		while (ip < iLen) {
111 			int l = Math.min(iLen - ip, blockLen);
112 			buf.append(encode(in, iOff + ip, l));
113 			buf.append(lineSeparator);
114 			ip += l;
115 		}
116 		return buf.toString();
117 	}

145 	/***
146 	 * Encodes a byte array into Base64 format. No blanks or line breaks are
147 	 * inserted in the output.
148 	 * 
149 	 * @param in
150 	 *            An array containing the data bytes to be encoded.
151 	 * @param iOff
152 	 *            Offset of the first byte in <code>in</code> to be processed.
153 	 * @param iLen
154 	 *            Number of bytes to process in <code>in</code>, starting at
155 	 *            <code>iOff</code>.
156 	 * @return A character array containing the Base64 encoded data.
157 	 */
158 	public static char[] encode(byte[] in, int iOff, int iLen) {
159 		int oDataLen = (iLen * 4 + 2) / 3; // output length without padding
160 		int oLen = ((iLen + 2) / 3) * 4; // output length including padding
161 		char[] out = new char[oLen];
162 		int ip = iOff;
163 		int iEnd = iOff + iLen;
164 		int op = 0;
165 		while (ip < iEnd) {
166 			int i0 = in[ip++] & 0xff;
167 			int i1 = ip < iEnd ? in[ip++] & 0xff : 0;
168 			int i2 = ip < iEnd ? in[ip++] & 0xff : 0;
169 			int o0 = i0 >>> 2;
170 			int o1 = ((i0 & 3) << 4) | (i1 >>> 4);
171 			int o2 = ((i1 & 0xf) << 2) | (i2 >>> 6);
172 			int o3 = i2 & 0x3F;
173 			out[op++] = map1[o0];
174 			out[op++] = map1[o1];
175 			out[op] = op < oDataLen ? map1[o2] : '=';
176 			op++;
177 			out[op] = op < oDataLen ? map1[o3] : '=';
178 			op++;
179 		}
180 		return out;
181 	}

197 	/***
198 	 * Decodes a byte array from Base64 format and ignores line separators, tabs
199 	 * and blanks. CR, LF, Tab and Space characters are ignored in the input
200 	 * data. This method is compatible with
201 	 * <code>sun.misc.BASE64Decoder.decodeBuffer(String)</code>.
202 	 * 
203 	 * @param s
204 	 *            A Base64 String to be decoded.
205 	 * @return An array containing the decoded data bytes.
206 	 * @throws IllegalArgumentException
207 	 *             If the input is not valid Base64 encoded data.
208 	 */
209 	public static byte[] decodeLines(String s) {
210 		char[] buf = new char[s.length()];
211 		int p = 0;
212 		for (int ip = 0; ip < s.length(); ip++) {
213 			char c = s.charAt(ip);
214 			if (c != ' ' && c != '\r' && c != '\n' && c != '\t')
215 				buf[p++] = c;
216 		}
217 		return decode(buf, 0, p);
218 	}

248 	/***
249 	 * Decodes a byte array from Base64 format. No blanks or line breaks are
250 	 * allowed within the Base64 encoded input data.
251 	 * 
252 	 * @param in
253 	 *            A character array containing the Base64 encoded data.
254 	 * @param iOff
255 	 *            Offset of the first character in <code>in</code> to be
256 	 *            processed.
257 	 * @param iLen
258 	 *            Number of characters to process in <code>in</code>, starting
259 	 *            at <code>iOff</code>.
260 	 * @return An array containing the decoded data bytes.
261 	 * @throws IllegalArgumentException
262 	 *             If the input is not valid Base64 encoded data.
263 	 */
264 	public static byte[] decode(char[] in, int iOff, int iLen) {
265 		if (iLen % 4 != 0)
266 			throw new IllegalArgumentException(
267 					"Length of Base64 encoded input string is not a multiple of 4.");
268 		while (iLen > 0 && in[iOff + iLen - 1] == '=')
269 			iLen--;
270 		int oLen = (iLen * 3) / 4;
271 		byte[] out = new byte[oLen];
272 		int ip = iOff;
273 		int iEnd = iOff + iLen;
274 		int op = 0;
275 		while (ip < iEnd) {
276 			int i0 = in[ip++];
277 			int i1 = in[ip++];
278 			int i2 = ip < iEnd ? in[ip++] : 'A';
279 			int i3 = ip < iEnd ? in[ip++] : 'A';
280 			if (i0 > 127 || i1 > 127 || i2 > 127 || i3 > 127)
281 				throw new IllegalArgumentException(
282 						"Illegal character in Base64 encoded data.");
283 			int b0 = map2[i0];
284 			int b1 = map2[i1];
285 			int b2 = map2[i2];
286 			int b3 = map2[i3];
287 			if (b0 < 0 || b1 < 0 || b2 < 0 || b3 < 0)
288 				throw new IllegalArgumentException(
289 						"Illegal character in Base64 encoded data.");
290 			int o0 = (b0 << 2) | (b1 >>> 4);
291 			int o1 = ((b1 & 0xf) << 4) | (b2 >>> 2);
292 			int o2 = ((b2 & 3) << 6) | b3;
293 			out[op++] = (byte) o0;
294 			if (op < oLen)
295 				out[op++] = (byte) o1;
296 			if (op < oLen)
297 				out[op++] = (byte) o2;
298 		}
299 		return out;
300 	}
