211 	 */
212 	public void addPostProcessVariant(String variantType, VariantSet variantSet){
213 		
214 		Map variantMap = new HashMap();
215 		variantMap.put(variantType, variantSet);
216 		addPostProcessVariant(variantMap);
217 	}
218 	
219 	/***
220 	 * Add a post process variant
221 	 * @param variantType the variant type
222 	 * @param variantSet the variant set
223 	 */
224 	public void addPostProcessVariant(Map variants){
225 		
226 		if(!searchingPostProcessorVariants){
227 			throw new BundlingProcessException("You are not allowed to define post process variants if we are not searching for post processor variants.");
228 		}
229 		postProcessVariants = VariantUtils.concatVariants(postProcessVariants, variants);
230 	}
