33  /***
34   * This class defines the URL rewriter for the Css post processor
35   *  
36   * @author Ibrahim Chaehoi
37   */
38  public class PostProcessorCssImageUrlRewriter extends CssImageUrlRewriter {
39  	
40  	/*** Logger */
41  	private static Logger LOGGER = Logger.getLogger(PostProcessorCssImageUrlRewriter.class);
42  	
43  	/*** The bundle processing status */
44  	private BundleProcessingStatus status;
45  	
46  	/***
47  	 * Constructor
48  	 * @param status the bundle processing status
49  	 */
50  	public PostProcessorCssImageUrlRewriter(BundleProcessingStatus status) {
51  		super(status.getJawrConfig());
52  		this.status = status;
53  	}
54  	
55  	
56  	/* (non-Javadoc)
57  	 * @see net.jawr.web.resource.bundle.css.CssImageUrlRewriter#getRewrittenImagePath(java.lang.String, java.lang.String, java.lang.String)
58  	 */
59  	protected String getRewrittenImagePath(String originalCssPath,
60  			String newCssPath, String url) throws IOException {
61  		
62  		JawrConfig jawrConfig = status.getJawrConfig();
63  		
64  		// Retrieve the image servlet mapping
65  		ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) jawrConfig.getContext().getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
66  		String imgServletPath = "";
67  		
68  		if(imgRsHandler != null){
69  			imgServletPath = PathNormalizer.asPath(imgRsHandler.getJawrConfig().getServletMapping());
70  		}
71  		
72  		String imgUrl = null;
73  		
74  		// Retrieve the current CSS file from which the CSS image is referenced
75  		String currentCss = originalCssPath;
76  		boolean generatedImg = false;
77  		if(imgRsHandler != null){
78  			GeneratorRegistry imgRsGeneratorRegistry = imgRsHandler.getJawrConfig().getGeneratorRegistry();
79  			generatedImg = imgRsGeneratorRegistry.isGeneratedImage(url);
80  		}
81  		
82  		boolean cssGeneratorIsHandleCssImage = isCssGeneratorHandlingCssImage(currentCss, status);
83  		
84  		String rootPath = currentCss;
85  		
86  		// If the CSS image is taken from the classpath, add the classpath cache prefix
87  		if(generatedImg || cssGeneratorIsHandleCssImage){
88  			
89  			String tempUrl = url;
90  			
91  			// If it's a classpath CSS, the url of the CSS image is defined relatively to it.
92  			if(cssGeneratorIsHandleCssImage && !generatedImg){
93  				tempUrl = PathNormalizer.concatWebPath(rootPath, url);
94  			}
95  
96  			// generate image cache URL
97  			imgUrl = rewriteURL(status, tempUrl, imgServletPath, newCssPath, imgRsHandler);
98  		}else{
99  			
100 			if(jawrConfig.getGeneratorRegistry().isPathGenerated(rootPath)){
101 				rootPath = rootPath.substring(rootPath.indexOf(GeneratorRegistry.PREFIX_SEPARATOR)+1);
102 			}
103 				
104 			// Generate the image URL from the current CSS path
105 			imgUrl = PathNormalizer.concatWebPath(rootPath, url);
106 			imgUrl = rewriteURL(status, imgUrl, imgServletPath, newCssPath, imgRsHandler);
107 		}
108 		
109 		// This following condition should never be true. 
110 		// If it does, it means that the image path is wrongly defined.
111 		if(imgUrl == null){
112 			LOGGER.error("The CSS image path for '"+url+"' defined in '"+currentCss+"' is out of the application context. Please check your CSS file.");
113 		}
114 		
115 		return imgUrl;
116 	}

118 	/***
119 	 * Checks if the Css generator associated to the Css resource path handle also the Css image resources. 
120 	 * @param currentCss the CSS resource path
121 	 * @param status the status
122 	 * @return true if the Css generator associated to the Css resource path handle also the Css image resources. 
123 	 */
124 	private boolean isCssGeneratorHandlingCssImage(String currentCss, BundleProcessingStatus status) {
125 		return status.getJawrConfig().getGeneratorRegistry().isHandlingCssImage(currentCss);
126 	}
127 	
128 	/***
129 	 * Rewrites the image URL
130 	 * @param status the bundle processing status
131 	 * @param url the image URL
132 	 * @param imgServletPath the image servlet path
133 	 * @param newCssPath the new Css path
134 	 * @param imgRsHandler the image resource handler
135 	 * @return the rewritten image URL
136 	 * @throws IOException if an IOException occurs
137 	 */
138 	protected String rewriteURL(BundleProcessingStatus status, String url, String imgServletPath, String newCssPath, ImageResourcesHandler imgRsHandler) throws IOException {
139 		
140 		String imgUrl = addCacheBuster(status, url, imgRsHandler);
141 		
142 		// Add image servlet path in the URL, if it's defined
143 		if(StringUtils.isNotEmpty(imgServletPath)){
144 			imgUrl = imgServletPath+JawrConstant.URL_SEPARATOR+imgUrl;
145 		}
146 		
147 		imgUrl = PathNormalizer.asPath(imgUrl);
148 		return PathNormalizer.getRelativeWebPath(PathNormalizer
149 				.getParentPath(newCssPath), imgUrl);
150 	}
151 	
152 	/***
153 	 * Adds the cache buster to the CSS image
154 	 * @param status the bundle processing status
155 	 * @param url the URL of the image
156 	 * @param imgRsHandler the image resource handler
157 	 * @return the url of the CSS image with a cache buster
158 	 * @throws IOException if an IO exception occurs
159 	 */
160 	private String addCacheBuster(BundleProcessingStatus status, String url, ImageResourcesHandler imgRsHandler) throws IOException {
161 		
162 		// Try to retrieve the from the bundle processing cache
163 		Map imageMapping = (Map) status.getData(JawrConstant.POST_PROCESSING_CTX_JAWR_IMAGE_MAPPING);
164 		String newUrl = null;
165 		if(imageMapping != null){
166 			newUrl = (String) imageMapping.get(url);
167 			if(newUrl != null){
168 				return newUrl;
169 			}
170 		}
171 		
172 		// Try to retrieve the from the image resource handler cache
173 		if(imgRsHandler != null){
174 			newUrl = imgRsHandler.getCacheUrl(url);
175 			if(newUrl != null){
176 				return newUrl;
177 			}
178 			// Retrieve the new URL with the cache prefix
179 			try {
180 				newUrl = CheckSumUtils.getCacheBustedUrl(url, imgRsHandler.getRsReaderHandler(), imgRsHandler.getJawrConfig());
181 			} catch (ResourceNotFoundException e) {
182 				LOGGER.info("Impossible to define the checksum for the resource '"+url+"'. ");
183 				return url;
184 			} catch (IOException e) {
185 				LOGGER.info("Impossible to define the checksum for the resource '"+url+"'.");
186 				return url;
187 			}
188 			
189 			imgRsHandler.addMapping(url, newUrl);
190 			
191 		}else{
192 			newUrl = url;
193 		}
194 		
195 		// Set the result in a cache, so we will not search for it the next time
196 		if(imageMapping == null){
197 			imageMapping = new HashMap();
198 			status.putData(JawrConstant.POST_PROCESSING_CTX_JAWR_IMAGE_MAPPING, imageMapping);
199 		}
200 		imageMapping.put(url, newUrl);
201 		
202 		return newUrl;
203 	}
