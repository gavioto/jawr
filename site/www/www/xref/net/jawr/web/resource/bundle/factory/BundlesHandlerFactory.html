135 	/***
136 	 * Build a ResourceBundlesHandler. Must be invoked after setting at least the ResourceHandler.
137 	 * 
138 	 * @param jawrConfig the jawr config
139 	 * @return the resource bundles handler
140 	 * @throws DuplicateBundlePathException if two bundles are defined with the same path
141 	 * @throws BundleDependencyException if an error exists in the dependency definition
142 	 */
143 	public ResourceBundlesHandler buildResourceBundlesHandler()
144 			throws DuplicateBundlePathException, BundleDependencyException {
145 		if (LOGGER.isInfoEnabled())
146 			LOGGER.info("Building resources handler... ");
147 
148 		// Ensure state is correct
149 		if (null == jawrConfig)
150 			throw new IllegalStateException(
151 					"Must set the JawrConfig for this factory before invoking buildResourceBundlesHandler(). ");
152 
153 		if (null == resourceReaderHandler)
154 			throw new IllegalStateException(
155 					"Must set the resourceHandler for this factory before invoking buildResourceBundlesHandler(). ");
156 		if (useSingleResourceFactory && null == singleFileBundleName)
157 			throw new IllegalStateException(
158 					"Must set the singleFileBundleName when useSingleResourceFactory is set to true. Please check the documentation. ");
159 
160 		// Initialize custom postprocessors before using the factory to build the postprocessing chains
161 		if (null != customPostprocessors)
162 			chainFactory.setCustomPostprocessors(customPostprocessors);
163 
164 		// List of bundles
165 		List resourceBundles = new ArrayList();
166 
167 		boolean processBundle = !jawrConfig.getUseBundleMapping()
168 				|| !resourceBundleHandler.isExistingMappingFile();
169 		if (processBundle) {
170 			initResourceBundles(resourceBundles);
171 		} else {
172 			initResourceBundlesFromFullMapping(resourceBundles);
173 		}
174 
175 		// Build the postprocessor for bundles
176 		ResourceBundlePostProcessor processor = null;
177 		if (null == this.globalPostProcessorKeys){
178 			processor = this.chainFactory.buildDefaultProcessorChain();
179 		}else{
180 			processor = this.chainFactory
181 					.buildPostProcessorChain(globalPostProcessorKeys);
182 		}
183 		
184 		// Build the postprocessor to use on resources before adding them to the bundle.
185 		ResourceBundlePostProcessor unitProcessor = null;
186 		if (null == this.unitPostProcessorKeys){
187 			unitProcessor = this.chainFactory.buildDefaultUnitProcessorChain();
188 		}else{
189 			unitProcessor = this.chainFactory
190 					.buildPostProcessorChain(unitPostProcessorKeys);
191 		}
192 		
193 		// Build the postprocessor for bundles
194 		ResourceBundlePostProcessor compositeBundleProcessor = null;
195 		if (null == this.globalCompositePostProcessorKeys){
196 			compositeBundleProcessor = this.chainFactory.buildDefaultCompositeProcessorChain();
197 		}else{
198 			compositeBundleProcessor = this.chainFactory
199 					.buildPostProcessorChain(globalCompositePostProcessorKeys);
200 		}
201 		
202 		// Build the postprocessor to use on resources before adding them to the bundle.
203 		ResourceBundlePostProcessor compositeUnitProcessor = null;
204 		if (null == this.unitCompositePostProcessorKeys){
205 			compositeUnitProcessor = this.chainFactory.buildDefaultUnitCompositeProcessorChain();
206 		}else{
207 			compositeUnitProcessor = this.chainFactory
208 					.buildPostProcessorChain(unitCompositePostProcessorKeys);
209 		}
210 		
211 		
212 		// Build the resource type processor to use on resources.
213 		// Initialize custom postprocessors before using the factory to build the postprocessing chains
214 		if (null != customGlobalPreprocessors)
215 			resourceTypeChainFactory.setCustomGlobalPreprocessors(customGlobalPreprocessors);
216 
217 		GlobalPreprocessor resourceTypeProcessor = null;
218 		if (null == this.resourceTypeProcessorKeys)
219 			resourceTypeProcessor = this.resourceTypeChainFactory.buildDefaultProcessorChain();
220 		else
221 			resourceTypeProcessor = this.resourceTypeChainFactory
222 					.buildProcessorChain(resourceTypeProcessorKeys);
223 
224 		// Build the handler
225 		ResourceBundlesHandler collector = new ResourceBundlesHandlerImpl(
226 				resourceBundles, resourceReaderHandler, resourceBundleHandler, jawrConfig, processor,
227 				unitProcessor, compositeBundleProcessor, compositeUnitProcessor, resourceTypeProcessor);
228 
229 		// Use the cached proxy if specified when debug mode is off.
230 		if (useInMemoryCache && !jawrConfig.isDebugModeOn())
231 			collector = new CachedResourceBundlesHandler(collector);
232 
233 		collector.initAllBundles();
234 
235 		return collector;
236 	}

242 
243 		if (LOGGER.isInfoEnabled()){
244 			LOGGER.info("Building bundles from the full bundle mapping. The bundles will not be processed.");
245 		}
246 		Properties mappingProperties = resourceBundleHandler.getJawrBundleMapping();
247 		FullMappingPropertiesBasedBundlesHandlerFactory factory = new FullMappingPropertiesBasedBundlesHandlerFactory(resourceType,

253 	/***
254 	 * Initialize the resource bundles
255 	 * 
256 	 * @param resourceBundles the resource bundles
257 	 * @throws DuplicateBundlePathException if two bundles are defined with the same path 
258 	 * @throws BundleDependencyException if an error exists in the dependency definition
259 	 */
260 	private void initResourceBundles(List resourceBundles)
261 			throws DuplicateBundlePathException, BundleDependencyException  {
262 
263 		// Create custom defined bundles
264 		bundleDefinitionsWithDependencies = new HashSet();
265 		if (null != bundleDefinitions) {
266 			if (LOGGER.isInfoEnabled())
267 				LOGGER.info("Adding custom bundle definitions. ");
268 			for (Iterator it = bundleDefinitions.iterator(); it.hasNext();) {
269 				ResourceBundleDefinition def = (ResourceBundleDefinition) it
270 						.next();
271 
272 				// If this is a composite bundle
273 				if (def.isComposite()) {
274 					List childBundles = new ArrayList();
275 					for (Iterator childIterator = def.getChildren().iterator(); childIterator
276 							.hasNext();) {
277 						ResourceBundleDefinition child = (ResourceBundleDefinition) childIterator
278 								.next();
279 						childBundles.add(buildResourcebundle(child));
280 					}
281 					resourceBundles.add(buildCompositeResourcebundle(def,
282 							childBundles));
283 				} else
284 					resourceBundles.add(buildResourcebundle(def));
285 			}
286 		}
287 
288 		// Use the dirmapper if specified
289 		if (useDirMapperFactory) {
290 			if (LOGGER.isInfoEnabled())
291 				LOGGER.info("Using ResourceBundleDirMapper. ");
292 
293 			 ResourceBundleDirMapper dirFactory = new ResourceBundleDirMapper(
294 					baseDir, resourceReaderHandler, resourceBundles, fileExtension,
295 					excludedDirMapperDirs);
296 			Map mappings = dirFactory.getBundleMapping();
297 			for (Iterator it = mappings.entrySet().iterator(); it.hasNext();) {
298 				Entry entry = (Entry) it.next();
299 				resourceBundles.add(buildDirMappedResourceBundle((String) entry.getKey(),
300 						(String) entry.getValue()));
301 			}
302 		}
303 
304 		if (this.scanForOrphans) {
305 			// Add all orphan bundles
306 			OrphanResourceBundlesMapper orphanFactory = new OrphanResourceBundlesMapper(
307 					baseDir, resourceReaderHandler, jawrConfig.getGeneratorRegistry(),
308 					resourceBundles, fileExtension);
309 			List orphans = orphanFactory.getOrphansList();
310 
311 			// Orphans may be added separately or as one single resource bundle.
312 			if (useSingleResourceFactory) {
313 				// Add extension to the filename
314 				if (!singleFileBundleName.endsWith(fileExtension))
315 					singleFileBundleName += fileExtension;
316 
317 				if (LOGGER.isInfoEnabled())
318 					LOGGER
319 							.info("Building bundle of orphan resources with the name: "
320 									+ singleFileBundleName);
321 
322 				resourceBundles.add(buildOrphansResourceBundle(
323 						singleFileBundleName, orphans));
324 
325 			} else {
326 				if (LOGGER.isInfoEnabled())
327 					LOGGER.info("Creating mappings for orphan resources. ");
328 				for (Iterator it = orphans.iterator(); it.hasNext();) {
329 					resourceBundles.add(buildOrphanResourceBundle((String) it
330 							.next()));
331 				}
332 			}
333 		} else if (LOGGER.isDebugEnabled()) {
334 			LOGGER.debug("Skipping orphan file auto processing. ");
335 			if ("".equals(jawrConfig.getServletMapping()))
336 				LOGGER
337 						.debug("Note that there is no specified mapping for Jawr "
338 								+ "(it has been seet to serve *.js or *.css requests). "
339 								+ "The orphan files will become unreachable through the server.");
340 		}
341 		
342 		// Initialize bundle dependencies
343 		for (Iterator iterator = bundleDefinitionsWithDependencies.iterator(); iterator.hasNext();) {
344 			ResourceBundleDefinition definition = (ResourceBundleDefinition) iterator.next();
345 			JoinableResourceBundle bundle = getBundleFromName(definition.getBundleName(), resourceBundles);
346 			if(bundle != null){
347 				bundle.setDependencies(getBundleDependencies(definition, resourceBundles));
348 			}
349 		}
350 	}

352 	/***
353 	 * Returns a bundle from its name
354 	 * @param name the bundle name
355 	 * @param bundles the list of bundle
356 	 * @return a bundle from its name
357 	 */
358 	private JoinableResourceBundle getBundleFromName(String name, List bundles){
359 		
360 		JoinableResourceBundle bundle = null;
361 		for (Iterator iterator = bundles.iterator(); iterator.hasNext();) {
362 			JoinableResourceBundle aBundle = (JoinableResourceBundle) iterator.next();
363 			if(aBundle.getName().equals(name)){
364 				bundle = aBundle;
365 				break;
366 			}
367 		}
368 		
369 		return bundle;
370 	}

373 	 * Build a Composite resource bundle using a ResourceBundleDefinition
374 	 * 
375 	 * @param definition the bundle definition
376 	 * @param childBundles the list of child bundles
377 	 * @return a Composite resource bundle
378 	 */
379 	private JoinableResourceBundle buildCompositeResourcebundle(
380 			ResourceBundleDefinition definition, List childBundles) {
381 
382 		if (LOGGER.isDebugEnabled())
383 			LOGGER.debug("Init composite bundle with id:"
384 					+ definition.getBundleId());
385 
386 		InclusionPattern include = new InclusionPattern(definition.isGlobal(),
387 				definition.getInclusionOrder(), definition.isDebugOnly(),
388 				definition.isDebugNever());
389 
390 		CompositeResourceBundle composite = new CompositeResourceBundle(
391 				definition.getBundleId(), definition.getBundleName(), 
392 				childBundles, include, resourceReaderHandler, fileExtension,
393 				jawrConfig);
394 		if (null != definition.getBundlePostProcessorKeys())
395 			composite.setBundlePostProcessor(chainFactory
396 					.buildPostProcessorChain(definition
397 							.getBundlePostProcessorKeys()));
398 
399 		if (null != definition.getUnitaryPostProcessorKeys())
400 			composite.setUnitaryPostProcessor(chainFactory
401 					.buildPostProcessorChain(definition
402 							.getUnitaryPostProcessorKeys()));
403 
404 		if (null != definition.getIeConditionalExpression())
405 			composite.setExplorerConditionalExpression(definition
406 					.getIeConditionalExpression());
407 
408 		if (null != definition.getAlternateProductionURL())
409 			composite.setAlternateProductionURL(definition
410 					.getAlternateProductionURL());
411 		
412 		if (null != definition.getVariants())
413 			composite.setVariants(definition
414 					.getVariants());
415 		
416 		if (null != definition.getDependencies() && !definition.getDependencies().isEmpty())
417 			bundleDefinitionsWithDependencies.add(definition);
418 
419 		return composite;
420 	}

441 				resourceReaderHandler, jawrConfig.getGeneratorRegistry());
442 		if (null != definition.getBundlePostProcessorKeys())
443 			newBundle.setBundlePostProcessor(chainFactory
444 					.buildPostProcessorChain(definition
445 							.getBundlePostProcessorKeys()));
446 
447 		if (null != definition.getUnitaryPostProcessorKeys())
448 			newBundle.setUnitaryPostProcessor(chainFactory
449 					.buildPostProcessorChain(definition
450 							.getUnitaryPostProcessorKeys()));
451 
452 		if (null != definition.getIeConditionalExpression())
453 			newBundle.setExplorerConditionalExpression(definition

455 
456 		if (null != definition.getVariants())
457 			newBundle.setVariants(definition

459 		
460 		if (null != definition.getAlternateProductionURL())

463 
464 		if (null != definition.getDependencies() && !definition.getDependencies().isEmpty()){
465 			
466 			bundleDefinitionsWithDependencies.add(definition);
467 		}
468 			
469 		return newBundle;

472 	/***
473 	 * Returns the bundle dependencies from the resource bundle definition
474 	 * 
475 	 * @param definition the resource definition
476 	 * @param bundles the list of bundles
477 	 * 
478 	 * @throws BundleDependencyException  if an error exists in the dependency definition
479 	 */
480 	private List getBundleDependencies(ResourceBundleDefinition definition, List bundles) throws BundleDependencyException {
481 
482 		List dependencies = new ArrayList();
483 		List processedBundles = new ArrayList();
484 		if(definition.isGlobal() && definition.getDependencies() != null && !definition.getDependencies().isEmpty()){
485 			throw new BundleDependencyException(definition.getBundleName(), "The dependencies property is not allowed for global bundles. Please use the order property " +
486 					"to define the import order.");
487 		}
488 		initBundleDependencies(definition.getBundleName(), definition, dependencies, processedBundles, bundles);
489 		return dependencies;
490 	}

492 	/***
493 	 * Initialize the bundle dependencies
494 	 * 
495 	 * @param rootBundleDefinition the name of the bundle, whose is initalized
496 	 * @param definition the current resource bundle definition
497 	 * @param bundleDependencies the bundle dependencies
498 	 * @param processedBundles the list of bundles already processed during the dependency resolution
499 	 * @param bundles the list of reference bundles
500 	 * 
501 	 * @throws BundleDependencyException if an error exists in the dependency definition
502 	 */
503 	private void initBundleDependencies(String rootBundleDefinition, ResourceBundleDefinition definition,
504 			List bundleDependencies, List processedBundles, List bundles) throws BundleDependencyException {
505 
506 		List bundleDefDependencies = definition.getDependencies();
507 		if(definition.isGlobal()){
508 			if(LOGGER.isInfoEnabled()){
509 				LOGGER.info("The global bundle '"+definition.getBundleName()+"' belongs to the dependencies of '"+rootBundleDefinition+"'." +
510 						"As it's a global bundle, it will not be defined as part of the dependencies.");
511 			}
512 			return;
513 		}
514 		
515 		if (bundleDefDependencies != null && !bundleDefDependencies.isEmpty()) {
516 			if (processedBundles.contains(definition.getBundleName())) {
517 				throw new BundleDependencyException(rootBundleDefinition, "There is a circular dependency. The bundle in conflict is '"+definition.getBundleName()+"'");	
518 			
519 			} else {
520 			
521 				processedBundles.add(definition.getBundleName());
522 				for (Iterator iterator = bundleDefDependencies.iterator(); iterator
523 						.hasNext();) {
524 					String dependency = (String) iterator.next();
525 					for (Iterator itDep = bundleDefinitions.iterator(); itDep
526 							.hasNext();) {
527 						ResourceBundleDefinition dependencyBundle = (ResourceBundleDefinition) itDep
528 								.next();
529 						String dependencyBundleName = dependencyBundle.getBundleName();
530 						if (dependencyBundleName.equals(dependency)) {
531 
532 							if (!bundleDependencies.contains(dependencyBundleName)){
533 								
534 								if(!processedBundles.contains(dependencyBundleName)) {
535 									initBundleDependencies(rootBundleDefinition, dependencyBundle,
536 											bundleDependencies, processedBundles, bundles);
537 									bundleDependencies.add(getBundleFromName(dependencyBundleName, bundles));
538 								}else{
539 									throw new BundleDependencyException(rootBundleDefinition, "There is a circular dependency. The bundle in conflict is '"+dependencyBundleName+"'");
540 								}
541 							} else {
542 								if(LOGGER.isInfoEnabled()){
543 									LOGGER.info("The bundle '"
544 											+ dependencyBundle.getBundleId()
545 											+ "' occurs multiple time in the dependencies hierarchy of the bundle '"+rootBundleDefinition+"'.");	
546 								}
547 							}
548 						}
549 					}
550 				}
551 			}
552 		}
553 	}

571 	/***
572 	 * Generates the bundle ID from the bundle name
573 	 * 
574 	 * @param bundleId the bundle name
575 	 * @return the generated bundle ID
576 	 */
577 	private String generateBundleNameFromBundleId(String bundleId) {
578 		String bundleName = bundleId;
579 		if(bundleName.startsWith("/")){
580 			bundleName = bundleName.substring(1);
581 		}
582 		int idxExtension = FileNameUtils.indexOfExtension(bundleName);
583 		if(idxExtension != -1){
584 			bundleName = bundleName.substring(0, idxExtension);
585 		}
586 		return bundleName.replaceAll("/", "_");
587 	}

607 	 * Build a non-global, single-file resource bundle for orphans.

613 		String mapping = orphanPath;// .startsWith("/") ? orphanPath.substring(0) : orphanPath;
614
