83      /*
84  	 * (non-Javadoc)
85  	 * 
86  	 * @see javax.servlet.jsp.tagext.TagSupport#doStartTag()
87  	 */
88  	public int doStartTag() throws JspException {
89  
90  		// Renderer istance which takes care of generating the response
91  		this.renderer = createRenderer();
92  
93  		HttpServletRequest request = (HttpServletRequest) pageContext
94  				.getRequest();
95  
96  		// set the debug override
97  		RendererRequestUtils.setRequestDebuggable(request, renderer
98  				.getBundler().getConfig());
99  
100 		try {
101 			BundleRendererContext ctx = RendererRequestUtils
102 					.getBundleRendererContext(request, renderer);
103 			
104 			renderer.renderBundleLinks(src, ctx, pageContext.getOut());
105 			
106 		} catch (IOException ex) {
107 			throw new JspException(
108 					"Unexpected IOException when writing script tags for path "
109 							+ src, ex);
110 		}
111 
112 		return SKIP_BODY;
113 	}
114 	
115     /* (non-Javadoc)
116 	 * @see net.jawr.web.taglib.AbstractResourceBundleTag#createRenderer()
117 	 */
118 	protected BundleRenderer createRenderer() {
119 		if(null == pageContext.getServletContext().getAttribute(JawrConstant.CSS_CONTEXT_ATTRIBUTE))
120 			throw new IllegalStateException("ResourceBundlesHandler not present in servlet context. Initialization of Jawr either failed or never occurred.");
121 
122 		ResourceBundlesHandler rsHandler = (ResourceBundlesHandler) pageContext.getServletContext().getAttribute(JawrConstant.CSS_CONTEXT_ATTRIBUTE);
123 		return  new CSSHTMLBundleLinkRenderer(rsHandler, this.useRandomParam, this.media, this.alternate, this.displayAlternate, this.title);
124 	}
125

135 	}
