138 		// init registry

157 		if (null != mapping) {
158 			jawrConfig.setServletMapping(mapping);

166 		rsBundleHandler = initResourceBundleHandler();

176 		}

178 		if (jawrConfig.getUseBundleMapping()) {
179 			bundleMapping = rsBundleHandler.getJawrBundleMapping();

191 		if (LOGGER.isDebugEnabled()) {

195

203 	/***
204 	 * Initialize the image mapping of the image resources handler
205 	 * 
206 	 * @param imgRsHandler
207 	 *            the image resources handler
208 	 */
209 	private void initImageMapping(ImageResourcesHandler imgRsHandler) {
210 
211 		if (jawrConfig.getUseBundleMapping()
212 				&& rsBundleHandler.isExistingMappingFile()) {
213 
214 			// Initialize the image mapping
215 			Iterator mapIterator = bundleMapping.keySet().iterator();
216 			while (mapIterator.hasNext()) {
217 				String key = (String) mapIterator.next();
218 				imgRsHandler.addMapping(key, bundleMapping.getProperty(key));
219 			}
220 
221 		} else {
222 			// Create a resource handler to read files from the WAR archive or
223 			// exploded dir.
224 			String imageResourcesDefinition = jawrConfig
225 					.getImageResourcesDefinition();
226 			if (imageResourcesDefinition != null) {
227 
228 				StringTokenizer tokenizer = new StringTokenizer(
229 						imageResourcesDefinition, ",");
230 				while (tokenizer.hasMoreTokens()) {
231 					String pathMapping = tokenizer.nextToken();
232 
233 					// path is a generated image and ends with an image
234 					// extension
235 					if (generatorRegistry.isGeneratedImage(pathMapping)
236 							&& hasImageFileExtension(pathMapping)) {
237 
238 						addImagePath(imgRsHandler, pathMapping);
239 					}
240 					// path ends in /, the folder is included without subfolders
241 					else if (pathMapping.endsWith("/")) {
242 						addItemsFromDir(imgRsHandler, pathMapping, false);
243 					}
244 					// path ends in /, the folder is included with all
245 					// subfolders
246 					else if (pathMapping.endsWith("/**")) {
247 						addItemsFromDir(imgRsHandler, pathMapping.substring(0,
248 								pathMapping.lastIndexOf("**")), true);
249 					} else if (hasImageFileExtension(pathMapping)) {
250 						addImagePath(imgRsHandler, pathMapping);
251 					} else
252 						LOGGER
253 								.warn("Wrong mapping ["
254 										+ pathMapping
255 										+ "] for image bundle. Please check configuration. ");
256 				}
257 			}
258 		}
259 
260 		// Store the bundle mapping
261 		if (jawrConfig.getUseBundleMapping()
262 				&& !rsBundleHandler.isExistingMappingFile()) {
263 			rsBundleHandler.storeJawrBundleMapping(bundleMapping);
264 		}
265 
266 		if (LOGGER.isDebugEnabled())
267 			LOGGER.debug("Finish creation of map for image bundle");
268 	}

271 	 * Add an image path to the image map
272 	 * 
273 	 * @param imgRsHandler
274 	 *            the image resources handler
275 	 * @param imgPath
276 	 *            the image path
277 	 * @param classPathImg
278 	 *            the flag indicating if the image should be retrieved from
279 	 *            classpath
280 	 */
281 	private void addImagePath(ImageResourcesHandler imgRsHandler, String imgPath) {
282 
283 		try {
284 			String resultPath = CheckSumUtils.getCacheBustedUrl(imgPath,
285 					rsReaderHandler, jawrConfig);
286 			imgRsHandler.addMapping(imgPath, resultPath);
287 			bundleMapping.put(imgPath, resultPath);
288 		} catch (IOException e) {
289 			LOGGER.error(
290 					"An exception occurs while defining the mapping for the file : "
291 							+ imgPath, e);
292 		} catch (ResourceNotFoundException e) {
293 			LOGGER.error("Impossible to define the checksum for the resource '"
294 					+ imgPath
295 					+ "'. Unable to retrieve the content of the file.");
296 		}
297 	}

300 	 * Returns true of the path contains an image file extension
301 	 * 
302 	 * @param path
303 	 *            the path
304 	 * @return the image file extension
305 	 */
306 	private boolean hasImageFileExtension(String path) {
307 		boolean result = false;
308 
309 		int extFileIdx = path.lastIndexOf(".");
310 		if (extFileIdx != -1 && extFileIdx + 1 < path.length()) {
311 			String extension = path.substring(extFileIdx + 1);
312 			result = imgMimeMap.containsKey(extension);
313 		}
314 
315 		return result;
316 	}

318 	/***
319 	 * Adds all the resources within a path to the image map.
320 	 * 
321 	 * @param imgRsHandler
322 	 *            the image resources handler
323 	 * @param dirName
324 	 *            the directory name
325 	 * @param addSubDirs
326 	 *            boolean If subfolders will be included. In such case, every
327 	 *            folder below the path is included.
328 	 */
329 	private void addItemsFromDir(ImageResourcesHandler imgRsHandler,
330 			String dirName, boolean addSubDirs) {
331 		Set resources = rsReaderHandler.getResourceNames(dirName);
332 
333 		if (LOGGER.isDebugEnabled()) {
334 			LOGGER.debug("Adding " + resources.size() + " resources from path ["
335 					+ dirName + "] to image bundle");
336 		}
337 
338 		// Add remaining resources (remaining after sorting, or all if no sort
339 		// file present)
340 		List folders = new ArrayList();
341 		for (Iterator it = resources.iterator(); it.hasNext();) {
342 			String resourceName = (String) it.next();
343 			String resourcePath = PathNormalizer.joinPaths(dirName,
344 					resourceName);
345 			if (hasImageFileExtension(resourceName)) {
346 				addImagePath(imgRsHandler, resourcePath);
347 
348 				if (LOGGER.isDebugEnabled())
349 					LOGGER.debug("Added to item path list:"
350 							+ PathNormalizer.asPath(resourcePath));
351 			} else if (addSubDirs) {
352 
353 				try {
354 					if (rsReaderHandler.isDirectory(resourcePath)) {
355 						folders.add(resourceName);
356 					}
357 				} catch (InvalidPathException e) {
358 					if (LOGGER.isDebugEnabled())
359 						LOGGER
360 								.debug("Enable to define if the following resource is a directory : "
361 										+ PathNormalizer.asPath(resourcePath));
362 				}
363 			}
364 		}
365 
366 		// Add subfolders if requested. Subfolders are added last unless
367 		// specified in sorting file.
368 		if (addSubDirs) {
369 			for (Iterator it = folders.iterator(); it.hasNext();) {
370 				String folderName = (String) it.next();
371 				addItemsFromDir(imgRsHandler, PathNormalizer.joinPaths(dirName,
372 						folderName), true);
373 			}
374 		}
375 	}

377 	/***
378 	 * Handles a resource request.
379 	 * <ul>
380 	 * <li>If the request contains an If-Modified-Since header, the 304 status
381 	 * is set and no data is written to the response</li>
382 	 * <li>If the requested path begins with the gzip prefix, a gzipped version
383 	 * of the resource is served, with the corresponding content-encoding
384 	 * header.</li>
385 	 * <li>Otherwise, the resource is written as text to the response.</li>
386 	 * <li>If the resource is not found, the response satus is set to 404 and no
387 	 * response is written.</li>
388 	 * </ul>
389 	 * 
390 	 * @param requestedPath
391 	 * @param request
392 	 * @param response
393 	 * @throws ServletException
394 	 * @throws IOException
395 	 */
396 	public void processRequest(String requestedPath,
397 			HttpServletRequest request, HttpServletResponse response)
398 			throws ServletException, IOException {
399 
400 		// manual reload request
401 		if (this.jawrConfig.getRefreshKey().length() > 0 && null != request.getParameter("refreshKey")
402 				&& this.jawrConfig.getRefreshKey().equals(request.getParameter("refreshKey"))) {
403 			this.configChanged(propertiesSource.getConfigProperties());
404 		}
405 		
406 		if (LOGGER.isDebugEnabled())
407 			LOGGER.debug("Request received for path:" + requestedPath);
408 
409 		if(handleSpecificRequest(requestedPath, requestedPath, request, response)){
410 			return;
411 		}
412 		
413 		// Handle the strict mode
414 		boolean validBundle = isValidBundle(requestedPath);
415 
416 		// Set the content type
417 		response.setContentType(getContentType(requestedPath, request));
418 		
419 		boolean writeResponseHeader = false;
420 		
421 		// If debug mode is off, check for If-Modified-Since and If-none-match headers and set response caching headers.
422 		if (!this.jawrConfig.isDebugModeOn()) {
423 			// If a browser checks for changes, always respond 'no changes'.
424 			if (validBundle && (null != request.getHeader(IF_MODIFIED_SINCE_HEADER) || null != request.getHeader(IF_NONE_MATCH_HEADER))) {
425 				response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
426 				if (LOGGER.isDebugEnabled())
427 					LOGGER.debug("Returning 'not modified' header. ");
428 				return;
429 			}
430 
431 			if(validBundle){
432 				// Add caching headers
433 				setResponseHeaders(response);
434 			}else{
435 				
436 				writeResponseHeader = illegalBundleRequestHandler.writeResponseHeader(requestedPath, request, response);
437 				if(!writeResponseHeader){
438 					// Add caching headers
439 					setResponseHeaders(response);
440 				}
441 			}
442 		}
443 		
444 		// Returns the real file path
445 		String filePath = getRealFilePath(requestedPath);
446 
447 		try {
448 			if(isValidRequestedPath(filePath) && (validBundle || illegalBundleRequestHandler.canWriteContent(requestedPath, request))){
449 				writeContent(response, filePath);
450 			}else{
451 				if(!writeResponseHeader){
452 					LOGGER.error("Unable to load the image for the request URI : "
453 							+ request.getRequestURI());
454 					response.sendError(HttpServletResponse.SC_NOT_FOUND);
455 				}
456 			}
457 		} catch (Exception ex) {
458 
459 			LOGGER.error("Unable to load the image for the request URI : "
460 					+ request.getRequestURI(), ex);
461 			response.sendError(HttpServletResponse.SC_NOT_FOUND);
462 		}
463 
464 		if (LOGGER.isDebugEnabled())
465 			LOGGER.debug("request succesfully attended");
466 	}

512 	protected String getContentType(String filePath, HttpServletRequest request) {
513 		String requestUri = request.getRequestURI();
514 
515 		// Retrieve the extension
516 		String extension = getExtension(filePath);
517 		if (extension == null) {
518 
519 			LOGGER.error("No extension found for the request URI : " + requestUri);
520 			return null;
521 		}
522 
523 		String contentType = (String) imgMimeMap.get(extension);
524 		if (contentType == null) {
525 
526 			LOGGER.error("No image extension match the extension '" + extension
527 					+ "' for the request URI : " + requestUri);
528 			return null;
529 		}
530 		return contentType;

551 	/***
552 	 * Write the image content to the response
553 	 * 
554 	 * @param response
555 	 *            the response
556 	 * @param fileName
557 	 *            the filename
558 	 * @throws IOException
559 	 *             if an IO exception occurs.
560 	 */
561 	private void writeContent(HttpServletResponse response, String fileName)
562 			throws IOException {
563 
564 		OutputStream os = response.getOutputStream();
565 		InputStream is = null;
566 
567 		String resourceName = fileName;
568 		if (!jawrConfig.getGeneratorRegistry().isGeneratedImage(resourceName)
569 				&& !resourceName.startsWith("/")) {
570 			resourceName = "/" + resourceName;
571 		}
572 
573 		try {
574 			is = rsReaderHandler.getResourceAsStream(resourceName);
575 		} catch (ResourceNotFoundException e) {
576 			// Nothing to do here
577 		}
578 
579 		if (is == null) {
580 			response.setStatus(HttpServletResponse.SC_NOT_FOUND);
581 			if (LOGGER.isInfoEnabled())
582 				LOGGER
583 						.info("Received a request for a non existing image resource: "
584 								+ resourceName);
585 			return;
586 		}
587 
588 		try {
589 			IOUtils.copy(is, os);
590 		} catch (EOFException eofex) {
591 			LOGGER.debug("Browser cut off response", eofex);
592 		}
593 		finally{
594 			IOUtils.close(is);
595 		}
596 	}

598 	/***
599 	 * Removes the cache buster
600 	 * 
601 	 * @param fileName
602 	 *            the file name
603 	 * @return the file name without the cache buster.
604 	 */
605 	private String getRealFilePath(String fileName) {
606 
607 		String realFilePath = fileName;
608 		if (realFilePath.startsWith("/")) {
609 			realFilePath = realFilePath.substring(1);
610 		}
611 		
612 		Matcher matcher = cacheBusterPattern.matcher(realFilePath);
613 		StringBuffer result = new StringBuffer();
614 		if (matcher.find()) {
615 			matcher
616 					.appendReplacement(
617 							result,
618 							StringUtils.isEmpty(matcher
619 									.group(GENERATED_IMAGE_PREFIX_INDEX)) ? CACHE_BUSTER_STANDARD_IMAGE_REPLACE_PATTERN
620 									: CACHE_BUSTER_GENERATED_IMAGE_REPLACE_PATTERN);
621 			return result.toString();
622 		}
623 
624 		return realFilePath;
625 	}
