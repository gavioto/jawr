32  /***
33   * This class defines the post processor which encodes resources (background images) using base64 encoding.
34   * 
35   * @author Gerben Jorna
36   * @author Ibrahim Chaehoi
37   */
38  public class Base64ImageEncoderPostProcessor extends
39  	CSSURLPathRewriterPostProcessor implements ResourceBundlePostProcessor {
40  
41  	/*** The logger */
42  	protected static final Logger LOGGER = 
43  		Logger.getLogger(Base64ImageEncoderPostProcessor.class);
44  
45  	/*** Tab */
46  	protected static final String TAB = "\t";
47  
48  	/*** Boundary separator */
49  	protected static final String BOUNDARY_SEPARATOR = "JAWR_BASE64_ENCODED_IMAGES";
50  
51  	/*** The background property url value start */
52  	protected static final String PROPERTY_URL_VALUE_START = "url(";
53  	
54  	/*** The background property url value end */
55  	protected static final String PROPERTY_URL_VALUE_END = ")";
56  	
57  	/*** Boundary separator prefix */
58  	protected static final String BOUNDARY_SEPARATOR_PREFIX = "--";
59  		
60  	/*** The mhtml prefix */
61  	protected static final String MHTML_PREFIX = "mhtml:";

63  	/***
64  	 * Constructor
65  	 */
66  	public Base64ImageEncoderPostProcessor() {
67  		super(PostProcessFactoryConstant.BASE64_IMAGE_ENCODER);
68  	}

70  	/* (non-Javadoc)
71  	 * @see net.jawr.web.resource.bundle.postprocess.impl.CSSURLPathRewriterPostProcessor#createImageUrlRewriter(net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus)
72  	 */
73  	protected PostProcessorCssImageUrlRewriter createImageUrlRewriter(
74  			BundleProcessingStatus status) {
75

77  	}

79  	/* (non-Javadoc)
80  	 * @see net.jawr.web.resource.bundle.postprocess.impl.CSSURLPathRewriterPostProcessor#doPostProcessBundle(net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus, java.lang.StringBuffer)
81  	 */
82  	protected StringBuffer doPostProcessBundle(BundleProcessingStatus status, StringBuffer bundleData) throws IOException {
83  	
84  		if(LOGGER.isInfoEnabled()){
85  			LOGGER.info("Base64 encoding resources - '" + status.getLastPathAdded() + "'");
86  		}
87  		
88  		Map encodedResources = (Map) status.getData(JawrConstant.BASE64_ENCODED_RESOURCES);
89  		if(encodedResources == null){
90  			encodedResources = new HashMap();
91  			status.putData(JawrConstant.BASE64_ENCODED_RESOURCES, encodedResources);
92  		}
93  		
94  		StringBuffer sb = bundleData;
95  		if(!(status.getCurrentBundle().isComposite() && status.getProcessingType().equals(BundleProcessingStatus.BUNDLE_PROCESSING_TYPE))){
96  			sb = super.doPostProcessBundle(status, bundleData);
97  		}
98  		
99  		if(!encodedResources.isEmpty() && status.isSearchingPostProcessorVariants()){
100 			VariantSet variantSet = new VariantSet(JawrConstant.BROWSER_VARIANT_TYPE, "", new String[]{"", JawrConstant.BROWSER_IE6,JawrConstant.BROWSER_IE7});
101 			status.addPostProcessVariant(JawrConstant.BROWSER_VARIANT_TYPE, variantSet);
102 			variantSet = new VariantSet(JawrConstant.CONNECTION_TYPE_VARIANT_TYPE, "", new String[]{"", JawrConstant.SSL});
103 			status.addPostProcessVariant(JawrConstant.CONNECTION_TYPE_VARIANT_TYPE, variantSet);
104 		}
105 		
106 		if(!status.isSearchingPostProcessorVariants() && status.getCurrentBundle().isComposite() && status.getProcessingType().equals(BundleProcessingStatus.BUNDLE_PROCESSING_TYPE)){
107 				
108 			Map bundleVariants = status.getBundleVariants();
109 			if(bundleVariants != null){
110 				String browser = (String) bundleVariants.get(JawrConstant.BROWSER_VARIANT_TYPE);
111 				if(StringUtils.isNotEmpty(browser) && (JawrConstant.BROWSER_IE6.equals(browser) || JawrConstant.BROWSER_IE7.equals(browser))){
112 					prependBase64EncodedResources(sb, encodedResources);
113 				}
114 			}
115 		}
116 		
117 		if(LOGGER.isInfoEnabled()){
118 			LOGGER.info("Base64 Postprocessing done");
119 		}
120 		return sb;
121 	}

124 	/***
125 	 * Prepend the base64 encoded resources to the bundle data
126 	 * @param sb the string buffer containing the processed bundle data
127 	 * @param encodedImages a map of encoded images
128 	 */
129 	protected void prependBase64EncodedResources(StringBuffer sb, Map encodedImages) {
130 		Iterator it = encodedImages.entrySet().iterator();
131 		StringBuffer mhtml = new StringBuffer();
132 		String lineSeparator = net.jawr.web.util.StringUtils.LINE_SEPARATOR;
133 		mhtml.append("/*!" + lineSeparator);
134 		mhtml.append("Content-Type: multipart/related; boundary=\"" + BOUNDARY_SEPARATOR + "\"" + lineSeparator + lineSeparator);
135 		
136 		while (it.hasNext()) {
137 	        Map.Entry pair = (Map.Entry)it.next();
138 	        Base64EncodedResource encodedResource = (Base64EncodedResource) pair.getValue();
139 	        mhtml.append(BOUNDARY_SEPARATOR_PREFIX + BOUNDARY_SEPARATOR + lineSeparator);
140 	        mhtml.append("Content-Type:" + encodedResource.getType() + lineSeparator);
141 	        mhtml.append("Content-Location:" + encodedResource.getId() + lineSeparator);
142 	        mhtml.append("Content-Transfer-Encoding:base64" + lineSeparator + lineSeparator);
143 	        mhtml.append(encodedResource.getBase64Encoding() + lineSeparator + lineSeparator);
144 	    }		
145 		
146 		mhtml.append(BOUNDARY_SEPARATOR_PREFIX + BOUNDARY_SEPARATOR + BOUNDARY_SEPARATOR_PREFIX + lineSeparator);
147 		mhtml.append("*/" + lineSeparator + lineSeparator);
148         sb.insert(0, mhtml.toString());
149 	}
