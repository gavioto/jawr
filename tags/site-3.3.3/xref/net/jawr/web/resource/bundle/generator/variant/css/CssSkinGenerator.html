143 	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#createResource(net.jawr.web.resource.bundle.generator.GeneratorContext)
144 	 */
145 	public Reader createResource(GeneratorContext context) {
146 		
147 		Reader reader = null;
148 		ResourceReaderHandler readerHandler = context.getResourceReaderHandler();
149 		String path = context.getPath();
150 		
151 		String skinRootDir = getSkinRootDir(path, skinMapping.keySet());
152 		Map ctxVariantSetMap = (Map) skinMapping.get(skinRootDir);
153 		
154 		String skinVariantPath = path.substring(skinRootDir.length());
155 		String[] paths = skinVariantPath.split(JawrConstant.URL_SEPARATOR);
156 		
157 		VariantResourceReaderStrategy strategy = getVariantStrategy(context, ctxVariantSetMap);
158 		Map variantMap = null;
159 		do {
160 			variantMap = strategy.nextVariantMapConbination();
161 			if(variantMap != null){
162 				reader = getResourceReader(readerHandler, skinRootDir, paths, variantMap);
163 			}
164 		}while(variantMap != null && reader == null);
165 		
166 		if(!context.isProcessingBundle() && reader != null){
167 			
168 			// Rewrite the image URL
169 			StringWriter writer = new StringWriter();
170 			try {
171 				IOUtils.copy(reader, writer);
172 				CssImageUrlRewriter rewriter = new CssImageUrlRewriter(context.getConfig());
173 				String bundlePath = PathNormalizer.concatWebPath(context.getConfig().getServletMapping(), ResourceGenerator.CSS_DEBUGPATH);
174 				StringBuffer result = rewriter.rewriteUrl(path, bundlePath, writer.toString());
175 				reader = new StringReader(result.toString());
176 			} catch (IOException e) {
177 				throw new BundlingProcessException(e);
178 			}
179 		}
180 		
181 		return reader;
182 	}

225 	 * Returns the skin mapping from the Jawr config, here the resource mapping type is skin_locale 
226 	 * @param rsBrowser the resource browser
227 	 * @param config the jawr config
228 	 * @param skinMapping the skin mapping
229 	 * @param skinRootDirectories the skin root directories
230 	 */
231 	private void updateSkinMappingUsingTypeSkinLocale(ResourceBrowser rsBrowser,
232 			JawrConfig config, Map skinMapping, Set skinRootDirectories) {
233 		
234 		String defaultSkinName = null;
235 		String defaultLocaleName = null;
236 		
237 		// Check if there are no directory which is contained in another root directory
238 		for (Iterator itRootDir = skinRootDirectories.iterator(); itRootDir.hasNext();) {
239 			String defaultSkinDir = PathNormalizer.asDirPath((String) itRootDir.next());
240 			String defaultSkinDirName = PathNormalizer.getPathName(defaultSkinDir);
241 			String skinRootDir = PathNormalizer.getParentPath(defaultSkinDir);
242 			
243 			String skinName = null;
244 			String localeName = null;
245 			
246 			if(LocaleUtils.LOCALE_SUFFIXES.contains(defaultSkinDirName)){ // The rootDir point to a locale directory
247 				localeName = defaultSkinDirName;
248 				skinName = PathNormalizer.getPathName(skinRootDir);
249 				skinRootDir = PathNormalizer.getParentPath(skinRootDir);
250 				
251 			}else{
252 				skinName = defaultSkinDirName;
253 			}
254 			
255 			if(defaultSkinName == null){
256 				defaultSkinName = skinName;
257 			}else if(!defaultSkinName.equals(skinName)){
258 				throw new BundlingProcessException("The default skin for the skin root directories are not the same. Please check your configuration.");
259 			}
260 			
261 			if(defaultLocaleName == null){
262 				defaultLocaleName = localeName;
263 			}else if(defaultLocaleName != null && !defaultLocaleName.equals(localeName)){
264 				throw new BundlingProcessException("The default locale for the skin root directories are not the same. Please check your configuration.");
265 			}
266 			
267 			checkRootDirectoryNotOverlap(defaultSkinDir, skinRootDirectories);
268 			
269 			Map variantsMap = getVariants(rsBrowser, skinRootDir, skinName,
270 					localeName, true);
271 			
272 			skinMapping.put(skinRootDir, variantsMap);
273 		}
274 		
275 		CssSkinVariantResolver resolver = new CssSkinVariantResolver(defaultSkinName, config.getSkinCookieName());
276 		config.getGeneratorRegistry().registerVariantResolver(resolver);
277 	}

351 	/***
352 	 * Initialize the skinMapping from the parent path
353 	 * @param rsBrowser the resource browser
354 	 * @param rootDir the skin root dir path
355 	 * @param defaultSkinName the default skin name
356 	 * @param defaultLocaleName the default locale name
357 	 */
358 	private Map getVariants(ResourceBrowser rsBrowser, String rootDir,
359 			String defaultSkinName, String defaultLocaleName, boolean mappingSkinLocale) {
360 		Set paths = rsBrowser.getResourceNames(rootDir);
361 		Set skinNames = new HashSet();
362 		Set localeVariants = new HashSet();
363 		for (Iterator itPath = paths.iterator(); itPath.hasNext();) {
364 			String path = rootDir+ (String) itPath.next();
365 			if(rsBrowser.isDirectory(path)){
366 				String dirName = PathNormalizer.getPathName(path);
367 				if(mappingSkinLocale){
368 					skinNames.add(dirName);
369 					// check if there are locale variants for this skin,
370 					// and update the localeVariants if needed
371 					updateLocaleVariants(rsBrowser, path, localeVariants);
372 				}else{
373 					if(LocaleUtils.LOCALE_SUFFIXES.contains(dirName)){
374 						localeVariants.add(dirName);
375 						// check if there are skin variants for this locales,
376 						// and update the skinVariants if needed
377 						updateSkinVariants(rsBrowser, path, skinNames);
378 					}
379 				}
380 			}
381 		}
382 		
383 		// Initialize the variant mapping for the skin root directory
384 		return getVariants(defaultSkinName, skinNames, defaultLocaleName, localeVariants);
385 	}

453 	 * Check if there are no directory which is contained in another root directory,
454 	 * @param dir the root directory to check
455 	 * @param skinRootDirectories the root directories
456 	 */
457 	private void checkRootDirectoryNotOverlap(String dir,
458 			Set skinRootDirectories) {
459 		
460 		String rootDir = removeLocaleSuffixIfExist(dir);
461 		for (Iterator itSkinDir = skinRootDirectories.iterator(); itSkinDir.hasNext();) {
462 			String skinDir = PathNormalizer.asDirPath((String) itSkinDir.next());
463 			if(!skinDir.equals(dir)){
464 				skinDir = removeLocaleSuffixIfExist(skinDir);
465 				if(skinDir.startsWith(rootDir)){
466 					throw new BundlingProcessException("There is a misconfiguration. It is not allowed to have a skin root directory containing another one.");
467 				}
468 			}
469 		}
470 	}

473 	 * Removes from the directory path the locale suffix if it exists
474 	 * @param dir the directory path
475 	 * 
476 	 * <pre>
477      * removeLocaleSuffixIfExist( "/css/panel/default/" )         = "/css/panel/default/"
478      * removeLocaleSuffixIfExist( "/css/panel/default/en_US" )   = "/css/panel/default/"
479      * </pre>
480      * 
481 	 * @return the directory path, with the locale suffix removed if it exists
482 	 */
483 	private String removeLocaleSuffixIfExist(String dir){
484 		
485 		String result = dir;
486 		String dirName = PathNormalizer.getPathName(dir);
487 		if(LocaleUtils.LOCALE_SUFFIXES.contains(dirName)){
488 			result = dir.substring(0, dir.indexOf("/"+dirName)+1);
489 		}
490 		return result;
491 	}

514 	/***
515 	 * Returns the reader for the resource path define in parameter
516 	 * @param skinRootDir the root directory
517 	 * @param paths the array of path
518 	 * @param variantMap the variant map
519 	 * @return the reader
520 	 */
521 	private Reader getResourceReader(ResourceReaderHandler readerHandler, String skinRootDir, String[] paths,
522 			Map variantMap) {
523 		
524 		Reader reader = null;
525 		StringBuffer path = new StringBuffer(skinRootDir);
526 		String skinVariant = (String) variantMap.get(JawrConstant.SKIN_VARIANT_TYPE);
527 		String localeVariant = (String) variantMap.get(JawrConstant.LOCALE_VARIANT_TYPE);
528 		
529 		if(skinMappingType.equals(JawrConstant.SKIN_TYPE_MAPPING_SKIN_LOCALE)){
530 			paths[0] = skinVariant;
531 			if(localeVariant != null){
532 				paths[1] = localeVariant;
533 			}
534 		}else{
535 			paths[0] = localeVariant;
536 			if(skinVariant != null){
537 				paths[1] = skinVariant;
538 			}
539 		}
540 		
541 		for (int i = 0; i < paths.length; i++) {
542 			path.append(paths[i]);
543 			if(i+1 < paths.length){
544 				path.append(JawrConstant.URL_SEPARATOR);
545 			}
546 		}
547 		
548 		try {
549 			reader = readerHandler.getResource(path.toString());
550 		} catch (ResourceNotFoundException e) {
551 			// Nothing to do
552 		}
553 		
554 		return reader;
555 	}
