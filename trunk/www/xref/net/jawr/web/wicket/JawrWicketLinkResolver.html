14  package net.jawr.web.wicket;
15  
16  import org.apache.wicket.Component;
17  import org.apache.wicket.MarkupContainer;
18  import org.apache.wicket.markup.ComponentTag;
19  import org.apache.wicket.markup.MarkupStream;
20  import org.apache.wicket.markup.WicketTag;
21  import org.apache.wicket.markup.html.WebMarkupContainer;
22  import org.apache.wicket.markup.resolver.IComponentResolver;
23  
24  /***
25   * This class defines a tag resolver which handles &lt;wicket:jawr&gt; tags. 
26   * The tasks of this resolver will be to :
27   * 		- add a "transparent" WebMarkupContainer to transparently handling child components.
28   * 		- create the right component depending on the tag defined inside &lt;wicket:jawr&gt;
29   *
30   * @author Ibrahim Chaehoi
31   * 
32   */
33  public class JawrWicketLinkResolver implements IComponentResolver {
34  
35  	private static final String IMAGE_TAG_NAME = "image";
36  	private static final String IMG_TAG_NAME = "img";
37  	/*** The serial version UID */
38  	private static final long serialVersionUID = -2106412613442819122L;
39  
40  	/* (non-Javadoc)
41  	 * @see org.apache.wicket.markup.resolver.IComponentResolver#resolve(org.apache.wicket.MarkupContainer, org.apache.wicket.markup.MarkupStream, org.apache.wicket.markup.ComponentTag)
42  	 */
43  	public boolean resolve(final MarkupContainer container,
44  			final MarkupStream markupStream, final ComponentTag tag) {
45  		
46  		// Only component tags have the id == "_jawrAutolink_"
47  		String tagName = tag.getName();
48  		if (tag.getId().equals(JawrWicketLinkTagHandler.AUTOLINK_ID)) {
49  			
50  			// create the right component depending on the tag name
51  			WebMarkupContainer jawrTag = null;
52  			final String id = tag.getId()+ container.getPage().getAutoIndex();
53  			if(tagName.equalsIgnoreCase(IMG_TAG_NAME)){
54  				jawrTag = new JawrHtmlImageReference(id);
55  			}else if(tagName.equalsIgnoreCase("input") && tag.getAttribute("type").equals(IMAGE_TAG_NAME)){
56  				jawrTag = new JawrHtmlImageReference(id);
57  			}else if(tagName.equalsIgnoreCase("script")){
58  				jawrTag = new JawrJavascriptReference(id);
59  			}else if(tagName.equalsIgnoreCase("link")){
60  				jawrTag = new JawrStylesheetReference(id);
61  			}
62  			
63  			if(jawrTag != null){
64  				container.autoAdd(jawrTag, markupStream);
65  			}
66  			
67  			// Yes, we handled the tag
68  			return jawrTag != null;
69  		} else if (tag instanceof WicketTag) {
70  
71  			// For tag wicket:jawr
72  			if (tagName.equals("jawr")) {
73  
74  				final String id = tag.getId()
75  						+ container.getPage().getAutoIndex();
76  				final Component component = new WebMarkupContainer(id) {
77  					private static final long serialVersionUID = 1L;
78  
79  					/***
80  					 * @see org.apache.wicket.MarkupContainer#isTransparentResolver()
81  					 */
82  					public boolean isTransparentResolver() {
83  						return true;
84  					}
85  				};
86  				component.setRenderBodyOnly(true);
87  				container.autoAdd(component, markupStream);
88  
89  				// Yes, we handled the tag
90  				return true;
91  			}
92  		}
93  
94  		// We were not able to handle the tag
95  		return false;
96  	}
97  
98  }
