14  package net.jawr.web.context;
15  
16  import javax.management.ObjectName;
17  
18  /**
19   * This class defines the context for Jawr, it holds the context in a ThreadLocal object.
20   * 
21   * @author Matt Ruby
22   * @author Ibrahim Chaehoi
23   */
24  public final class ThreadLocalJawrContext {
25  
26  	/**
27  	 * debugOverride will allow us to override production mode on a request by request basis.
28  	 * ThreadLocal is used to hold the overridden status throughout a given request.
29  	 */
30  	private static ThreadLocal<JawrContext> jawrContext = new ThreadLocal<JawrContext>(){
31  
32  		/* (non-Javadoc)
33  		 * @see java.lang.ThreadLocal#initialValue()
34  		 */
35  		protected JawrContext initialValue() {
36  			return new JawrContext();
37  		}
38  	    
39  	};
40  	
41  	/**
42  	 * The debugOverride will be automatially set to false
43  	 */
44  	private ThreadLocalJawrContext() {
45  		
46  	}
47  	
48  	/**
49  	 * Returns the mbean object name of the Jawr config manager
50  	 * @return the mbean object name of the Jawr config manager
51  	 */
52  	public static ObjectName getJawrConfigMgrObjectName() {
53  		
54  		return jawrContext.get().getJawrConfigMgrObjectName();
55  	}
56  
57  	/**
58  	 * Sets the mbean object name of the Jawr config manager
59  	 * @param mbeanObjectName the mbean object name of the Jawr config manager
60  	 */
61  	public static void setJawrConfigMgrObjectName(ObjectName mbeanObjectName) {
62  
63  		jawrContext.get().setJawrConfigMgrObjectName(mbeanObjectName);
64  	}
65  	
66  	/**
67  	 * Get the flag stating that production mode should be overridden
68  	 * @return the flag stating that production mode should be overridden
69  	 */
70  	public static boolean isDebugOverriden() {
71  		
72  		return jawrContext.get().isDebugOverriden();
73  	}
74  
75  	/**
76  	 * Set the override flag that will live only for this request
77  	 * @param override the flag to set
78  	 */
79  	public static void setDebugOverriden(boolean override) {
80  
81  		jawrContext.get().setDebugOverriden(override);
82  	}
83  	
84  	/**
85  	 * Returns the flag indicating that we are using making a bundle processing at build time
86  	 * @return the flag indicating that we are using making a bundle processing at build time
87  	 */
88  	public static boolean isBundleProcessingAtBuildTime() {
89  		return jawrContext.get().isBundleProcessingAtBuildTime();
90  	}
91  
92  	/**
93  	 * Sets the flag indicating that we are using making a bundle processing at build time
94  	 * @param bundleProcessingAtBuildTime the flag to set
95  	 */
96  	public static void setBundleProcessingAtBuildTime(boolean bundleProcessingAtBuildTime) {
97  		jawrContext.get().setBundleProcessingAtBuildTime(bundleProcessingAtBuildTime);
98  	}
99  	
100 	/**
101 	 * Returns the current request
102 	 * @return the request
103 	 */
104 	public static String getRequestURL() {
105 		return jawrContext.get().getRequestURL();
106 	}
107 
108 	/**
109 	 * Sets the request
110 	 * @param request the request to set
111 	 */
112 	public static void setRequest(String requestURL) {
113 		jawrContext.get().setRequestURL(requestURL);
114 	}
115 	
116 	/**
117 	 * Sets the mbean object name
118 	 * @param mbeanObjectName the mbean object name
119 	 */
120 	public static void reset() {
121 
122 		jawrContext.remove();
123 	}
124 	
125 }
