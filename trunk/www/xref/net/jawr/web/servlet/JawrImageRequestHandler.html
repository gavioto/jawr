131 	protected void initializeJawrConfig(Properties props)
132 			throws ServletException {
133 		// Initialize config
134 		if (null != jawrConfig)
135 			jawrConfig.invalidate();
136 
137 		jawrConfig = createJawrConfig(props);
138 
139 		jawrConfig.setContext(servletContext);
140 		jawrConfig.setGeneratorRegistry(generatorRegistry);
141 
142 		// Set the content type to be used for every request.
143 		contentType = JawrConstant.IMG_TYPE;
144 
145 		// Set mapping, to be used by the tag lib to define URLs that point to
146 		// this servlet.
147 		String mapping = (String) initParameters
148 				.get(JawrConstant.SERVLET_MAPPING_PROPERTY_NAME);
149 		if (null != mapping) {
150 			jawrConfig.setServletMapping(mapping);
151 		}
152 
153 		// Initialize the resource handler
154 		rsReaderHandler = initResourceReaderHandler();
155 		rsBundleHandler = initResourceBundleHandler();
156 
157 		// Initialize custom generators
158 		PropertiesConfigHelper propertiesHelper = new PropertiesConfigHelper(
159 				props, resourceType);
160 		Iterator generators = propertiesHelper.getCommonPropertyAsSet(
161 				PropertiesBundleConstant.CUSTOM_GENERATORS).iterator();
162 		while (generators.hasNext()) {
163 			String generatorClass = (String) generators.next();
164 			generatorRegistry.registerGenerator(generatorClass);
165 		}
166 
167 		if (jawrConfig.getUseBundleMapping()) {
168 			bundleMapping = rsBundleHandler.getJawrBundleMapping();
169 		} else {
170 			bundleMapping = new Properties();
171 		}
172 
173 		ImageResourcesHandler imgRsHandler = new ImageResourcesHandler(
174 				jawrConfig, rsReaderHandler, rsBundleHandler);
175 		initImageMapping(imgRsHandler);
176 
177 		servletContext.setAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE,
178 				imgRsHandler);
179 
180 		if (log.isDebugEnabled()) {
181 			log.debug("Configuration read. Current config:");
182 			log.debug(jawrConfig);
183 		}
184 
185 		// Warn when in debug mode
186 		if (jawrConfig.isDebugModeOn()) {
187 			log
188 					.warn("Jawr initialized in DEVELOPMENT MODE. Do NOT use this mode in production or integration servers. ");
189 		}
190 	}

192 	/***
193 	 * Initialize the image mapping of the image resources handler
194 	 * 
195 	 * @param imgRsHandler
196 	 *            the image resources handler
197 	 */
198 	private void initImageMapping(ImageResourcesHandler imgRsHandler) {
199 
200 		if (jawrConfig.getUseBundleMapping()
201 				&& rsBundleHandler.isExistingMappingFile()) {
202 
203 			// Initialize the image mapping
204 			Iterator mapIterator = bundleMapping.keySet().iterator();
205 			while (mapIterator.hasNext()) {
206 				String key = (String) mapIterator.next();
207 				imgRsHandler.addMapping(key, bundleMapping.getProperty(key));
208 			}
209 
210 		} else {
211 			// Create a resource handler to read files from the WAR archive or
212 			// exploded dir.
213 			String imageResourcesDefinition = jawrConfig
214 					.getImageResourcesDefinition();
215 			if (imageResourcesDefinition != null) {
216 
217 				StringTokenizer tokenizer = new StringTokenizer(
218 						imageResourcesDefinition, ",");
219 				while (tokenizer.hasMoreTokens()) {
220 					String pathMapping = tokenizer.nextToken();
221 
222 					// path is a generated image and ends with an image
223 					// extension
224 					if (generatorRegistry.isGeneratedImage(pathMapping)
225 							&& hasImageFileExtension(pathMapping)) {
226 
227 						addImagePath(imgRsHandler, pathMapping);
228 					}
229 					// path ends in /, the folder is included without subfolders
230 					else if (pathMapping.endsWith("/")) {
231 						addItemsFromDir(imgRsHandler, pathMapping, false);
232 					}
233 					// path ends in /, the folder is included with all
234 					// subfolders
235 					else if (pathMapping.endsWith("/**")) {
236 						addItemsFromDir(imgRsHandler, pathMapping.substring(0,
237 								pathMapping.lastIndexOf("**")), true);
238 					} else if (hasImageFileExtension(pathMapping)) {
239 						addImagePath(imgRsHandler, pathMapping);
240 					} else
241 						log
242 								.warn("Wrong mapping ["
243 										+ pathMapping
244 										+ "] for image bundle. Please check configuration. ");
245 				}
246 			}
247 		}
248 
249 		// Store the bundle mapping
250 		if (jawrConfig.getUseBundleMapping()
251 				&& !rsBundleHandler.isExistingMappingFile()) {
252 			rsBundleHandler.storeJawrBundleMapping(bundleMapping);
253 		}
254 
255 		if (log.isDebugEnabled())
256 			log.debug("Finish creation of map for image bundle");
257 	}

259 	/***
260 	 * Add an image path to the image map
261 	 * 
262 	 * @param imgRsHandler
263 	 *            the image resources handler
264 	 * @param imgPath
265 	 *            the image path
266 	 * @param classPathImg
267 	 *            the flag indicating if the image should be retrieved from
268 	 *            classpath
269 	 */
270 	private void addImagePath(ImageResourcesHandler imgRsHandler, String imgPath) {
271 
272 		try {
273 			String resultPath = CheckSumUtils.getCacheBustedUrl(imgPath,
274 					rsReaderHandler, jawrConfig);
275 			imgRsHandler.addMapping(imgPath, resultPath);
276 			bundleMapping.put(imgPath, resultPath);
277 		} catch (IOException e) {
278 			log.error(
279 					"An exception occurs while defining the mapping for the file : "
280 							+ imgPath, e);
281 		} catch (ResourceNotFoundException e) {
282 			log.error("Impossible to define the checksum for the resource '"
283 					+ imgPath
284 					+ "'. Unable to retrieve the content of the file.");
285 		}
286 	}

288 	/***
289 	 * Returns true of the path contains an image file extension
290 	 * 
291 	 * @param path
292 	 *            the path
293 	 * @return the image file extension
294 	 */
295 	private boolean hasImageFileExtension(String path) {
296 		boolean result = false;
297 
298 		int extFileIdx = path.lastIndexOf(".");
299 		if (extFileIdx != -1 && extFileIdx + 1 < path.length()) {
300 			String extension = path.substring(extFileIdx + 1);
301 			result = imgMimeMap.containsKey(extension);
302 		}
303 
304 		return result;
305 	}

307 	/***
308 	 * Adds all the resources within a path to the image map.
309 	 * 
310 	 * @param imgRsHandler
311 	 *            the image resources handler
312 	 * @param dirName
313 	 *            the directory name
314 	 * @param addSubDirs
315 	 *            boolean If subfolders will be included. In such case, every
316 	 *            folder below the path is included.
317 	 */
318 	private void addItemsFromDir(ImageResourcesHandler imgRsHandler,
319 			String dirName, boolean addSubDirs) {
320 		Set resources = rsReaderHandler.getResourceNames(dirName);
321 
322 		if (log.isDebugEnabled()) {
323 			log.debug("Adding " + resources.size() + " resources from path ["
324 					+ dirName + "] to image bundle");
325 		}
326 
327 		// Add remaining resources (remaining after sorting, or all if no sort
328 		// file present)
329 		List folders = new ArrayList();
330 		for (Iterator it = resources.iterator(); it.hasNext();) {
331 			String resourceName = (String) it.next();
332 			String resourcePath = PathNormalizer.joinPaths(dirName,
333 					resourceName);
334 			if (hasImageFileExtension(resourceName)) {
335 				addImagePath(imgRsHandler, resourcePath);
336 
337 				if (log.isDebugEnabled())
338 					log.debug("Added to item path list:"
339 							+ PathNormalizer.asPath(resourcePath));
340 			} else if (addSubDirs) {
341 
342 				try {
343 					if (rsReaderHandler.isDirectory(resourcePath)) {
344 						folders.add(resourceName);
345 					}
346 				} catch (InvalidPathException e) {
347 					if (log.isDebugEnabled())
348 						log
349 								.debug("Enable to define if the following resource is a directory : "
350 										+ PathNormalizer.asPath(resourcePath));
351 				}
352 			}
353 		}
354 
355 		// Add subfolders if requested. Subfolders are added last unless
356 		// specified in sorting file.
357 		if (addSubDirs) {
358 			for (Iterator it = folders.iterator(); it.hasNext();) {
359 				String folderName = (String) it.next();
360 				addItemsFromDir(imgRsHandler, PathNormalizer.joinPaths(dirName,
361 						folderName), true);
362 			}
363 		}
364 	}

366 	/***
367 	 * Handles a resource request.
368 	 * <ul>
369 	 * <li>If the request contains an If-Modified-Since header, the 304 status
370 	 * is set and no data is written to the response</li>
371 	 * <li>If the requested path begins with the gzip prefix, a gzipped version
372 	 * of the resource is served, with the corresponding content-encoding
373 	 * header.</li>
374 	 * <li>Otherwise, the resource is written as text to the response.</li>
375 	 * <li>If the resource is not found, the response satus is set to 404 and no
376 	 * response is written.</li>
377 	 * </ul>
378 	 * 
379 	 * @param requestedPath
380 	 * @param request
381 	 * @param response
382 	 * @throws ServletException
383 	 * @throws IOException
384 	 */
385 	public void processRequest(String requestedPath,
386 			HttpServletRequest request, HttpServletResponse response)
387 			throws ServletException, IOException {
388 
389 		if (log.isDebugEnabled())
390 			log.debug("Request received for path:" + requestedPath);
391 
392 		// Retrieve the file path
393 		//String filePath = getFilePath(request);
394 		if (requestedPath == null) {
395 			response.sendError(HttpServletResponse.SC_NOT_FOUND);
396 			return;
397 		}
398 
399 		// Retrieve the content type
400 		String imgContentType = getContentType(request, requestedPath);
401 		if (imgContentType == null) {
402 			response.sendError(HttpServletResponse.SC_NOT_FOUND);
403 			return;
404 		}
405 
406 		// Set the content type
407 		response.setContentType(imgContentType);
408 		// If debug mode is off, check for If-Modified-Since and If-none-match
409 		// headers and set response caching headers.
410 		if (!this.jawrConfig.isDebugModeOn()) {
411 			// If a browser checks for changes, always respond 'no changes'.
412 			if (null != request.getHeader(IF_MODIFIED_SINCE_HEADER)
413 					|| null != request.getHeader(IF_NONE_MATCH_HEADER)) {
414 				response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
415 				if (log.isDebugEnabled())
416 					log.debug("Returning 'not modified' header. ");
417 				return;
418 			}
419 
420 			// Add caching headers
421 			setResponseHeaders(response);
422 		}
423 
424 		// Returns the real file path
425 		String filePath = getRealFilePath(requestedPath);
426 
427 		try {
428 			if (isValidRequestedPath(filePath)) {
429 				writeContent(response, filePath);
430 			} else {
431 				log.error("Unable to load the image for the request URI : "
432 						+ request.getRequestURI());
433 				response.sendError(HttpServletResponse.SC_NOT_FOUND);
434 			}
435 
436 		} catch (Exception ex) {
437 
438 			log.error("Unable to load the image for the request URI : "
439 					+ request.getRequestURI(), ex);
440 			response.sendError(HttpServletResponse.SC_NOT_FOUND);
441 		}
442 
443 		if (log.isDebugEnabled())
444 			log.debug("request succesfully attended");
445 	}

448 	 * Returns the content type for the image
449 	 * 
450 	 * @param request
451 	 *            the request
452 	 * @param filePath
453 	 *            the image file path
454 	 * @return the content type of the image
455 	 */
456 	private String getContentType(HttpServletRequest request, String filePath) {
457 		String requestUri = request.getRequestURI();
458 
459 		// Retrieve the extension
460 		String extension = getExtension(filePath);
461 		if (extension == null) {
462 
463 			log.error("No extension found for the request URI : " + requestUri);
464 			return null;
465 		}
466 
467 		String contentType = (String) imgMimeMap.get(extension);
468 		if (contentType == null) {
469 
470 			log.error("No image extension match the extension '" + extension
471 					+ "' for the request URI : " + requestUri);
472 			return null;
473 		}
474 		return contentType;
475 	}

480 	 * @param requestedPath
481 	 *            the requested path

484 	protected boolean isValidRequestedPath(String requestedPath) {
485 
486 		boolean result = true;
487 		if (requestedPath.startsWith(JawrConstant.WEB_INF_DIR_PREFIX)
488 				|| requestedPath.startsWith(JawrConstant.META_INF_DIR_PREFIX)) {
489 			result = false;

495 	/***
496 	 * Write the image content to the response
497 	 * 
498 	 * @param response
499 	 *            the response
500 	 * @param fileName
501 	 *            the filename
502 	 * @throws IOException
503 	 *             if an IO exception occurs.
504 	 */
505 	private void writeContent(HttpServletResponse response, String fileName)
506 			throws IOException {
507 
508 		OutputStream os = response.getOutputStream();
509 		InputStream is = null;
510 
511 		if (!jawrConfig.getGeneratorRegistry().isGeneratedImage(fileName)
512 				&& !fileName.startsWith("/")) {
513 			fileName = "/" + fileName;
514 		}
515 
516 		try {
517 			is = rsReaderHandler.getResourceAsStream(fileName);
518 		} catch (ResourceNotFoundException e) {
519 			// Nothing to do here
520 		}
521 
522 		if (is == null) {
523 			response.setStatus(HttpServletResponse.SC_NOT_FOUND);
524 			if (log.isInfoEnabled())
525 				log
526 						.info("Received a request for a non existing image resource: "
527 								+ fileName);
528 			return;
529 		}
530 
531 		try {
532 			IOUtils.copy(is, os);
533 		} catch (EOFException eofex) {
534 			log.debug("Browser cut off response", eofex);
535 		}
536 		finally{
537 			IOUtils.close(is);
538 		}
539 	}

541 	/***
542 	 * Removes the cache buster
543 	 * 
544 	 * @param fileName
545 	 *            the file name
546 	 * @return the file name without the cache buster.
547 	 */
548 	private String getRealFilePath(String fileName) {
549 
550 		if (fileName.startsWith("/")) {
551 			fileName = fileName.substring(1);
552 		}
553 		Matcher matcher = cacheBusterPattern.matcher(fileName);
554 		StringBuffer result = new StringBuffer();
555 		if (matcher.find()) {
556 			matcher
557 					.appendReplacement(
558 							result,
559 							StringUtils.isEmpty(matcher
560 									.group(GENERATED_IMAGE_PREFIX_INDEX)) ? CACHE_BUSTER_STANDARD_IMAGE_REPLACE_PATTERN
561 									: CACHE_BUSTER_GENERATED_IMAGE_REPLACE_PATTERN);
562 			return result.toString();
563 		}
564 
565 		return fileName;
566 	}
