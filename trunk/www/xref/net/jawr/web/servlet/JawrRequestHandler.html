168 	/***
169 	 * Reads the properties file and initializes all configuration using the ServletConfig object. If applicable, a ConfigChangeListenerThread will be
170 	 * started to listen to changes in the properties configuration.
171 	 * 
172 	 * @param servletContext ServletContext
173 	 * @param servletConfig ServletConfig
174 	 * @throws ServletException if an exception occurs
175 	 */
176 	public JawrRequestHandler(ServletContext context, ServletConfig config) throws ServletException {
177 		this.initParameters = new HashMap();
178 		Enumeration params = config.getInitParameterNames();
179 		while (params.hasMoreElements()) {
180 			String param = (String) params.nextElement();
181 			initParameters.put(param, config.getInitParameter(param));
182 		}
183 		initParameters.put("handlerName", config.getServletName());
184 
185 		if (log.isInfoEnabled())
186 			log.info("Initializing jawr config for servlet named " + config.getServletName());
187 		
188 		initRequestHandler(context, null);
189 
190 	}

207 	/***
208 	 * Initialize the request handler
209 	 * @param context the servlet context
210 	 * @param configProps the configuration properties
211 	 * @throws ServletException if an exception occurs
212 	 */
213 	private void initRequestHandler(ServletContext context, Properties configProps) throws ServletException {
214 		
215 		long initialTime = System.currentTimeMillis();
216 		if (log.isInfoEnabled())
217 			log.info("Initializing jawr config for request handler named " + getInitParameter("handlerName"));
218 
219 		this.imgMimeMap = MIMETypesSupport.getSupportedProperties(this);
220 		this.servletContext = context;
221 		this.overrideProperties = configProps;
222 		resourceType = getInitParameter("type");
223 		resourceType = null == resourceType ? "js" : resourceType;
224 
225 		String configLocation = getInitParameter("configLocation");
226 		String configPropsSourceClass = getInitParameter("configPropertiesSourceClass");
227 		if (null == configProps && null == configLocation && null == configPropsSourceClass)
228 			throw new ServletException("Neither configLocation nor configPropertiesSourceClass init params were set."
229 					+ " You must set at least the configLocation param. Please check your web.xml file");
230 
231 		// Initialize the config properties source that will provide with all configuration options.
232 		ConfigPropertiesSource propsSrc;
233 
234 		// Load a custom class to set config properties
235 		if (null != configPropsSourceClass) {
236 			propsSrc = (ConfigPropertiesSource) ClassLoaderResourceUtils.buildObjectInstance(configPropsSourceClass);
237 			if (propsSrc instanceof ServletContextAware) {
238 				((ServletContextAware) propsSrc).setServletContext(context);
239 			}
240 		} else if(configLocation == null && configProps != null){
241 			
242 			// configuration retrieved from the in memory configuration properties
243 			propsSrc = new PropsConfigPropertiesSource(configProps);
244 			
245 		}else{
246 			// Default config properties source, reads from a .properties file in the classpath.
247 			propsSrc = new PropsFilePropertiesSource();
248 		}
249 
250 		// If a custom properties source is a subclass of PropsFilePropertiesSource, we hand it the configLocation param.
251 		// This affects the standard one as well.
252 		if (propsSrc instanceof PropsFilePropertiesSource)
253 			((PropsFilePropertiesSource) propsSrc).setConfigLocation(configLocation);
254 
255 		// Read properties from properties source
256 		Properties props = propsSrc.getConfigProperties();
257 		// override the properties if needed
258 		if(this.overrideProperties != null){
259 			props.putAll(overrideProperties);
260 		}
261 		
262 		// init registry
263 		generatorRegistry = new GeneratorRegistry(resourceType);
264 
265 		// hang onto the propertiesSource for manual reloads
266 		this.propertiesSource = propsSrc;
267 
268 		// Initialize config
269 		initializeJawrConfig(props);
270 
271 		// Initialize the properties reloading checker daemon if specified
272 		if(!ThreadLocalJawrContext.isBundleProcessingAtBuildTime() && null != props.getProperty(CONFIG_RELOAD_INTERVAL)) {
273 			int interval = Integer.valueOf(props.getProperty(CONFIG_RELOAD_INTERVAL)).intValue();
274 			log.warn("Jawr started with configuration auto reloading on. "
275 					+ "Be aware that a daemon thread will be checking for changes to configuration every " + interval + " seconds.");
276 
277 			this.configChangeListenerThread = new ConfigChangeListenerThread(propsSrc, this.overrideProperties, 
278 					this, interval);
279 			configChangeListenerThread.start();
280 		}
281 
282 		// initialize the jmx Bean
283 		if(isJmxEnabled()){
284 				JmxUtils.initJMXBean(this, servletContext, resourceType, jawrConfig.getConfigProperties());
285 		}
286 		
287 		if (log.isInfoEnabled()) {
288 			long totaltime = System.currentTimeMillis() - initialTime;
289 			log.info("Init method succesful. jawr started in " + (totaltime / 1000) + " seconds....");
290 		}
291 	}

310 	/***
311 	 * Initialize the Jawr config
312 	 * 
313 	 * @param props the properties
314 	 * @throws ServletException if an exception occurs
315 	 */
316 	protected void initializeJawrConfig(Properties props) throws ServletException {
317 		// Initialize config
318 		if (null != jawrConfig)
319 			jawrConfig.invalidate();
320 
321 		createJawrConfig(props);
322 		
323 		jawrConfig.setContext(servletContext);
324 		jawrConfig.setGeneratorRegistry(generatorRegistry);
325 
326 		// Set the content type to be used for every request.
327 		contentType = "text/";
328 		contentType += "js".equals(resourceType) ? "javascript" : "css";
329 		contentType += "; charset=" + jawrConfig.getResourceCharset().name();
330 
331 		// Set mapping, to be used by the tag lib to define URLs that point to this servlet.
332 		String mapping = (String) initParameters.get("mapping");
333 		if (null != mapping)
334 			jawrConfig.setServletMapping(mapping);
335 
336 		if (jawrConfig.isCssClasspathImageHandledByClasspathCss() && resourceType.equals("css")) {
337 			ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) servletContext.getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
338 			if (imgRsHandler == null) {
339 				log.error("You are using the CSS classpath image feature, but the JAWR Image servlet is yet initialized.\n"
340 						+ "The JAWR Image servlet must be initialized before the JAWR CSS servlet.\n"
341 						+ "Please check you web application configuration.");
342 			}
343 		}
344 
345 		if (log.isDebugEnabled()) {
346 			log.debug("Configuration read. Current config:");
347 			log.debug(jawrConfig);
348 		}
349 
350 		// Create a resource handler to read files from the WAR archive or exploded dir.
351 		rsReaderHandler = initResourceReaderHandler();
352 		ResourceBundleHandler rsBundleHandler = initResourceBundleHandler();
353 		PropertiesBasedBundlesHandlerFactory factory = new PropertiesBasedBundlesHandlerFactory(props, resourceType, rsReaderHandler, rsBundleHandler, jawrConfig
354 				.getGeneratorRegistry());
355 		try {
356 			bundlesHandler = factory.buildResourceBundlesHandler(jawrConfig);
357 		} catch (DuplicateBundlePathException e) {
358 			throw new ServletException(e);
359 		} catch (BundleDependencyException e) {
360 			throw new ServletException(e);
361 		}
362 
363 		if (resourceType.equals(JawrConstant.JS_TYPE))
364 			servletContext.setAttribute(JawrConstant.JS_CONTEXT_ATTRIBUTE, bundlesHandler);
365 		else
366 			servletContext.setAttribute(JawrConstant.CSS_CONTEXT_ATTRIBUTE, bundlesHandler);
367 
368 		this.clientSideScriptRequestHandler = new ClientSideHandlerScriptRequestHandler(bundlesHandler, jawrConfig);
369 
370 		if (log.isDebugEnabled()) {
371 			log.debug("content type set to: " + contentType);
372 		}
373 
374 		// Warn when in debug mode
375 		if (jawrConfig.isDebugModeOn()) {
376 			log.warn("Jawr initialized in DEVELOPMENT MODE. Do NOT use this mode in production or integration servers. ");
377 		}
378 	}

429 	/***
430 	 * Handles a resource request by getting the requested path from the request object and invoking processRequest.
431 	 * 
432 	 * @param request the request
433 	 * @param response the response
434 	 * @throws ServletException if a servlet exception occurs
435 	 * @throws IOException if an IO exception occurs.
436 	 */
437 	public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
438 		
439 		try{
440 			// Initialize the Thread local for the Jawr context
441 			if(isJmxEnabled()){
442 				ThreadLocalJawrContext.setJawrConfigMgrObjectName(JmxUtils.getMBeanObjectName(request.getContextPath(), resourceType));
443 			}
444 			
445 			RendererRequestUtils.setRequestDebuggable(request, jawrConfig);
446 			
447 			String requestedPath = "".equals(jawrConfig.getServletMapping()) ? request.getServletPath() : request.getPathInfo();
448 			processRequest(requestedPath, request, response);
449 			
450 		} catch (Exception e) {
451 			
452 			if (log.isDebugEnabled()){
453 				log.debug("ServletException : ", e);
454 			}
455 			
456 			throw new ServletException(e);
457 		}finally{
458 			
459 			// Reset the Thread local for the Jawr context
460 			ThreadLocalJawrContext.reset();
461 		}
462 		
463 	}

465 	/***
466 	 * Handles a resource request.
467 	 * <ul>
468 	 * <li>If the request contains an If-Modified-Since header, the 304 status is set and no data is written to the response</li>
469 	 * <li>If the requested path begins with the gzip prefix, a gzipped version of the resource is served, with the corresponding content-encoding
470 	 * header.</li>
471 	 * <li>Otherwise, the resource is written as text to the response.</li>
472 	 * <li>If the resource is not found, the response satus is set to 404 and no response is written.</li>
473 	 * </ul>
474 	 * 
475 	 * @param requestedPath the requested path
476 	 * @param request the request
477 	 * @param response the response
478 	 * @throws ServletException if a servlet exception occurs
479 	 * @throws IOException if an IO exception occurs
480 	 */
481 	public void processRequest(String requestedPath, HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
482 
483 		// manual reload request
484 		if (this.jawrConfig.getRefreshKey().length() > 0 && null != request.getParameter("refreshKey")
485 				&& this.jawrConfig.getRefreshKey().equals(request.getParameter("refreshKey"))) {
486 			this.configChanged(propertiesSource.getConfigProperties());
487 		}
488 
489 		if (log.isDebugEnabled())
490 			log.debug("Request received for path:" + requestedPath);
491 
492 		if (CLIENTSIDE_HANDLER_REQ_PATH.equals(requestedPath)) {
493 			this.clientSideScriptRequestHandler.handleClientSideHandlerRequest(request, response);
494 			return;
495 		}
496 
497 		// CSS images would be requested through this handler in case servletMapping is used
498 		// if( this.jawrConfig.isDebugModeOn() && !("".equals(this.jawrConfig.getServletMapping())) && null == request.getParameter(GENERATION_PARAM)) {
499 		if (JawrConstant.CSS_TYPE.equals(resourceType) && 
500 				!JawrConstant.CSS_TYPE.equals(getExtension(requestedPath)) &&
501 				this.imgMimeMap.containsKey(getExtension(requestedPath))) {
502 
503 			if (null == bundlesHandler.resolveBundleForPath(requestedPath)) {
504 				if (log.isDebugEnabled())
505 					log.debug("Path '" + requestedPath + "' does not belong to a bundle. Forwarding request to the server. ");
506 				request.getRequestDispatcher(requestedPath).forward(request, response);
507 				return;
508 			}
509 		}
510 
511 		// If debug mode is off, check for If-Modified-Since and If-none-match headers and set response caching headers.
512 		if (!this.jawrConfig.isDebugModeOn()) {
513 			// If a browser checks for changes, always respond 'no changes'.
514 			if (null != request.getHeader(IF_MODIFIED_SINCE_HEADER) || null != request.getHeader(IF_NONE_MATCH_HEADER)) {
515 				response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
516 				if (log.isDebugEnabled())
517 					log.debug("Returning 'not modified' header. ");
518 				return;
519 			}
520 
521 			// Add caching headers
522 			setResponseHeaders(response);
523 		} else if (null != request.getParameter(GENERATION_PARAM))
524 			requestedPath = request.getParameter(GENERATION_PARAM);
525 
526 		// By setting content type, the response writer will use appropiate encoding
527 		response.setContentType(contentType);
528 
529 		try {
530 			// Send gzipped resource if user agent supports it.
531 			if (requestedPath.startsWith(BundleRenderer.GZIP_PATH_PREFIX)) {
532 				requestedPath = "/" + requestedPath.substring(BundleRenderer.GZIP_PATH_PREFIX.length(), requestedPath.length());
533 				response.setHeader("Content-Encoding", "gzip");
534 				bundlesHandler.streamBundleTo(requestedPath, response.getOutputStream());
535 			} else {
536 
537 				// In debug mode, we take in account the image generated from a StreamGenerator like classpath Image generator
538 				// The following code will rewrite the URL path for the generated images,
539 				// because in debug mode, we are retrieving the CSS ressources directly from the webapp
540 				// and if the CSS contains generated images, we should rewrite the URL.
541 				ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) servletContext.getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
542 				if (imgRsHandler != null && this.jawrConfig.isDebugModeOn() && resourceType.equals(JawrConstant.CSS_TYPE)) {
543 
544 					// Write the content of the CSS in the Stringwriter
545 					Writer writer = new StringWriter();
546 					Reader rd = rsReaderHandler.getResource(requestedPath);
547 					if(rd == null){
548 						throw new ResourceNotFoundException(requestedPath);
549 					}
550 					IOUtils.copy(rd, writer);
551 					String content = writer.toString();
552 
553 					String imageServletMapping = imgRsHandler.getJawrConfig().getServletMapping();
554 					
555 					if (imageServletMapping == null) {
556 						imageServletMapping = "";
557 					}
558 
559 					// Define the replacement pattern for the generated image (like jar:img/myImg.png)
560 					String relativeRootUrlPath = getRootRelativeCssUrlPath(request, requestedPath);
561 					String replacementPattern = PathNormalizer.normalizePath("$1" + relativeRootUrlPath + imageServletMapping + "/$4_cbDebug/$7$8");
562 					
563 					Matcher matcher = GENERATED_IMG_PATTERN.matcher(content);
564 
565 					// Rewrite the images define in the classpath, to point to the image servlet
566 					StringBuffer result = new StringBuffer();
567 					while (matcher.find()) {
568 						matcher.appendReplacement(result, replacementPattern);
569 					}
570 					matcher.appendTail(result);
571 					Writer out = response.getWriter();
572 					out.write(result.toString());
573 				} else {
574 
575 					if(isValidRequestedPath(requestedPath)){
576 						Writer out = response.getWriter();

641 		String originalRequestPath = "".equals(jawrConfig.getServletMapping()) ? request.getServletPath() : request.getPathInfo();
642 		// Deals with Jawr generated resource path containing /jawr_generator.css
643 		if(originalRequestPath.startsWith(ResourceGenerator.CSS_DEBUGPATH)){

645 		}
