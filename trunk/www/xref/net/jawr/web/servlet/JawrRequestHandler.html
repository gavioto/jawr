164 	/***
165 	 * Reads the properties file and initializes all configuration using the ServletConfig object. If applicable, a ConfigChangeListenerThread will be
166 	 * started to listen to changes in the properties configuration.
167 	 * 
168 	 * @param servletContext ServletContext
169 	 * @param servletConfig ServletConfig
170 	 * @throws ServletException if an exception occurs
171 	 */
172 	public JawrRequestHandler(ServletContext context, ServletConfig config) throws ServletException {
173 		this.initParameters = new HashMap();
174 		Enumeration params = config.getInitParameterNames();
175 		while (params.hasMoreElements()) {
176 			String param = (String) params.nextElement();
177 			initParameters.put(param, config.getInitParameter(param));
178 		}
179 		initParameters.put("handlerName", config.getServletName());
180 
181 		if (LOGGER.isInfoEnabled())
182 			LOGGER.info("Initializing jawr config for servlet named " + config.getServletName());
183 		
184 		initRequestHandler(context, null);
185 
186 	}

203 	/***
204 	 * Initialize the request handler
205 	 * @param context the servlet context
206 	 * @param configProps the configuration properties
207 	 * @throws ServletException if an exception occurs
208 	 */
209 	private void initRequestHandler(ServletContext context, Properties configProps) throws ServletException {
210 		
211 		long initialTime = System.currentTimeMillis();
212 		if (LOGGER.isInfoEnabled())
213 			LOGGER.info("Initializing jawr config for request handler named " + getInitParameter("handlerName"));
214 
215 		this.imgMimeMap = MIMETypesSupport.getSupportedProperties(this);
216 		this.servletContext = context;
217 		this.overrideProperties = configProps;
218 		resourceType = getInitParameter("type");
219 		resourceType = null == resourceType ? "js" : resourceType;
220 
221 		String configLocation = getInitParameter("configLocation");
222 		String configPropsSourceClass = getInitParameter("configPropertiesSourceClass");
223 		if (null == configProps && null == configLocation && null == configPropsSourceClass)
224 			throw new ServletException("Neither configLocation nor configPropertiesSourceClass init params were set."
225 					+ " You must set at least the configLocation param. Please check your web.xml file");
226 
227 		// Initialize the config properties source that will provide with all configuration options.
228 		ConfigPropertiesSource propsSrc;
229 
230 		// Load a custom class to set config properties
231 		if (null != configPropsSourceClass) {
232 			propsSrc = (ConfigPropertiesSource) ClassLoaderResourceUtils.buildObjectInstance(configPropsSourceClass);
233 			if (propsSrc instanceof ServletContextAware) {
234 				((ServletContextAware) propsSrc).setServletContext(context);
235 			}
236 		} else if(configLocation == null && configProps != null){
237 			
238 			// configuration retrieved from the in memory configuration properties
239 			propsSrc = new PropsConfigPropertiesSource(configProps);
240 			
241 		}else{
242 			// Default config properties source, reads from a .properties file in the classpath.
243 			propsSrc = new PropsFilePropertiesSource();
244 		}
245 
246 		// If a custom properties source is a subclass of PropsFilePropertiesSource, we hand it the configLocation param.
247 		// This affects the standard one as well.
248 		if (propsSrc instanceof PropsFilePropertiesSource)
249 			((PropsFilePropertiesSource) propsSrc).setConfigLocation(configLocation);
250 
251 		// Read properties from properties source
252 		Properties props = propsSrc.getConfigProperties();
253 		// override the properties if needed
254 		if(this.overrideProperties != null){
255 			props.putAll(overrideProperties);
256 		}
257 		
258 		// hang onto the propertiesSource for manual reloads
259 		this.propertiesSource = propsSrc;
260 
261 		// Initialize config
262 		initializeJawrConfig(props);
263 
264 		// Initialize the properties reloading checker daemon if specified
265 		if(!ThreadLocalJawrContext.isBundleProcessingAtBuildTime() && null != props.getProperty(CONFIG_RELOAD_INTERVAL)) {
266 			int interval = Integer.valueOf(props.getProperty(CONFIG_RELOAD_INTERVAL)).intValue();
267 			LOGGER.warn("Jawr started with configuration auto reloading on. "
268 					+ "Be aware that a daemon thread will be checking for changes to configuration every " + interval + " seconds.");
269 
270 			this.configChangeListenerThread = new ConfigChangeListenerThread(propsSrc, this.overrideProperties, 
271 					this, interval);
272 			configChangeListenerThread.start();
273 		}
274 
275 		// initialize the jmx Bean
276 		if(JmxUtils.isJmxEnabled()){
277 				JmxUtils.initJMXBean(this, servletContext, resourceType, jawrConfig.getConfigProperties());
278 		}
279 		
280 		if (LOGGER.isInfoEnabled()) {
281 			long totaltime = System.currentTimeMillis() - initialTime;
282 			LOGGER.info("Init method succesful. jawr started in " + (totaltime / 1000) + " seconds....");
283 		}
284 	}

295 	/***
296 	 * Initialize the Jawr config
297 	 * 
298 	 * @param props the properties
299 	 * @throws ServletException if an exception occurs
300 	 */
301 	protected void initializeJawrConfig(Properties props) throws ServletException {
302 		
303 		// init registry
304 		generatorRegistry = new GeneratorRegistry(resourceType);
305 
306 		// Initialize config
307 		if (null != jawrConfig){
308 			jawrConfig.invalidate();
309 		}
310 		
311 		createJawrConfig(props);
312 		
313 		jawrConfig.setContext(servletContext);
314 		jawrConfig.setGeneratorRegistry(generatorRegistry);
315 
316 		// Set the content type to be used for every request.
317 		contentType = "text/";
318 		contentType += "js".equals(resourceType) ? "javascript" : "css";
319 		contentType += "; charset=" + jawrConfig.getResourceCharset().name();
320 
321 		// Set mapping, to be used by the tag lib to define URLs that point to this servlet.
322 		String mapping = (String) initParameters.get("mapping");
323 		if (null != mapping)
324 			jawrConfig.setServletMapping(mapping);
325 
326 		if (jawrConfig.isCssClasspathImageHandledByClasspathCss() && resourceType.equals("css")) {
327 			ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) servletContext.getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
328 			if (imgRsHandler == null) {
329 				LOGGER.error("You are using the CSS classpath image feature, but the JAWR Image servlet is yet initialized.\n"
330 						+ "The JAWR Image servlet must be initialized before the JAWR CSS servlet.\n"
331 						+ "Please check you web application configuration.");
332 			}
333 		}
334 
335 		if (LOGGER.isDebugEnabled()) {
336 			LOGGER.debug("Configuration read. Current config:");
337 			LOGGER.debug(jawrConfig);
338 		}
339 
340 		// Initialize the IllegalBundleRequest handler
341 		initIllegalBundleRequestHandler();
342 		
343 		// Create a resource handler to read files from the WAR archive or exploded dir.
344 		rsReaderHandler = initResourceReaderHandler();
345 		ResourceBundleHandler rsBundleHandler = initResourceBundleHandler();
346 		PropertiesBasedBundlesHandlerFactory factory = new PropertiesBasedBundlesHandlerFactory(props, resourceType, rsReaderHandler, rsBundleHandler, jawrConfig
347 				.getGeneratorRegistry());
348 		try {
349 			bundlesHandler = factory.buildResourceBundlesHandler(jawrConfig);
350 		} catch (DuplicateBundlePathException e) {
351 			throw new ServletException(e);
352 		} catch (BundleDependencyException e) {
353 			throw new ServletException(e);
354 		}
355 
356 		if (resourceType.equals(JawrConstant.JS_TYPE))
357 			servletContext.setAttribute(JawrConstant.JS_CONTEXT_ATTRIBUTE, bundlesHandler);
358 		else
359 			servletContext.setAttribute(JawrConstant.CSS_CONTEXT_ATTRIBUTE, bundlesHandler);
360 
361 		this.clientSideScriptRequestHandler = new ClientSideHandlerScriptRequestHandler(bundlesHandler, jawrConfig);
362 
363 		if (LOGGER.isDebugEnabled()) {
364 			LOGGER.debug("content type set to: " + contentType);
365 		}
366 
367 		// Warn when in debug mode
368 		if (jawrConfig.isDebugModeOn()) {
369 			LOGGER.warn("Jawr initialized in DEVELOPMENT MODE. Do NOT use this mode in production or integration servers. ");
370 		}
371 	}

435 	 * Handles a resource request by getting the requested path from the request object and invoking processRequest.

437 	 * @param request the request

442 	public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
443 		
444 		try{
445 			
446 			String requestedPath = "".equals(jawrConfig.getServletMapping()) ? request.getServletPath() : request.getPathInfo();
447 			processRequest(requestedPath, request, response);
448 			
449 		} catch (Exception e) {
450 			
451 			if (LOGGER.isDebugEnabled()){
452 				LOGGER.debug("ServletException : ", e);
453 			}
454 			
455 			throw new ServletException(e);

459 	/***
460 	 * Handles a resource request.
461 	 * <ul>
462 	 * <li>If the request contains an If-Modified-Since header, the 304 status is set and no data is written to the response</li>
463 	 * <li>If the requested path begins with the gzip prefix, a gzipped version of the resource is served, with the corresponding content-encoding
464 	 * header.</li>
465 	 * <li>Otherwise, the resource is written as text to the response.</li>
466 	 * <li>If the resource is not found, the response satus is set to 404 and no response is written.</li>
467 	 * </ul>
468 	 * 
469 	 * @param requestedPath the requested path
470 	 * @param request the request
471 	 * @param response the response
472 	 * @throws ServletException if a servlet exception occurs
473 	 * @throws IOException if an IO exception occurs
474 	 */
475 	public void processRequest(String requestedPath, HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
476 
477 		try{
478 			// Initialize the Thread local for the Jawr context
479 			if(JmxUtils.isJmxEnabled()){
480 				ThreadLocalJawrContext.setJawrConfigMgrObjectName(JmxUtils.getMBeanObjectName(request.getContextPath(), resourceType));
481 			}
482 			
483 			ThreadLocalJawrContext.setRequest(request.getRequestURL().toString());
484 			
485 			RendererRequestUtils.setRequestDebuggable(request, jawrConfig);
486 			
487 			// manual reload request
488 			if (this.jawrConfig.getRefreshKey().length() > 0 && null != request.getParameter("refreshKey")
489 					&& this.jawrConfig.getRefreshKey().equals(request.getParameter("refreshKey"))) {
490 				this.configChanged(propertiesSource.getConfigProperties());
491 			}
492 	
493 			if (LOGGER.isDebugEnabled())
494 				LOGGER.debug("Request received for path:" + requestedPath);
495 	
496 			String contentType = getContentType(requestedPath, request);
497 			
498 			if(handleSpecificRequest(requestedPath, contentType, request, response)){
499 				return;
500 			}
501 	
502 			// Handle the strict mode
503 			boolean validBundle = isValidBundle(requestedPath);
504 			processRequest(requestedPath, request, response, contentType, validBundle);
505 		}finally{
506 			
507 			// Reset the Thread local for the Jawr context
508 			ThreadLocalJawrContext.reset();
509 		}
510 	}

512 	/***
513 	 * Handle the specific requests 
514 	 * @param requestedPath the requested path
515 	 * @param contentType the content type
516 	 * @param request the request
517 	 * @param response the response
518 	 * @return true if the request has been processed
519 	 * @throws ServletException if a servlet exception occurs 
520 	 * @throws IOException if a IO exception occurs 
521 	 */
522 	protected boolean handleSpecificRequest(String requestedPath, String contentType,
523 			HttpServletRequest request, HttpServletResponse response)
524 			throws ServletException, IOException {
525 		
526 		boolean processed = false;
527 		if (CLIENTSIDE_HANDLER_REQ_PATH.equals(requestedPath)) {
528 			this.clientSideScriptRequestHandler.handleClientSideHandlerRequest(request, response);
529 			processed = true;
530 		}else{
531 			
532 			// CSS images would be requested through this handler in case servletMapping is used
533 			// if( this.jawrConfig.isDebugModeOn() && !("".equals(this.jawrConfig.getServletMapping())) && null == request.getParameter(GENERATION_PARAM)) {
534 			if (JawrConstant.CSS_TYPE.equals(resourceType) && 
535 					!JawrConstant.CSS_TYPE.equals(getExtension(requestedPath)) &&
536 					this.imgMimeMap.containsKey(getExtension(requestedPath))) {
537 
538 				if (null == bundlesHandler.resolveBundleForPath(requestedPath)) {
539 					if (LOGGER.isDebugEnabled())
540 						LOGGER.debug("Path '" + requestedPath + "' does not belong to a bundle. Forwarding request to the server. ");
541 					request.getRequestDispatcher(requestedPath).forward(request, response);
542 					processed = true;
543 				}
544 			}
545 		}
546 
547 		return processed;
548 	}

550 	/***
551 	 * Returns true if the bundle is a valid bundle
552 	 * @param requestedPath the requested path
553 	 * @return true if the bundle is a valid bundle
554 	 */
555 	protected boolean isValidBundle(String requestedPath) {
556 		boolean validBundle = true;
557 		if(!jawrConfig.isDebugModeOn() && jawrConfig.isStrictMode()){
558 			validBundle = bundlesHandler.containsValidBundleHashcode(requestedPath);
559 		}
560 		return validBundle;
561 	}

563 	/***
564 	 * Process the request
565 	 * @param requestedPath the requested path
566 	 * @param request the request
567 	 * @param response the response
568 	 * @param validBundle the flag indicating if the requested bundle is a valid bundle
569 	 * @throws IOException if an IOException occurs
570 	 */
571 	protected void processRequest(String requestedPath,
572 			HttpServletRequest request, HttpServletResponse response,
573 			String contentType, boolean validBundle) throws IOException {
574 		
575 		boolean writeResponseHeader = false;
576 		
577 		if (this.jawrConfig.isDebugModeOn() && null != request.getParameter(GENERATION_PARAM))
578 			requestedPath = request.getParameter(GENERATION_PARAM);
579 		
580 		// If debug mode is off, check for If-Modified-Since and If-none-match headers and set response caching headers.
581 		if (!this.jawrConfig.isDebugModeOn()) {
582 			// If a browser checks for changes, always respond 'no changes'.
583 			if (validBundle && (null != request.getHeader(IF_MODIFIED_SINCE_HEADER) || null != request.getHeader(IF_NONE_MATCH_HEADER))) {
584 				response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
585 				if (LOGGER.isDebugEnabled())
586 					LOGGER.debug("Returning 'not modified' header. ");
587 				return;
588 			}
589 
590 			if(validBundle){
591 				// Add caching headers
592 				setResponseHeaders(response);
593 			}else{
594 				
595 				writeResponseHeader = illegalBundleRequestHandler.writeResponseHeader(requestedPath, request, response);
596 				if(!writeResponseHeader){
597 					// Add caching headers
598 					setResponseHeaders(response);
599 				}
600 			}
601 		}
602 		
603 		try {
604 			if(validBundle || illegalBundleRequestHandler.canWriteContent(requestedPath, request)){
605 				// By setting content type, the response writer will use appropriate encoding
606 				response.setContentType(contentType);
607 				writeContent(requestedPath, request, response);
608 				if (LOGGER.isDebugEnabled())
609 					LOGGER.debug("request succesfully attended");
610 			}else{
611 				if(!writeResponseHeader){
612 					logBundleNotFound(requestedPath);
613 					response.sendError(HttpServletResponse.SC_NOT_FOUND);
614 				}
615 			}
616 		} catch (EOFException eofex) {
617 			LOGGER.debug("Browser cut off response", eofex);
618 		} catch (ResourceNotFoundException e) {
619 			logBundleNotFound(requestedPath);
620 			response.setStatus(HttpServletResponse.SC_NOT_FOUND);
621 			return;
622 		}
623 	}

650 	 * @throws ResourceNotFoundException if the resource is not found
651 	 */
652 	protected void writeContent(String requestedPath, HttpServletRequest request,
653 			HttpServletResponse response) throws IOException, ResourceNotFoundException {
654 		
655 		// Send gzipped resource if user agent supports it.
656 		if (requestedPath.startsWith(BundleRenderer.GZIP_PATH_PREFIX)) {
657 			requestedPath = "/" + requestedPath.substring(BundleRenderer.GZIP_PATH_PREFIX.length(), requestedPath.length());
658 			response.setHeader("Content-Encoding", "gzip");
659 			bundlesHandler.streamBundleTo(requestedPath, response.getOutputStream());
660 		} else {
661 
662 			// In debug mode, we take in account the image generated from a StreamGenerator like classpath Image generator
663 			// The following code will rewrite the URL path for the generated images,
664 			// because in debug mode, we are retrieving the CSS ressources directly from the webapp
665 			// and if the CSS contains generated images, we should rewrite the URL.
666 			ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) servletContext.getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
667 			if (imgRsHandler != null && this.jawrConfig.isDebugModeOn() && resourceType.equals(JawrConstant.CSS_TYPE)) {
668 
669 				handleGeneratedCssInDebugMode(requestedPath, request,
670 						response, imgRsHandler);
671 			} else {
672 
673 				if(isValidRequestedPath(requestedPath)){
674 					Writer out = response.getWriter();
675 					bundlesHandler.writeBundleTo(requestedPath, out);	
676 				}else{
677 					throw new ResourceNotFoundException(requestedPath);
678 				}
679 			}
680 		}

685 	 * Handle the generated CSS content in debug mode.
686 	 * @param requestedPath the request path
687 	 * @param request the request
688 	 * @param response the response
689 	 * @param imgRsHandler the image resource handler
690 	 * @throws ResourceNotFoundException if the resource is not found
691 	 * @throws IOException if an IO exception occurs
692 	 */
693 	private void handleGeneratedCssInDebugMode(String requestedPath,
694 			HttpServletRequest request, HttpServletResponse response,
695 			ImageResourcesHandler imgRsHandler)
696 			throws ResourceNotFoundException, IOException {
697 		
698 		// Write the content of the CSS in the Stringwriter
699 		Writer writer = new StringWriter();
700 		Reader rd = rsReaderHandler.getResource(requestedPath);
701 		if(rd == null){
702 			throw new ResourceNotFoundException(requestedPath);
703 		}
704 		IOUtils.copy(rd, writer);
705 		String content = writer.toString();
706 
707 		String imageServletMapping = imgRsHandler.getJawrConfig().getServletMapping();
708 		
709 		if (imageServletMapping == null) {
710 			imageServletMapping = "";
711 		}
712 
713 		// Define the replacement pattern for the generated image (like jar:img/myImg.png)
714 		String relativeRootUrlPath = PathNormalizer.getRootRelativePath(getRequestPath(request));
715 		String replacementPattern = PathNormalizer.normalizePath("$1" + relativeRootUrlPath + imageServletMapping + "/$4_cbDebug/$7$8");
716 		
717 		Matcher matcher = GENERATED_IMG_PATTERN.matcher(content);
718 
719 		// Rewrite the images define in the classpath, to point to the image servlet
720 		StringBuffer result = new StringBuffer();
721 		while (matcher.find()) {
722 			matcher.appendReplacement(result, replacementPattern);
723 		}
724 		matcher.appendTail(result);
725 		Writer out = response.getWriter();
726 		out.write(result.toString());
727 	}
