46  	/***
47  	 * Returns the bundle renderer context.
48  	 * 
49  	 * @param request the request
50  	 * @param resourceType the resource type
51  	 * @return the bundle renderer context.
52  	 */
53  	public static BundleRendererContext getBundleRendererContext(HttpServletRequest request, BundleRenderer renderer) {
54  		String bundleRendererCtxAttributeName = BUNDLE_RENDERER_CONTEXT_ATTR_PREFIX+renderer.getResourceType();
55  		
56  		BundleRendererContext ctx = (BundleRendererContext) request.getAttribute(bundleRendererCtxAttributeName);
57  		if(ctx == null){
58  			String localeKey = renderer.getBundler().getConfig().getLocaleResolver().resolveLocaleCode(request);
59  	         boolean isGzippable = isRequestGzippable(request,renderer.getBundler().getConfig());
60  	         ctx = new BundleRendererContext(request.getContextPath(), localeKey, isGzippable,
61  	                 isSslRequest(request));
62  	         request.setAttribute(bundleRendererCtxAttributeName, ctx);
63  		}
64  		
65  		return ctx;
66  		
67  	}

81  	/***
82  	 * Determines wether gzip is suitable for the current request given the current config.
83  	 * 
84  	 * @param req
85  	 * @param jawrConfig
86  	 * @return
87  	 */
88  	public static boolean isRequestGzippable(HttpServletRequest req, JawrConfig jeesConfig) {
89  		boolean rets;
90  		// If gzip is completely off, return false.
91  		if (!jeesConfig.isGzipResourcesModeOn())
92  			rets = false;
93  		else if (req.getHeader("Accept-Encoding") != null && req.getHeader("Accept-Encoding").indexOf("gzip") != -1) {
94  
95  			// If gzip for IE6 or less is off, the user agent is checked to avoid compression.
96  			if (!jeesConfig.isGzipResourcesForIESixOn()) {
97  				String agent = req.getHeader("User-Agent");
98  				if (log.isDebugEnabled())
99  					log.debug("User-Agent for this request:" + agent);
100 
101 				if (null != agent && agent.indexOf("MSIE") != -1) {
102 					rets = agent.indexOf("MSIE 4") == -1 && agent.indexOf("MSIE 5") == -1 && agent.indexOf("MSIE 6") == -1;
103 					if (log.isDebugEnabled())
104 						log.debug("Gzip enablement for IE executed, with result:" + rets);
105 				} else
106 					rets = true;
107 			} else
108 				rets = true;
109 		} else
110 			rets = false;
111 		return rets;
112 	}

114 	/***
115 	 * Determines wether to override the debug settings. Sets the debugOverride status on ThreadLocalJawrContext
116 	 * 
117 	 * @param req the request
118 	 * @param jawrConfig the jawr config
119 	 * 
120 	 */
121 	public static void setRequestDebuggable(HttpServletRequest req, JawrConfig jawrConfig) {
122 
123 		// make sure we have set an overrideKey
124 		// make sure the overrideKey exists in the request
125 		// lastly, make sure the keys match
126 		if (jawrConfig.getDebugOverrideKey().length() > 0 && null != req.getParameter(JawrConstant.OVERRIDE_KEY_PARAMETER_NAME)
127 				&& jawrConfig.getDebugOverrideKey().equals(req.getParameter(JawrConstant.OVERRIDE_KEY_PARAMETER_NAME))) {
128 			ThreadLocalJawrContext.setDebugOverriden(true);
129 		} else {
130 			ThreadLocalJawrContext.setDebugOverriden(false);
131 		}
132 
133 		// Inherit the debuggable property of the session if the session is a debuggable one
134 		inheritSessionDebugProperty(req);
135 
136 	}

138 	/***
139 	 * Sets a request debuggable if the session is a debuggable session.
140 	 * 
141 	 * @param req the request
142 	 */
143 	public static void inheritSessionDebugProperty(HttpServletRequest request) {
144 
145 		HttpSession session = request.getSession(false);
146 		if (session != null) {
147 			String sessionId = session.getId();
148 
149 			JawrApplicationConfigManager appConfigMgr = (JawrApplicationConfigManager) session.getServletContext().getAttribute(
150 					JawrConstant.JAWR_APPLICATION_CONFIG_MANAGER);
151 
152 			// If the session ID is a debuggable session ID, activate debug mode for the request.
153 			if (appConfigMgr != null && appConfigMgr.isDebugSessionId(sessionId)) {
154 				ThreadLocalJawrContext.setDebugOverriden(true);
155 			}
156 		}
157 	}

170 	public static String getRenderedUrl(String newUrl, JawrConfig jawrConfig,
171 			String contextPath, boolean sslRequest) {
172 		String contextPathOverride = getContextPathOverride(sslRequest, jawrConfig);
173 		// If the contextPathOverride is not null and we are in production mode,
174 		// or if we are in debug mode but we should use the contextPathOverride even in debug mode
175 		// then use the contextPathOverride
176 		if(contextPathOverride != null && 
177 				((jawrConfig.isDebugModeOn() && jawrConfig.isUseContextPathOverrideInDebugMode()) ||
178 				!jawrConfig.isDebugModeOn())) {
179 			
180 				String override = contextPathOverride;
181 				// Blank override, create url relative to path
182 				if ("".equals(override)) {
183 					newUrl = newUrl.substring(1);
184 				} else
185 					newUrl = PathNormalizer.joinPaths(override, newUrl);
186 		} else
187 			newUrl = PathNormalizer.joinPaths(contextPath, newUrl);
188 		return newUrl;
189 	}

191 	/***
192 	 * Returns the context path depending on the request mode (SSL or not)
193 	 * 
194 	 * @param isSslRequest teh flag indicating that the request is an SSL request
195 	 * @return the context path depending on the request mode
196 	 */
197 	private static String getContextPathOverride(boolean isSslRequest, JawrConfig config) {
198 		String contextPathOverride = null;
199 		if (isSslRequest) {
200 			contextPathOverride = config.getContextPathSslOverride();
201 		} else {
202 			contextPathOverride = config.getContextPathOverride();
203 		}
204 		return contextPathOverride;
205 	}
