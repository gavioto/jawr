116 	/***
117 	 * <p>
118 	 * Counts how many times the substring appears in the larger String.
119 	 * </p>
120 	 * 
121 	 * <p>
122 	 * A <code>null</code> or empty ("") String input returns <code>0</code>.
123 	 * </p>
124 	 * 
125 	 * <pre>
126 	 * StringUtils.countMatches(null, *)       = 0
127 	 * StringUtils.countMatches(&quot;&quot;, *)         = 0
128 	 * StringUtils.countMatches(&quot;abba&quot;, null)  = 0
129 	 * StringUtils.countMatches(&quot;abba&quot;, &quot;&quot;)    = 0
130 	 * StringUtils.countMatches(&quot;abba&quot;, &quot;a&quot;)   = 2
131 	 * StringUtils.countMatches(&quot;abba&quot;, &quot;ab&quot;)  = 1
132 	 * StringUtils.countMatches(&quot;abba&quot;, &quot;xxx&quot;) = 0
133 	 * </pre>
134 	 * 
135 	 * @param str the String to check, may be null
136 	 * @param sub the substring to count, may be null
137 	 * @return the number of occurrences, 0 if either String is <code>null</code>
138 	 */
139 	public static int countMatches(String str, String sub) {
140 		if (isEmpty(str) || isEmpty(sub)) {
141 			return 0;
142 		}
143 		int count = 0;
144 		int idx = 0;
145 		while ((idx = str.indexOf(sub, idx)) != -1) {
146 			count++;
147 			idx += sub.length();
148 		}
149 		return count;
150 	}

154 	/***
155 	 * <p>
156 	 * Remove the last character from a String.
157 	 * </p>
158 	 * 
159 	 * <p>
160 	 * If the String ends in <code>\r\n</code>, then remove both of them.
161 	 * </p>
162 	 * 
163 	 * <pre>
164 	 * StringUtils.chop(null)          = null
165 	 * StringUtils.chop(&quot;&quot;)            = &quot;&quot;
166 	 * StringUtils.chop(&quot;abc \r&quot;)      = &quot;abc &quot;
167 	 * StringUtils.chop(&quot;abc\n&quot;)       = &quot;abc&quot;
168 	 * StringUtils.chop(&quot;abc\r\n&quot;)     = &quot;abc&quot;
169 	 * StringUtils.chop(&quot;abc&quot;)         = &quot;ab&quot;
170 	 * StringUtils.chop(&quot;abc\nabc&quot;)    = &quot;abc\nab&quot;
171 	 * StringUtils.chop(&quot;a&quot;)           = &quot;&quot;
172 	 * StringUtils.chop(&quot;\r&quot;)          = &quot;&quot;
173 	 * StringUtils.chop(&quot;\n&quot;)          = &quot;&quot;
174 	 * StringUtils.chop(&quot;\r\n&quot;)        = &quot;&quot;
175 	 * </pre>
176 	 * 
177 	 * @param str the String to chop last character from, may be null
178 	 * @return String without last character, <code>null</code> if null String input
179 	 */
180 	public static String chop(String str) {
181 		if (str == null) {
182 			return null;
183 		}
184 		int strLen = str.length();
185 		if (strLen < 2) {
186 			return EMPTY;
187 		}
188 		int lastIdx = strLen - 1;
189 		String ret = str.substring(0, lastIdx);
190 		char last = str.charAt(lastIdx);
191 		if (last == LF) {
192 			if (ret.charAt(lastIdx - 1) == CR) {
193 				return ret.substring(0, lastIdx - 1);
194 			}
195 		}
196 		return ret;
197 	}

199 	/***
200      * <p>Removes <code>separator</code> from the end of
201      * <code>str</code> if it's there, otherwise leave it alone.</p>
202      *
203      * <p>NOTE: This method changed in version 2.0.
204      * It now more closely matches Perl chomp.
205      * For the previous behavior, use {@link #substringBeforeLast(String, String)}.
206      * This method uses {@link String#endsWith(String)}.</p>
207      *
208      * <pre>
209      * StringUtils.chomp(null, *)         = null
210      * StringUtils.chomp("", *)           = ""
211      * StringUtils.chomp("foobar", "bar") = "foo"
212      * StringUtils.chomp("foobar", "baz") = "foobar"
213      * StringUtils.chomp("foo", "foo")    = ""
214      * StringUtils.chomp("foo ", "foo")   = "foo "
215      * StringUtils.chomp(" foo", "foo")   = " "
216      * StringUtils.chomp("foo", "foooo")  = "foo"
217      * StringUtils.chomp("foo", "")       = "foo"
218      * StringUtils.chomp("foo", null)     = "foo"
219      * </pre>
220      *
221      * @param str  the String to chomp from, may be null
222      * @param separator  separator String, may be null
223      * @return String without trailing separator, <code>null</code> if null String input
224      */
225     public static String chomp(String str, String separator) {
226         if (isEmpty(str) || separator == null) {
227             return str;
228         }
229         if (str.endsWith(separator)) {
230             return str.substring(0, str.length() - separator.length());
231         }
232         return str;
233     }
234 
235     /***
236      * <p>Remove a value if and only if the String ends with that value.</p>
237      *
238      * @param str  the String to chomp from, must not be null
239      * @param sep  the String to chomp, must not be null
240      * @return String without chomped ending
241      * @throws NullPointerException if str or sep is <code>null</code>
242      */
243     public static String chompLast(String str, String sep) {
244         if (str.length() == 0) {
245             return str;
246         }
247         String sub = str.substring(str.length() - sep.length());
248         if (sep.equals(sub)) {
249             return str.substring(0, str.length() - sep.length());
250         }
251         return str;
252     }
