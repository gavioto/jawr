34  /***
35   * ResourceBundlesHandler wrapper implementation that uses a ConcurrentHashMap to cache the resources. 
36   * Each resource is loaded only once from the ResourceBundlesHandler, then it is stored in cache and 
37   * retrieved from there in subsequent calls. Every method call not related to retrieving data is delegated
38   * to the wrapped implementation. 
39   * 
40   * @author Jordi Hernández Sellés
41   * @author Ibrahim Chaehoi
42   *
43   */
44  public class CachedResourceBundlesHandler implements ResourceBundlesHandler {
45  	
46  	/*** The resource bundle handler */
47  	private ResourceBundlesHandler rsHandler;
48  	
49  	/*** The cache map for text resource */
50  	private Map textCache;
51  	
52  	/*** The cache map for zip resource */
53  	private Map gzipCache;

55  	/***
56  	 * Build a cached wrapper around the supplied ResourceBundlesHandler. 
57  	 * @param rsHandler
58  	 */
59  	public CachedResourceBundlesHandler(ResourceBundlesHandler rsHandler) {
60  		super();
61  		this.rsHandler = rsHandler;
62  		this.textCache = ConcurrentCollectionsFactory.buildConcurrentHashMap();
63  		this.gzipCache = ConcurrentCollectionsFactory.buildConcurrentHashMap();
64  	}

101 	/* (non-Javadoc)
102 	 * @see net.jawr.web.resource.bundle.ResourceBundlesHandler#streamBundleTo(java.lang.String, java.io.OutputStream)
103 	 */
104 	public void streamBundleTo(String bundlePath, OutputStream out)
105 			throws ResourceNotFoundException {
106 		
107 		try {
108 			ByteBuffer gzip = (ByteBuffer) gzipCache.get(bundlePath);
109 			
110 			// If it's not cached yet
111 			if(null == gzip) {				
112 				// Stream the stored data
113 				ByteArrayOutputStream baOs = new ByteArrayOutputStream();
114 				BufferedOutputStream bfOs = new BufferedOutputStream(baOs);
115 				rsHandler.streamBundleTo(bundlePath,bfOs);
116 				
117 				// Copy the data into the ByteBuffer
118 				bfOs.close();
119 				gzip = ByteBuffer.wrap(baOs.toByteArray());
120 				
121 				// Cache the ByteBuffer
122 				gzipCache.put(bundlePath,gzip);
123 			}
124 			
125 			// Write bytes to the outputstream
126 			int max = gzip.capacity();
127 			for(int x = 0; x < max;x++)
128 				out.write(gzip.get(x)); // Use absolute get method
129 		
130 		} catch (IOException e) {
131 			throw new RuntimeException("Unexpected IOException writing bundle[" + bundlePath + "]",e);
132 		}
133 
134 	}

136 	/* (non-Javadoc)
137 	 * @see net.jawr.web.resource.bundle.ResourceBundlesHandler#writeBundleTo(java.lang.String, java.io.Writer)
138 	 */
139 	public void writeBundleTo(String bundlePath, Writer writer)
140 			throws ResourceNotFoundException {
141 		String text = (String) textCache.get(bundlePath);
142 		try {
143 			// If it's not cached yet
144 			if(null == text) {
145 				String charsetName = rsHandler.getConfig().getResourceCharset().name();
146 				ByteArrayOutputStream baOs = new ByteArrayOutputStream();
147 			    WritableByteChannel wrChannel = Channels.newChannel(baOs);
148 			    Writer tempWriter = Channels.newWriter(wrChannel, charsetName);
149 			    rsHandler.writeBundleTo(bundlePath, tempWriter);
150 			    text = baOs.toString(charsetName);
151 			    textCache.put(bundlePath,text);
152 			}
153 			
154 			// Write the text to the outputstream
155 			writer.write(text);
156 			writer.flush();
157 			
158 		} catch (IOException e) {
159 			throw new RuntimeException("Unexpected IOException writing bundle[" + bundlePath + "]",e);
160 		}
161 	}
