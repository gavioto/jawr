33  	/***
34  	 * Attempots to load a resource from the classpath, either usinf the caller's class loader or the current thread's 
35  	 * context classloader. 
36  	 * 
37  	 * @param resourcePath
38  	 * @param source
39  	 * @return
40  	 * @throws FileNotFoundException
41  	 */
42  	public static InputStream getResourceAsStream(String resourcePath, Object source) throws FileNotFoundException {
43  		
44  		// Try the current classloader
45  		InputStream is = source.getClass().getResourceAsStream(resourcePath);
46  		
47  		// Weblogic 10 likes this one better..
48  		if(null == is) {
49  			ClassLoader cl = source.getClass().getClassLoader();
50  			if(null != cl)
51  				is = cl.getResourceAsStream(resourcePath);
52  		}
53  		
54  		// If current classloader failed, try with the Threads context classloader. If that fails ott, the resource is either not on the 
55  		// classpath or inaccessible from the current context. 
56  		
57  		if(null == is) {
58  			is = Thread.currentThread().getContextClassLoader().getResourceAsStream(resourcePath);
59  		}
60  		
61  		// Try to retrieve by URL
62  		if(null == is) {
63  			try {
64  				URL url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
65  				
66  				// Last chance, hack in the classloader
67  				if(null == url) {
68  					ClassLoader threadClassLoader = Thread.currentThread().getContextClassLoader();
69  					try {
70  						 Thread.currentThread().setContextClassLoader(source.getClass().getClassLoader());
71  						 url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
72  					}
73  					finally {
74  						 Thread.currentThread().setContextClassLoader(threadClassLoader);
75  					}
76  					
77  				}
78  				if(null == url){
79  					throw new FileNotFoundException( resourcePath + " could not be found. ");
80  				}
81  				
82  				is = new FileInputStream(new File(url.getFile()));			
83  			} 
84  			catch (IOException e) {
85  				throw new FileNotFoundException( resourcePath + " could not be found. ");
86  			}
87  		}
88  		
89  		// Everything failed... exception is trown. 
90  		if(null == is)
91  			throw new FileNotFoundException( resourcePath + " could not be found. ");
92  		return is;
93  	}

96  	 * Attempts to find the URL of a resource from the classpath, either usinf the caller's class loader or the current thread's 
97  	 * context classloader. 
98  	 * 
99  	 * @param resourcePath the resource path
100 	 * @param source the object
101 	 * @return the URL.
102 	 * @throws ResourceNotFoundException if the resource is not found
103 	 */
104 	public static URL getResourceURL(String resourcePath, Object source) throws ResourceNotFoundException {
105 		
106 		// Try the current classloader
107 		URL url = source.getClass().getResource(resourcePath);
108 		
109 		// Weblogic 10 likes this one better..
110 		if(null == url) {
111 			ClassLoader cl = source.getClass().getClassLoader();
112 			if(null != cl)
113 				url = cl.getResource(resourcePath);
114 		}
115 		
116 		// If current classloader failed, try with the Threads context classloader. If that fails ott, the resource is either not on the 
117 		// classpath or inaccessible from the current context. 
118 		
119 		if(null == url) {
120 			url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
121 		}
122 		
123 		// Try to retrieve by URL
124 		if(null == url) {
125 				url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
126 				
127 				// Last chance, hack in the classloader
128 				if(null == url) {
129 					ClassLoader threadClassLoader = Thread.currentThread().getContextClassLoader();
130 					try {
131 						 Thread.currentThread().setContextClassLoader(source.getClass().getClassLoader());
132 						 url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
133 					}
134 					finally {
135 						 Thread.currentThread().setContextClassLoader(threadClassLoader);
136 					}
137 					
138 				}
139 				if(null == url){
140 					throw new ResourceNotFoundException( resourcePath + " could not be found. ");
141 				}
142 		}
143 		
144 		// Everything failed... exception is trown. 
145 		if(null == url)
146 			throw new ResourceNotFoundException( resourcePath + " could not be found. ");
147 		return url;

151 	 * Builds a class instance using reflection, by using its classname. The class must have a zero-arg constructor. 
152 	 * @param classname the class to build an instance of. 
153 	 * @return
154 	 */
155 	public static Object buildObjectInstance(String classname) {
156 		Object rets = null;
157 		Class clazz = locateClass(classname);
158 		try {
159 			rets = clazz.newInstance();
160 		}catch(Exception e) {
161 				throw new RuntimeException(e.getMessage() 
162 											+ " [The custom class " 
163 											+ classname 
164 											+ " could not be instantiated, check wether it is available on the classpath and" 
165 											+ " verify that it has a zero-arg constructor].\n" 
166 											+ " The specific error message is: " + e.getClass().getName() + ":" + e.getMessage(),e);			
167 		}
168 		return rets;
169 	}

171 	private static Class locateClass(String classname) {
172 		Class clazz = null;
173 		try {
174 			clazz = Class.forName(classname);
175 			
176 		} catch (Exception e) {
177 			// Try the second approach 
178 		}
179 		if(null == clazz) {
180 			try {
181 				clazz = Class.forName(classname, true, new ClassLoaderResourceUtils().getClass().getClassLoader());
182 			} catch (Exception e) {
183 				throw new RuntimeException(e.getMessage() 
184 											+ " [The custom class " 
185 											+ classname 
186 											+ " could not be instantiated, check wether it is available on the classpath and" 
187 											+ " verify that it has a zero-arg constructor].\n" 
188 											+ " The specific error message is: " + e.getClass().getName() + ":" + e.getMessage(),e);
189 		  }
190 		}
191 		return clazz;
192 	}

194 	/***
195 	 * Builds a class instance using reflection, by using its classname. The class must have a zero-arg constructor. 
196 	 * @param classname the class to build an instance of. 
197 	 * @return
198 	 */
199 	public static Object buildObjectInstance(String classname, Object[] params) {
200 		Object rets = null;
201 		
202 		Class[] paramTypes = new Class[params.length];
203 		for(int x = 0; x < params.length; x++){
204 			paramTypes[x] = params[x].getClass();
205 		}
206 		
207 		try {
208 			Class clazz = locateClass(classname);
209 			rets = clazz.getConstructor(paramTypes).newInstance(params);
210 			
211 		} catch (Exception e) {
212 			throw new RuntimeException(e.getMessage() 
213 										+ " [The custom class " 
214 										+ classname 
215 										+ " could not be instantiated, check wether it is available on the classpath and" 
216 										+ " verify that it has a zero-arg constructor].\n" 
217 										+ " The specific error message is: " + e.getClass().getName() + ":" + e.getMessage(),e);
218 		}
219 		return rets;
220 	}
