14  package net.jawr.web.resource.bundle.global.preprocessor.css.smartsprites;
15  
16  import java.io.InputStream;
17  import java.io.Reader;
18  
19  import net.jawr.web.JawrConstant;
20  import net.jawr.web.config.JawrConfig;
21  import net.jawr.web.resource.bundle.generator.GeneratorRegistry;
22  import net.jawr.web.resource.handler.reader.FileSystemResourceReader;
23  import net.jawr.web.resource.handler.reader.PathPrefixedServletContextResourceReader;
24  import net.jawr.web.resource.handler.reader.ResourceReader;
25  import net.jawr.web.resource.handler.reader.StreamResourceReader;
26  import net.jawr.web.resource.handler.reader.TextResourceReader;
27  
28  /***
29   * This class defines the resource reader for the resources generated by the smartsprites
30   * global preprocessor : the generated sprites and the modified CSS files.
31   *   
32   * @author Ibrahim Chaehoi
33   *
34   */
35  public class CssSmartSpritesResourceReader implements TextResourceReader, StreamResourceReader {
36  
37  	/*** The smartsprites resource reader */
38  	private ResourceReader resourceReader;
39  	
40  	/*** The jawr config */
41  	private JawrConfig jawrConfig;
42  	
43  	/***
44  	 * Constructor
45  	 * @param tempDir the smartsprites temporary directory
46  	 * @param jawrConfig the jawr config
47  	 */
48  	public CssSmartSpritesResourceReader(String tempDir, JawrConfig jawrConfig) {
49  		
50  		this.jawrConfig = jawrConfig;
51  		if(jawrConfig.isWorkingDirectoryInWebApp()){
52  			resourceReader = new PathPrefixedServletContextResourceReader(jawrConfig.getContext(), jawrConfig, tempDir+JawrConstant.CSS_SMARTSPRITES_TMP_DIR);
53  		}else{
54  			resourceReader = new FileSystemResourceReader(tempDir+JawrConstant.CSS_SMARTSPRITES_TMP_DIR, jawrConfig);
55  		}
56  		
57  	}
58  	
59  	/* (non-Javadoc)
60  	 * @see net.jawr.web.resource.handler.ResourceReader#getResource(java.lang.String)
61  	 */
62  	public Reader getResource(String resourceName) {
63  		
64  		Reader rd = null;
65  		if(!jawrConfig.isDebugModeOn()){
66  			((TextResourceReader) resourceReader).getResource(resourceName);
67  		}
68  		
69  		return rd;
70  	}
71  
72  	/* (non-Javadoc)
73  	 * @see net.jawr.web.resource.handler.ResourceReader#getResource(java.lang.String, boolean)
74  	 */
75  	public Reader getResource(String resourceName, boolean processingBundle) {
76  		
77  		Reader rd = null;
78  		if(processingBundle){
79  			String path = resourceName;
80  			if(jawrConfig.getGeneratorRegistry().isPathGenerated(path)){
81  				path = JawrConstant.SPRITE_GENERATED_CSS_DIR+path.replace(":", "/");
82  			}
83  			rd = ((TextResourceReader) resourceReader).getResource(path, processingBundle);
84  		}
85  		
86  		return rd;
87  	}
88  
89  	/* (non-Javadoc)
90  	 * @see net.jawr.web.resource.handler.stream.StreamResourceReader#getResourceAsStream(java.lang.String)
91  	 */
92  	public InputStream getResourceAsStream(String resourceName) {
93  		
94  		return getResourceAsStream(resourceName, false);
95  	}
96  
97  	/* (non-Javadoc)
98  	 * @see net.jawr.web.resource.handler.stream.StreamResourceReader#getResourceAsStream(java.lang.String, boolean)
99  	 */
100 	public InputStream getResourceAsStream(String resourceName,
101 			boolean processingBundle) {
102 		
103 		String path = resourceName;
104 		GeneratorRegistry generatorRegistry = jawrConfig.getGeneratorRegistry();
105 		if(generatorRegistry.isGeneratedImage(path)){
106 			path = JawrConstant.SPRITE_GENERATED_IMG_DIR+path.replace(":", "/");
107 		}
108 		return ((StreamResourceReader) resourceReader).getResourceAsStream(path, processingBundle);
109 	}
