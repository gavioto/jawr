70  	 * @see net.jawr.web.resource.bundle.renderer.BundleRenderer#renderBundleLinks(java.lang.String, net.jawr.web.resource.bundle.renderer.BundleRendererContext, javax.servlet.jsp.Writer)
71  	 */
72  	public void renderBundleLinks(String requestedPath, BundleRendererContext ctx,
73  			Writer out) throws IOException {
74  		
75  		boolean debugOn = bundler.getConfig().isDebugModeOn();
76  		JoinableResourceBundle bundle = bundler.resolveBundleForPath(requestedPath);
77  
78  		if (null == bundle)
79  			return;
80  
81  		// If the global bundles had been added before, it will not be included again.
82  		if(!ctx.isGlobalBundleAdded()){
83  			renderGlobalBundleLinks(ctx, out, debugOn);
84  		}
85  		
86  		// If there is a fixed URL for production mode it is rendered and method returns.  
87      	if(!debugOn && null != bundle.getAlternateProductionURL()){
88      		if(ctx.getIncludedBundles().add(bundle.getId()))
89      			out.write(renderLink(bundle.getAlternateProductionURL()));
90      		return;
91      	}
92      	
93          renderBundleLinks(bundle, requestedPath, ctx, out, debugOn, true);
94  		
95          if (debugOn) {
96  			addComment("Finished adding members resolved by " + requestedPath, out);
97  		}
98  	}

100 	/***
101 	 * Renders the links for a bundle
102 	 * @param bundle the bundle
103 	 * @param requestedPath thed requested path
104 	 * @param ctx the renderer context
105 	 * @param out the writer
106 	 * @param debugOn the debug flag
107 	 * @param renderDependencyLinks the flag indicating if we must render the dependency links
108 	 * @throws IOException if an IOException occurs
109 	 */
110 	private void renderBundleLinks(JoinableResourceBundle bundle,
111 			String requestedPath, BundleRendererContext ctx, Writer out,
112 			boolean debugOn, boolean renderDependencyLinks) throws IOException {
113 		if (debugOn) {
114 			addComment("Start adding members resolved by '" + requestedPath + "'. Bundle id is: '" + bundle.getId() + "'", out);
115 		}
116 
117 		// If DWR is being used, add a path var to the page
118 		if (null != bundler.getConfig().getDwrMapping() && ctx.getIncludedBundles().add(ID_SCRIPT_DWR_PATH)) {
119 
120 			String contextPath = ctx.getContextPath();
121 			StringBuffer sb = DWRParamWriter.buildRequestSpecificParams(contextPath, PathNormalizer.joinPaths(contextPath, bundler.getConfig()
122 					.getDwrMapping()));
123 			out.write(sb.toString());
124 		}
125 
126 		// Include the bundle if it has not been included yet
127 		if(ctx.getIncludedBundles().add(bundle.getId())){
128 			
129 			if(renderDependencyLinks){
130 				renderBundleDependenciesLinks(requestedPath, ctx, out, debugOn,
131 						bundle.getDependencies());
132 			}
133 			
134 			// Retrieve the name or names of bundle(s) that belong to/with the requested path.
135 			if(isForcedToRenderIeCssBundleInDebug(ctx, debugOn)){
136 	        	
137 				ResourceBundlePathsIterator it = bundler.getBundlePaths(false, bundle.getId(), new ConditionalCommentRenderer(out), ctx.getVariantKey());
138 		        while(it.hasNext()){
139 					String bundlePath = it.nextPath();
140 					renderIeCssBundleLink(ctx, out, bundlePath);
141 				}
142 	        }else{
143 	        	ResourceBundlePathsIterator it = bundler.getBundlePaths(bundle.getId(), new ConditionalCommentRenderer(out), ctx.getVariantKey());
144 	        	renderBundleLinks(it, ctx, debugOn, out);
145 	        }
146 		}else{
147 			if (debugOn) {
148 				addComment("The bundle '" + bundle.getId() + "' is already included in the page.", out);
149 			}
150 		}
151 	}

153 	/***
154 	 * Renders the links for the global bundles
155 	 * @param ctx the context
156 	 * @param out the writer
157 	 * @param debugOn the debug flag
158 	 * @throws IOException if an IOException occurs.
159 	 */
160 	private void renderGlobalBundleLinks(BundleRendererContext ctx, Writer out,
161 			boolean debugOn) throws IOException {
162 		
163 		if (debugOn) {
164 			addComment("Start adding global members.", out);
165 		}
166 		if(isForcedToRenderIeCssBundleInDebug(ctx, debugOn)){
167 			ResourceBundlePathsIterator resourceBundleIterator = bundler.getGlobalResourceBundlePaths(false, new ConditionalCommentRenderer(out), ctx.getVariantKey());
168 			while(resourceBundleIterator.hasNext()){
169 				String globalBundlePath = resourceBundleIterator.nextPath();
170 				renderIeCssBundleLink(ctx, out, globalBundlePath);
171 			}
172 		}else{
173 			
174 			ResourceBundlePathsIterator resourceBundleIterator = bundler.getGlobalResourceBundlePaths(new ConditionalCommentRenderer(out), ctx.getVariantKey());
175 			renderBundleLinks(resourceBundleIterator,
176 					ctx, debugOn, out);
177 		}
178 		
179 		ctx.setGlobalBundleAdded(true);
180 		if (debugOn) {
181 			addComment("Finished adding global members.", out);
182 		}
183 	}

185 	/***
186 	 * Renders the bundle links for the bundle dependencies
187 	 * @param requestedPath the request path
188 	 * @param ctx the context
189 	 * @param out the writer
190 	 * @param debugOn the debug flag
191 	 * @param dependencies the dependencies
192 	 * @throws IOException if an IOException occurs.
193 	 */
194 	private void renderBundleDependenciesLinks(String requestedPath,
195 			BundleRendererContext ctx, Writer out, boolean debugOn,
196 			List dependencies) throws IOException {
197 		
198 		if(dependencies != null && !dependencies.isEmpty()){
199 			for (Iterator iterator = dependencies.iterator(); iterator
200 					.hasNext();) {
201 				JoinableResourceBundle dependencyBundle = (JoinableResourceBundle) iterator.next();
202 				if(debugOn){
203 					addComment("Start adding dependency '"+dependencyBundle.getId()+"'", out);
204 				}
205 				renderBundleLinks(dependencyBundle, requestedPath, ctx, out, debugOn, false);
206 				if(debugOn){
207 					addComment("Finished adding dependency '"+dependencyBundle.getId()+"'", out);
208 				}
209 			}
210 		}
211 	}

226 	/***
227 	 * Renders the CSS link to retrieve the CSS bundle for IE in debug mode.
228 	 * @param ctx the context
229 	 * @param out the writer
230 	 * @param bundle the bundle
231 	 * @throws IOException if an IOException occurs
232 	 */
233 	private void renderIeCssBundleLink(BundleRendererContext ctx, Writer out,
234 			String bundlePath) throws IOException {
235 		Random randomSeed = new Random();
236 		int random = randomSeed.nextInt();
237 		if (random < 0)
238 			random *= -1;
239 		String path = GeneratorRegistry.IE_CSS_GENERATOR_PREFIX+GeneratorRegistry.PREFIX_SEPARATOR+bundlePath;
240 		path = PathNormalizer.createGenerationPath(path, bundler.getConfig().getGeneratorRegistry())+"&d="+random;
241 		out.write(createBundleLink(path, ctx.getContextPath(), ctx.isSslRequest()));
242 	}

246 	/***
247 	 * Renders the bundle links for the resource iterator passed in parameter
248 	 * @param it the iterator on the bundles 
249 	 * @param contextPath the context path
250 	 * @param includedBundles the included bundles
251 	 * @param useGzip the flag indicating if we use gzip or not
252 	 * @param isSslRequest the flag indicating if it's an SSL request or not
253 	 * @param debugOn the flag indicating if we are in debug mode or not
254 	 * @param out the output writer
255 	 * @throws IOException if an IO exception occurs
256 	 */
257 	private void renderBundleLinks(ResourceBundlePathsIterator it,
258 			BundleRendererContext ctx, boolean debugOn, Writer out) throws IOException {
259 	
260 		String contextPath = ctx.getContextPath();
261 		boolean useGzip = ctx.isUseGzip();
262 		boolean isSslRequest = ctx.isSslRequest();
263 		
264 		// Add resources to the page as links.
265 		Random randomSeed = new Random();
266 		while (it.hasNext()) {
267 			String resourceName = it.nextPath();
268 
269 			if( resourceName != null){
270 				// In debug mode, all the resources are included separately and use a random parameter to avoid caching.
271 				// If useRandomParam is set to false, the links are created without the random parameter.
272 				if (debugOn && useRandomParam && !bundler.getConfig().getGeneratorRegistry().isPathGenerated(resourceName)) {
273 					int random = randomSeed.nextInt();
274 					if (random < 0)
275 						random *= -1;
276 					out.write(createBundleLink(resourceName + "?d=" + random, contextPath, isSslRequest));
277 				} else if (!debugOn && useGzip){
278 					out.write(createGzipBundleLink(resourceName, contextPath, isSslRequest));
279 				}else{
280 					out.write(createBundleLink(resourceName, contextPath, isSslRequest));
281 				}
282 				
283 				if(debugOn && !ctx.getIncludedResources().add(resourceName)){
284 					addComment("The resource '" + resourceName + "' is already included in the page.", out);
285 				}
286 			}
287 		}
288 	}

304 	 * Creates a link to a bundle in the page, prepending the gzip prefix to its identifier.
305 	 * 
306 	 * @param resourceName the resource name
307 	 * @param contextPath the context path
308 	 * @return the link to the gzip bundle in the page
309 	 */
310 	protected String createGzipBundleLink(String resourceName, String contextPath, boolean isSslRequest) {
311 		// remove '/' from start of name
312 		resourceName = resourceName.substring(1, resourceName.length());
313 		return createBundleLink(BundleRenderer.GZIP_PATH_PREFIX + resourceName, contextPath, isSslRequest);
314 	}

316 	/***
317 	 * Creates a link to a bundle in the page.
318 	 * 
319 	 * @param bundleId the bundle ID
320 	 * @param contextPath the context path
321 	 * @return the link to a bundle in the page
322 	 */
323 	protected String createBundleLink(String bundleId, String contextPath, boolean isSslRequest) {
324 
325 		// When debug mode is on and the resource is generated the path must include a parameter
326 		if (bundler.getConfig().isDebugModeOn() && bundler.getConfig().getGeneratorRegistry().isPathGenerated(bundleId)) {
327 			bundleId = PathNormalizer.createGenerationPath(bundleId, bundler.getConfig().getGeneratorRegistry());
328 		}
329 		String fullPath = PathNormalizer.joinPaths(bundler.getConfig().getServletMapping(), bundleId);
330 
331 		fullPath = RendererRequestUtils.getRenderedUrl(fullPath, bundler.getConfig(), contextPath, isSslRequest);
332 		
333 		// allow debugOverride to pass through on the generated urls
334 		if (ThreadLocalJawrContext.isDebugOverriden()) {
335 			fullPath = PathNormalizer.addGetParameter(fullPath, "overrideKey", bundler.getConfig().getDebugOverrideKey());
336 		}
337 
338 		return renderLink(fullPath);
339 	}
