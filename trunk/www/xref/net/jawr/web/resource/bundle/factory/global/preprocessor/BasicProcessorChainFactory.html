30  /***
31   * This class defines the global preprocessor factory.
32   * 
33   * @author Ibrahim Chaehoi
34   * 
35   */
36  public class BasicProcessorChainFactory implements
37  		GlobalPreprocessorChainFactory {
38  
39  	/*** The user-defined preprocessors */
40  	private Map customPreprocessors = new HashMap();
41  
42  	/*
43  	 * (non-Javadoc)
44  	 * 
45  	 * @seenet.jawr.web.resource.bundle.factory.global.preprocessor.
46  	 * GlobalPreprocessorChainFactory#setCustomGlobalPreprocessors(java.util.Map)
47  	 */
48  	public void setCustomGlobalPreprocessors(Map keysClassNames) {
49  		
50  		for(Iterator it = keysClassNames.entrySet().iterator(); it.hasNext();){
51  			
52  			Entry entry = (Entry) it.next();
53  			GlobalPreprocessor customGlobalPreprocessor = 
54  				(GlobalPreprocessor) ClassLoaderResourceUtils.buildObjectInstance((String) entry.getValue());
55  			
56  			String key = (String) entry.getKey();			
57  			customPreprocessors.put(key, new CustomGlobalPreprocessorChainedWrapper(key, customGlobalPreprocessor));
58  		}		
59  	}
60  
61  	/*
62  	 * (non-Javadoc)
63  	 * 
64  	 * @see
65  	 * net.jawr.web.resource.bundle.factory.processor.ProcessorChainFactory#
66  	 * buildDefaultProcessorChain()
67  	 */
68  	public GlobalPreprocessor buildDefaultProcessorChain() {
69  
70  		return new EmptyGlobalPreprocessor();
71  	}
72  
73  	/*
74  	 * (non-Javadoc)
75  	 * 
76  	 * @see
77  	 * net.jawr.web.resource.bundle.factory.processor.ProcessorChainFactory#
78  	 * buildProcessorChain(java.lang.String)
79  	 */
80  	public GlobalPreprocessor buildProcessorChain(String processorKeys) {
81  
82  		if (null == processorKeys)
83  			return null;
84  		else if (JawrConstant.EMPTY_GLOBAL_PREPROCESSOR_ID
85  				.equals(processorKeys))
86  			return new EmptyGlobalPreprocessor();
87  
88  		StringTokenizer tk = new StringTokenizer(processorKeys, ",");
89  
90  		AbstractChainedGlobalPreprocessor chain = null;
91  		while (tk.hasMoreTokens())
92  			chain = addOrCreateChain(chain, tk.nextToken());
93  
94  		return chain;
95  	}
96  
97  	/***
98  	 * Creates an AbstractChainedGlobalPreprocessor. If the supplied
99  	 * chain is null, the new chain is returned. Otherwise it is added to the
100 	 * existing chain.
101 	 * 
102 	 * @param chain
103 	 *            the chained post processor
104 	 * @param key
105 	 *            the id of the post processor
106 	 * @return the chained post processor, with the new post processor.
107 	 */
108 	private AbstractChainedGlobalPreprocessor addOrCreateChain(
109 			AbstractChainedGlobalPreprocessor chain, String key) {
110 
111 		AbstractChainedGlobalPreprocessor toAdd;
112 
113 		if (customPreprocessors.get(key) == null) {
114 			toAdd = buildProcessorByKey(key);
115 		} else{
116 			toAdd = (AbstractChainedGlobalPreprocessor) customPreprocessors
117 				.get(key);
118 		}
119 		
120 		AbstractChainedGlobalPreprocessor newChainResult = null;
121 		if (chain == null) {
122 			newChainResult = toAdd;
123 		}else{
124 			chain.addNextProcessor(toAdd);
125 			newChainResult = chain;
126 		}
127 
128 		return newChainResult;
129 	}
130 
131 	/***
132 	 * Build the global preprocessor from the ID given in parameter
133 	 * 
134 	 * @param key the ID of the preprocessor
135 	 * @return a global preprocessor
136 	 */
137 	private AbstractChainedGlobalPreprocessor buildProcessorByKey(String key) {
138 
139 		AbstractChainedGlobalPreprocessor processor = null;
140 
141 		if (key.equals(JawrConstant.GLOBAL_CSS_SMARTSPRITES_PREPROCESSOR_ID)) {
142 			processor = new CssSmartSpritesGlobalPreprocessor();
143 		}
144 
145 		return processor;
146 	}
