14  package net.jawr.web.resource.bundle.renderer;
15  
16  import net.jawr.web.JawrConstant;
17  import net.jawr.web.resource.bundle.handler.ResourceBundlesHandler;
18  
19  /***
20   * Renderer that creates css link tags. 
21   * 
22   * @author Jordi Hernández Sellés
23   * @author Ibrahim Chaehoi
24   */
25  public class CSSHTMLBundleLinkRenderer extends AbstractBundleLinkRenderer implements BundleRenderer{
26      
27  	/*** The start tag */
28  	private static final String PRE_TAG = "<link rel=\"stylesheet\" type=\"text/css\" media=\"";
29      
30  	/*** The HREF prefix */
31  	private static final String MID_TAG = "\" href=\"";
32      
33  	/*** The end tag */
34  	private static final String POST_TAG = "\" />\n";
35      
36  	/*** The end HTML tag */
37  	private static final String POST_HTML_TAG = "\" >\n";
38      
39  	/*** The end XHTML tag */
40  	private static final String POST_XHTML_EXT_TAG = "\" ></link>\n";
41      
42  	/*** The XHTML flavor */
43  	public static final String FLAVORS_XHTML = "xhtml";
44  
45  	/*** The XHTML extended flavor */
46  	public static final String FLAVORS_XHTML_EXTENDED = "xhtml_ext";
47      
48  	/*** The HTML flavor */
49  	public static final String FLAVORS_HTML = "html";
50      
51  	/*** The closing tag flavor */
52      private static String closingFlavor = POST_TAG;
53      
54      /*** The media attribute */
55      private String media;
56      
57      /***
58       * Constructor
59       * @param bundler the bundler
60       * @param useRandomParam the flag indicating if we use the random flag
61       * @param media the media
62       */
63      public CSSHTMLBundleLinkRenderer(ResourceBundlesHandler bundler, boolean useRandomParam, String media) {
64          super(bundler, useRandomParam);
65          
66          this.media = null == media ? "screen" : media;
67      }
68      
69      /***
70       * Utility method to get the closing tag value based on 
71       * a config parameter. 
72       * @param flavor the flavor
73       * @return the closing tag
74       */
75      public static void setClosingTag(String flavor) {
76      	
77      	if(FLAVORS_XHTML_EXTENDED.equalsIgnoreCase(flavor)) {
78          	closingFlavor = POST_XHTML_EXT_TAG;
79          }
80          else if(FLAVORS_HTML.equalsIgnoreCase(flavor))
81          	closingFlavor = POST_HTML_TAG;
82          else closingFlavor = POST_TAG;
83      	
84      }
85      
86      /* (non-Javadoc)
87  	 * @see net.jawr.web.resource.bundle.renderer.BundleRenderer#getResourceType()
88  	 */
89  	public String getResourceType() {
90  		return JawrConstant.CSS_TYPE;
91  	}
92  
93      /* (non-Javadoc)
94       * @see net.jawr.web.resource.bundle.renderer.AbstractBundleLinkRenderer#createBundleLink(java.lang.String, java.lang.String)
95       */
96      protected String renderLink(String fullPath) {
97      	
98          StringBuffer sb = new StringBuffer(PRE_TAG);
99  		sb.append(media).append(MID_TAG)
100 						.append(fullPath)	
101 						.append(closingFlavor); 
102         return sb.toString();
103     }
