35  /***
36   * This factory is used to build JoinableResourceBundle from the generated properties mapping file, 
37   * which contains all calculated information about the bundle.
38   * 
39   * @author Ibrahim Chaehoi
40   * 
41   */
42  public class FullMappingPropertiesBasedBundlesHandlerFactory {
43  
44  	/*** The post processor chain factory */
45  	private PostProcessorChainFactory chainFactory;

47  	/*** The resource type */
48  	private String resourceType;

51  	private ResourceHandler rsHandler;

53  	/***
54  	 * Create a PropertiesBasedBundlesHandlerFactory using the specified properties.
55  	 * 
56  	 * @param resourceType js or css
57  	 * @param rsHandler ResourceHandler to access files.
58  	 * @param generatorRegistry the generator registry
59  	 * @param chainFactory the post processor chain factory
60  	 */
61  	public FullMappingPropertiesBasedBundlesHandlerFactory(String resourceType, 
62  			ResourceHandler rsHandler,
63  			PostProcessorChainFactory chainFactory) {
64  
65  		this.resourceType = resourceType;
66  		this.chainFactory = chainFactory;
67  		this.rsHandler = rsHandler;
68  	}

70  	/***
71  	 * Returns the list of joinable resource bundle
72  	 * 
73  	 * @return the list of joinable resource bundle
74  	 */
75  	public List getResourceBundles(Properties properties) {
76  		
77  		PropertiesConfigHelper props = new PropertiesConfigHelper(properties, resourceType);
78  		String fileExtension = "." + resourceType;
79  
80  		// Initialize custom bundles
81  		List customBundles = new ArrayList();
82  		// Check if we should use the bundle names property or
83  		// find the bundle name using the bundle id declaration :
84  		// jawr.<type>.bundle.<name>.id
85  		if (null != props
86  				.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_NAMES)) {
87  			StringTokenizer tk = new StringTokenizer(
88  					props
89  							.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_NAMES),
90  					",");
91  			while (tk.hasMoreTokens()) {
92  				customBundles.add(buildJoinableResourceBundle(props, tk.nextToken()
93  						.trim(), fileExtension, rsHandler));
94  			}
95  		} else {
96  			Iterator bundleNames = props.getPropertyBundleNameSet().iterator();
97  			while (bundleNames.hasNext()) {
98  				customBundles.add(buildJoinableResourceBundle(props, 
99  						(String) bundleNames.next(), fileExtension,
100 						rsHandler));
101 			}
102 		}
103 		
104 		return customBundles;
105 	}

107 	/***
108 	 * Create a JoinableResourceBundle based on the properties file.
109 	 * 
110 	 * @param props the properties config helper
111 	 * @param bundleName the bundle name
112 	 * @param rsHandler the resource handler
113 	 * @return BundleDefinition
114 	 */
115 	private JoinableResourceBundle buildJoinableResourceBundle(
116 			PropertiesConfigHelper props, String bundleName, String fileExtension,
117 			ResourceHandler rsHandler) {
118 
119 		// Id for the bundle
120 		String bundleId = props.getCustomBundleProperty(bundleName,
121 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_ID);
122 
123 		InclusionPattern inclusionPattern = getInclusionPattern(props, bundleName);
124 		JoinableResourceBundleImpl bundle = new JoinableResourceBundleImpl(
125 				bundleId, bundleName, fileExtension, inclusionPattern,
126 				rsHandler);
127 
128 		// Override bundle postprocessor
129 		String bundlePostProcessors = props.getCustomBundleProperty(bundleName,
130 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_POSTPROCESSOR);
131 		if (StringUtils.isNotEmpty(bundlePostProcessors)) {
132 			bundle.setBundlePostProcessor(chainFactory
133 					.buildPostProcessorChain(bundlePostProcessors));
134 		}
135 
136 		// Override unitary postprocessor
137 		String unitaryPostProcessors = props
138 				.getCustomBundleProperty(
139 						bundleName,
140 						PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_FILE_POSTPROCESSOR);
141 		if (StringUtils.isNotEmpty(unitaryPostProcessors)) {
142 			bundle.setUnitaryPostProcessor(chainFactory
143 					.buildPostProcessorChain(unitaryPostProcessors));
144 		}
145 
146 		// Set conditional comment for IE, in case one is specified
147 		String explorerConditionalCondition = props
148 				.getCustomBundleProperty(
149 						bundleName,
150 						PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_IE_CONDITIONAL_EXPRESSION);
151 		if (StringUtils.isNotEmpty(explorerConditionalCondition)) {
152 			bundle
153 					.setExplorerConditionalExpression(explorerConditionalCondition);
154 		}
155 
156 		// Sets the alternate URL for production mode.
157 		String alternateProductionURL = props
158 				.getCustomBundleProperty(
159 						bundleName,
160 						PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_PRODUCTION_ALT_URL);
161 		if (StringUtils.isNotEmpty(alternateProductionURL)) {
162 			bundle
163 					.setAlternateProductionURL(props
164 							.getCustomBundleProperty(
165 									bundleName,
166 									PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_PRODUCTION_ALT_URL));
167 		}
168 		
169 		// Sets the licence path lists.
170 		Set licencePathList = props
171 				.getCustomBundlePropertyAsSet(
172 						bundleName,
173 						PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_LICENCE_PATH_LIST);
174 		if (!licencePathList.isEmpty()) {
175 			bundle.setLicensesPathList(licencePathList);
176 		}
177 
178 		String mappingsProperty = props.getCustomBundleProperty(bundleName,
179 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_MAPPINGS);
180 		if (null == mappingsProperty) {
181 			throw new IllegalArgumentException(
182 					"No mappings were defined for the bundle with name:"
183 							+ bundleName
184 							+ ". Please specify at least one in configuration. ");
185 		}
186 		
187 		// Add the mappings
188 		List mappings = new ArrayList();
189 		StringTokenizer tk = new StringTokenizer(mappingsProperty, ",");
190 		while (tk.hasMoreTokens()) {
191 			String mapping = tk.nextToken().trim();
192 			mappings.add(mapping);
193 		}
194 
195 		bundle.setMappings(mappings);
196 		
197 		Set localeKeys = new HashSet();
198 		Set tmpLocales = props.getCustomBundlePropertyAsSet(bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_LOCALE_VARIANTS);
199 		for (Iterator iterator = tmpLocales.iterator(); iterator.hasNext();) {
200 			String localVariantKey = (String) iterator.next();
201 			if(StringUtils.isNotEmpty(localVariantKey)){
202 				localeKeys.add(localVariantKey);
203 				String hashcode = props.getCustomBundleProperty(bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_HASHCODE_VARIANT+localVariantKey);
204 				bundle.setBundleDataHashCode(localVariantKey, hashcode);
205 			}
206 		}
207 		
208 		if(!localeKeys.isEmpty()){
209 			bundle.setLocaleVariantKeys(Collections.list(Collections
210 					.enumeration(localeKeys)));
211 		}
212 		
213 		String hashcode = props.getCustomBundleProperty(bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_HASHCODE);
214 		bundle.setBundleDataHashCode(null, hashcode);
215 	
216 		
217 		return bundle;
218 	}

220 	/***
221 	 * Returns the inclusion pattern for a bundle
222 	 * 
223 	 * @param props the properties helper
224 	 * @param bundleName the bundle name
225 	 * @return the inclusion pattern for a bundle
226 	 */
227 	private InclusionPattern getInclusionPattern(PropertiesConfigHelper props, String bundleName) {
228 		// Wether it's global or not
229 		boolean isGlobal = Boolean
230 				.valueOf(
231 						props
232 								.getCustomBundleProperty(
233 										bundleName,
234 										PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_GLOBAL_FLAG,
235 										"false")).booleanValue();
236 
237 		// Set order if its a global bundle
238 		int order = 0;
239 		if (isGlobal) {
240 			order = Integer.parseInt(props.getCustomBundleProperty(bundleName,
241 					PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_ORDER, "0"));
242 		}
243 
244 		// Use only with debug mode on
245 		boolean isDebugOnly = Boolean
246 				.valueOf(
247 						props
248 								.getCustomBundleProperty(
249 										bundleName,
250 										PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEBUGONLY,
251 										"false")).booleanValue();
252 
253 		// Use only with debug mode off
254 		boolean isDebugNever = Boolean
255 				.valueOf(
256 						props
257 								.getCustomBundleProperty(
258 										bundleName,
259 										PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEBUGNEVER,
260 										"false")).booleanValue();
261 
262 		return new InclusionPattern(isGlobal, order, isDebugOnly, isDebugNever);
263 	}
