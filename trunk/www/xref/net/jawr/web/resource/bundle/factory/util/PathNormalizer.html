39  	/***
40  	 * Removes the URL prefix defined in the configuration from a path. If the prefix contains a variant information, it adds it to the name.
41  	 * 
42  	 * @param path the path
43  	 * @return the path without the prefix
44  	 */
45  	public static String removeVariantPrefixFromPath(String path) {
46  		// Remove first slash
47  		path = path.substring(1, path.length());
48  
49  		// eval the existence of a suffix
50  		String prefix = path.substring(0, path.indexOf("/"));
51  
52  		// The prefix also contains variant information after a '.'
53  		if (prefix.indexOf('.') != -1) {
54  			String suffix = '_' + prefix.substring(prefix.indexOf('.') + 1) + path.substring(path.lastIndexOf('.'));
55  			path = path.substring(path.indexOf("/"), path.lastIndexOf('.')) + suffix;
56  		} else
57  			path = path.substring(path.indexOf("/"), path.length());
58  		return path;
59  	}

91  	/***
92  	 * Normalizes two paths and joins them as a single path. 
93  	 * @param prefix
94  	 * @param path
95  	 * @return
96  	 */
97  	public static final String joinPaths(String prefix,String path) {
98  		
99  		if(prefix.startsWith(JawrConstant.HTTP_URL_PREFIX) || prefix.startsWith(JawrConstant.HTTPS_URL_PREFIX) || prefix.startsWith("//")){
100 			return joinDomainToPath(prefix, path);
101 		}
102 		
103 		prefix = PathNormalizer.normalizePath(prefix);
104 		path = PathNormalizer.normalizePath(path);
105 		StringBuffer sb = new StringBuffer(JawrConstant.URL_SEPARATOR);
106 		if(!"".equals(prefix))
107 			sb.append(prefix).append(JawrConstant.URL_SEPARATOR);
108 		sb.append(path);
109 		
110 		return sb.toString();
111 		
112 	}

132 	/***
133 	 * Removes leading and trailing separators from a path, and removes 
134 	 * double separators (// is replaced by /). 
135 	 * @param path
136 	 * @return
137 	 */
138 	public static final String normalizePath(String path) {
139 		path = path.replaceAll("//", JawrConstant.URL_SEPARATOR);
140 		StringTokenizer tk = new StringTokenizer(path,JawrConstant.URL_SEPARATOR);
141 		StringBuffer sb = new StringBuffer();
142 		while(tk.hasMoreTokens()) {
143 			sb.append(tk.nextToken());
144 			if(tk.hasMoreTokens())
145 				sb.append(JawrConstant.URL_SEPARATOR);
146 		}
147 		return sb.toString();
148 		
149 	}

152 	 * Normalizes all the paths in a Set. 
153 	 * @param paths
154 	 * @return
155 	 */
156 	public static final Set normalizePaths(Set paths) {
157 		Set ret = new HashSet();
158 		for(Iterator it = paths.iterator();it.hasNext();) {
159 			String path = normalizePath((String)it.next());
160 			ret.add(path);
161 		}
162 		return ret;
163 	}

217     /***
218      * Determines the parent path of a filename or a directory.
219      * 
220      * <pre>
221      * PathUtils.getParentPath( null )                           = ""
222      * PathUtils.getParentPath( "" )                             = ""
223      * PathUtils.getParentPath( "/" )                            = "/"
224      * PathUtils.getParentPath( "/usr/local/" )                  = "/usr/local/"
225      * PathUtils.getRelativePath( "/usr/local/bin/java.sh" )     = ""/usr/local/bin/"
226      * </pre>
227      * 
228      * @param path the path
229      * @return the parent path.
230      */
231     public static String getParentPath(String path){
232     
233     	if(StringUtils.isEmpty(path)){
234     		return "";
235     	}
236     	
237     	if(path.length() > 1 && path.endsWith(JawrConstant.URL_SEPARATOR)){
238     		path = path.substring(0, path.length()-2);
239     	}
240     	int index = path.lastIndexOf(JawrConstant.URL_SEPARATOR);
241     	if(index > 0){
242     		return path.substring(0, index+1);
243     	}
244     	return JawrConstant.URL_SEPARATOR; 
245     }

247     /***
248      * Determines the relative path of a filename from a base directory.
249      * This method is useful in building relative links within pages of
250      * a web site.  It provides similar functionality to Anakia's
251      * <code>$relativePath</code> context variable.  The arguments to
252      * this method may contain either forward or backward slashes as
253      * file separators.  The relative path returned is formed using
254      * forward slashes as it is expected this path is to be used as a
255      * link in a web page (again mimicking Anakia's behavior).
256      * <p/>
257      * This method is thread-safe.
258      * <br/>
259      * <pre>
260      * PathUtils.getRelativePath( null, null )                                   = ""
261      * PathUtils.getRelativePath( null, "/usr/local/java/bin" )                  = ""
262      * PathUtils.getRelativePath( "/usr/local/", null )                          = ""
263      * PathUtils.getRelativePath( "/usr/local/", "/usr/local/java/bin" )         = ".."
264      * PathUtils.getRelativePath( "/usr/local/", "/usr/local/java/bin/java.sh" ) = "../.."
265      * PathUtils.getRelativePath( "/usr/local/java/bin/java.sh", "/usr/local/" ) = ""
266      * </pre>
267      *
268      * @param basedir The base directory.
269      * @param filename The filename that is relative to the base
270      * directory.
271      * @return The relative path of the filename from the base
272      * directory.  This value is not terminated with a forward slash.
273      * A zero-length string is returned if: the filename is not relative to
274      * the base directory, <code>basedir</code> is null or zero-length,
275      * or <code>filename</code> is null or zero-length.
276      */
277     public static final String getRelativePath( String basedir, String filename )
278     {
279         basedir = uppercaseDrive(basedir);
280         filename = uppercaseDrive(filename);
281 
282         /*
283          * Verify the arguments and make sure the filename is relative
284          * to the base directory.
285          */
286         if ( basedir == null || basedir.length() == 0 || filename == null
287             || filename.length() == 0 || !filename.startsWith( basedir ) )
288         {
289             return "";
290         }
291 
292         /*
293          * Normalize the arguments.  First, determine the file separator
294          * that is being used, then strip that off the end of both the
295          * base directory and filename.
296          */
297         String separator = determineSeparator( filename );
298         basedir = StringUtils.chompLast( basedir, separator );
299         filename = StringUtils.chompLast( filename, separator );
300 
301         /*
302          * Remove the base directory from the filename to end up with a
303          * relative filename (relative to the base directory).  This
304          * filename is then used to determine the relative path.
305          */
306         String relativeFilename = filename.substring( basedir.length() );
307 
308         return determineRelativePath( relativeFilename, separator );
309     }

311     /***
312      * Concatenates a filename to a base web path.
313      * If the base path doesn't end with "/", it will consider as base path the parent folder of the base path passed as parameter.
314      * 
315      * <pre>
316      * PathUtils.concatWebPath("", null));								   			= null
317 	 * PathUtils.concatWebPath(null, null));								   		= null
318 	 * PathUtils.concatWebPath(null, ""));								   			= null
319 	 * PathUtils.concatWebPath(null, "a"));								   			= null			
320 	 * PathUtils.concatWebPath(null, "/a"));								   		= "/a" 
321  	 * PathUtils.concatWebPath( "/css/folder/subfolder/", "icons/img.png" )      	= "/css/folder/subfolder/icons/img.png"
322      * PathUtils.concatWebPath( "/css/folder/subfolder/style.css", "icons/img.png") = "/css/folder/subfolder/icons/img.png"
323      * PathUtils.concatWebPath( "/css/folder/", "../icons/img.png" )      			= "/css/icons/img.png"
324      * PathUtils.concatWebPath( "/css/folder/style.css", "../icons/img.png" ) 		= "/css/icons/img.png"
325      * </pre>
326      * 
327      * @param basePath the base path
328      * @param fullFilenameToAdd the file name to add
329      * @return the concatenated path, or null if invalid
330      */
331     public static String concatWebPath(String basePath, String fullFilenameToAdd){
332      	
333     	if(fullFilenameToAdd == null || basePath == null && (fullFilenameToAdd.length() == 0
334     			|| fullFilenameToAdd.charAt(0) != JawrConstant.URL_SEPARATOR_CHAR)){
335     		return null;
336     	}
337     	
338     	if(basePath == null){
339     		basePath = "";
340     	}
341     	// If the basePath is pointing to a file, set the base path to the parent directory
342     	if(basePath.length() > 1 && basePath.charAt(basePath.length()-1) != '/'){
343     		basePath = getParentPath(basePath);
344     	}
345     	
346     	int len = basePath.length();
347     	String fullPath = null;
348         if (len == 0) {
349         	return doNormalizeIgnoreOtherSeparator(fullFilenameToAdd, true);
350         }
351         
352     	char ch = basePath.charAt(len - 1);
353         if (ch == JawrConstant.URL_SEPARATOR_CHAR) {
354         	fullPath = basePath + fullFilenameToAdd;
355         } else {
356         	fullPath = basePath + '/' + fullFilenameToAdd;
357         }
358         
359         return doNormalizeIgnoreOtherSeparator(fullPath, true); 
360         
361     }

363     /***
364      * Internal method to perform the normalization.
365      *
366      * @param filename  the filename
367      * @param keepSeparator  true to keep the final separator
368      * @return the normalized filename
369      */
370     private static String doNormalizeIgnoreOtherSeparator(String filename, boolean keepSeparator) {
371         if (filename == null) {
372             return null;
373         }
374         int size = filename.length();
375         if (size == 0) {
376             return filename;
377         }
378         int prefix = 0;
379 //        int prefix = getPrefixLength(filename);
380 //        if (prefix < 0) {
381 //            return null;
382 //        }
383         
384         char[] array = new char[size + 2];  // +1 for possible extra slash, +2 for arraycopy
385         filename.getChars(0, filename.length(), array, 0);
386         
387         // add extra separator on the end to simplify code below
388         boolean lastIsDirectory = true;
389         if (array[size - 1] != JawrConstant.URL_SEPARATOR_CHAR) {
390             array[size++] = JawrConstant.URL_SEPARATOR_CHAR;
391             lastIsDirectory = false;
392         }
393         
394         // adjoining slashes
395         for (int i = prefix + 1; i < size; i++) {
396             if (array[i] == JawrConstant.URL_SEPARATOR_CHAR && array[i - 1] == JawrConstant.URL_SEPARATOR_CHAR) {
397                 System.arraycopy(array, i, array, i - 1, size - i);
398                 size--;
399                 i--;
400             }
401         }
402         
403         // dot slash
404         for (int i = prefix + 1; i < size; i++) {
405             if (array[i] == JawrConstant.URL_SEPARATOR_CHAR && array[i - 1] == '.' &&
406                     (i == prefix + 1 || array[i - 2] == JawrConstant.URL_SEPARATOR_CHAR)) {
407                 if (i == size - 1) {
408                     lastIsDirectory = true;
409                 }
410                 System.arraycopy(array, i + 1, array, i - 1, size - i);
411                 size -=2;
412                 i--;
413             }
414         }
415         
416         // double dot slash
417         outer:
418         for (int i = prefix + 2; i < size; i++) {
419             if (array[i] == JawrConstant.URL_SEPARATOR_CHAR && array[i - 1] == '.' && array[i - 2] == '.' &&
420                     (i == prefix + 2 || array[i - 3] == JawrConstant.URL_SEPARATOR_CHAR)) {
421                 if (i == prefix + 2) {
422                     return null;
423                 }
424                 if (i == size - 1) {
425                     lastIsDirectory = true;
426                 }
427                 int j;
428                 for (j = i - 4 ; j >= prefix; j--) {
429                     if (array[j] == JawrConstant.URL_SEPARATOR_CHAR) {
430                         // remove b/../ from a/b/../c
431                         System.arraycopy(array, i + 1, array, j + 1, size - i);
432                         size -= (i - j);
433                         i = j + 1;
434                         continue outer;
435                     }
436                 }
437                 // remove a/../ from a/../c
438                 System.arraycopy(array, i + 1, array, prefix, size - i);
439                 size -= (i + 1 - prefix);
440                 i = prefix + 1;
441             }
442         }
443         
444         if (size <= 0) {  // should never be less than 0
445             return "";
446         }
447         if (size <= prefix) {  // should never be less than prefix
448             return new String(array, 0, size);
449         }
450         if (lastIsDirectory && keepSeparator) {
451             return new String(array, 0, size);  // keep trailing separator
452         }
453         return new String(array, 0, size - 1);  // lose trailing separator
454     }

456     /***
457      * This method can calculate the relative path between two pathes on a web site.
458      * <br/>
459      * <pre>
460      * PathUtils.getRelativeWebPath( null, null )                                          = ""
461      * PathUtils.getRelativeWebPath( null, "http://plexus.codehaus.org/" )                 = ""
462      * PathUtils.getRelativeWebPath( "http://plexus.codehaus.org/", null )                 = ""
463      * PathUtils.getRelativeWebPath( "http://plexus.codehaus.org/",
464      *                      "http://plexus.codehaus.org/plexus-utils/index.html" )        = "plexus-utils/index.html"
465      * PathUtils.getRelativeWebPath( "http://plexus.codehaus.org/plexus-utils/index.html",
466      *                      "http://plexus.codehaus.org/"                                 = "../../"
467      * </pre>
468      *
469      * @param oldPath
470      * @param newPath
471      * @return a relative web path from <code>oldPath</code>.
472      */
473     public static final String getRelativeWebPath( final String oldPath, final String newPath )
474     {
475         if ( StringUtils.isEmpty( oldPath ) || StringUtils.isEmpty( newPath ) )
476         {
477             return "";
478         }
479 
480         String resultPath = buildRelativePath( newPath, oldPath, '/' );
481 
482         if ( newPath.endsWith( "/" ) && !resultPath.endsWith( "/" ) )
483         {
484             return resultPath + "/";
485         }
486 
487         return resultPath;
488     }

490     /***
491      * This method can calculate the relative path between two pathes on a file system.
492      * <br/>
493      * <pre>
494      * PathUtils.getRelativeFilePath( null, null )                                   = ""
495      * PathUtils.getRelativeFilePath( null, "/usr/local/java/bin" )                  = ""
496      * PathUtils.getRelativeFilePath( "/usr/local", null )                           = ""
497      * PathUtils.getRelativeFilePath( "/usr/local", "/usr/local/java/bin" )          = "java/bin"
498      * PathUtils.getRelativeFilePath( "/usr/local", "/usr/local/java/bin/" )         = "java/bin"
499      * PathUtils.getRelativeFilePath( "/usr/local/java/bin", "/usr/local/" )         = "../.."
500      * PathUtils.getRelativeFilePath( "/usr/local/", "/usr/local/java/bin/java.sh" ) = "java/bin/java.sh"
501      * PathUtils.getRelativeFilePath( "/usr/local/java/bin/java.sh", "/usr/local/" ) = "../../.."
502      * PathUtils.getRelativeFilePath( "/usr/local/", "/bin" )                        = "../../bin"
503      * PathUtils.getRelativeFilePath( "/bin", "/usr/local/" )                        = "../usr/local"
504      * </pre>
505      * Note: On Windows based system, the <code>/</code> character should be replaced by <code>\</code> character.
506      *
507      * @param oldPath
508      * @param newPath
509      * @return a relative file path from <code>oldPath</code>.
510      */
511     public static final String getRelativeFilePath( final String oldPath, final String newPath )
512     {
513         if ( StringUtils.isEmpty( oldPath ) || StringUtils.isEmpty( newPath ) )
514         {
515             return "";
516         }
517 
518         // normalise the path delimiters
519         String fromPath = new File( oldPath ).getPath();
520         String toPath = new File( newPath ).getPath();
521 
522         // strip any leading slashes if its a windows path
523         if ( toPath.matches( "^//[a-zA-Z]:" ) )
524         {
525             toPath = toPath.substring( 1 );
526         }
527         if ( fromPath.matches( "^//[a-zA-Z]:" ) )
528         {
529             fromPath = fromPath.substring( 1 );
530         }
531 
532         // lowercase windows drive letters.
533         if ( fromPath.startsWith( ":", 1 ) )
534         {
535             fromPath = Character.toLowerCase( fromPath.charAt( 0 ) ) + fromPath.substring( 1 );
536         }
537         if ( toPath.startsWith( ":", 1 ) )
538         {
539             toPath = Character.toLowerCase( toPath.charAt( 0 ) ) + toPath.substring( 1 );
540         }
541 
542         // check for the presence of windows drives. No relative way of
543         // traversing from one to the other.
544         if ( ( toPath.startsWith( ":", 1 ) && fromPath.startsWith( ":", 1 ) )
545                         && ( !toPath.substring( 0, 1 ).equals( fromPath.substring( 0, 1 ) ) ) )
546         {
547             // they both have drive path element but they dont match, no
548             // relative path
549             return null;
550         }
551 
552         if ( ( toPath.startsWith( ":", 1 ) && !fromPath.startsWith( ":", 1 ) )
553                         || ( !toPath.startsWith( ":", 1 ) && fromPath.startsWith( ":", 1 ) ) )
554         {
555             // one has a drive path element and the other doesnt, no relative
556             // path.
557             return null;
558         }
559 
560         String resultPath = buildRelativePath( toPath, fromPath, File.separatorChar );
561 
562         if ( newPath.endsWith( File.separator ) && !resultPath.endsWith( File.separator ) )
563         {
564             return resultPath + File.separator;
565         }
566 
567         return resultPath;
568     }

574     /***
575      * Determines the relative path of a filename.  For each separator
576      * within the filename (except the leading if present), append the
577      * "../" string to the return value.
578      *
579      * @param filename The filename to parse.
580      * @param separator The separator used within the filename.
581      * @return The relative path of the filename.  This value is not
582      * terminated with a forward slash.  A zero-length string is
583      * returned if: the filename is zero-length.
584      */
585     private static final String determineRelativePath( String filename,
586                                                        String separator )
587     {
588         if ( filename.length() == 0 )
589         {
590             return "";
591         }
592 
593 
594         /*
595          * Count the slashes in the relative filename, but exclude the
596          * leading slash.  If the path has no slashes, then the filename
597          * is relative to the current directory.
598          */
599         int slashCount = StringUtils.countMatches( filename, separator ) - 1;
600         if ( slashCount <= 0 )
601         {
602             return ".";
603         }
604 
605         /*
606          * The relative filename contains one or more slashes indicating
607          * that the file is within one or more directories.  Thus, each
608          * slash represents a "../" in the relative path.
609          */
610         StringBuffer sb = new StringBuffer();
611         for ( int i = 0; i < slashCount; i++ )
612         {
613             sb.append( "../" );
614         }
615 
616         /*
617          * Finally, return the relative path but strip the trailing
618          * slash to mimic Anakia's behavior.
619          */
620         return StringUtils.chop( sb.toString() );
621     }

623     /***
624      * Helper method to determine the file separator (forward or
625      * backward slash) used in a filename.  The slash that occurs more
626      * often is returned as the separator.
627      *
628      * @param filename The filename parsed to determine the file
629      * separator.
630      * @return The file separator used within <code>filename</code>.
631      * This value is either a forward or backward slash.
632      */
633     private static final String determineSeparator( String filename )
634     {
635         int forwardCount = StringUtils.countMatches( filename, "/" );
636         int backwardCount = StringUtils.countMatches( filename, "//" );

659     private static final String buildRelativePath( String toPath,  String fromPath, final char separatorChar )
660     {
661         // use tokeniser to traverse paths and for lazy checking
662         StringTokenizer toTokeniser = new StringTokenizer( toPath, String.valueOf( separatorChar ) );
663         StringTokenizer fromTokeniser = new StringTokenizer( fromPath, String.valueOf( separatorChar ) );
664 
665         int count = 0;
666 
667         // walk along the to path looking for divergence from the from path
668         while ( toTokeniser.hasMoreTokens() && fromTokeniser.hasMoreTokens() )
669         {
670             if ( separatorChar == '//' )
671             {
672                 if ( !fromTokeniser.nextToken().equalsIgnoreCase( toTokeniser.nextToken() ) )
673                 {
674                     break;
675                 }
676             }
677             else
678             {
679                 if ( !fromTokeniser.nextToken().equals( toTokeniser.nextToken() ) )
680                 {
681                     break;
682                 }
683             }
684 
685             count++;
686         }
687 
688         // reinitialise the tokenisers to count positions to retrieve the
689         // gobbled token
690 
691         toTokeniser = new StringTokenizer( toPath, String.valueOf( separatorChar ) );
692         fromTokeniser = new StringTokenizer( fromPath, String.valueOf( separatorChar ) );
693 
694         while ( count-- > 0 )
695         {
696             fromTokeniser.nextToken();
697             toTokeniser.nextToken();
698         }
699 
700         StringBuffer relativePath = new StringBuffer();
701 
702         // add back refs for the rest of from location.
703         while ( fromTokeniser.hasMoreTokens() )
704         {
705             fromTokeniser.nextToken();
706 
707             relativePath.append("..");
708 
709             if ( fromTokeniser.hasMoreTokens() )
710             {
711             	relativePath.append(separatorChar);
712             }
713         }
714 
715         if ( relativePath.length() != 0 && toTokeniser.hasMoreTokens() )
716         {
717         	relativePath.append(separatorChar);
718         }
719 
720         // add fwd fills for whatevers left of newPath.
721         while ( toTokeniser.hasMoreTokens() )
722         {
723         	relativePath.append(toTokeniser.nextToken());
724 
725             if ( toTokeniser.hasMoreTokens() )
726             {
727             	relativePath.append(separatorChar);
728             }
729         }
730         return relativePath.toString();
731     }
