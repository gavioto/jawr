14  package net.jawr.web.resource.bundle.postprocess.impl.yui;
15  
16  import java.io.BufferedReader;
17  import java.io.IOException;
18  import java.io.StringReader;
19  
20  import net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus;
21  
22  import org.apache.log4j.Logger;
23  import org.mozilla.javascript.ErrorReporter;
24  import org.mozilla.javascript.EvaluatorException;
25  
26  /***
27   * Implementation of Rhino's ErrorReporter, used as a callback to log errors encuntered while 
28   * parsing a javascript file. 
29   * 
30   * @author Jordi Hernández Sellés
31   * @author Ibrahim Chaehoi
32   */
33  public class YUIErrorReporter implements ErrorReporter {
34  	
35  	/*** The logger */
36  	private static final Logger log = Logger.getLogger(YUIErrorReporter.class);
37  	
38  	/*** The bundle processing status */
39  	private BundleProcessingStatus status;
40  	
41  	/*** The bundle content */
42  	private StringBuffer bundleData;
43  	
44  	/*** The error line */
45  	private int errorLine;
46  	
47  	/*** The YUI error message */
48  	private String yuiErrorMessage;
49  
50  	/***
51  	 * Constructor
52  	 * @param status Current bundling status. 
53  	 * @param bundleData Contents of the bundle, used when an error occurs to display the conlifcting line. 
54  	 */
55  	public YUIErrorReporter(BundleProcessingStatus status,StringBuffer bundleData) {
56  		super();
57  		this.status = status;
58  		this.bundleData = bundleData;
59  	}
60  
61  
62  
63  	/* (non-Javadoc)
64  	 * @see org.mozilla.javascript.ErrorReporter#error(java.lang.String, java.lang.String, int, java.lang.String, int)
65  	 */
66  	public void error(String message, String sourceName, int line, String lineSource, int lineOffset) {
67  		
68  		// Only log the first error...
69  		if(this.errorLine < 1) {
70  			this.errorLine = line;
71  			this.yuiErrorMessage = message;
72  		}
73  		else if(this.errorLine == line) {
74  			this.yuiErrorMessage += "] [" + message;
75  		}
76  	}
77  
78  	
79  	/***
80  	 * Creates an EvaluatorException that will be thrown by Rhino. 
81  	 * 
82  	 * @param message a String describing the error
83  	 * @param sourceName a String describing the JavaScript source where the error occured; typically a filename or URL
84  	 * @param line the line number associated with the error
85  	 * @param lineSource the text of the line (may be null)
86  	 * @param lineOffset the offset into lineSource where problem was detected
87  	 * 
88  	 * @return an EvaluatorException that will be thrown.
89  	 */
90  	public EvaluatorException runtimeError(String message, String sourceName, int line, String lineSource, int lineOffset) {
91  		StringBuffer errorMsg = new StringBuffer("YUI failed to minify the bundle with id: '" + status.getCurrentBundle().getId() + "'.\n");
92  		errorMsg.append("YUI error message(s):[").append(this.yuiErrorMessage).append("]\n");
93  		errorMsg.append("The error happened at this point in your javascript: \n");
94  		errorMsg.append("_______________________________________________\n...\n");
95  		
96  		BufferedReader rd = new BufferedReader(new StringReader(bundleData.toString()));
97  		String s;
98  		int totalLines = 0;
99  		int start = this.errorLine - 10;
100 		
101 		try {
102 			while((s=rd.readLine())!=null){
103 				totalLines++;
104 				if(totalLines >= start && totalLines <= this.errorLine){
105 					errorMsg.append(s);
106 					if(totalLines == this.errorLine)
107 						errorMsg.append(" <-- ERROR");
108 					errorMsg.append("\n");
109 				}
110 			}
111 		} catch (IOException e) {
112 			errorMsg.append("[Jawr suffered an IOException while attempting to show the faulty script]");
113 		}
114 		
115 		errorMsg.append("_______________________________________________");
116 		errorMsg.append("\nIf you can't find the error, try to check the scripts using JSLint (http://www.jslint.com/) to find the conflicting part of the code.\n");
117 
118 		return new EvaluatorException(errorMsg.toString(),status.getCurrentBundle().getName(),this.errorLine);
119 	}
120 
121 	/* (non-Javadoc)
122 	 * @see org.mozilla.javascript.ErrorReporter#warning(java.lang.String, java.lang.String, int, java.lang.String, int)
123 	 */
124 	public void warning(String message, String sourceName, int line, String lineSource, int lineOffset) {
125 		if(log.isDebugEnabled())
126 			log.debug(message);
127 	}

129 }
