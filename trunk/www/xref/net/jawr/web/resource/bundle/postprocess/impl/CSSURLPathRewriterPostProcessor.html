35  /***
36   * Single file postprocessor used to rewrite CSS URLs according to the new relative locations of the references when
37   * added to a bundle. Since the path changes, the URLs must be rewritten accordingly.  
38   * URLs in css files are expected to be according to the css spec (see http://www.w3.org/TR/REC-CSS2/syndata.html#value-def-uri). 
39   * Thus, single double, or no quotes enclosing the url are allowed (and remain as they are after rewriting). Escaped parens and quotes 
40   * are allowed within the url.   
41   *  
42   * @author Jordi Hernández Sellés
43   * @author Ibrahim Chaehoi
44   */
45  public class CSSURLPathRewriterPostProcessor extends
46  		AbstractChainedResourceBundlePostProcessor {
47  	
48  	/*** This variable is used to fake the gzip prefix for the full bundle path */
49  	private static final String FAKE_BUNDLE_PREFIX = "/prefix/";
50  
51  	/*** Logger */
52  	private static Logger log = Logger.getLogger(CSSURLPathRewriterPostProcessor.class);
53  	
54  	/*** The URL separator */
55  	private static final String URL_SEPARATOR = "/";
56  
57  	/*** The url pattern */
58  	private static final Pattern urlPattern = Pattern.compile(	"url//(//s*" // 'url(' and any number of whitespaces 
59  																+ "((//////))|[^)])*" // any sequence of characters, except an unescaped ')'
60  																+ "//s*//)",  // Any number of whitespaces, then ')'
61  																Pattern.CASE_INSENSITIVE); // works with 'URL('
62  	
63  	/***
64  	 * Constructor
65  	 */
66  	public CSSURLPathRewriterPostProcessor() {
67  		super(PostProcessFactoryConstant.URL_PATH_REWRITER);
68  	}

70  	/* (non-Javadoc)
71  	 * @see net.jawr.web.resource.bundle.postprocess.impl.AbstractChainedResourceBundlePostProcessor#doPostProcessBundle(net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus, java.lang.StringBuffer)
72  	 */
73  	protected StringBuffer doPostProcessBundle(BundleProcessingStatus status,
74  			StringBuffer bundleData) throws IOException {
75  		
76  		String data = bundleData.toString();
77  		
78  		JawrConfig jawrConfig = status.getJawrConfig();
79  		
80  		// Retrieve the full bundle path, so we will be able to define the relative path for the css images
81  		String bundleName = getFinalFullBundlePath(status, jawrConfig);
82  		
83  		// Rewrite each css image url path
84  		Matcher matcher = urlPattern.matcher(data);
85  		StringBuffer sb = new StringBuffer();
86  		while(matcher.find()) {
87  		
88  			String url = getUrlPath(matcher.group(), bundleName, status);
89  			matcher.appendReplacement(sb, RegexUtil.adaptReplacementToMatcher(url));
90  		}
91  		matcher.appendTail(sb);
92  		return sb;
93  	}

96  	/***
97  	 * Transform a matched url so it points to the proper relative path with respect to the given path.  
98  	 * @param match the matched URL
99  	 * @param fullBundlePath the full bundle path
100 	 * @param status the bundle processing status
101 	 * @return the image URL path
102 	 * @throws IOException if an IO exception occurs
103 	 */
104 	private String getUrlPath(String match, String fullBundlePath, BundleProcessingStatus status) throws IOException {
105 
106 		JawrConfig jawrConfig = status.getJawrConfig();
107 				
108 		// Retrieve the image servlet mapping
109 		ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) jawrConfig.getContext().getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
110 		String classPathImgServletPath = "";
111 		
112 		if(imgRsHandler != null){
113 			classPathImgServletPath = PathNormalizer.asPath(imgRsHandler.getJawrConfig().getServletMapping());
114 		}
115 		
116 		String url = match.substring(match.indexOf('(')+1,match.lastIndexOf(')'))
117 					.trim();
118 
119 		// To keep quotes as they are, first they are checked and removed. 
120 		String quoteStr = "";
121 		if( url.startsWith("'") || url.startsWith("\"")) {
122 			quoteStr = url.charAt(0)+"";
123 			url = url.substring(1,url.length()-1);
124 		}
125 		
126 		// Check if the URL is absolute, if it is return it as is. 
127 		int firstSlash = url.indexOf('/');
128 		if(0 == firstSlash || (firstSlash != -1 && url.charAt(++firstSlash) == '/')){
129 			StringBuffer sb = new StringBuffer("url(");
130 			sb.append(quoteStr).append(url).append(quoteStr).append(")");
131 			return sb.toString();
132 		}
133 		
134 		// Check if the URL is embedded data (RFC2397), if it is return it as is
135 		if(url.trim().toLowerCase().startsWith("data:")) {
136 			StringBuffer sb = new StringBuffer("url(");
137 			sb.append(quoteStr).append(url).append(quoteStr).append(")");
138 			return sb.toString();
139 		}
140 
141 		if(url.startsWith(URL_SEPARATOR))
142 			url = url.substring(1,url.length());
143 		else if(url.startsWith("./"))
144 			url = url.substring(2,url.length());
145 		
146 		// Here we generate the full path of the CSS image
147 		// to be able to define the relative path from the full bundle path
148 		String fullImgPath = getFinalFullImagePath(url, classPathImgServletPath, status, imgRsHandler);
149 		String imgUrl = PathNormalizer.getRelativeWebPath(PathNormalizer.getParentPath(fullBundlePath), fullImgPath);
150 		
151 		// Start rendering the result, starting by the initial quote, if any. 
152 		StringBuffer urlPrefix = new StringBuffer("url(").append(quoteStr);
153 		return PathNormalizer.normalizePath(urlPrefix.append(imgUrl).append(quoteStr).append(")").toString());
154 	}

157 	/***
158 	 * Returns the full path for the CSS bundle, taking in account the css servlet path if defined, 
159 	 * the caching prefix, and the url context path overriden
160 	 *   
161 	 * @param status the status
162 	 * @param jawrConfig the jawr configuration
163 	 * @return the full bundle path
164 	 */
165 	private String getFinalFullBundlePath(BundleProcessingStatus status, JawrConfig jawrConfig) {
166 
167 		String fullBundlePath = null;
168 		String bundleName = status.getCurrentBundle().getId();
169 		
170 		// Generation the bundle prefix
171 		String bundlePrefix = "";
172 		if(!bundleName.equals(ResourceGenerator.CSS_DEBUGPATH)){
173 			bundlePrefix = FAKE_BUNDLE_PREFIX;
174 		}
175 		
176 		// Add path reference for the servlet mapping if it exists 
177 		if(! "".equals(jawrConfig.getServletMapping())){
178 			bundlePrefix = PathNormalizer.asPath(jawrConfig.getServletMapping()+bundlePrefix)+"/";
179 			
180 		}  
181 		
182 		// Concatenate the bundle prefix and the bundle name
183 		fullBundlePath = PathNormalizer.concatWebPath(bundlePrefix, bundleName);
184 		
185 		return fullBundlePath;
186 	}

189 	/***
190 	 * Returns the full path of the CSS image, taking in account the css servlet path if defined, 
191 	 * the caching prefix, and the url context path overriden.
192 	 * 
193 	 * @param url the image url
194 	 * @param imgServletPath the image servlet path
195 	 * @param status the status
196 	 * @param imgRsHandler the image Resource handler
197 	 * @return the full image path from the web application context path
198 	 * @throws IOException if an IOException occurs
199 	 */
200 	private String getFinalFullImagePath(String url, String imgServletPath, BundleProcessingStatus status,
201 			ImageResourcesHandler imgRsHandler) throws IOException {
202 		
203 		String imgUrl = null;
204 		
205 		// Retrieve the current CSS file from which the CSS image is referenced
206 		String currentCss = status.getLastPathAdded();
207 		
208 		boolean classpathImg = url.startsWith(JawrConstant.CLASSPATH_RESOURCE_PREFIX);
209 		boolean classpathCss = isClassPathCss(currentCss, status);
210 		
211 		String rootPath = currentCss;
212 		
213 		// If the CSS image is taken from the classpath, add the classpath cache prefix
214 		if(classpathImg || classpathCss){
215 			
216 			String tempUrl = url;
217 			
218 			// If it's a classpath CSS, the url of the CSS image is defined relatively to it.
219 			if(classpathCss){
220 				tempUrl = PathNormalizer.concatWebPath(rootPath, url);
221 			}
222 
223 			// generate image cache URL
224 			String cacheUrl = addCacheBuster(status, tempUrl, imgRsHandler);
225 			imgUrl = cacheUrl;
226 		}else{
227 			
228 			// Generate the image URL from the current CSS path
229 			imgUrl = PathNormalizer.concatWebPath(rootPath, url);
230 			
231 			// If the image is part of the images cached by the Jawr image servlet
232 			// use the image cache url
233 			// Note : the Jawr image servlet must always be initialized before the CSS one
234 			String imgCacheUrl = null;
235 			if(imgRsHandler != null){
236 				imgCacheUrl = imgRsHandler.getCacheUrl(imgUrl);
237 				if(imgCacheUrl == null){
238 					imgCacheUrl = addCacheBuster(status, imgUrl, imgRsHandler);
239 				}
240 				imgUrl = imgCacheUrl;
241 			}
242 		}
243 		
244 		// This following condition should never be true. 
245 		// If it does, it means that the image path is wrongly defined.
246 		if(imgUrl == null){
247 			log.error("The CSS image path for '"+url+"' defined in '"+currentCss+"' is out of the application context. Please check your CSS file.");
248 		}
249 		
250 		// Add image servlet path in the URL, if it's defined
251 		if(StringUtils.isNotEmpty(imgServletPath)){
252 			imgUrl = imgServletPath+URL_SEPARATOR+imgUrl;
253 		}
254 		
255 		return PathNormalizer.asPath(imgUrl);
256 	}

259 	/***
260 	 * Checks if the Css path in parameter is a classpath CSS. 
261 	 * @param currentCss the CSS 
262 	 * @param status the status
263 	 * @return true if if the Css path in parameter is a classpath CSS. 
264 	 */
265 	private boolean isClassPathCss(String currentCss, BundleProcessingStatus status) {
266 		return currentCss.startsWith(JawrConstant.CLASSPATH_RESOURCE_PREFIX) && status.getJawrConfig().isUsingClasspathCssImageServlet();
267 	}

269 	/***
270 	 * Adds the cache buster to the CSS image
271 	 * @param status the bundle processing status
272 	 * @param url the URL of the image
273 	 * @param imgRsHandler the image resource handler
274 	 * @return the url of the CSS image with a cache buster
275 	 * @throws IOException if an IO exception occurs
276 	 */
277 	private String addCacheBuster(BundleProcessingStatus status, String url, ImageResourcesHandler imgRsHandler) throws IOException {
278 		
279 		// Try to retrieve the from the bundle processing cache
280 		String newUrl = status.getImageMapping(url);
281 		if(newUrl != null){
282 			return newUrl;
283 		}
284 		
285 		// Try to retrieve the from the image resource handler cache
286 		if(imgRsHandler != null){
287 			newUrl = imgRsHandler.getCacheUrl(url);
288 			if(newUrl != null){
289 				return newUrl;
290 			}
291 			// Retrieve the new URL with the cache prefix
292 			try {
293 				newUrl = CheckSumUtils.getCacheBustedUrl(url, status.getRsHandler(), status.getJawrConfig());
294 			} catch (ResourceNotFoundException e) {
295 				log.info("Impossible to define the checksum for the resource '"+url+"'. ");
296 				return url;
297 			} catch (IOException e) {
298 				log.info("Impossible to define the checksum for the resource '"+url+"'.");
299 				return url;
300 			}
301 			
302 			imgRsHandler.addMapping(url, newUrl);
303 			
304 		}else{
305 			newUrl = url;
306 		}
307 		
308 		
309 		// Set the result in a cache, so we will not search for it the next time
310 		status.setImageMapping(url, newUrl);
311 		
312 		return newUrl;
313 	}
