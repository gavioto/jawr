34  /***
35   * Single file postprocessor used to rewrite CSS URLs according to the new relative locations of the references when
36   * added to a bundle. Since the path changes, the URLs must be rewritten accordingly.  
37   * URLs in css files are expected to be according to the css spec (see http://www.w3.org/TR/REC-CSS2/syndata.html#value-def-uri). 
38   * Thus, single double, or no quotes enclosing the url are allowed (and remain as they are after rewriting). Escaped parens and quotes 
39   * are allowed within the url.   
40   *  
41   * @author Jordi Hernández Sellés
42   * @author Ibrahim Chaehoi
43   */
44  public class CSSURLPathRewriterPostProcessor extends
45  		AbstractChainedResourceBundlePostProcessor {
46  	
47  	/*** This variable is used to fake the gzip prefix for the full bundle path */
48  	private static final String FAKE_BUNDLE_PREFIX = "/prefix/";
49  
50  	/*** Logger */
51  	private static Logger log = Logger.getLogger(CSSURLPathRewriterPostProcessor.class);
52  	
53  	/*** The URL separator */
54  	private static final String URL_SEPARATOR = "/";
55  
56  	/*** The url pattern */
57  	private static final Pattern urlPattern = Pattern.compile(	"url//(//s*" // 'url(' and any number of whitespaces 
58  																+ "((//////))|[^)])*" // any sequence of characters, except an unescaped ')'
59  																+ "//s*//)",  // Any number of whitespaces, then ')'
60  																Pattern.CASE_INSENSITIVE); // works with 'URL('
61  	
62  	/***
63  	 * Constructor
64  	 */
65  	public CSSURLPathRewriterPostProcessor() {
66  		super(PostProcessFactoryConstant.URL_PATH_REWRITER);
67  	}

69  	/* (non-Javadoc)
70  	 * @see net.jawr.web.resource.bundle.postprocess.impl.AbstractChainedResourceBundlePostProcessor#doPostProcessBundle(net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus, java.lang.StringBuffer)
71  	 */
72  	protected StringBuffer doPostProcessBundle(BundleProcessingStatus status,
73  			StringBuffer bundleData) throws IOException {
74  		
75  		String data = bundleData.toString();
76  		
77  		JawrConfig jawrConfig = status.getJawrConfig();
78  		
79  		// Retrieve the full bundle path, so we will be able to define the relative path for the css images
80  		String bundleName = getFinalFullBundlePath(status, jawrConfig);
81  		
82  		// Rewrite each css image url path
83  		Matcher matcher = urlPattern.matcher(data);
84  		StringBuffer sb = new StringBuffer();
85  		while(matcher.find()) {
86  		
87  			String url = getUrlPath(matcher.group(), bundleName, status);
88  			matcher.appendReplacement(sb, RegexUtil.adaptReplacementToMatcher(url));
89  		}
90  		matcher.appendTail(sb);
91  		return sb;
92  	}

95  	/***
96  	 * Transform a matched url so it points to the proper relative path with respect to the given path.  
97  	 * @param match the matched URL
98  	 * @param fullBundlePath the full bundle path
99  	 * @param status the bundle processing status
100 	 * @return the image URL path
101 	 * @throws IOException if an IO exception occurs
102 	 */
103 	private String getUrlPath(String match, String fullBundlePath, BundleProcessingStatus status) throws IOException {
104 
105 		JawrConfig jawrConfig = status.getJawrConfig();
106 				
107 		// Retrieve the image servlet mapping
108 		ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) jawrConfig.getContext().getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
109 		String classPathImgServletPath = "";
110 		
111 		if(imgRsHandler != null){
112 			classPathImgServletPath = PathNormalizer.asPath(imgRsHandler.getJawrConfig().getServletMapping());
113 		}
114 		
115 		String url = match.substring(match.indexOf('(')+1,match.lastIndexOf(')'))
116 					.trim();
117 
118 		// To keep quotes as they are, first they are checked and removed. 
119 		String quoteStr = "";
120 		if( url.startsWith("'") || url.startsWith("\"")) {
121 			quoteStr = url.charAt(0)+"";
122 			url = url.substring(1,url.length()-1);
123 		}
124 		
125 		// Check if the URL is absolute, if it is return it as is. 
126 		int firstSlash = url.indexOf('/');
127 		if(0 == firstSlash || (firstSlash != -1 && url.charAt(++firstSlash) == '/')){
128 			StringBuffer sb = new StringBuffer("url(");
129 			sb.append(quoteStr).append(url).append(quoteStr).append(")");
130 			return sb.toString();
131 		}
132 		
133 		// Check if the URL is embedded data (RFC2397), if it is return it as is
134 		if(url.trim().toLowerCase().startsWith("data:")) {
135 			StringBuffer sb = new StringBuffer("url(");
136 			sb.append(quoteStr).append(url).append(quoteStr).append(")");
137 			return sb.toString();
138 		}
139 
140 		if(url.startsWith(URL_SEPARATOR))
141 			url = url.substring(1,url.length());
142 		else if(url.startsWith("./"))
143 			url = url.substring(2,url.length());
144 		
145 		// Here we generate the full path of the CSS image
146 		// to be able to define the relative path from the full bundle path
147 		String fullImgPath = getFinalFullImagePath(url, classPathImgServletPath, status, imgRsHandler);
148 		String imgUrl = PathNormalizer.getRelativeWebPath(PathNormalizer.getParentPath(fullBundlePath), fullImgPath);
149 		
150 		// Start rendering the result, starting by the initial quote, if any. 
151 		StringBuffer urlPrefix = new StringBuffer("url(").append(quoteStr);
152 		return PathNormalizer.normalizePath(urlPrefix.append(imgUrl).append(quoteStr).append(")").toString());
153 	}

156 	/***
157 	 * Returns the full path for the CSS bundle, taking in account the css servlet path if defined, 
158 	 * the caching prefix, and the url context path overriden
159 	 *   
160 	 * @param status the status
161 	 * @param jawrConfig the jawr configuration
162 	 * @return the full bundle path
163 	 */
164 	private String getFinalFullBundlePath(BundleProcessingStatus status, JawrConfig jawrConfig) {
165 
166 		String fullBundlePath = null;
167 		String bundleName = status.getCurrentBundle().getId();
168 		
169 		// Generation the bundle prefix
170 		String bundlePrefix = "";
171 		if(!bundleName.equals(ResourceGenerator.CSS_DEBUGPATH)){
172 			bundlePrefix = FAKE_BUNDLE_PREFIX;
173 		}
174 		
175 		// Add path reference for the servlet mapping if it exists 
176 		if(! "".equals(jawrConfig.getServletMapping())){
177 			bundlePrefix = PathNormalizer.asPath(jawrConfig.getServletMapping()+bundlePrefix)+"/";
178 			
179 		}  
180 		
181 		// Concatenate the bundle prefix and the bundle name
182 		fullBundlePath = PathNormalizer.concatWebPath(bundlePrefix, bundleName);
183 		
184 		return fullBundlePath;
185 	}

188 	/***
189 	 * Returns the full path of the CSS image, taking in account the css servlet path if defined, 
190 	 * the caching prefix, and the url context path overriden.
191 	 * 
192 	 * @param url the image url
193 	 * @param imgServletPath the image servlet path
194 	 * @param status the status
195 	 * @param imgRsHandler the image Resource handler
196 	 * @return the full image path from the web application context path
197 	 * @throws IOException if an IOException occurs
198 	 */
199 	private String getFinalFullImagePath(String url, String imgServletPath, BundleProcessingStatus status,
200 			ImageResourcesHandler imgRsHandler) throws IOException {
201 		
202 		String imgUrl = null;
203 		
204 		// Retrieve the current CSS file from which the CSS image is referenced
205 		String currentCss = status.getLastPathAdded();
206 		
207 		boolean classpathImg = url.startsWith(JawrConstant.CLASSPATH_RESOURCE_PREFIX);
208 		boolean classpathCss = isClassPathCss(currentCss, status);
209 		
210 		String rootPath = currentCss;
211 		
212 		// If the CSS image is taken from the classpath, add the classpath cache prefix
213 		if(classpathImg || classpathCss){
214 			
215 			String tempUrl = url;
216 			
217 			// If it's a classpath CSS, the url of the CSS image is defined relatively to it.
218 			if(classpathCss){
219 				tempUrl = PathNormalizer.concatWebPath(rootPath, url);
220 			}
221 
222 			// generate image cache URL
223 			String cacheUrl = addCacheBuster(status, tempUrl, imgRsHandler);
224 			imgUrl = cacheUrl;
225 		}else{
226 			
227 			// Generate the image URL from the current CSS path
228 			imgUrl = PathNormalizer.concatWebPath(rootPath, url);
229 			
230 			// If the image is part of the images cached by the Jawr image servlet
231 			// use the image cache url
232 			// Note : the Jawr image servlet must always be initialized before the CSS one
233 			String imgCacheUrl = null;
234 			if(imgRsHandler != null){
235 				imgCacheUrl = imgRsHandler.getCacheUrl(imgUrl);
236 				if(imgCacheUrl == null){
237 					imgCacheUrl = addCacheBuster(status, imgUrl, imgRsHandler);
238 				}
239 				imgUrl = imgCacheUrl;
240 			}
241 		}
242 		
243 		// This following condition should never be true. 
244 		// If it does, it means that the image path is wrongly defined.
245 		if(imgUrl == null){
246 			log.error("The CSS image path for '"+url+"' defined in '"+currentCss+"' is out of the application context. Please check your CSS file.");
247 		}
248 		
249 		// Add image servlet path in the URL, if it's defined
250 		if(StringUtils.isNotEmpty(imgServletPath)){
251 			imgUrl = imgServletPath+URL_SEPARATOR+imgUrl;
252 		}
253 		
254 		return PathNormalizer.asPath(imgUrl);
255 	}

258 	/***
259 	 * Checks if the Css path in parameter is a classpath CSS. 
260 	 * @param currentCss the CSS 
261 	 * @param status the status
262 	 * @return true if if the Css path in parameter is a classpath CSS. 
263 	 */
264 	private boolean isClassPathCss(String currentCss, BundleProcessingStatus status) {
265 		return currentCss.startsWith(JawrConstant.CLASSPATH_RESOURCE_PREFIX) && status.getJawrConfig().isUsingClasspathCssImageServlet();
266 	}

268 	/***
269 	 * Adds the cache buster to the CSS image
270 	 * @param status the bundle processing status
271 	 * @param url the URL of the image
272 	 * @param imgRsHandler the image resource handler
273 	 * @return the url of the CSS image with a cache buster
274 	 * @throws IOException if an IO exception occurs
275 	 */
276 	private String addCacheBuster(BundleProcessingStatus status, String url, ImageResourcesHandler imgRsHandler) throws IOException {
277 		
278 		// Try to retrieve the from the bundle processing cache
279 		String newUrl = status.getImageMapping(url);
280 		if(newUrl != null){
281 			return newUrl;
282 		}
283 		
284 		// Try to retrieve the from the image resource handler cache
285 		if(imgRsHandler != null){
286 			newUrl = imgRsHandler.getCacheUrl(url);
287 			if(newUrl != null){
288 				return newUrl;
289 			}
290 			// Retrieve the new URL with the cache prefix
291 			newUrl = CheckSumUtils.getCacheBustedUrl(url, status.getRsHandler(), status.getJawrConfig());
292 			if(imgRsHandler != null){
293 				imgRsHandler.addMapping(url, newUrl);
294 			}
295 		}else{
296 			newUrl = url;
297 		}
298 		
299 		
300 		// Set the result in a cache, so we will not search for it the next time
301 		status.setImageMapping(url, newUrl);
302 		
303 		return newUrl;
304 	}
