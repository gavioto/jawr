108 	 * Constructor
109 	 * 
110 	 * @param rsBrowser
111 	 *            the resource browser
112 	 * @param config
113 	 *            the jawr config
114 	 * @param initSkinMapping
115 	 *            the flag indicating if we should initialize the skin mapping
116 	 *            or not.
117 	 */
118 	public CssSkinGenerator(ResourceBrowser rsBrowser, JawrConfig config,
119 			boolean initSkinMapping) {
120 		this.rsBrowser = rsBrowser;
121 		cssSkinResolver = (AbstractCssSkinVariantResolver) config
122 				.getGeneratorRegistry().getVariantResolver(
123 						JawrConstant.SKIN_VARIANT_TYPE);
124 		if (cssSkinResolver == null) {
125 			cssSkinResolver = new CssSkinVariantResolver();
126 			config.getGeneratorRegistry().registerVariantResolver(
127 					cssSkinResolver);
128 		}
129 		if (initSkinMapping) {
130 			String skinMappingType = config
131 					.getProperty(JawrConstant.SKIN_TYPE_MAPPING_CONFIG_PARAM);
132 			if (StringUtils.isNotEmpty(skinMappingType)) {
133 				if (skinMappingType
134 						.equals(JawrConstant.SKIN_TYPE_MAPPING_LOCALE_SKIN)
135 						|| skinMappingType
136 								.equals(JawrConstant.SKIN_TYPE_MAPPING_SKIN_LOCALE)) {
137 					this.skinMappingType = skinMappingType;
138 				} else {
139 					throw new IllegalArgumentException("The value for the '"
140 							+ JawrConstant.SKIN_TYPE_MAPPING_LOCALE_SKIN + "' "
141 							+ "property [" + skinMappingType + "] is invalid. "
142 							+ "Please check the docs for valid values ");
143 				}
144 			}
145 			this.skinMapping = getSkinMapping(rsBrowser, config);
146 			resourceProviderStrategyClass = CssSkinVariantResourceProviderStrategy.class;
147 		}
148 	}

229 		if (skinRootDir != null) {

232 					.hasNext();) {

241 	@Override
242 	protected Reader generateResourceForBundle(GeneratorContext context) {
243 		
244 		Reader reader = null;
245 		ResourceReaderHandler readerHandler = context
246 				.getResourceReaderHandler();
247 		String path = context.getPath();
248 
249 		String skinRootDir = getSkinRootDir(path, skinMapping.keySet());
250 		Map<String, VariantSet> ctxVariantSetMap = skinMapping.get(skinRootDir);
251 
252 		String skinVariantPath = path.substring(skinRootDir.length());
253 		String[] paths = skinVariantPath.split(JawrConstant.URL_SEPARATOR);
254 
255 		VariantResourceReaderStrategy strategy = getVariantStrategy(context,
256 				ctxVariantSetMap);
257 		Map<String, String> variantMap = null;
258 		do {
259 			variantMap = strategy.nextVariantMapConbination();
260 			if (variantMap != null) {
261 				reader = getResourceReader(readerHandler, skinRootDir, paths,
262 						variantMap);
263 			}
264 		} while (variantMap != null && reader == null);
265 		
266 		return reader;
267 	}

329 	 * Returns the skin mapping from the Jawr config
330 	 * 
331 	 * @param the
332 	 *            resource browser
333 	 * @param config
334 	 *            the Jawr config
335 	 * @return the skin mapping
336 	 */
337 	public Map<String, Map<String, VariantSet>> getSkinMapping(
338 			ResourceBrowser rsBrowser, JawrConfig config) {
339 
340 		Map<String, Map<String, VariantSet>> skinMapping = new HashMap<String, Map<String, VariantSet>>();
341 		PropertiesConfigHelper props = new PropertiesConfigHelper(
342 				config.getConfigProperties(), JawrConstant.CSS_TYPE);
343 		Set<String> skinRootDirectories = props
344 				.getPropertyAsSet(JawrConstant.SKIN_DEFAULT_ROOT_DIRS);
345 		if (skinMappingType.equals(JawrConstant.SKIN_TYPE_MAPPING_SKIN_LOCALE)) {
346 			updateSkinMappingUsingTypeSkinLocale(rsBrowser, config,
347 					skinMapping, skinRootDirectories);
348 		} else {
349 			updateSkinMappingUsingTypeLocaleSkin(rsBrowser, config,
350 					skinMapping, skinRootDirectories);
351 		}
352 
353 		return skinMapping;
354 	}

356 	/**
357 	 * Returns the skin mapping from the Jawr config, here the resource mapping
358 	 * type is skin_locale
359 	 * 
360 	 * @param rsBrowser
361 	 *            the resource browser
362 	 * @param config
363 	 *            the jawr config
364 	 * @param skinMapping
365 	 *            the skin mapping
366 	 * @param skinRootDirectories
367 	 *            the skin root directories
368 	 */
369 	private void updateSkinMappingUsingTypeSkinLocale(
370 			ResourceBrowser rsBrowser, JawrConfig config,
371 			Map<String, Map<String, VariantSet>> skinMapping,
372 			Set<String> skinRootDirectories) {
373 
374 		String defaultSkinName = null;
375 		String defaultLocaleName = null;
376 
377 		// Check if there are no directory which is contained in another root
378 		// directory
379 		for (Iterator<String> itRootDir = skinRootDirectories.iterator(); itRootDir
380 				.hasNext();) {
381 			String defaultSkinDir = PathNormalizer.asDirPath(itRootDir.next());
382 			String defaultSkinDirName = PathNormalizer
383 					.getPathName(defaultSkinDir);
384 			String skinRootDir = PathNormalizer.getParentPath(defaultSkinDir);
385 
386 			String skinName = null;
387 			String localeName = null;
388 
389 			if (LocaleUtils.LOCALE_SUFFIXES.contains(defaultSkinDirName)) { // The
390 																			// rootDir
391 																			// point
392 																			// to
393 																			// a
394 																			// locale
395 																			// directory
396 				localeName = defaultSkinDirName;
397 				skinName = PathNormalizer.getPathName(skinRootDir);
398 				skinRootDir = PathNormalizer.getParentPath(skinRootDir);
399 
400 			} else {
401 				skinName = defaultSkinDirName;
402 			}
403 
404 			if (defaultSkinName == null) {
405 				defaultSkinName = skinName;
406 			} else if (!defaultSkinName.equals(skinName)) {
407 				throw new BundlingProcessException(
408 						"The default skin for the skin root directories are not the same. Please check your configuration.");
409 			}
410 
411 			if (defaultLocaleName == null) {
412 				defaultLocaleName = localeName;
413 			} else if (defaultLocaleName != null
414 					&& !defaultLocaleName.equals(localeName)) {
415 				throw new BundlingProcessException(
416 						"The default locale for the skin root directories are not the same. Please check your configuration.");
417 			}
418 
419 			checkRootDirectoryNotOverlap(defaultSkinDir, skinRootDirectories);
420 
421 			Map<String, VariantSet> variantsMap = getVariants(rsBrowser,
422 					skinRootDir, skinName, localeName, true);
423 
424 			skinMapping.put(skinRootDir, variantsMap);
425 		}
426 
427 		// Init resolver
428 		cssSkinResolver.setDefaultSkin(defaultSkinName);
429 		cssSkinResolver.setSkinCookieName(config.getSkinCookieName());
430 	}

540 	/**
541 	 * Initialize the skinMapping from the parent path
542 	 * 
543 	 * @param rsBrowser
544 	 *            the resource browser
545 	 * @param rootDir
546 	 *            the skin root dir path
547 	 * @param defaultSkinName
548 	 *            the default skin name
549 	 * @param defaultLocaleName
550 	 *            the default locale name
551 	 */
552 	private Map<String, VariantSet> getVariants(ResourceBrowser rsBrowser,
553 			String rootDir, String defaultSkinName, String defaultLocaleName,
554 			boolean mappingSkinLocale) {
555 		Set<String> paths = rsBrowser.getResourceNames(rootDir);
556 		Set<String> skinNames = new HashSet<String>();
557 		Set<String> localeVariants = new HashSet<String>();
558 		for (Iterator<String> itPath = paths.iterator(); itPath.hasNext();) {
559 			String path = rootDir + itPath.next();
560 			if (rsBrowser.isDirectory(path)) {
561 				String dirName = PathNormalizer.getPathName(path);
562 				if (mappingSkinLocale) {
563 					skinNames.add(dirName);
564 					// check if there are locale variants for this skin,
565 					// and update the localeVariants if needed
566 					updateLocaleVariants(rsBrowser, path, localeVariants);
567 				} else {
568 					if (LocaleUtils.LOCALE_SUFFIXES.contains(dirName)) {
569 						localeVariants.add(dirName);
570 						// check if there are skin variants for this locales,
571 						// and update the skinVariants if needed
572 						updateSkinVariants(rsBrowser, path, skinNames);
573 					}
574 				}
575 			}
576 		}
577 
578 		// Initialize the variant mapping for the skin root directory
579 		return getVariants(defaultSkinName, skinNames, defaultLocaleName,
580 				localeVariants);
581 	}

596 	private Map<String, VariantSet> getVariants(String defaultSkin,
597 			Set<String> skinNames, String defaultLocaleName,
598 			Set<String> localeVariants) {
599 
600 		Map<String, VariantSet> skinVariants = new HashMap<String, VariantSet>();
601 		if (!skinNames.isEmpty()) {
602 			skinVariants.put(JawrConstant.SKIN_VARIANT_TYPE, new VariantSet(
603 					JawrConstant.SKIN_VARIANT_TYPE, defaultSkin, skinNames));
604 		}
605 		if (!localeVariants.isEmpty()) {
606 
607 			skinVariants.put(JawrConstant.LOCALE_VARIANT_TYPE, new VariantSet(
608 					JawrConstant.LOCALE_VARIANT_TYPE, defaultLocaleName,
609 					localeVariants));
610 		}
611 
612 		return skinVariants;
613 	}

645 	 *            the resource browser
646 	 * @param path
647 	 *            the skin path
648 	 * @param skinVariants
649 	 *            the set of skin variants to update
650 	 */
651 	private void updateSkinVariants(ResourceBrowser rsBrowser, String path,
652 			Set<String> skinVariants) {
653 
654 		Set<String> skinPaths = rsBrowser.getResourceNames(path);
655 		for (Iterator<String> itSkinPath = skinPaths.iterator(); itSkinPath
656 				.hasNext();) {
657 			String skinPath = path + itSkinPath.next();
658 			if (rsBrowser.isDirectory(skinPath)) {
659 				String skinDirName = PathNormalizer.getPathName(skinPath);
660 				skinVariants.add(skinDirName);
661 			}
662 		}
663 	}

665 	/**
666 	 * Check if there are no directory which is contained in another root
667 	 * directory,
668 	 * 
669 	 * @param dir
670 	 *            the root directory to check
671 	 * @param skinRootDirectories
672 	 *            the root directories
673 	 */
674 	private void checkRootDirectoryNotOverlap(String dir,
675 			Set<String> skinRootDirectories) {
676 
677 		String rootDir = removeLocaleSuffixIfExist(dir);
678 		for (Iterator<String> itSkinDir = skinRootDirectories.iterator(); itSkinDir
679 				.hasNext();) {
680 			String skinDir = PathNormalizer.asDirPath(itSkinDir.next());
681 			if (!skinDir.equals(dir)) {
682 				skinDir = removeLocaleSuffixIfExist(skinDir);
683 				if (skinDir.startsWith(rootDir)) {
684 					throw new BundlingProcessException(
685 							"There is a misconfiguration. It is not allowed to have a skin root directory containing another one.");
686 				}
687 			}
688 		}
689 	}

691 	/**
692 	 * Removes from the directory path the locale suffix if it exists
693 	 * 
694 	 * @param dir
695 	 *            the directory path
696 	 * 
697 	 *            <pre>
698 	 * removeLocaleSuffixIfExist( "/css/panel/default/" )         = "/css/panel/default/"
699 	 * removeLocaleSuffixIfExist( "/css/panel/default/en_US" )   = "/css/panel/default/"
700 	 * </pre>
701 	 * 
702 	 * @return the directory path, with the locale suffix removed if it exists
703 	 */
704 	private String removeLocaleSuffixIfExist(String dir) {
705 
706 		String result = dir;
707 		String dirName = PathNormalizer.getPathName(dir);
708 		if (LocaleUtils.LOCALE_SUFFIXES.contains(dirName)) {
709 			result = dir.substring(0, dir.indexOf("/" + dirName) + 1);
710 		}
711 		return result;
712 	}

740 	/**
741 	 * Returns the reader for the resource path define in parameter
742 	 * 
743 	 * @param skinRootDir
744 	 *            the root directory
745 	 * @param paths
746 	 *            the array of path
747 	 * @param variantMap
748 	 *            the variant map
749 	 * @return the reader
750 	 */
751 	private Reader getResourceReader(ResourceReaderHandler readerHandler,
752 			String skinRootDir, String[] paths, Map<String, String> variantMap) {
753 
754 		Reader reader = null;
755 		StringBuffer path = new StringBuffer(skinRootDir);
756 		String skinVariant = (String) variantMap
757 				.get(JawrConstant.SKIN_VARIANT_TYPE);
758 		String localeVariant = (String) variantMap
759 				.get(JawrConstant.LOCALE_VARIANT_TYPE);
760 
761 		if (skinMappingType.equals(JawrConstant.SKIN_TYPE_MAPPING_SKIN_LOCALE)) {
762 			paths[0] = skinVariant;
763 			if (localeVariant != null) {
764 				paths[1] = localeVariant;
765 			}
766 		} else {
767 			paths[0] = localeVariant;
768 			if (skinVariant != null) {
769 				paths[1] = skinVariant;
770 			}
771 		}
772 
773 		for (int i = 0; i < paths.length; i++) {
774 			path.append(paths[i]);
775 			if (i + 1 < paths.length) {
776 				path.append(JawrConstant.URL_SEPARATOR);
777 			}
778 		}
779 
780 		try {
781 			reader = readerHandler.getResource(path.toString());
782 		} catch (ResourceNotFoundException e) {
783 			// Nothing to do
784 		}
785 
786 		return reader;
787 	}
