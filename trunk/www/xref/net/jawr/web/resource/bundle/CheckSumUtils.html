50  	/***
51  	 * Return the cache busted url associated to the url passed in parameter 
52  	 * @param url the url path to the resource file
53  	 * @param is the resource input stream 
54  	 * @param jawrConfig the jawrConfig
55  	 * @return the cache busted url
56  	 * @throws IOException if an IO exception occurs.
57  	 */
58  	public static String getCacheBustedUrl(String url, ResourceHandler rsHandler, JawrConfig jawrConfig, boolean fromClasspath) throws IOException{
59  		
60  		String checksum = null;
61  		String classpathResourceUrl = null;
62  		InputStream is = null;
63  		try {
64  			
65  			if(fromClasspath){
66  				classpathResourceUrl = url;
67  				is = ClassLoaderResourceUtils.getResourceAsStream(classpathResourceUrl, CheckSumUtils.class);
68  			}else{
69  				url = PathNormalizer.asPath(url);
70  				is = rsHandler.getResourceAsStream(url);
71  			}
72  			
73  			checksum = CheckSumUtils.getChecksum(is, jawrConfig.getImageHashAlgorithm());
74  		}catch(IOException e){
75  			log.error("An exception occurs while defining the mapping for the resource : "+url, e);
76  		} catch (ResourceNotFoundException e) {
77  			log.error("An exception occurs while defining the mapping for the resource : "+url, e);
78  		}
79  		finally {
80  			IOUtils.close(is);
81  		}
82  		
83  		String result = "";
84  		
85  		if(fromClasspath){
86  			result = JawrConstant.CLASSPATH_CACHE_BUSTER_PREFIX;
87  		}else{
88  			result = JawrConstant.CACHE_BUSTER_PREFIX;
89  		}
90  		
91  		result = result+checksum;
92  		if(fromClasspath){
93  			url = classpathResourceUrl;
94  		}
95  		if(!url.startsWith("/")){
96  			result = result + "/";
97  		}
98  		// Add the cache buster extension
99  		return PathNormalizer.asPath(result+url);
100 	}

140 	/***
141 	 * Returns the checksum value of the input stream taking in count the algorithm passed in parameter
142 	 * @param is the input stream
143 	 * @param algorithm the checksum algorithm
144 	 * @return the checksum value
145 	 * @throws IOException if an exception occurs.
146 	 */
147 	public static String getChecksum(InputStream is, String algorithm) throws IOException {
148 	
149 		if(algorithm.equals(CRC32_ALGORITHM)){
150 			return getCRC32Checksum(is);
151 		}else if(algorithm.equals(MD5_ALGORITHM)){
152 			return getMD5Checksum(is);
153 		}else{
154 			throw new RuntimeException("The checksum algorithm '"+algorithm+"' is not supported.\n" +
155 					"The only supported algorithm are 'CRC32' or 'MD5'.");
156 		}
157 	}

159 	/***
160 	 * Returns the CRC 32 Checksum of the input stream
161 	 * 
162 	 * @param is the input stream
163 	 * 
164 	 * @return the CRC 32 checksum of the input stream
165 	 * @throws IOException if an IO exception occurs
166 	 */
167 	public static String getCRC32Checksum(InputStream is) throws IOException {
168 
169 		Checksum checksum = new CRC32();
170 
171 		byte[] bytes = new byte[1024];
172 		int len = 0;
173 
174 		while ((len = is.read(bytes)) >= 0) {
175 			checksum.update(bytes, 0, len);
176 		}
177 
178 		return Long.toString(checksum.getValue());
179 	}

181 	/***
182 	 * Returns the MD5 Checksum of the input stream

184 	 * @param is the input stream

186 	 * @return the Checksum of the input stream
187 	 * @throws IOException if an IO exception occurs
188 	 */
189 	public static String getMD5Checksum(InputStream is) throws IOException {
190 
191 		byte[] digest = null;
192 		try {
193 			MessageDigest md = MessageDigest.getInstance(MD5_ALGORITHM);
194 			is = new DigestInputStream(is, md);
195 			// read stream to EOF as normal...
196 			while (is.read() != -1) {
197 
198 			}
199 			digest = md.digest();
200 		} catch (NoSuchAlgorithmException e) {
201 			throw new RuntimeException("MD5 algorithm needs to be installed", e);
202 		}
203 
204 		return new BigInteger(1, digest).toString(16);
205 	}
