39  /***
40   * This class defines the image URL rewriter for the base64 image post processor.
41   * This postprocessor will apply the standard URL rewriting process if the image URL
42   * is annotated as "jawr:base64-skip" or if the brwoser is IE7, because
43   * the MHTML is not properly supported with IE7 on Windows Vista and Windows 7
44   * Please check the following link for more info.
45   * 
46   * http://www.phpied.com/data-uris-mhtml-ie7-win7-vista-blues/#vista
47   * 
48   * @author Ibrahim Chaehoi
49   */
50  public class Base64PostProcessorCssImageUrlRewriter extends
51  		PostProcessorCssImageUrlRewriter {
52  
53  	/*** The logger */
54  	private Logger LOGGER = Logger
55  			.getLogger(Base64PostProcessorCssImageUrlRewriter.class);
56  
57  	/*** The data prefix */
58  	private static final String DATA_PREFIX = "data:";
59  
60  	/*** The mhtml prefix */
61  	private static final String MHTML_PREFIX = "mhtml:";
62  
63  	/*** The annotation to skip or force the base64 encoding (jawr:base64 or jawr:base64-skip ) */
64  	private static final Pattern ANNOTATION_BASE64_PATTERN = Pattern.compile("jawr(?://s)*:(?://s)*(base64)(-skip)?");
65  
66  	/*** The annotation group in the URL pattern */
67  	private static final int ANNOTATION_GROUP = 9;
68  
69  	/*** The url pattern */
70  	private static final Pattern URL_WITH_ANNOTATION_PATTERN = Pattern.compile(
71  			"((" + URL_REGEXP + "//s*)+)" + "([^;]*);?"
72  					+ "//s*(///*//*(?:.|[//n//r])*?//*/)?", // Any number of
73  			// whitespaces and then
74  			// an annotation
75  
76  			Pattern.CASE_INSENSITIVE); // works with 'URL('
77  
78  	/*** The default max file size */
79  	private static final int MAX_LENGTH_FILE = 30000;
80  
81  	/*** The maximum image file size authorized to be encoded in base64 */
82  	private int maxFileSize;
83  	
84  	/*** The map of encoded resources */
85  	private Map encodedResources = null;
86  	
87  	/*** The flag which determine if we must encode by default or not */
88  	private boolean encodeByDefault;
89  	
90  	/*** The flag indicating if we must encode the sprites or not */
91  	private boolean encodeSprite;
92  
93  	/*** The flag indicating if we must skip the base64 encoding */
94  	private boolean skipBase64Encoding;

96  	/***
97  	 * Constructor
98  	 * 
99  	 * @param status
100 	 *            the bundle processing status
101 	 */
102 	public Base64PostProcessorCssImageUrlRewriter(BundleProcessingStatus status) {
103 		super(status);
104 		encodedResources = (Map) status
105 				.getData(JawrConstant.BASE64_ENCODED_RESOURCES);
106 		maxFileSize = MAX_LENGTH_FILE;
107 		Properties configProperties = status.getJawrConfig()
108 		.getConfigProperties();
109 		String maxLengthProperty = (String) configProperties.get(
110 						JawrConstant.BASE64_MAX_IMG_FILE_SIZE);
111 		
112 		if (StringUtils.isNotEmpty(maxLengthProperty)) {
113 			maxFileSize = Integer.parseInt(maxLengthProperty);
114 		}
115 		
116 		String strEncodeByDefault = (String) configProperties.get(
117 				JawrConstant.BASE64_ENCODE_BY_DEFAULT);
118 		encodeByDefault = getBooleanValue(strEncodeByDefault, true);
119 				
120 		String strEncodeSprite = (String) configProperties.get(
121 				JawrConstant.BASE64_ENCODE_SPRITE);
122 		encodeSprite = getBooleanValue(strEncodeSprite, false);
123 		
124 		LOGGER.debug("max file length: " + maxFileSize);
125 	}

127 	/***
128 	 * Returns the boolean value of the string passed in parameter or the default value 
129 	 * if the string is null
130 	 * @param strVal the string value
131 	 * @param defaultValue the default value
132 	 * @return the boolean value of the string passed in parameter or the default value 
133 	 * if the string is null
134 	 */
135 	private boolean getBooleanValue(String strVal, boolean defaultValue){
136 		
137 		boolean result = defaultValue;
138 		if(strVal != null){
139 			result = Boolean.valueOf(strVal).booleanValue();
140 		}
141 		
142 		return result;
143 	}

145 	/***
146 	 * Rewrites the image URL
147 	 * 
148 	 * @param originalCssPath
149 	 *            the original CSS path
150 	 * @param newCssPath
151 	 *            the new CSS path
152 	 * @param originalCssContent
153 	 *            the original CSS content
154 	 * @return the new CSS content with image path rewritten
155 	 * @throws IOException
156 	 */
157 	public StringBuffer rewriteUrl(String originalCssPath, String newCssPath,
158 			String originalCssContent) throws IOException {
159 
160 		// Rewrite each css image url path
161 		Matcher matcher = URL_WITH_ANNOTATION_PATTERN
162 				.matcher(originalCssContent);
163 		StringBuffer sb = new StringBuffer();
164 
165 		while (matcher.find()) {
166 
167 			String annotation = matcher.group(ANNOTATION_GROUP);
168 			if(StringUtils.isNotEmpty(annotation)){
169 				Matcher annotationMatcher = ANNOTATION_BASE64_PATTERN.matcher(annotation);
170 				if(annotationMatcher.find()){
171 					skipBase64Encoding = annotationMatcher.group(2) != null;
172 				}
173 			}else{
174 				// If no annotation use the default encoding mode
175 				skipBase64Encoding = !encodeByDefault;
176 			}
177 			
178 			StringBuffer sbUrl = new StringBuffer();
179 			Matcher urlMatcher = URL_PATTERN.matcher(matcher.group());
180 			while (urlMatcher.find()) {
181 				
182 				String url = urlMatcher.group();
183 				
184 				// Skip sprite encoding if it is configured so
185 				if(!encodeSprite && url.indexOf(GeneratorRegistry.SPRITE_GENERATOR_PREFIX+GeneratorRegistry.PREFIX_SEPARATOR) != -1){
186 					skipBase64Encoding = true;
187 				}
188 				
189 				if(LOGGER.isDebugEnabled()){
190 					LOGGER.debug("Skip encoding image resource : "+url);
191 				}
192 				
193 				url = getUrlPath(url, originalCssPath,
194 						newCssPath);
195 
196 				urlMatcher.appendReplacement(sbUrl, RegexUtil
197 						.adaptReplacementToMatcher(url));
198 			}
199 			urlMatcher.appendTail(sbUrl);
200 			matcher.appendReplacement(sb, RegexUtil
201 					.adaptReplacementToMatcher(sbUrl.toString()));
202 
203 		}
204 		matcher.appendTail(sb);
205 
206 		return sb;
207 	}

209 	/*
210 	 * (non-Javadoc)
211 	 * 
212 	 * @seenet.jawr.web.resource.bundle.postprocess.impl.
213 	 * PostProcessorCssImageUrlRewriter
214 	 * #rewriteURL(net.jawr.web.resource.bundle.postprocess
215 	 * .BundleProcessingStatus, java.lang.String, java.lang.String,
216 	 * java.lang.String, net.jawr.web.resource.ImageResourcesHandler)
217 	 */
218 	protected String rewriteURL(BundleProcessingStatus status, String url,
219 			String imgServletPath, String newCssPath,
220 			ImageResourcesHandler imgRsHandler) throws IOException {
221 
222 		String imgUrl = url;
223 		String browser = status.getVariant(JawrConstant.BROWSER_VARIANT_TYPE);
224 
225 		// Skip base64 encoding if it has ben deactivated
226 		if (skipBase64Encoding) {
227 			imgUrl = super.rewriteURL(status, imgUrl, imgServletPath,
228 					newCssPath, imgRsHandler);
229 		} else {
230 
231 			LOGGER.info("Encoding resource: " + url);
232 			try {
233 				InputStream is = imgRsHandler.getRsReaderHandler()
234 						.getResourceAsStream(url);
235 
236 				String fileExtension = FileNameUtils.getExtension(url);
237 				String fileMimeType = (String) MIMETypesSupport
238 						.getSupportedProperties(this).get(fileExtension);
239 				ByteArrayOutputStream out = new ByteArrayOutputStream();
240 				IOUtils.copy(is, out, true);
241 
242 				int size = out.size();
243 				if (size > maxFileSize) { // Check file size 
244 
245 					LOGGER.warn("File content length of '"+url+"' exceeds maximum file length: "
246 							+ size + " > " + maxFileSize);
247 				} else {
248 
249 					byte[] data = out.toByteArray();
250 					StringBuffer s = new StringBuffer(encodeInBase64(data));
251 
252 					Base64EncodedResource encodedImage = new Base64EncodedResource();
253 					encodedImage.setId(url.hashCode());
254 					encodedImage.setType(fileMimeType);
255 					encodedImage.setBase64Encoding(s);
256 
257 					encodedResources.put(encodedImage.getId(), encodedImage);
258 
259 					// For IE under IE8, use MHTML
260 					if (JawrConstant.BROWSER_IE6.equals(browser) || JawrConstant.BROWSER_IE7.equals(browser)) {
261 
262 						/***
263 						 * For Internet Explorer 6 and 7, the url must be mhtml: followed
264 						 * by an absolute url. However, this URL is not known at
265 						 * post process time. So we make add a place holder which will
266 						 * be resolved at runtime.
267 						 */
268 						imgUrl = MHTML_PREFIX+JawrConstant.JAWR_BUNDLE_PATH_PLACEHOLDER+"!"+encodedImage.getId();
269 
270 					} else {
271 						imgUrl = DATA_PREFIX + fileMimeType + ";base64," + s;
272 					}
273 				}
274 			} catch (IOException e) {
275 				LOGGER.error(e);
276 
277 			} catch (ResourceNotFoundException e) {
278 				LOGGER.error("The resource '" + e.getRequestedPath()
279 						+ "' has not been found.");
280 			} catch (Throwable e) {
281 				LOGGER.error(e);
282 			}
283 		}
284 
285 		return imgUrl;
286 	}

292 	 * @return the base64 encoded string of the data
