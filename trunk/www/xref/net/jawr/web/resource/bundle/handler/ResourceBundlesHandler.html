14  package net.jawr.web.resource.bundle.handler;
15  
16  import java.io.OutputStream;
17  import java.io.Writer;
18  import java.util.List;
19  
20  import net.jawr.web.config.JawrConfig;
21  import net.jawr.web.exception.ResourceNotFoundException;
22  import net.jawr.web.resource.bundle.JoinableResourceBundle;
23  import net.jawr.web.resource.bundle.iterator.ConditionalCommentCallbackHandler;
24  import net.jawr.web.resource.bundle.iterator.ResourceBundlePathsIterator;
25  
26  /***
27   * Main interface to work with resource bundles. It helps in resolving groups of resources
28   * which are served as a single one, and provides methods to generate urls that point to either
29   * the full bundle or its individual resources. 
30   * 
31   * @author Jordi Hernández Sellés
32   * @author Ibrahim Chaehoi
33   *
34   */
35  public interface ResourceBundlesHandler {
36  	
37  	/***
38  	 * Determines which bundle corresponds to a path. The path may be
39  	 * a directory or file path. This path will not include any prefix, it is intended
40  	 * to be the path normally used for a tag library. 
41  	 * @param path
42  	 * @return String The bundle ID that can be used to retrieve it.  
43  	 */
44  	public JoinableResourceBundle resolveBundleForPath(String path);
45  	
46  	/***
47  	 * Returns true if the bundle Id is the Id a global resource bundle
48  	 * @param resourceBundleId the resource bundle ID 
49  	 * @return the global resource bundle path iterator
50  	 */
51  	public boolean isGlobalResourceBundle(String resourceBundleId);
52  	
53  	/***
54  	 * Returns the global resource bundle path iterator
55  	 * @param commentCallbackHandler the comment callback handler
56  	 * @param variantKey the variant key
57  	 * @return the global resource bundle path iterator
58  	 */
59  	public ResourceBundlePathsIterator getGlobalResourceBundlePaths(ConditionalCommentCallbackHandler commentCallbackHandler, String variantKey);
60  	
61  	/***
62  	 * Returns an ordered list of the paths to use when accesing a resource bundle. 
63  	 * Each implementation may return one or several paths depending on wether all resources
64  	 * are unified into one or several bundles. The paths returned should include the prefix
65  	 * that uniquely identify the bundle contents. 
66  	 * 
67  	 * @param bundleId
68  	 * @return
69  	 */
70  	public ResourceBundlePathsIterator getBundlePaths(String bundleId, 
71  														ConditionalCommentCallbackHandler commentCallbackHandler, 
72  														String variantKey);
73  	
74  	/***
75  	 * Writes data using the supplied writer, representing a unified bundle of resources. 
76  	 * @param bundlePath
77  	 * @return
78  	 */
79  	public void writeBundleTo(String bundlePath, Writer writer) throws ResourceNotFoundException;
80  	
81  	/***
82  	 * Writes the bytes of a bundle to the specified OutputStream.
83  	 * This method is used to copy the gzip data in the output stream. 
84  	 * @param bundlePath the bundle path
85  	 * @param out the output stream
86  	 */
87  	public void streamBundleTo(String bundlePath, OutputStream out) throws ResourceNotFoundException;
88  	
89  	/***
90  	 * Returns the context bundles
91  	 * @return the context bundles
92  	 */
93  	public List getContextBundles();
94  	
95  	/***
96  	 * Generates all file bundles so that they will be ready to attend requests. 
97  	 */
98  	public void initAllBundles();
99  	
100 	/***
101 	 * Retrieves the configuration for this bundler
102 	 * @return
103 	 */
104 	public JawrConfig getConfig();
105 	
106 	
107 	/***
108 	 * Returns the client side handler generator
109 	 * @return the client side handler generator
110 	 */
111 	public ClientSideHandlerGenerator getClientSideHandler();
112 	
113 	
114 }
