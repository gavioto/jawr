145 	/***
146 	 * Splits the bundles in two lists, one for global lists and other for the remaining bundles.
147 	 */
148 	private void splitBundlesByType(List bundles) {
149 		// Temporary lists (CopyOnWriteArrayList does not support sort())
150 		List tmpGlobal = new ArrayList();
151 		List tmpContext = new ArrayList();
152 
153 		for (Iterator it = bundles.iterator(); it.hasNext();) {
154 			JoinableResourceBundle bundle = (JoinableResourceBundle) it.next();
155 
156 			// Exclude/include debug only scripts
157 			if (config.isDebugModeOn() && bundle.getInclusionPattern().isExcludeOnDebug())
158 				continue;
159 			else if (!config.isDebugModeOn() && bundle.getInclusionPattern().isIncludeOnDebug())
160 				continue;
161 
162 			if (bundle.getInclusionPattern().isGlobal())
163 				tmpGlobal.add(bundle);
164 			else
165 				tmpContext.add(bundle);
166 		}
167 
168 		// Sort the global bundles
169 		Collections.sort(tmpGlobal, new GlobalResourceBundleComparator());
170 
171 		globalBundles = ConcurrentCollectionsFactory.buildCopyOnWriteArrayList();
172 		globalBundles.addAll(tmpGlobal);
173 
174 		contextBundles = ConcurrentCollectionsFactory.buildCopyOnWriteArrayList();
175 		contextBundles.addAll(tmpContext);
176 	}

208 
209 		List bundles = new ArrayList();
210 		
211 		// if the path did not correspond to a global bundle, find the requested one.
212 		if(!isGlobalResourceBundle(bundleId)){
213 			for (Iterator it = contextBundles.iterator(); it.hasNext();) {
214 				JoinableResourceBundle bundle = (JoinableResourceBundle) it.next();
215 				if (bundle.getId().equals(bundleId)) {
216 					bundles.add(bundle);
217 					break;
218 				}
219 			}
220 		}
221 		
222 		return getBundleIterator(bundles, commentCallbackHandler, variantKey);
223 	}

242 	/***
243 	 * Removes the URL prefix defined in the configuration from a path. If the prefix contains a variant information, it adds it to the name.
244 	 * 
245 	 * @param path the path
246 	 * @return the path without the prefix
247 	 */
248 	private String removePrefixFromPath(String path) {
249 		// Remove first slash
250 		path = path.substring(1, path.length());
251 
252 		// eval the existence of a suffix
253 		String prefix = path.substring(0, path.indexOf("/"));
254 
255 		// The prefix also contains variant information after a '.'
256 		if (prefix.indexOf('.') != -1) {
257 			String suffix = '_' + prefix.substring(prefix.indexOf('.') + 1) + path.substring(path.lastIndexOf('.'));
258 			path = path.substring(path.indexOf("/"), path.lastIndexOf('.')) + suffix;
259 		} else
260 			path = path.substring(path.indexOf("/"), path.length());
261 		return path;
262 	}

265 	 * (non-Javadoc)
266 	 * 
267 	 * @see net.jawr.web.resource.bundle.ResourceCollector#writeBundleTo(java.lang.String, java.io.Writer)
268 	 */
269 	public void writeBundleTo(String bundlePath, Writer writer) throws ResourceNotFoundException {
270 		
271 		Reader rd = null;
272 
273 		// If debug mode is on, resources are retrieved one by one.
274 		if (config.isDebugModeOn()){
275 			
276 			rd = resourceHandler.getResource(bundlePath);
277 		}else{
278 			// Prefixes are used only in production mode
279 			bundlePath = removePrefixFromPath(bundlePath);
280 			rd = resourceHandler.getResourceBundleReader(bundlePath);
281 		}
282 
283 		try {
284 			IOUtils.copy(rd, writer);
285 			rd.close();
286 			writer.flush();
287 		} catch (IOException e) {
288 			throw new RuntimeException("Unexpected IOException writing bundle[" + bundlePath + "]", e);
289 		}
290 	}

296 	 */
297 	public void streamBundleTo(String bundlePath, OutputStream out) throws ResourceNotFoundException {
298 
299 		// Remove prefix, which are used only in production mode
300 		bundlePath = removePrefixFromPath(bundlePath);
301 
302 		try {
303 		
304 			ReadableByteChannel data = resourceHandler.getResourceBundleChannel(bundlePath);
305 			WritableByteChannel outChannel = Channels.newChannel(out);
306 			IOUtils.copy(data, outChannel);
307 
308 		} catch (IOException e) {
309 			throw new RuntimeException("Unexpected IOException writing bundle [" + bundlePath + "]", e);
310 		}
311 	}

322 	/*
323 	 * (non-Javadoc)
324 	 * 
325 	 * @see net.jawr.web.resource.bundle.ResourceCollector#initAllBundles()
326 	 */
327 	public void initAllBundles() {
328 
329 		if (config.getUseBundleMapping()) {
330 			bundleMapping = resourceHandler.getJawrBundleMapping();
331 		}
332 
333 		// Run through every bundle
334 		boolean mappingFileExists = resourceHandler.isExistingMappingFile();
335 		boolean processBundle = !config.getUseBundleMapping() || !mappingFileExists;
336 		for (Iterator itCol = bundles.iterator(); itCol.hasNext();) {
337 			JoinableResourceBundle bundle = (JoinableResourceBundle) itCol.next();
338 			if (bundle instanceof CompositeResourceBundle)
339 				joinAndStoreCompositeResourcebundle((CompositeResourceBundle) bundle, processBundle);
340 			else
341 				joinAndStoreBundle(bundle, processBundle);
342 			
343 			if(config.getUseBundleMapping() && !mappingFileExists){
344 				JoinableResourceBundlePropertySerializer.serializeInProperties(bundle, resourceHandler.getResourceType(), bundleMapping);
345 			}
346 		}
347 
348 		if (config.getUseBundleMapping() && !mappingFileExists) {
349 			resourceHandler.storeJawrBundleMapping(bundleMapping);
350 			
351 				
352 			if(resourceHandler.getResourceType().equals(JawrConstant.CSS_TYPE)){
353 				// Retrieve the image servlet mapping
354 				ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) config.getContext().getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
355 				if(imgRsHandler != null){
356 					// Here we update the image mapping if we are using the build time bundle processor
357 					JawrConfig imgJawrConfig = imgRsHandler.getJawrConfig();
358 					
359 					// If we use the full image bundle mapping and the jawr working directory is not located inside the web application
360 					// We store the image bundle maping which now contains the mapping for CSS images
361 					String jawrWorkingDirectory = imgJawrConfig.getJawrWorkingDirectory();
362 					if(imgJawrConfig.getUseBundleMapping() && (jawrWorkingDirectory == null || !jawrWorkingDirectory.startsWith(JawrConstant.URL_SEPARATOR))){
363 						
364 						// Store the bundle mapping
365 						Properties props = new Properties();
366 						props.putAll(imgRsHandler.getImageMap());
367 						imgRsHandler.getRsHandler().storeJawrBundleMapping(props);
368 						
369 					}
370 				}
371 			}
372 		}
373 	}

375 	/***
376 	 * Joins the members of a composite bundle in all its variants, storing in a separate file for each variant.
377 	 * 
378 	 * @param composite the composite resource bundle
379 	 * @param processBundle the flag indicating if we should process the bundle or not
380 	 */
381 	private void joinAndStoreCompositeResourcebundle(CompositeResourceBundle composite, boolean processBundle) {
382 		JoinableResourceBundleContent store = null;
383 
384 		// Collect all variant names from child bundles
385 		Set variants = new HashSet();
386 		for (Iterator it = composite.getChildBundles().iterator(); it.hasNext();) {
387 			JoinableResourceBundle childbundle = (JoinableResourceBundle) it.next();
388 			if (null != childbundle.getLocaleVariantKeys())
389 				variants.addAll(childbundle.getLocaleVariantKeys());
390 		}
391 
392 		// Process all variants
393 		for (Iterator vars = variants.iterator(); vars.hasNext();) {
394 
395 			String variant = (String) vars.next();
396 			store = new JoinableResourceBundleContent();
397 			for (Iterator it = composite.getChildBundles().iterator(); it.hasNext();) {
398 				JoinableResourceBundle childbundle = (JoinableResourceBundle) it.next();
399 				store.append(joinandPostprocessBundle(childbundle, variant, processBundle));
400 			}
401 			String name = LocaleUtils.getLocalizedBundleName(composite.getId(), variant);
402 			if (processBundle) {
403 				resourceHandler.storeBundle(name, store);
404 				initBundleDataHashcode(composite, store, variant);
405 			} 
406 		}
407 
408 		// Create the default bundle (the non variant one)
409 		store = new JoinableResourceBundleContent();
410 		for (Iterator it = composite.getChildBundles().iterator(); it.hasNext();) {
411 			JoinableResourceBundle childbundle = (JoinableResourceBundle) it.next();
412 			store.append(joinandPostprocessBundle(childbundle, null, processBundle));
413 		}
414 
415 		// Store the collected resources as a single file, both in text and gzip formats.
416 		if (processBundle) {
417 
418 			resourceHandler.storeBundle(composite.getId(), store);
419 			// Set the data hascode in the bundle, in case the prefix needs to be generated
420 			initBundleDataHashcode(composite, store, null);
421 		} 
422 
423 	}

426 	 * Initialize the bundle data hashcode and initialize the bundle mapping if needed
427 	 * @param bundle the bundle
428 	 * @param store the data to store

430 	private void initBundleDataHashcode(JoinableResourceBundle bundle, JoinableResourceBundleContent store, String localeVariant) {
431 		int bundleHashcode = store.toString().hashCode();
432 		bundle.setBundleDataHashCode(localeVariant, bundleHashcode);
433 	}

435 	/***
436 	 * Joins the members of a bundle and stores it
437 	 * 
438 	 * @param bundle the bundle
439 	 * @param the flag indicating if we should process the bundle or not
440 	 */
441 	private void joinAndStoreBundle(JoinableResourceBundle bundle, boolean processBundle) {
442 		JoinableResourceBundleContent store = null;
443 
444 		// Process the locale specific variants
445 		if (null != bundle.getLocaleVariantKeys()) {
446 			for (Iterator it = bundle.getLocaleVariantKeys().iterator(); it.hasNext();) {
447 				String variantKey = (String) it.next();
448 				String name = LocaleUtils.getLocalizedBundleName(bundle.getId(), variantKey);
449 				store = joinandPostprocessBundle(bundle, variantKey, processBundle);
450 				if (processBundle) {
451 					resourceHandler.storeBundle(name, store);
452 					initBundleDataHashcode(bundle, store, variantKey);
453 					
454 				} 
455 			}
456 		}
457 
458 		// Store the collected resources as a single file, both in text and gzip formats.
459 		store = joinandPostprocessBundle(bundle, null, processBundle);
460 
461 		if (processBundle) {
462 			resourceHandler.storeBundle(bundle.getId(), store);
463 			// Set the data hascode in the bundle, in case the prefix needs to be generated
464 			initBundleDataHashcode(bundle, store, null);
465 		} 
466 	}

468 	/***
469 	 * Reads all the members of a bundle and executes all associated postprocessors.
470 	 * 
471 	 * @param bundle the bundle
472 	 * @param variantKey the variant key
473 	 * @param the flag indicating if we should process the bundle or not
474 	 * @return the resource bundle content, where all postprocessors have been executed
475 	 */
476 	private JoinableResourceBundleContent joinandPostprocessBundle(JoinableResourceBundle bundle, String variantKey, boolean processBundle) {
477 
478 		JoinableResourceBundleContent bundleContent = new JoinableResourceBundleContent();
479 
480 		// Don't bother with the bundle if it is excluded because of the inclusion pattern
481 		// or if we don't process the bundle at start up
482 		if ((bundle.getInclusionPattern().isExcludeOnDebug() && config.isDebugModeOn())
483 				|| (bundle.getInclusionPattern().isIncludeOnDebug() && !config.isDebugModeOn())
484 				|| !processBundle)
485 			return bundleContent;
486 
487 		StringBuffer bundleData = new StringBuffer();
488 		StringBuffer store = null;
489 
490 		BundleProcessingStatus status = new BundleProcessingStatus(bundle, resourceHandler, config);
491 
492 		try {
493 			// Run through all the files belonging to the bundle
494 			for (Iterator it = bundle.getItemPathList(variantKey).iterator(); it.hasNext();) {
495 
496 				// File is first created in memory using a stringwriter.
497 				StringWriter writer = new StringWriter();
498 				BufferedWriter bwriter = new BufferedWriter(writer);
499 
500 				String path = (String) it.next();
501 				if (log.isDebugEnabled())
502 					log.debug("Adding file [" + path + "] to bundle " + bundle.getId());
503 
504 				// Get a reader on the resource, with appropiate encoding
505 				Reader rd;
506 				try {
507 					rd = resourceHandler.getResource(path, true);
508 				} catch (ResourceNotFoundException e) {
509 					// If a mapped file does not exist, a warning is issued and process continues normally.
510 					log.warn("A mapped resource was not found: [" + path + "]. Please check your configuration");
511 					continue;
512 				}
513 
514 				// Update the status.
515 				status.setLastPathAdded(path);
516 
517 				// Make a buffered reader, to read line by line.
518 				BufferedReader bRd = new BufferedReader(rd);
519 				String line;
520 
521 				// Write each line and the corresponding new line.
522 				while ((line = bRd.readLine()) != null) {
523 					bwriter.write(line);
524 					bwriter.newLine();
525 				}
526 				bRd.close();
527 				bwriter.close();
528 
529 				// Do unitary postprocessing.
530 				if (null != bundle.getUnitaryPostProcessor()) {
531 					StringBuffer resourceData = bundle.getUnitaryPostProcessor().postProcessBundle(status, writer.getBuffer());
532 
533 					// Set the CSS Classpath resource data for the debug mode
534 					initializeCssClasspathMap(bundleContent, status, writer.getBuffer());
535 
536 					bundleData.append(resourceData);
537 				} else if (null != this.unitaryPostProcessor) {
538 					if (log.isDebugEnabled())
539 						log.debug("POSTPROCESSING UNIT:" + status.getLastPathAdded());
540 					StringBuffer resourceData = this.unitaryPostProcessor.postProcessBundle(status, writer.getBuffer());
541 					
542 					// Set the CSS Classpath resource data for the debug mode
543 					initializeCssClasspathMap(bundleContent, status, writer.getBuffer());
544 
545 					bundleData.append(resourceData);
546 				} else
547 					bundleData.append(writer.getBuffer());
548 			}
549 
550 			// Post process bundle as needed
551 			if (null != bundle.getBundlePostProcessor())
552 				store = bundle.getBundlePostProcessor().postProcessBundle(status, bundleData);
553 			else if (null != this.postProcessor)
554 				store = this.postProcessor.postProcessBundle(status, bundleData);
555 			else
556 				store = bundleData;
557 
558 		} catch (IOException e) {
559 			throw new RuntimeException("Unexpected IOException generating collected file [" + bundle.getId() + "].", e);
560 		}
561 
562 		bundleContent.setContent(store);
563 		return bundleContent;
564 	}

566 	/***
567 	 * Initialize the bundle content for CSS classpath.
568 	 * 
569 	 * @param bundleContent the bundle content
570 	 * @param status the status
571 	 * @param resourceData the resource data
572 	 * @param resourceBundlePostProcessor the resourceBundle post processor
573 	 */
574 	private void initializeCssClasspathMap(JoinableResourceBundleContent bundleContent, BundleProcessingStatus status, StringBuffer resourceData) {
575 
576 		// Set the CSS Classpath resource data for the debug mode
577 		String filePath = status.getLastPathAdded();
578 		if (filePath.startsWith(JawrConstant.CLASSPATH_RESOURCE_PREFIX)) {
579 		
580 			// Here we create a new context where the bundle name is the Jawr generator CSS path
581 			// The version of the CSS classpath for debug mode will be different compare to the standard one
582 			JoinableResourceBundle tempBundle = new JoinableResourceBundleImpl(ResourceGenerator.CSS_DEBUGPATH, null, null, null, null);
583 			BundleProcessingStatus tempStatus = new BundleProcessingStatus(tempBundle, status.getRsHandler(), status.getJawrConfig());
584 			
585 			tempStatus.setLastPathAdded(status.getLastPathAdded());
586 			CSSURLPathRewriterPostProcessor resourceBundlePostProcessor = new CSSURLPathRewriterPostProcessor();
587 			resourceData = resourceBundlePostProcessor.postProcessBundle(tempStatus, resourceData);
588 			bundleContent.putCssClasspathDebugContent(filePath.substring(JawrConstant.CLASSPATH_RESOURCE_PREFIX.length()), resourceData.toString());
589 		}
590 	}

597 	public JoinableResourceBundle resolveBundleForPath(String path) {
598 
599 		JoinableResourceBundle theBundle = null;
600 		for (Iterator it = bundles.iterator(); it.hasNext();) {
601 			JoinableResourceBundle bundle = (JoinableResourceBundle) it.next();
602 			if (bundle.getId().equals(path) || bundle.belongsToBundle(path)) {
603 				theBundle = bundle;
604 				break;
605 			}
606 		}
607 		return theBundle;
