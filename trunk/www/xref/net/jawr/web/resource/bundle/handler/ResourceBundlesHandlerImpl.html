198 	/***
199 	 * Splits the bundles in two lists, one for global lists and other for the
200 	 * remaining bundles.
201 	 */
202 	private void splitBundlesByType(List bundles) {
203 		// Temporary lists (CopyOnWriteArrayList does not support sort())
204 		List tmpGlobal = new ArrayList();
205 		List tmpContext = new ArrayList();
206 
207 		for (Iterator it = bundles.iterator(); it.hasNext();) {
208 			JoinableResourceBundle bundle = (JoinableResourceBundle) it.next();
209 
210 			// Exclude/include debug only scripts
211 			if (config.isDebugModeOn()
212 					&& bundle.getInclusionPattern().isExcludeOnDebug())
213 				continue;
214 			else if (!config.isDebugModeOn()
215 					&& bundle.getInclusionPattern().isIncludeOnDebug())
216 				continue;
217 
218 			if (bundle.getInclusionPattern().isGlobal())
219 				tmpGlobal.add(bundle);
220 			else
221 				tmpContext.add(bundle);
222 		}
223 
224 		// Sort the global bundles
225 		Collections.sort(tmpGlobal, new GlobalResourceBundleComparator());
226 
227 		globalBundles = ConcurrentCollectionsFactory
228 				.buildCopyOnWriteArrayList();
229 		globalBundles.addAll(tmpGlobal);
230 
231 		contextBundles = ConcurrentCollectionsFactory
232 				.buildCopyOnWriteArrayList();
233 		contextBundles.addAll(tmpContext);
234 	}

243

339 			ConditionalCommentCallbackHandler commentCallbackHandler,
340 			Map variants) {
341 
342 		List bundles = new ArrayList();
343 
344 		// if the path did not correspond to a global bundle, find the requested
345 		// one.
346 		if (!isGlobalResourceBundle(bundleId)) {
347 			for (Iterator it = contextBundles.iterator(); it.hasNext();) {
348 				JoinableResourceBundle bundle = (JoinableResourceBundle) it
349 						.next();
350 				if (bundle.getId().equals(bundleId)) {
351 
352 					bundles.add(bundle);
353 					break;
354 				}
355 			}
356 		}
357 
358 		return getBundleIterator(debugMode, bundles, commentCallbackHandler,
359 				variants);
360 	}

388 	 * (non-Javadoc)
389 	 * 
390 	 * @see
391 	 * net.jawr.web.resource.bundle.ResourceCollector#writeBundleTo(java.lang
392 	 * .String, java.io.Writer)
393 	 */
394 	public void writeBundleTo(String bundlePath, Writer writer)
395 			throws ResourceNotFoundException {
396 
397 		Reader rd = null;
398 
399 		try {
400 			
401 			// If debug mode is on, resources are retrieved one by one.
402 			if (config.isDebugModeOn()) {
403 
404 				rd = resourceHandler.getResource(bundlePath);
405 			} else {
406 				// Prefixes are used only in production mode
407 				String path = PathNormalizer.removeVariantPrefixFromPath(bundlePath);
408 				rd = resourceBundleHandler.getResourceBundleReader(path);
409 				if(liveProcessBundles.contains(path)){
410 					rd = processInLive(rd);
411 				}
412 			}
413 			
414 			IOUtils.copy(rd, writer);
415 			writer.flush();
416 		} catch (IOException e) {
417 			throw new BundlingProcessException("Unexpected IOException writing bundle["
418 					+ bundlePath + "]", e);
419 		}finally{
420 			IOUtils.close(rd);
421 		}
422 	}

424 	/***
425 	 * Process the bundle content in live
426 	 * @param reader the reader
427 	 * @return the processed bundle content
428 	 * @throws IOException if an IOException occured
429 	 */
430 	private StringReader processInLive(Reader reader)
431 			throws IOException {
432 		
433 		String requestURL = ThreadLocalJawrContext.getRequestURL();
434 		StringWriter swriter = new StringWriter(); 
435 		IOUtils.copy(reader, swriter, true);
436 		String updatedContent = swriter.getBuffer().toString();
437 		if(requestURL != null){
438 			
439 			updatedContent = updatedContent.replaceAll(JawrConstant.JAWR_BUNDLE_PATH_PLACEHOLDER_PATTERN, requestURL);
440 		}
441 		return new StringReader(updatedContent);
442 	}

444 	/*
445 	 * (non-Javadoc)
446 	 * 
447 	 * @see
448 	 * net.jawr.web.resource.bundle.ResourceBundlesHandler#streamBundleTo(java
449 	 * .lang.String, java.io.OutputStream)
450 	 */
451 	public void streamBundleTo(String bundlePath, OutputStream out)
452 			throws ResourceNotFoundException {
453 
454 		// Remove prefix, which are used only in production mode
455 		String path = PathNormalizer.removeVariantPrefixFromPath(bundlePath);
456 		ReadableByteChannel data = null;
457 		try {
458 			if(liveProcessBundles.contains(path)){
459 				
460 				Reader rd = resourceBundleHandler.getResourceBundleReader(path);
461 				StringReader strRd = processInLive(rd);
462 				StringWriter strWriter = new StringWriter();
463 				IOUtils.copy(strRd, strWriter);
464 				
465 				ByteArrayOutputStream bos = new ByteArrayOutputStream();
466 				GZIPOutputStream gzOut = new GZIPOutputStream(bos);
467 				byte[] byteData = strWriter.getBuffer().toString().getBytes(
468 						config.getResourceCharset().name());
469 				gzOut.write(byteData, 0, byteData.length);
470 				gzOut.close();
471 				ByteArrayInputStream bis = new ByteArrayInputStream(bos.toByteArray());
472 				data = Channels.newChannel(bis);
473 			
474 			}else{
475 				data = resourceBundleHandler.getResourceBundleChannel(path);
476 			}
477 			
478 			
479 			WritableByteChannel outChannel = Channels.newChannel(out);
480 			IOUtils.copy(data, outChannel);
481 		
482 		} catch (IOException e) {
483 			throw new BundlingProcessException(
484 					"Unexpected IOException writing bundle [" + path
485 							+ "]", e);
486 		}finally{
487 			IOUtils.close(data);
488 		}
489 	}

500 	/*
501 	 * (non-Javadoc)
502 	 * 
503 	 * @see net.jawr.web.resource.bundle.ResourceCollector#initAllBundles()
504 	 */
505 	public void initAllBundles() {
506 
507 		if (config.getUseBundleMapping()) {
508 			bundleMapping = resourceBundleHandler.getJawrBundleMapping();
509 		}
510 
511 		// Run through every bundle
512 		boolean mappingFileExists = resourceBundleHandler
513 				.isExistingMappingFile();
514 		boolean processBundleFlag = !config.getUseBundleMapping()
515 				|| !mappingFileExists;
516 
517 		if (resourceTypeProcessor != null) {
518 			GlobalPreprocessingContext ctx = new GlobalPreprocessingContext(
519 					config, resourceHandler, processBundleFlag);
520 			resourceTypeProcessor.processBundles(ctx, bundles);
521 		}
522 
523 		for (Iterator itCol = bundles.iterator(); itCol.hasNext();) {
524 			JoinableResourceBundle bundle = (JoinableResourceBundle) itCol
525 					.next();
526 
527 			boolean processBundle = processBundleFlag;
528 			if (!ThreadLocalJawrContext.isBundleProcessingAtBuildTime()
529 					&& null != bundle.getAlternateProductionURL()) {
530 				if (LOGGER.isDebugEnabled()) {
531 					LOGGER
532 							.debug("No bundle generated for '"
533 									+ bundle.getId()
534 									+ "' because a production URL is defined for this bundle.");
535 				}
536 				processBundle = false;
537 			}
538 			if (bundle instanceof CompositeResourceBundle)
539 				joinAndStoreCompositeResourcebundle(
540 						(CompositeResourceBundle) bundle, processBundle);
541 			else
542 				joinAndStoreBundle(bundle, processBundle);
543 
544 			if (config.getUseBundleMapping() && !mappingFileExists) {
545 				JoinableResourceBundlePropertySerializer.serializeInProperties(
546 						bundle, resourceBundleHandler.getResourceType(),
547 						bundleMapping);
548 			}
549 		}
550 
551 		if (config.getUseBundleMapping() && !mappingFileExists) {
552 			resourceBundleHandler.storeJawrBundleMapping(bundleMapping);
553 
554 			if (resourceBundleHandler.getResourceType().equals(
555 					JawrConstant.CSS_TYPE)) {
556 				// Retrieve the image servlet mapping
557 				ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) config
558 						.getContext().getAttribute(
559 								JawrConstant.IMG_CONTEXT_ATTRIBUTE);
560 				if (imgRsHandler != null) {
561 					// Here we update the image mapping if we are using the
562 					// build time bundle processor
563 					JawrConfig imgJawrConfig = imgRsHandler.getJawrConfig();
564 
565 					// If we use the full image bundle mapping and the jawr
566 					// working directory is not located inside the web
567 					// application
568 					// We store the image bundle maping which now contains the
569 					// mapping for CSS images
570 					String jawrWorkingDirectory = imgJawrConfig
571 							.getJawrWorkingDirectory();
572 					if (imgJawrConfig.getUseBundleMapping()
573 							&& (jawrWorkingDirectory == null || !jawrWorkingDirectory
574 									.startsWith(JawrConstant.URL_SEPARATOR))) {
575 
576 						// Store the bundle mapping
577 						Properties props = new Properties();
578 						props.putAll(imgRsHandler.getImageMap());
579 						imgRsHandler.getRsBundleHandler()
580 								.storeJawrBundleMapping(props);
581 
582 					}
583 				}
584 			}
585 		}
586 	}

588 	/***
589 	 * Joins the members of a composite bundle in all its variants, storing in a
590 	 * separate file for each variant.
591 	 * 
592 	 * @param composite
593 	 *            the composite resource bundle
594 	 * @param processBundle
595 	 *            the flag indicating if we should process the bundle or not
596 	 */
597 	private void joinAndStoreCompositeResourcebundle(
598 			CompositeResourceBundle composite, boolean processBundle) {
599 
600 		BundleProcessingStatus status = new BundleProcessingStatus(BundleProcessingStatus.FILE_PROCESSING_TYPE, composite,
601 				resourceHandler, config);
602 		
603 		// Collect all variant names from child bundles
604 		Map compositeBundleVariants = new HashMap();
605 		for (Iterator it = composite.getChildBundles().iterator(); it.hasNext();) {
606 			JoinableResourceBundle childbundle = (JoinableResourceBundle) it
607 					.next();
608 			if (null != childbundle.getVariants())
609 				compositeBundleVariants = VariantUtils.concatVariants(compositeBundleVariants, childbundle.getVariants());
610 		}
611 		composite.setVariants(compositeBundleVariants);
612 		status.setSearchingPostProcessorVariants(true);
613 		joinAndPostProcessBundle(composite, status, processBundle);
614 
615 		Map postProcessVariants = status.getPostProcessVariants();
616 		if(!postProcessVariants.isEmpty()){
617 			if(LOGGER.isDebugEnabled()){
618 				LOGGER.debug("Post process variants found for bundle "+composite.getId()+":"+ postProcessVariants);
619 			}
620 			Map newVariants =  VariantUtils.concatVariants(composite.getVariants(), postProcessVariants);
621 			composite.setVariants(newVariants);
622 			status.setSearchingPostProcessorVariants(false);
623 			joinAndPostProcessBundle(composite, status, processBundle);
624 		}
625 		
626 	}

628 	/***
629 	 * Joins and post process the variant composite bundle
630 	 * @param composite the composite bundle
631 	 * @param status the status
632 	 * @param compositeBundleVariants the variants
633 	 * @param processBundle the flag indicating if we must process the bundle or not.
634 	 */
635 	private void joinAndPostProcessBundle(CompositeResourceBundle composite,
636 			BundleProcessingStatus status, boolean processBundle) {
637 		JoinableResourceBundleContent store;
638 		
639 		List allVariants = VariantUtils.getAllVariants(composite.getVariants());
640 		// Add the default bundle variant (the non variant one)
641 		allVariants.add(null);
642 		// Process all variants
643 		for (Iterator vars = allVariants.iterator(); vars.hasNext();) {
644 
645 			Map variants = (Map) vars.next();
646 			status.setBundleVariants(variants);
647 			store = new JoinableResourceBundleContent();
648 			for (Iterator it = composite.getChildBundles().iterator(); it
649 					.hasNext();) {
650 				JoinableResourceBundle childbundle = (JoinableResourceBundle) it
651 						.next();
652 				
653 				JoinableResourceBundleContent childContent = joinAndPostprocessBundle(childbundle, variants,
654 						status, processBundle);
655 				// Do unitary postprocessing.
656 				status.setProcessingType(BundleProcessingStatus.FILE_PROCESSING_TYPE);
657 				StringBuffer content = executeUnitaryPostProcessing(composite, status, childContent.getContent(), this.unitaryCompositePostProcessor);
658 				childContent.setContent(content);
659 				store.append(childContent);

842 	/***
843 	 * Reads all the members of a bundle and executes all associated
844 	 * postprocessors.
845 	 * 
846 	 * @param bundle
847 	 *            the bundle
848 	 * @param variants
849 	 *            the variant map
850 	 * @param the bundling processing status           
851 	 * @param the
852 	 *            flag indicating if we should process the bundle or not
853 	 * @return the resource bundle content, where all postprocessors have been
854 	 *         executed
855 	 */
856 	private JoinableResourceBundleContent joinAndPostprocessBundle(
857 			JoinableResourceBundle bundle, Map variants,
858 			BundleProcessingStatus status, boolean processBundle) {
859 
860 		JoinableResourceBundleContent bundleContent = new JoinableResourceBundleContent();
861 
862 		// Don't bother with the bundle if it is excluded because of the
863 		// inclusion pattern
864 		// or if we don't process the bundle at start up
865 		if ((bundle.getInclusionPattern().isExcludeOnDebug() && config
866 				.isDebugModeOn())
867 				|| (bundle.getInclusionPattern().isIncludeOnDebug() && !config
868 						.isDebugModeOn()) || !processBundle)
869 			return bundleContent;
870 
871 		StringBuffer bundleData = new StringBuffer();
872 		StringBuffer store = null;
873 
874 		try {
875 			// Run through all the files belonging to the bundle
876 			for (Iterator it = bundle.getItemPathList(variants).iterator(); it
877 					.hasNext();) {
878 
879 				// File is first created in memory using a stringwriter.
880 				StringWriter writer = new StringWriter();
881 				BufferedWriter bwriter = new BufferedWriter(writer);
882 
883 				String path = (String) it.next();
884 				if (LOGGER.isDebugEnabled())
885 					LOGGER.debug("Adding file [" + path + "] to bundle "
886 							+ bundle.getId());
887 
888 				// Get a reader on the resource, with appropiate encoding
889 				Reader rd = null;
890 
891 				try {
892 					rd = resourceHandler.getResource(path, true);
893 				} catch (ResourceNotFoundException e) {
894 					// If a mapped file does not exist, a warning is issued and
895 					// process continues normally.
896 					LOGGER.warn("A mapped resource was not found: [" + path
897 							+ "]. Please check your configuration");
898 					continue;
899 				}
900 
901 				// Update the status.
902 				status.setLastPathAdded(path);
903 
904 				// Copy the content
905 				IOUtils.copy(rd, bwriter, true);
906 				
907 				// Add new line at the end if it doesn't exist
908 				if(!writer.getBuffer().toString().endsWith(StringUtils.LINE_SEPARATOR)){
909 					writer.getBuffer().append(StringUtils.LINE_SEPARATOR);
910 				}
911 				
912 				// Do unitary postprocessing.
913 				status.setProcessingType(BundleProcessingStatus.FILE_PROCESSING_TYPE);
914 				bundleData.append(executeUnitaryPostProcessing(bundle, status, writer.getBuffer(), this.unitaryPostProcessor));
915 			}
916 
917 			// Post process bundle as needed
918 			store = executeBundlePostProcessing(bundle, status, bundleData);
919 
920 		} catch (IOException e) {
921 			throw new BundlingProcessException(
922 					"Unexpected IOException generating collected file ["
923 							+ bundle.getId() + "].", e);
924 		}
925 
926 		bundleContent.setContent(store);
927 		return bundleContent;
928 	}
