39  	/***
40  	 * Returns the localized bundle name
41  	 * 
42  	 * @param bundleName
43  	 * @param localeKey
44  	 * @return the localized bundle name
45  	 */
46  	public static String getLocalizedBundleName(String bundleName, String localeKey) {
47  
48  		String newName = bundleName;
49  		int idxSeparator = bundleName.lastIndexOf('.');
50  		if (StringUtils.isNotEmpty(localeKey) && idxSeparator != -1) {
51  			newName = bundleName.substring(0, idxSeparator);
52  			newName += '_' + localeKey;
53  			newName += bundleName.substring(idxSeparator);
54  		}
55  
56  		return newName;
57  	}

59  	/***
60  	 * Returns the list of available locale suffixes for a message resource bundle
61  	 * 
62  	 * @param messageBundlePath the resource bundle path
63  	 * @return the list of available locale suffixes for a message resource bundle
64  	 */
65  	public static List getAvailableLocaleSuffixesForBundle(String messageBundlePath) {
66  
67  		int idxNameSpace = messageBundlePath.indexOf("(");
68  		int idxFilter = messageBundlePath.indexOf("[");
69  		int idx = -1;
70  		if(idxNameSpace != -1 && idxFilter != -1){
71  			idx = Math.min(idxNameSpace, idxFilter);
72  		}else if(idxNameSpace != -1 && idxFilter == -1){
73  			idx = idxNameSpace;
74  		}else if(idxNameSpace == -1 && idxFilter != -1){
75  			idx = idxFilter;
76  		}
77  		
78  		String messageBundle = null;
79  		if(idx > 0){
80  			messageBundle = messageBundlePath.substring(0, idx);  
81  		}else{
82  			messageBundle = messageBundlePath;
83  		}
84  		return getAvailableLocaleSuffixes(messageBundle);
85  	}

87  	/***
88  	 * Returns the list of available locale suffixes for a message resource bundle
89  	 * 
90  	 * @param messageBundle the resource bundle path
91  	 * @return the list of available locale suffixes for a message resource bundle
92  	 */
93  	public static List getAvailableLocaleSuffixes(String messageBundle) {
94  		List availableLocaleSuffixes = new ArrayList();
95  		Locale[] availableLocales = Locale.getAvailableLocales();
96  	
97  		addSuffixIfAvailable(messageBundle, availableLocaleSuffixes, null);
98  	
99  		for (int i = 0; i < availableLocales.length; i++) {
100 			Locale locale = availableLocales[i];
101 			addSuffixIfAvailable(messageBundle, availableLocaleSuffixes, locale);
102 		}
103 	
104 		return availableLocaleSuffixes;
105 	}
106 	/***
107 	 * Add the locale suffix if the message resource bundle file exists.
108 	 * 
109 	 * @param messageBundlePath the message resource bundle path
110 	 * @param availableLocaleSuffixes the list of available locale suffix to update
111 	 * @param locale the locale to check.
112 	 */
113 	private static void addSuffixIfAvailable(String messageBundlePath, List availableLocaleSuffixes, Locale locale) {
114 		String localMsgResourcePath = toBundleName(messageBundlePath, locale) + MSG_RESOURCE_BUNDLE_SUFFIX;
115 		URL resourceUrl = null;
116 		try {
117 			resourceUrl = ClassLoaderResourceUtils.getResourceURL(localMsgResourcePath, LocaleUtils.class);
118 		} catch (Exception e) {
119 			// Nothing to do
120 		}
121 
122 		if (resourceUrl != null) {
123 
124 			String suffix = localMsgResourcePath.substring(messageBundlePath.length());
125 			if (suffix.length() > 0) {
126 
127 				if (suffix.length() == MSG_RESOURCE_BUNDLE_SUFFIX.length()) {
128 					suffix = "";
129 				} else {
130 					// remove the "_" before the suffix "_en_US" => "en_US"
131 					suffix = suffix.substring(1, suffix.length() - MSG_RESOURCE_BUNDLE_SUFFIX.length());
132 				}
133 			}
134 			availableLocaleSuffixes.add(suffix);
135 		}
136 	}

138 	/***
139 	 * Converts the given <code>baseName</code> and <code>locale</code> to the bundle name. This method is called from the default implementation of
140 	 * the {@link #newBundle(String, Locale, String, ClassLoader, boolean) newBundle} and
141 	 * {@link #needsReload(String, Locale, String, ClassLoader, ResourceBundle, long) needsReload} methods.
142 	 * 
143 	 * <p>
144 	 * This implementation returns the following value:
145 	 * 
146 	 * <pre>
147 	 * baseName + &quot;_&quot; + language + &quot;_&quot; + country + &quot;_&quot; + variant
148 	 * </pre>
149 	 * 
150 	 * where <code>language</code>, <code>country</code> and <code>variant</code> are the language, country and variant values of <code>locale</code>,
151 	 * respectively. Final component values that are empty Strings are omitted along with the preceding '_'. If all of the values are empty strings,
152 	 * then <code>baseName</code> is returned.
153 	 * 
154 	 * <p>
155 	 * For example, if <code>baseName</code> is <code>"baseName"</code> and <code>locale</code> is <code>Locale("ja",&nbsp;"",&nbsp;"XX")</code>, then
156 	 * <code>"baseName_ja_&thinsp;_XX"</code> is returned. If the given locale is <code>Locale("en")</code>, then <code>"baseName_en"</code> is
157 	 * returned.
158 	 * 
159 	 * <p>
160 	 * Overriding this method allows applications to use different conventions in the organization and packaging of localized resources.
161 	 * 
162 	 * @param baseName the base name of the resource bundle, a fully qualified class name
163 	 * @param locale the locale for which a resource bundle should be loaded
164 	 * @return the bundle name for the resource bundle
165 	 * @exception NullPointerException if <code>baseName</code> or <code>locale</code> is <code>null</code>
166 	 */
167 	public static String toBundleName(String baseName, Locale locale) {
168 		
169 		baseName = baseName.replace('.', '/');
170 		if (locale == null) {
171 			return baseName;
172 		}
173 
174 		String language = locale.getLanguage();
175 		String country = locale.getCountry();
176 		String variant = locale.getVariant();
177 
178 		if (language == "" && country == "" && variant == "") {
179 			return baseName;
180 		}
181 
182 		StringBuffer sb = new StringBuffer(baseName);
183 		sb.append('_');
184 		if (variant != "") {
185 			sb.append(language).append('_').append(country).append('_').append(variant);
186 		} else if (country != "") {
187 			sb.append(language).append('_').append(country);
188 		} else {
189 			sb.append(language);
190 		}
191 		return sb.toString();
192 
193 	}
