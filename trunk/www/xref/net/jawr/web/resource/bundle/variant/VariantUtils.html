39  	/***
40  	 * Returns all variant keys from the map of variant set
41  	 * @param variants the map of variant set
42  	 * @return all variant keys
43  	 */
44  	public static List getAllVariants(Map variantSets){
45  		
46  		List variantKeys = new ArrayList();
47  		if(variantSets != null){
48  			for (Iterator itVariantSets = new TreeMap(variantSets).entrySet().iterator(); itVariantSets.hasNext();) {
49  				
50  				Entry variantEntry = (Entry) itVariantSets.next();
51  				String variantType = (String) variantEntry.getKey();
52  				Collection variantList = (Collection) variantEntry.getValue();
53  				if(variantKeys.isEmpty()){
54  					variantKeys = getVariants(null, variantType, variantList);
55  				}else{
56  					
57  					List tmpResult = new ArrayList();
58  					for (Iterator itCurVariantKeys = variantKeys.iterator(); itCurVariantKeys
59  							.hasNext();) {
60  						Map curVariant = (Map) itCurVariantKeys.next();
61  						tmpResult.addAll(getVariants(curVariant, variantType, variantList));
62  					}
63  					variantKeys = tmpResult;
64  				}
65  			}
66  		}
67  		
68  		return variantKeys;
69  	}

71  	/***
72  	 * Returns the list of variant maps, which are initialize with the current map values and 
73   	 * each element of the list contains a element of the variant list with the variant type as key  
74  	 * @param curVariant the current variant map
75  	 * @param variantType the variant type
76  	 * @param variantList the variant list
77  	 * @return the list of variant maps
78  	 */
79  	private static List getVariants(Map curVariant, String variantType, Collection variantList) {
80  		
81  		List variants = new ArrayList();
82  		for (Iterator itCurVariant = variantList.iterator(); itCurVariant.hasNext();) {
83  			
84  			String variant = (String) itCurVariant.next();
85  			
86  			Map map = new HashMap();
87  			if(curVariant != null){
88  				map.putAll(curVariant);
89  			}
90  			map.put(variantType, variant);
91  			variants.add(map);
92  		}
93  		return variants;
94  	}

109 				
110 				List tmpResult = new ArrayList();
111 				for (Iterator itCurVariantKeys = variantKeys.iterator(); itCurVariantKeys
112 						.hasNext();) {
113 					String curVariantKey = (String) itCurVariantKeys.next();
114 					tmpResult.addAll(getVariantKeys(curVariantKey+JawrConstant.VARIANT_SEPARATOR_CHAR, variantList));
115 				}
116 				variantKeys = tmpResult;
117 			}
118 		}
119 		
120 		return variantKeys;

124 	 * Returns the variant keys
125 	 * @param variantKeyPrefix the variant key prefix
126 	 * @param variants The variants
127 	 */
128 	private static List getVariantKeys(String variantKeyPrefix, Collection variants) {
129 		
130 		List variantKeys = new ArrayList();
131 		for (Iterator itCurVariant = variants.iterator(); itCurVariant.hasNext();) {
132 			
133 			String variant = (String) itCurVariant.next();
134 			if(variant == null){
135 				variant = "";
136 			}
137 			if(variantKeyPrefix == null){
138 				variantKeys.add(variant);
139 			}else{
140 				variantKeys.add(variantKeyPrefix+variant);
141 			}
142 		}
143 		return variantKeys;
144 	}

161 	/***
162 	 * Resolves a registered path from a locale key, using the same algorithm used to locate ResourceBundles.
163 	 * 
164 	 * @param variantKey the requested variant key
165 	 * @return the variant key to use
166 	 */
167 	public static String getVariantKey(Map curVariants, Set variantTypes) {
168 		
169 		String variantKey = "";
170 		if(curVariants != null && variantTypes != null){
171 			
172 			Map tempVariants = new TreeMap(curVariants);
173 			StringBuffer variantKeyBuf = new StringBuffer();
174 			for (Iterator it = tempVariants.entrySet().iterator(); it
175 					.hasNext();) {
176 				Entry entry = (Entry) it.next();
177 				if(variantTypes.contains(entry.getKey())){
178 					String value = (String) entry.getValue();
179 					if(value == null){
180 						value = "";
181 					}
182 					variantKeyBuf.append(value+JawrConstant.VARIANT_SEPARATOR_CHAR);
183 				}
184 			}
185 			
186 			variantKey = variantKeyBuf.toString();
187 			if(StringUtils.isNotEmpty(variantKey) && variantKey.charAt(variantKey.length()-1) == JawrConstant.VARIANT_SEPARATOR_CHAR){
188 				variantKey = variantKey.substring(0, variantKey.length()-1);
189 			}
190 		}
191 		
192 		return variantKey;
193 	}

196 	 * Get the bundle name taking in account the variant key
197 	 * @param bundleName the bundle name
198 	 * @param variantKey the variant key
199 	 * @return the variant bundle name
200 	 */
201 	public static String getVariantBundleName(String bundleName, String variantKey){
202 		
203 		String newName = bundleName;
204 		if (StringUtils.isNotEmpty(variantKey)){
205 			int idxSeparator = bundleName.lastIndexOf('.');
206 			if(idxSeparator != -1) {
207 				newName = bundleName.substring(0, idxSeparator);
208 				newName += JawrConstant.VARIANT_SEPARATOR_CHAR + variantKey;
209 				newName += bundleName.substring(idxSeparator);
210 			}else{
211 				newName += JawrConstant.VARIANT_SEPARATOR_CHAR + variantKey;
212 			}
213 		}
214 
215 		return newName;
216 	}

230 	/***
231 	 * Concatenates 2 map of variant sets.
232 	 * @param variantSet1 the first map 
233 	 * @param variantSet2 the second map 
234 	 * @return the concatenated variant map
235 	 */
236 	public static Map concatVariants(Map variantSet1, Map variantSet2) {
237 		
238 		Map result = new HashMap();
239 		if(!isEmpty(variantSet1) && isEmpty(variantSet2)){
240 			result.putAll(variantSet1);
241 		}else if(isEmpty(variantSet1) && !isEmpty(variantSet2)){
242 			result.putAll(variantSet2);
243 		}else if(!isEmpty(variantSet1) && !isEmpty(variantSet2)){
244 			
245 			Set keySet = new HashSet();
246 			keySet.addAll(variantSet1.keySet());
247 			keySet.addAll(variantSet2.keySet());
248 			for (Iterator iterator = keySet.iterator(); iterator
249 					.hasNext();) {
250 				String variantType = (String) iterator.next();
251 				VariantSet variants1 = (VariantSet) variantSet1.get(variantType);
252 				VariantSet variants2 = (VariantSet) variantSet2.get(variantType);
253 				Set	variants = new HashSet();
254 				String defaultVariant = null;
255 				
256 				if(variants1 != null && variants2 != null &&
257 						!variants1.hasSameDefaultVariant(variants2)){
258 					throw new BundlingProcessException("For the variant type '"+variantType+"', the variant sets defined in your bundles don't have the same default value.");
259 				}
260 				
261 				if(variants1 != null){
262 					variants.addAll(variants1);
263 					defaultVariant = variants1.getDefaultVariant(); 
264 				}
265 				if(variants2 != null){
266 					variants.addAll(variants2);
267 					defaultVariant = variants2.getDefaultVariant();
268 				}
269 				
270 				VariantSet variantSet = new VariantSet(variantType, defaultVariant, variants);
271 				result.put(variantType, variantSet);
272 			}
273 		}
274 		
275 		return result;
276 	}
