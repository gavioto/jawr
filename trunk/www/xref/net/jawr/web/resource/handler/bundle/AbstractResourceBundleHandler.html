185 	 * Initialize the temporary directories
186 	 * 
187 	 * @param tempDirRoot the temporary directory root
188 	 * @param createTempSubDir the flag indicating if we should create the temporary directory.
189 	 */
190 	private void initTempDirectory(String tempDirRoot, boolean createTempSubDir) {
191 
192 		tempDirPath = tempDirRoot;
193 
194 		if (createTempSubDir) {
195 			tempDirPath = tempDirPath + File.separator + TEMP_SUBDIR;
196 		}
197 
198 		// In windows, pathnames with spaces are returned as %20
199 		if (tempDirPath.indexOf("%20") != -1)
200 			tempDirPath = tempDirPath.replaceAll("%20", " ");
201 
202 		this.textDirPath = tempDirPath + File.separator + TEMP_TEXT_SUBDIR;
203 		this.gzipDirPath = tempDirPath + File.separator + TEMP_GZIP_SUBDIR;
204 		this.cssClasspathDirPath = tempDirPath + File.separator
205 				+ TEMP_CSS_CLASSPATH_SUBDIR;
206 
207 		if (createTempSubDir) {
208 			try {
209 				createDir(tempDirPath);
210 				createDir(textDirPath);
211 				createDir(gzipDirPath);
212 				createDir(cssClasspathDirPath);
213 			} catch (IOException e) {
214 				throw new RuntimeException(
215 						"Unexpected IOException creating temporary jawr directory",
216 						e);
217 			}
218 		}
219 	}

288 	public InputStream getTemporaryResourceAsStream(String resourceName)
289 			throws ResourceNotFoundException {
290 		InputStream is = null;
291 		if (useFileSystemTempDir) {
292 			File file = new File(resourceName);
293 			try {
294 				is = new FileInputStream(file);
295 			} catch (FileNotFoundException e) {
296 				throw new ResourceNotFoundException(resourceName);
297 			}
298 		}else{
299 			is = doGetResourceAsStream(resourceName);
300 		}
301 		
302 		if (is == null) {
303 			throw new ResourceNotFoundException(resourceName);
304 		}
305 		
306 		return is;

427 	public void storeBundle(String bundleName,
428 			JoinableResourceBundleContent bundleResourcesContent) {
429 
430 		// Text version
431 		String bundleContent = bundleResourcesContent.getContent().toString();
432 		storeBundle(bundleName, bundleContent, false, textDirPath);
433 
434 		// binary version
435 		storeBundle(bundleName, bundleContent, true, gzipDirPath);
436 
437 		// Store Css classpath debug files
438 		Map cssClasspathDebugContentMap = bundleResourcesContent
439 				.getCssClasspathDebugContentMap();
440 		Iterator entryIterator = cssClasspathDebugContentMap.entrySet().iterator();
441 		while (entryIterator.hasNext()) {
442 			Entry entry =  (Entry) entryIterator.next();
443 			String filePath = (String) entry.getKey();
444 			storeBundle(filePath, (String) entry
445 					.getValue(), false, cssClasspathDirPath);
446 		}
447 	}

449 	/***
450 	 * Stores a resource bundle either in text or binary gzipped format.
451 	 * 
452 	 * @param bundleName the bundle name
453 	 * @param bundledResources the bundledRessources
454 	 * @param gzipFile a fag defining if the file is gzipped or not
455 	 * @param rootDir the root directory
456 	 */
457 	private void storeBundle(String bundleName, String bundledResources,
458 			boolean gzipFile, String rootdir) {
459 		if (log.isDebugEnabled()) {
460 			String msg = "Storing a generated "
461 					+ (gzipFile ? "and gzipped" : "")
462 					+ " bundle with an id of:" + bundleName;
463 			log.debug(msg);
464 		}
465 
466 		try {
467 			// Create subdirs if needed
468 			if (bundleName.indexOf('/') != -1) {
469 				StringTokenizer tk = new StringTokenizer(bundleName, "/");
470 				StringBuffer pathName = new StringBuffer(rootdir);
471 				while (tk.hasMoreTokens()) {
472 					String name = tk.nextToken();
473 					if (tk.hasMoreTokens()) {
474 						pathName.append(File.separator+name);
475 						createDir(pathName.toString());
476 					}
477 				}
478 				bundleName = bundleName.replace('/', File.separatorChar);
479 			}
480 
481 			File store = createNewFile(rootdir + File.separator + bundleName);
482 
483 			if (gzipFile) {
484 				FileOutputStream fos = new FileOutputStream(store);
485 				GZIPOutputStream gzOut = new GZIPOutputStream(fos);
486 				byte[] data = bundledResources.toString().getBytes(
487 						charset.name());
488 				gzOut.write(data, 0, data.length);
489 				gzOut.close();
490 			} else {
491 				FileOutputStream fos = new FileOutputStream(store);
492 				FileChannel channel = fos.getChannel();
493 				Writer wr = Channels.newWriter(channel, charset.newEncoder(),
494 						-1);
495 				wr.write(bundledResources.toString());
496 				wr.close();
497 			}
498 		} catch (IOException e) {
499 			e.printStackTrace();
500 			throw new RuntimeException(
501 					"Unexpected IOException creating temporary jawr file", e);
502 		}
503 	}

505 	/***
506 	 * Creates a directory. If dir is note created for some reason a runtimeexception is thrown.
507 	 * 
508 	 * @param dir
509 	 * @throws IOException
510 	 */
511 	private File createDir(String path) throws IOException {
512 		// In windows, pathnames with spaces are returned as %20
513 		if (path.indexOf("%20") != -1)
514 			path = path.replaceAll("%20", " ");
515 		File dir = new File(path);
516 		if (!dir.exists() && !dir.mkdirs())
517 			throw new RuntimeException(
518 					"Error creating temporary jawr directory with path:"
519 							+ dir.getPath());
520 
521 		if (log.isDebugEnabled())
522 			log.debug("Created dir: " + dir.getCanonicalPath());
523 		return dir;
524 	}

533 	private File createNewFile(String path) throws IOException {
534

536 		if (path.indexOf("%20") != -1)
537 			path = path.replaceAll("%20", " ");
538 		File newFile = new File(path);

541 			throw new RuntimeException("Unable to create a temporary file at "
