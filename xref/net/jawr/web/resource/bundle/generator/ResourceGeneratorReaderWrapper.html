32  /***
33   * This class defines the wrapper class for resource generator in text resource reader.
34   * 
35   * @author Ibrahim Chaehoi
36   */
37  public class ResourceGeneratorReaderWrapper implements TextResourceReader {
38  
39  	/*** The resource generator wrapped */
40  	private ResourceGenerator generator;
41  	
42  	/*** The resource handler */
43  	private ResourceReaderHandler rsHandler;
44  	
45  	/*** The Jawr config */
46  	private JawrConfig config;
47  	
48  	/***
49  	 * Constructor
50  	 * @param generator the generator
51  	 */
52  	public ResourceGeneratorReaderWrapper(ResourceGenerator generator, ResourceReaderHandler rsHandler, JawrConfig config) {
53  		this.generator = generator;
54  		this.config = config;
55  		this.rsHandler = rsHandler;
56  	}
57  	
58  	/* (non-Javadoc)
59  	 * @see net.jawr.web.resource.handler.ResourceReader#getResource(java.lang.String)
60  	 */
61  	public Reader getResource(String resourceName) {
62  		
63  		return getResource(resourceName, false);
64  	}
65  
66  	/* (non-Javadoc)
67  	 * @see net.jawr.web.resource.handler.ResourceReader#getResource(java.lang.String, boolean)
68  	 */
69  	public Reader getResource(String resourceName, boolean processingBundle) {
70  		
71  		Locale locale = null;
72  		String path = resourceName;
73  		// Remove generator prefix
74  		path = path.substring((generator.getMappingPrefix()+GeneratorRegistry.PREFIX_SEPARATOR).length());
75  		
76  		Map contextVariants = new HashMap();
77  		Map variantSets = new HashMap();
78  		if(generator instanceof VariantResourceGenerator || generator instanceof LocaleAwareResourceGenerator){
79  			
80  			int variantSuffixIdx = path.indexOf("@");
81  			if(variantSuffixIdx != -1){
82  				String variantKey = null;
83  
84  				variantKey = FileNameUtils.getBaseName(path.substring(path.indexOf('@')+1));
85  					// Remove variant suffix
86  					String extension = FileNameUtils.getExtension(path);
87  					path = path.substring(0, variantSuffixIdx);
88  					if(StringUtils.isNotEmpty(extension)){
89  						path = path+"."+extension;
90  					}
91  				String[] variants = variantKey.split("@");
92  				if(generator instanceof VariantResourceGenerator){
93  					variantSets = ((VariantResourceGenerator)generator).getAvailableVariants(path);
94  				}else{ // instanceof LocaleAwareResourceGenerator
95  					List availableLocales = ((LocaleAwareResourceGenerator)generator).getAvailableLocales(path);
96  					if(availableLocales != null){
97  						variantSets.put(JawrConstant.LOCALE_VARIANT_TYPE, availableLocales);
98  					}
99  				}
100 				
101 				ArrayList variantTypes = new ArrayList(variantSets.keySet());
102 				Collections.sort(variantTypes);
103 				int nbVariants = variants.length;
104 				
105 				for(int i = 0; i < nbVariants; i++){
106 					String variantType = (String) variantTypes.get(i);
107 					String variantValue = variants[i];
108 					contextVariants.put(variantType, variantValue);
109 					if(variantType == JawrConstant.LOCALE_VARIANT_TYPE){
110 						// Resourcebundle should be doing this for me...
111 						String[] params = variantValue.split("_");			
112 						switch(params.length) {
113 							case 3:
114 								locale = new Locale(params[0],params[1],params[2]);
115 								break;
116 							case 2: 
117 								locale = new Locale(params[0],params[1]);
118 								break;
119 							default:
120 								locale = new Locale(variantValue);
121 						}
122 					}
123 				}
124 			}
125 		}
126 		
127 		GeneratorContext context = new GeneratorContext(config, path);
128 		context.setVariantMap(contextVariants);
129 		context.setVariantSets(variantSets);
130 		
131 		context.setLocale(locale);
132 		context.setResourceReaderHandler(rsHandler);
133 		context.setProcessingBundle(processingBundle);
134 		
135 		return generator.createResource(context);
136 	}
