38  /***
39   * A generator that creates a script from message bundles.
40   * The generated script can be used to reference the message literals easily from javascript.  
41   * 
42   * @author Jordi Hernández Sellés
43   * @author Ibrahim Chaehoi
44   *
45   */
46  public class ResourceBundleMessagesGenerator extends AbstractJavascriptGenerator implements ResourceGenerator, VariantResourceGenerator, ConfigurationAwareResourceGenerator {
47  	
48  	/*** The logger */
49  	private static final Logger LOGGER = Logger.getLogger(ResourceBundleMessagesGenerator.class);
50  	
51  	/*** The grails message generator */
52  	private static final String GRAILS_MESSAGE_CREATOR = "net.jawr.web.resource.bundle.locale.message.GrailsMessageBundleScriptCreator";
53  	
54  	/*** The resource path prefix for grails i18n messages */
55  	private static final String GRAILS_APP_I18N_RESOURCE_PREFIX = "grails-app.i18n.";
56  
57  	/*** The servlet context */
58  	private ServletContext servletContext;
59  	
60  	/*** The flag indicating if we are in a grails context */
61  	private boolean grailsContext;
62  	
63  	/*** The flag indicating if we are in a grails war is deployed */
64  	private boolean grailsWarDeployed;
65  	
66  	/* (non-Javadoc)
67  	 * @see net.jawr.web.resource.bundle.generator.ConfigurationAwareResourceGenerator#setConfig(net.jawr.web.config.JawrConfig)
68  	 */
69  	public void setConfig(JawrConfig config){
70  		servletContext = config.getContext();
71  		grailsContext = servletContext.getAttribute(JawrConstant.GRAILS_WAR_DEPLOYED) != null;
72  		if(grailsContext){
73  			grailsWarDeployed = ((Boolean)servletContext.getAttribute(JawrConstant.GRAILS_WAR_DEPLOYED)).booleanValue();
74  		}
75  	}
76  	
77  	/* (non-Javadoc)
78  	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#createResource(java.lang.String, java.nio.charset.Charset)
79  	 */
80  	public Reader createResource(GeneratorContext context) {
81  		MessageBundleScriptCreator creator = null;
82  		// In grails apps, the generator uses a special implementation
83  		if(grailsContext){
84  			if(LOGGER.isDebugEnabled())
85  				LOGGER.debug("Using grails messages generator. ");
86  			// Loading this way prevents unwanted dependencies in non grails applications. 
87  			Object[] param = {context};
88  			creator = (MessageBundleScriptCreator) ClassLoaderResourceUtils.buildObjectInstance(GRAILS_MESSAGE_CREATOR,param);
89  		}
90  		else {
91  			if(LOGGER.isDebugEnabled())
92  				LOGGER.debug("Using standard messages generator. ");
93  			creator = new MessageBundleScriptCreator(context);
94  		}
95  		
96  		return creator.createScript(context.getCharset());
97  	}
98  
99  	/* (non-Javadoc)
100 	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#getMappingPrefix()
101 	 */
102 	public String getMappingPrefix() {
103 		return GeneratorRegistry.MESSAGE_BUNDLE_PREFIX;
104 	}
105 	
106 	/* (non-Javadoc)
107 	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#getDebugModeBuildTimeGenerationPath(java.lang.String)
108 	 */
109 	public String getDebugModeBuildTimeGenerationPath(String path) {
110 		
111 		String debugPath = path.replaceFirst(GeneratorRegistry.PREFIX_SEPARATOR, JawrConstant.URL_SEPARATOR);
112 		if(debugPath.endsWith("@")){
113 			debugPath = debugPath.replaceAll("@", "");
114 		}else{
115 			debugPath = debugPath.replaceAll("@", "_");
116 			debugPath = debugPath.replaceAll("//|", "_");
117 		}
118 		return debugPath+"."+JawrConstant.JS_TYPE;
119 	}
120 
121 	/* (non-Javadoc)
122 	 * @see net.jawr.web.resource.handler.LocaleAwareResourceReader#getAvailableLocales(java.lang.String)
123 	 */
124 	public List getAvailableLocales(String resource) {
125 		
126 		List availableLocales = null;
127 		if(grailsContext && grailsWarDeployed && resource.startsWith(GRAILS_APP_I18N_RESOURCE_PREFIX)){
128 			availableLocales = LocaleUtils.getAvailableLocaleSuffixesForBundle(resource, servletContext); 
129 		}else{
130 			availableLocales = LocaleUtils.getAvailableLocaleSuffixesForBundle(resource);
131 		}
132 		return availableLocales;
133 	}
134 
135 	/* (non-Javadoc)
136 	 * @see net.jawr.web.resource.bundle.generator.variant.VariantResourceGenerator#getAvailableVariants(java.lang.String)
137 	 */
138 	public Map getAvailableVariants(String resource) {
139 		
140 		List localeVariants = getAvailableLocales(resource);
141 		if(localeVariants.isEmpty()){
142 			throw new BundlingProcessException("Enable to find the resource bundle : "+resource);
143 		}
144 		Map variants = new HashMap();
145 		VariantSet variantSet = new VariantSet(JawrConstant.LOCALE_VARIANT_TYPE, "", localeVariants);
146 		variants.put(JawrConstant.LOCALE_VARIANT_TYPE, variantSet);
147 		return variants;
148 	}
