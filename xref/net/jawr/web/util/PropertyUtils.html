25  /***
26   * Utility methods for using Java Reflection APIs to facilitate generic
27   * property getter and setter operations on Java objects. 
28   * This is a slightly modified version of PropertyUtilsBean.
29   * 
30   * @author Ibrahim Chaehoi
31   *
32   */
33  public class PropertyUtils {
34  
35  	/*** The logger */
36  	private static final Logger LOGGER = Logger.getLogger(PropertyUtils.class);

38  	/***
39       * Return the value of the specified simple property of the specified
40       * bean, with string conversions.
41       *
42       * @param bean Bean whose property is to be extracted
43       * @param name Name of the property to be extracted
44       * @return the value of the specified simple property of the specified
45       * bean, with string conversions.
46       * 
47       * @exception IllegalAccessException if the caller does not have
48       *  access to the property accessor method
49       * @exception IllegalArgumentException if <code>bean</code> or
50       *  <code>name</code> is null
51       * @exception IllegalArgumentException if the property name
52       *  is nested or indexed
53       * @exception InvocationTargetException if the property accessor method
54       *  throws an exception
55       * @exception NoSuchMethodException if an accessor method for this
56       *  propety cannot be found
57       */
58      public static String getProperty(Object bean, String name) throws NoSuchMethodException, IllegalAccessException, InvocationTargetException{
59  	
60  		String value = null;
61  		
62  		// Retrieve the property getter method for the specified property
63  		PropertyDescriptor descriptor = getPropertyDescriptor(bean, name);
64          
65          if (descriptor == null) {
66              throw new NoSuchMethodException("Unknown property '" +
67                      name + "'");
68          }
69          
70          Method readMethod = descriptor.getReadMethod();
71          if (readMethod == null) {
72              throw new NoSuchMethodException("Property '" + name +
73                      "' has no getter method");
74          }
75  
76          // Call the property getter and return the value
77          Object result = (Object) invokeMethod(readMethod, bean, new Object[0]);
78          if(result != null){
79          	value = result.toString();
80          }
81          
82  		return value;
83  	}

85      /***
86       * <p>Retrieve the property descriptor for the specified property of the
87       * specified bean, or return <code>null</code> if there is no such
88       * descriptor.  This method resolves indexed and nested property
89       * references in the same manner as other methods in this class, except
90       * that if the last (or only) name element is indexed, the descriptor
91       * for the last resolved property itself is returned.</p>
92       *
93       * @param bean Bean for which a property descriptor is requested
94       * @param name Possibly indexed and/or nested name of the property for
95       *  which a property descriptor is requested
96       *
97       * @exception IllegalAccessException if the caller does not have
98       *  access to the property accessor method
99       * @exception IllegalArgumentException if <code>bean</code> or
100      *  <code>name</code> is null
101      * @exception IllegalArgumentException if a nested reference to a
102      *  property returns null
103      * @exception InvocationTargetException if the property accessor method
104      *  throws an exception
105      * @exception NoSuchMethodException if an accessor method for this
106      *  propety cannot be found
107      */
108     private static PropertyDescriptor getPropertyDescriptor(Object bean,
109 			String name) {
110 		PropertyDescriptor descriptor = null;
111 		PropertyDescriptor descriptors[] = getPropertyDescriptors(bean.getClass());
112         if (descriptors != null) {
113             
114             for (int i = 0; i < descriptors.length; i++) {
115                 if (name.equals(descriptors[i].getName()))
116                 	descriptor = descriptors[i];
117             }
118         }
119 		return descriptor;
120 	}

122 	/***
123      * <p>Retrieve the property descriptors for the specified class,
124      * introspecting and caching them the first time a particular bean class
125      * is encountered.</p>
126      *
127      * @param beanClass Bean class for which property descriptors are requested
128      *
129      * @exception IllegalArgumentException if <code>beanClass</code> is null
130      */
131     public static PropertyDescriptor[]
132             getPropertyDescriptors(Class beanClass) {
133 
134         if (beanClass == null) {
135             throw new IllegalArgumentException("No bean class specified");
136         }
137 
138         // Look up any cached descriptors for this bean class
139         PropertyDescriptor descriptors[] = null;
140         
141         // Introspect the bean and cache the generated descriptors
142         BeanInfo beanInfo = null;
143         try {
144             beanInfo = Introspector.getBeanInfo(beanClass);
145         } catch (IntrospectionException e) {
146             return (new PropertyDescriptor[0]);
147         }
148         descriptors = beanInfo.getPropertyDescriptors();
149         if (descriptors == null) {
150             descriptors = new PropertyDescriptor[0];
151         }
152         
153         return (descriptors);
154 
155     }

157     /***
158      * Set the value of the specified simple property of the specified bean,
159      * with no type conversions.
160      *
161      * @param bean Bean whose property is to be modified
162      * @param name Name of the property to be modified
163      * @param value Value to which the property should be set
164      *
165      * @exception IllegalAccessException if the caller does not have
166      *  access to the property accessor method
167      * @exception IllegalArgumentException if <code>bean</code> or
168      *  <code>name</code> is null
169      * @exception IllegalArgumentException if the property name is
170      *  nested or indexed
171      * @exception InvocationTargetException if the property accessor method
172      *  throws an exception
173      * @exception NoSuchMethodException if an accessor method for this
174      *  propety cannot be found
175      */
176     public static void setProperty(Object bean,
177                                          String name, Object value)
178             throws IllegalAccessException, InvocationTargetException,
179             NoSuchMethodException {
180 
181         if (bean == null) {
182             throw new IllegalArgumentException("No bean specified");
183         }
184         if (name == null) {
185             throw new IllegalArgumentException("No name specified");
186         }
187 
188         // Retrieve the property setter method for the specified property
189         PropertyDescriptor descriptor =
190                 getPropertyDescriptor(bean, name);
191         if (descriptor == null) {
192             throw new NoSuchMethodException("Unknown property '" +
193                     name + "'");
194         }
195         Method writeMethod = descriptor.getWriteMethod();
196         if (writeMethod == null) {
197             throw new NoSuchMethodException("Property '" + name +
198                     "' has no setter method");
199         }
200 
201         // Call the property setter method
202         Object values[] = new Object[1];
203         values[0] = value;
204         
205         invokeMethod(writeMethod, bean, values);
206 
207     }

209     /***
210      * This utility method just catches and wraps IllegalArgumentException.
211      * @param method the method to call
212      * @param bean the bean
213      * @param values the values
214      * @return the returned value of the method
215      * @throws IllegalAccessException if an exception occurs
216      * @throws InvocationTargetException if an exception occurs
217      */
218     private static Object invokeMethod(
219                         Method method, 
220                         Object bean, 
221                         Object[] values) 
222                             throws
223                                 IllegalAccessException,
224                                 InvocationTargetException {
225         try {
226             
227             return method.invoke(bean, values);
228         
229         } catch (IllegalArgumentException e) {
230             
231             LOGGER.error("Method invocation failed.", e);
232             throw new IllegalArgumentException(
233                 "Cannot invoke " + method.getDeclaringClass().getName() + "." 
234                 + method.getName() + " - " + e.getMessage());
235             
236         }
237     }
