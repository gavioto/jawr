144     /***
145      * Copies a file to a new location.
146      * <p>
147      * This method copies the contents of the specified source file
148      * to the specified destination file.
149      * The directory holding the destination file is created if it does not exist.
150      * If the destination file exists, then this method will overwrite it.
151      *
152      * @param srcFile  an existing file to copy, must not be <code>null</code>
153      * @param destFile  the new file, must not be <code>null</code>
154      * @param preserveFileDate  true if the file date of the copy
155      *  should be the same as the original
156      *
157      * @throws NullPointerException if source or destination is <code>null</code>
158      * @throws IOException if source or destination is invalid
159      * @throws IOException if an IO error occurs during copying
160      * @see #copyFileToDirectory(File, File, boolean)
161      */
162     public static void copyFile(File srcFile, File destFile,
163             boolean preserveFileDate) throws IOException {
164         if (srcFile == null) {
165             throw new IllegalArgumentException("Source must not be null");
166         }
167         if (destFile == null) {
168             throw new IllegalArgumentException("Destination must not be null");
169         }
170         if (srcFile.exists() == false) {
171             throw new FileNotFoundException("Source '" + srcFile + "' does not exist");
172         }
173         if (srcFile.isDirectory()) {
174             throw new IOException("Source '" + srcFile + "' exists but is a directory");
175         }
176         if (srcFile.getCanonicalPath().equals(destFile.getCanonicalPath())) {
177             throw new IOException("Source '" + srcFile + "' and destination '" + destFile + "' are the same");
178         }
179         if (destFile.getParentFile() != null && destFile.getParentFile().exists() == false) {
180             if (destFile.getParentFile().mkdirs() == false) {
181                 throw new IOException("Destination '" + destFile + "' directory cannot be created");
182             }
183         }
184         if (destFile.exists() && destFile.canWrite() == false) {
185             throw new IOException("Destination '" + destFile + "' exists but is read-only");
186         }
187         doCopyFile(srcFile, destFile, preserveFileDate);
188     }

190     /***
191      * Internal copy file method.
192      * 
193      * @param srcFile  the validated source file, must not be <code>null</code>
194      * @param destFile  the validated destination file, must not be <code>null</code>
195      * @param preserveFileDate  whether to preserve the file date
196      * @throws IOException if an error occurs
197      */
198     private static void doCopyFile(File srcFile, File destFile, boolean preserveFileDate) throws IOException {
199         if (destFile.exists() && destFile.isDirectory()) {
200             throw new IOException("Destination '" + destFile + "' exists but is a directory");
201         }
202 
203         FileInputStream input = new FileInputStream(srcFile);
204         FileOutputStream output = new FileOutputStream(destFile);
205         IOUtils.copy(input, output, true);
206         
207         if (srcFile.length() != destFile.length()) {
208             throw new IOException("Failed to copy full contents from '" +
209                     srcFile + "' to '" + destFile + "'");
210         }
211         if (preserveFileDate) {
212             destFile.setLastModified(srcFile.lastModified());
213         }
214     }

308     /***
309      * Copies a filtered directory to a new location.
310      * <p>
311      * This method copies the contents of the specified source directory
312      * to within the specified destination directory.
313      * <p>
314      * The destination directory is created if it does not exist.
315      * If the destination directory did exist, then this method merges
316      * the source with the destination, with the source taking precedence.
317      *
318      * <h4>Example: Copy directories only</h4> 
319      *  <pre>
320      *  // only copy the directory structure
321      *  FileUtils.copyDirectory(srcDir, destDir, DirectoryFileFilter.DIRECTORY, false);
322      *  </pre>
323      *
324      * <h4>Example: Copy directories and txt files</h4>
325      *  <pre>
326      *  // Create a filter for ".txt" files
327      *  IOFileFilter txtSuffixFilter = FileFilterUtils.suffixFileFilter(".txt");
328      *  IOFileFilter txtFiles = FileFilterUtils.andFileFilter(FileFileFilter.FILE, txtSuffixFilter);
329      *
330      *  // Create a filter for either directories or ".txt" files
331      *  FileFilter filter = FileFilterUtils.orFileFilter(DirectoryFileFilter.DIRECTORY, txtFiles);
332      *
333      *  // Copy using the filter
334      *  FileUtils.copyDirectory(srcDir, destDir, filter, false);
335      *  </pre>
336      * 
337      * @param srcDir  an existing directory to copy, must not be <code>null</code>
338      * @param destDir  the new directory, must not be <code>null</code>
339      * @param filter  the filter to apply, null means copy all directories and files
340      * @param preserveFileDate  true if the file date of the copy
341      *  should be the same as the original
342      *
343      * @throws NullPointerException if source or destination is <code>null</code>
344      * @throws IOException if source or destination is invalid
345      * @throws IOException if an IO error occurs during copying
346      * @since Commons IO 1.4
347      */
348     public static void copyDirectory(File srcDir, File destDir,
349             FileFilter filter, boolean preserveFileDate) throws IOException {
350         if (srcDir == null) {
351             throw new IllegalArgumentException("Source must not be null");
352         }
353         if (destDir == null) {
354             throw new IllegalArgumentException("Destination must not be null");
355         }
356         if (srcDir.exists() == false) {
357             throw new FileNotFoundException("Source '" + srcDir + "' does not exist");
358         }
359         if (srcDir.isDirectory() == false) {
360             throw new IOException("Source '" + srcDir + "' exists but is not a directory");
361         }
362         if (srcDir.getCanonicalPath().equals(destDir.getCanonicalPath())) {
363             throw new IOException("Source '" + srcDir + "' and destination '" + destDir + "' are the same");
364         }
365 
366         // Cater for destination being directory within the source directory (see IO-141)
367         List exclusionList = null;
368         if (destDir.getCanonicalPath().startsWith(srcDir.getCanonicalPath())) {
369             File[] srcFiles = filter == null ? srcDir.listFiles() : srcDir.listFiles(filter);
370             if (srcFiles != null && srcFiles.length > 0) {
371                 exclusionList = new ArrayList(srcFiles.length);
372                 for (int i = 0; i < srcFiles.length; i++) {
373                     File copiedFile = new File(destDir, srcFiles[i].getName());
374                     exclusionList.add(copiedFile.getCanonicalPath());
375                 }
376             }
377         }
378         doCopyDirectory(srcDir, destDir, filter, preserveFileDate, exclusionList);
379     }

381     /***
382      * Internal copy directory method.
383      * 
384      * @param srcDir  the validated source directory, must not be <code>null</code>
385      * @param destDir  the validated destination directory, must not be <code>null</code>
386      * @param filter  the filter to apply, null means copy all directories and files
387      * @param preserveFileDate  whether to preserve the file date
388      * @param exclusionList  List of files and directories to exclude from the copy, may be null
389      * @throws IOException if an error occurs
390      * @since Commons IO 1.1
391      */
392     private static void doCopyDirectory(File srcDir, File destDir, FileFilter filter,
393             boolean preserveFileDate, List exclusionList) throws IOException {
394         if (destDir.exists()) {
395             if (destDir.isDirectory() == false) {
396                 throw new IOException("Destination '" + destDir + "' exists but is not a directory");
397             }
398         } else {
399             if (destDir.mkdirs() == false) {
400                 throw new IOException("Destination '" + destDir + "' directory cannot be created");
401             }
402             if (preserveFileDate) {
403                 destDir.setLastModified(srcDir.lastModified());
404             }
405         }
406         if (destDir.canWrite() == false) {
407             throw new IOException("Destination '" + destDir + "' cannot be written to");
408         }
409         // recurse
410         File[] files = filter == null ? srcDir.listFiles() : srcDir.listFiles(filter);
411         if (files == null) {  // null if security restricted
412             throw new IOException("Failed to list contents of " + srcDir);
413         }
414         for (int i = 0; i < files.length; i++) {
415             File copiedFile = new File(destDir, files[i].getName());
416             if (exclusionList == null || !exclusionList.contains(files[i].getCanonicalPath())) {
417                 if (files[i].isDirectory()) {
418                     doCopyDirectory(files[i], copiedFile, filter, preserveFileDate, exclusionList);
419                 } else {
420                     doCopyFile(files[i], copiedFile, preserveFileDate);
421                 }
422             }
423         }
424     }

427      * Returns the resource names contained in a directory, and
428      * for directory resource, a trailing '/' is added
429      * @param dir the directory
430      * @return the resource names
431      */
432     public static Set getResourceNames(File dir){
433     	     
434     	Set resourceNames = new HashSet();
435 		
436     	// If the path is not valid throw an exception
437 		String[] resArray = dir.list();
438 		String absolutePath = dir.getAbsolutePath();
439 		
440 		if(resArray != null){
441 			// Make the returned dirs end with '/', to match a servletcontext behavior. 
442 			for (int i = 0; i < resArray.length; i++) {
443 				if(new File(dir, resArray[i]).isDirectory())
444 					resArray[i] += '/';
445 			}
446 			resourceNames.addAll(Arrays.asList(resArray));
447 		}
448 		return resourceNames;
449     }
