141 	/***
142 	 * Initialize the image mapping of the image resources handler
143 	 * 
144 	 * @param imgRsHandler the image resources handler
145 	 */
146 	private void initImageMapping(ImageResourcesHandler imgRsHandler) {
147 
148 		if (jawrConfig.getUseBundleMapping() && rsHandler.isExistingMappingFile()) {
149 
150 			// Initialize the image mapping
151 			Iterator mapIterator = bundleMapping.keySet().iterator();
152 			while (mapIterator.hasNext()) {
153 				String key = (String) mapIterator.next();
154 				imgRsHandler.addMapping(key, bundleMapping.getProperty(key));
155 			}
156 			
157 		} else {
158 			// Create a resource handler to read files from the WAR archive or exploded dir.
159 			String imageResourcesDefinition = jawrConfig.getImageResourcesDefinition();
160 			if (imageResourcesDefinition != null) {
161 
162 				StringTokenizer tokenizer = new StringTokenizer(imageResourcesDefinition, ",");
163 				while (tokenizer.hasMoreTokens()) {
164 					String pathMapping = tokenizer.nextToken();
165 
166 					// path in the classpath and ends with an image extension
167 					if (pathMapping.startsWith("jar:") && hasImageFileExtension(pathMapping)) {
168 
169 						addImagePath(imgRsHandler, pathMapping);
170 					}
171 					// path ends in /, the folder is included without subfolders
172 					else if (pathMapping.endsWith("/")) {
173 						addItemsFromDir(imgRsHandler, pathMapping, false);
174 					}
175 					// path ends in /, the folder is included with all subfolders
176 					else if (pathMapping.endsWith("/**")) {
177 						addItemsFromDir(imgRsHandler, pathMapping.substring(0, pathMapping.lastIndexOf("**")), true);
178 					} else if (hasImageFileExtension(pathMapping)) {
179 						addImagePath(imgRsHandler, pathMapping);
180 					} else
181 						log.warn("Wrong mapping [" + pathMapping + "] for image bundle. Please check configuration. ");
182 				}
183 			}
184 		}
185 
186 		// Store the bundle mapping
187 		if (jawrConfig.getUseBundleMapping() && !rsHandler.isExistingMappingFile()) {
188 			rsHandler.storeJawrBundleMapping(bundleMapping);
189 		}
190 		
191 		if (log.isDebugEnabled())
192 			log.debug("Finish creation of map for image bundle");
193 	}

196 	 * Add an image path to the image map
197 	 * 
198 	 * @param imgRsHandler the image resources handler
199 	 * @param imgPath the image path
200 	 * @param classPathImg the flag indicating if the image should be retrieved from classpath
201 	 */
202 	private void addImagePath(ImageResourcesHandler imgRsHandler, String imgPath) {
203 
204 		try {
205 			String resultPath = CheckSumUtils.getCacheBustedUrl(imgPath, rsHandler, jawrConfig);
206 			imgRsHandler.addMapping(imgPath, resultPath);
207 			bundleMapping.put(imgPath, resultPath);
208 		} catch (IOException e) {
209 			log.error("An exception occurs while defining the mapping for the file : " + imgPath, e);
210 		}
211 	}

213 	/***
214 	 * Returns true of the path contains an image file extension
215 	 * 
216 	 * @param path the path
217 	 * @return the image file extension
218 	 */
219 	private boolean hasImageFileExtension(String path) {
220 		boolean result = false;
221 
222 		int extFileIdx = path.lastIndexOf(".");
223 		if (extFileIdx != -1 && extFileIdx + 1 < path.length()) {
224 			String extension = path.substring(extFileIdx + 1);
225 			result = imgMimeMap.containsKey(extension);
226 		}
227 
228 		return result;
229 	}

231 	/***
232 	 * Adds all the resources within a path to the image map.
233 	 * 
234 	 * @param imgRsHandler the image resources handler
235 	 * @param dirName the directory name
236 	 * @param addSubDirs boolean If subfolders will be included. In such case, every folder below the path is included.
237 	 */
238 	private void addItemsFromDir(ImageResourcesHandler imgRsHandler, String dirName, boolean addSubDirs) {
239 		Set resources = rsHandler.getResourceNames(dirName);
240 
241 		if (log.isDebugEnabled()) {
242 			log.debug("Adding " + resources.size() + " resources from path [" + dirName + "] to image bundle");
243 		}
244 
245 		// Add remaining resources (remaining after sorting, or all if no sort file present)
246 		List folders = new ArrayList();
247 		for (Iterator it = resources.iterator(); it.hasNext();) {
248 			String resourceName = (String) it.next();
249 			String resourcePath = PathNormalizer.joinPaths(dirName, resourceName);
250 			if (hasImageFileExtension(resourceName)) {
251 				addImagePath(imgRsHandler, resourcePath);
252 
253 				if (log.isDebugEnabled())
254 					log.debug("Added to item path list:" + PathNormalizer.asPath(resourcePath));
255 			} else if (addSubDirs) {
256 
257 				try {
258 					if (rsHandler.isDirectory(resourcePath)) {
259 						folders.add(resourceName);
260 					}
261 				} catch (InvalidPathException e) {
262 					if (log.isDebugEnabled())
263 						log.debug("Enable to define if the following resource is a directory : " + PathNormalizer.asPath(resourcePath));
264 				}
265 			}
266 		}
267 
268 		// Add subfolders if requested. Subfolders are added last unless specified in sorting file.
269 		if (addSubDirs) {
270 			for (Iterator it = folders.iterator(); it.hasNext();) {
271 				String folderName = (String) it.next();
272 				addItemsFromDir(imgRsHandler, PathNormalizer.joinPaths(dirName, folderName), true);
273 			}
274 		}
275 	}

277 	/***
278 	 * Handles a resource request.
279 	 * <ul>
280 	 * <li>If the request contains an If-Modified-Since header, the 304 status is set and no data is written to the response</li>
281 	 * <li>If the requested path begins with the gzip prefix, a gzipped version of the resource is served, with the corresponding content-encoding
282 	 * header.</li>
283 	 * <li>Otherwise, the resource is written as text to the response.</li>
284 	 * <li>If the resource is not found, the response satus is set to 404 and no response is written.</li>
285 	 * </ul>
286 	 * 
287 	 * @param requestedPath
288 	 * @param request
289 	 * @param response
290 	 * @throws ServletException
291 	 * @throws IOException
292 	 */
293 	public void processRequest(String requestedPath, HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
294 
295 		if (log.isDebugEnabled())
296 			log.debug("Request received for path:" + requestedPath);
297 
298 		// Retrieve the file path
299 		String filePath = getFilePath(request);
300 		if (filePath == null) {
301 			response.sendError(HttpServletResponse.SC_NOT_FOUND);
302 			return;
303 		}
304 
305 		// Retrieve the content type
306 		String imgContentType = getContentType(request, filePath);
307 		if (imgContentType == null) {
308 			response.sendError(HttpServletResponse.SC_NOT_FOUND);
309 			return;
310 		}
311 
312 		// Set the content type
313 		response.setContentType(imgContentType);
314 		// If debug mode is off, check for If-Modified-Since and If-none-match headers and set response caching headers.
315 		if (!this.jawrConfig.isDebugModeOn()) {
316 			// If a browser checks for changes, always respond 'no changes'.
317 			if (null != request.getHeader(IF_MODIFIED_SINCE_HEADER) || null != request.getHeader(IF_NONE_MATCH_HEADER)) {
318 				response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
319 				if (log.isDebugEnabled())
320 					log.debug("Returning 'not modified' header. ");
321 				return;
322 			}
323 
324 			// Add caching headers
325 			setResponseHeaders(response);
326 		}
327 
328 		boolean isClasspathImg = isClasspathImage(filePath);
329 
330 		// Returns the real file path
331 		filePath = getRealFilePath(filePath);
332 
333 		// Read the file from the classpath and send it in the outputStream
334 		try {
335 			writeContent(response, filePath, isClasspathImg);
336 
337 			// Set the content length, and the content type based on the file extension
338 			response.setContentType(contentType);
339 
340 		} catch (Exception ex) {
341 
342 			log.error("Unable to load the image for the request URI : " + request.getRequestURI(), ex);
343 			response.sendError(HttpServletResponse.SC_NOT_FOUND);
344 		}
345 
346 		if (log.isDebugEnabled())
347 			log.debug("request succesfully attended");
348 	}

351 	 * Returns the content type for the image
352 	 * 
353 	 * @param request the request
354 	 * @param filePath the image file path
355 	 * @return the content type of the image
356 	 */
357 	private String getContentType(HttpServletRequest request, String filePath) {
358 		String requestUri = request.getRequestURI();
359 
360 		// Retrieve the extension
361 		String extension = getExtension(filePath);
362 		if (extension == null) {
363 
364 			log.error("No extension found for the request URI : " + requestUri);
365 			return null;
366 		}
367 
368 		String contentType = (String) imgMimeMap.get(extension);
369 		if (contentType == null) {
370 
371 			log.error("No image extension match the extension '" + extension + "' for the request URI : " + requestUri);
372 			return null;
373 		}
374 		return contentType;
375 	}

392 	 * Write the image content to the response
393 	 * 
394 	 * @param response the response
395 	 * @param fileName the filename
396 	 * @throws IOException if an IO exception occurs.
397 	 */
398 	private void writeContent(HttpServletResponse response, String fileName, boolean fromClasspath) throws IOException {
399 
400 		int length = 0;
401 
402 		OutputStream os = response.getOutputStream();
403 		InputStream is = null;
404 
405 		if (fromClasspath) {
406 			if (fileName.startsWith("/")) {
407 				fileName = fileName.substring(1);
408 			}
409 			is = ClassLoaderResourceUtils.getResourceAsStream(fileName, this);
410 		} else {
411 			try {
412 				if (!fileName.startsWith("/")) {
413 					fileName = "/" + fileName;
414 				}
415 				is = rsHandler.getResourceAsStream(fileName);
416 			} catch (ResourceNotFoundException e) {
417 				// Nothing to do
418 			}
419 		}
420 
421 		if (is == null) {
422 			response.setStatus(HttpServletResponse.SC_NOT_FOUND);
423 			if (log.isInfoEnabled())
424 				log.info("Received a request for a non existing image resource: " + fileName);
425 			return;
426 		}
427 
428 		IOUtils.copy(is, os, true);
429 
430 		response.setContentLength(length);
431 	}

433 	/***
434 	 * Removes the cache buster
435 	 * 
436 	 * @param fileName the file name
437 	 * @return the file name without the cache buster.
438 	 */
439 	private String getRealFilePath(String fileName) {
440 
441 		if (fileName.startsWith("/")) {
442 			fileName = fileName.substring(1);
443 		}
444 		Matcher matcher = cacheBusterPattern.matcher(fileName);
445 		StringBuffer result = new StringBuffer();
446 		if (matcher.find()) {
447 			matcher.appendReplacement(result, CACHE_BUSTER_REPLACE_PATTERN);
448 			return result.toString();
449 		}
450 
451 		return fileName;
452 	}
