39  	/*** The logger */
40  	private static final Logger log = Logger.getLogger(RendererRequestUtils.class.getName());

42  	/*** The bundle renderer context attribute name */
43  	private static final String BUNDLE_RENDERER_CONTEXT_ATTR_PREFIX = "net.jawr.web.resource.renderer.BUNDLE_RENDERER_CONTEXT";
44  	
45  	/***
46  	 * Returns the bundle renderer context.
47  	 * 
48  	 * @param request the request
49  	 * @param resourceType the resource type
50  	 * @return the bundle renderer context.
51  	 */
52  	public static BundleRendererContext getBundleRendererContext(HttpServletRequest request, BundleRenderer renderer) {
53  		String bundleRendererCtxAttributeName = BUNDLE_RENDERER_CONTEXT_ATTR_PREFIX+renderer.getResourceType();
54  		
55  		BundleRendererContext ctx = (BundleRendererContext) request.getAttribute(bundleRendererCtxAttributeName);
56  		if(ctx == null){
57  			String localeKey = renderer.getBundler().getConfig().getLocaleResolver().resolveLocaleCode(request);
58  	         boolean isGzippable = isRequestGzippable(request,renderer.getBundler().getConfig());
59  	         ctx = new BundleRendererContext(request.getContextPath(), localeKey, isGzippable,
60  	                 isSslRequest(request));
61  	         request.setAttribute(bundleRendererCtxAttributeName, ctx);
62  		}
63  		
64  		return ctx;
65  		
66  	}

69  	 * Sets the bundle renderer context.

75  	public static void setBundleRendererContext(ServletRequest request, String resourceType, BundleRendererContext ctx) {
76  		String globalBundleAddedAttributeName = BUNDLE_RENDERER_CONTEXT_ATTR_PREFIX+resourceType;

80  	/***
81  	 * Determines wether gzip is suitable for the current request given the current config.
82  	 * 
83  	 * @param req
84  	 * @param jawrConfig
85  	 * @return
86  	 */
87  	public static boolean isRequestGzippable(HttpServletRequest req, JawrConfig jeesConfig) {
88  		boolean rets;
89  		// If gzip is completely off, return false.
90  		if (!jeesConfig.isGzipResourcesModeOn())
91  			rets = false;
92  		else if (req.getHeader("Accept-Encoding") != null && req.getHeader("Accept-Encoding").indexOf("gzip") != -1) {
93  
94  			// If gzip for IE6 or less is off, the user agent is checked to avoid compression.
95  			if (!jeesConfig.isGzipResourcesForIESixOn()) {
96  				String agent = req.getHeader("User-Agent");
97  				if (log.isDebugEnabled())
98  					log.debug("User-Agent for this request:" + agent);
99  
100 				if (null != agent && agent.indexOf("MSIE") != -1) {
101 					rets = agent.indexOf("MSIE 4") == -1 && agent.indexOf("MSIE 5") == -1 && agent.indexOf("MSIE 6") == -1;
102 					if (log.isDebugEnabled())
103 						log.debug("Gzip enablement for IE executed, with result:" + rets);
104 				} else
105 					rets = true;
106 			} else
107 				rets = true;
108 		} else
109 			rets = false;
110 		return rets;
111 	}

113 	/***
114 	 * Determines wether to override the debug settings. Sets the debugOverride status on ThreadLocalJawrContext
115 	 * 
116 	 * @param req the request
117 	 * @param jawrConfig the jawr config
118 	 * 
119 	 */
120 	public static void setRequestDebuggable(HttpServletRequest req, JawrConfig jawrConfig) {
121 
122 		// make sure we have set an overrideKey
123 		// make sure the overrideKey exists in the request
124 		// lastly, make sure the keys match
125 		if (jawrConfig.getDebugOverrideKey().length() > 0 && null != req.getParameter(JawrConstant.OVERRIDE_KEY_PARAMETER_NAME)
126 				&& jawrConfig.getDebugOverrideKey().equals(req.getParameter(JawrConstant.OVERRIDE_KEY_PARAMETER_NAME))) {
127 			ThreadLocalJawrContext.setDebugOverriden(true);
128 		} else {
129 			ThreadLocalJawrContext.setDebugOverriden(false);
130 		}
131 
132 		// Inherit the debuggable property of the session if the session is a debuggable one
133 		inheritSessionDebugProperty(req);
134 
135 	}
136 
137 	/***
138 	 * Sets a request debuggable if the session is a debuggable session.
139 	 * 
140 	 * @param req the request
141 	 */
142 	public static void inheritSessionDebugProperty(HttpServletRequest request) {
143 
144 		HttpSession session = request.getSession();
145 		if (session != null) {
146 			String sessionId = session.getId();
147 
148 			JawrApplicationConfigManager appConfigMgr = (JawrApplicationConfigManager) session.getServletContext().getAttribute(
149 					JawrConstant.JAWR_APPLICATION_CONFIG_MANAGER);
150 
151 			// If the session ID is a debuggable session ID, activate debug mode for the request.
152 			if (appConfigMgr != null && appConfigMgr.isDebugSessionId(sessionId)) {
153 				ThreadLocalJawrContext.setDebugOverriden(true);
154 			}
155 		}
156 	}

158 	/***
159 	 * Returns true if the request URL is a SSL request (https://) 
160 	 * @param request the request
161 	 * @return true if the request URL is a SSL request
162 	 */
163 	public static boolean isSslRequest(HttpServletRequest request) {
164 		
165 		String requestUrl = request.getRequestURL().toString();
166 		return requestUrl.toLowerCase().startsWith(JawrConstant.HTTPS_URL_PREFIX);
167 	}
