112 	/***
113 	 * Reads the properties file and initializes all configuration using the ServletConfig object. If applicable, a ConfigChangeListenerThread will be
114 	 * started to listen to changes in the properties configuration.
115 	 * 
116 	 * @param servletContext ServletContext
117 	 * @param servletConfig ServletConfig
118 	 * @throws ServletException
119 	 */
120 	public JawrRequestHandler(ServletContext context, ServletConfig config) throws ServletException {
121 		this.imgMimeMap = MIMETypesSupport.getSupportedProperties(this);
122 		this.initParameters = new HashMap();
123 		Enumeration params = config.getInitParameterNames();
124 		while (params.hasMoreElements()) {
125 			String param = (String) params.nextElement();
126 			initParameters.put(param, config.getInitParameter(param));
127 		}
128 		initParameters.put("handlerName", config.getServletName());
129 
130 		if (log.isInfoEnabled())
131 			log.info("Initializing jawr config for servlet named " + config.getServletName());
132 		long initialTime = System.currentTimeMillis();
133 		this.servletContext = context;
134 
135 		resourceType = config.getInitParameter("type");
136 		resourceType = null == resourceType ? "js" : resourceType;
137 
138 		String configLocation = config.getInitParameter("configLocation");
139 		String configPropsSourceClass = config.getInitParameter("configPropertiesSourceClass");
140 		if (null == configLocation && null == configPropsSourceClass)
141 			throw new ServletException("Neither configLocation nor configPropertiesSourceClass init params were set."
142 					+ " You must set at least the configLocation param. Please check your web.xml file");
143 
144 		// Initialize the config properties source that will provide with all configuration options.
145 		ConfigPropertiesSource propsSrc;
146 
147 		// Load a custom class to set config properties
148 		if (null != configPropsSourceClass) {
149 			propsSrc = (ConfigPropertiesSource) ClassLoaderResourceUtils.buildObjectInstance(configPropsSourceClass);
150 			if (propsSrc instanceof ServletContextAware) {
151 				((ServletContextAware) propsSrc).setServletContext(context);
152 			}
153 		} else {
154 			// Default config properties source, reads from a .properties file in the classpath.
155 			propsSrc = new PropsFilePropertiesSource();
156 		}
157 
158 		// If a custom properties source is a subclass of PropsFilePropertiesSource, we hand it the configLocation param.
159 		// This affects the standard one as well.
160 		if (propsSrc instanceof PropsFilePropertiesSource)
161 			((PropsFilePropertiesSource) propsSrc).setConfigLocation(configLocation);
162 
163 		// Read properties from properties source
164 		Properties props = propsSrc.getConfigProperties();
165 
166 		// init registry
167 		generatorRegistry = new GeneratorRegistry(resourceType);
168 
169 		// hang onto the propertiesSource for manual reloads
170 		this.propertiesSource = propsSrc;
171 
172 		// Initialize config
173 		initializeJawrConfig(props);
174 
175 		// Initialize the properties reloading checker daemon if specified
176 		if(!ThreadLocalJawrContext.isBundleProcessingAtBuildTime() && null != props.getProperty(CONFIG_RELOAD_INTERVAL)) {
177 			int interval = Integer.valueOf(props.getProperty(CONFIG_RELOAD_INTERVAL)).intValue();
178 			log.warn("Jawr started with configuration auto reloading on. "
179 					+ "Be aware that a daemon thread will be checking for changes to configuration every " + interval + " seconds.");
180 
181 			this.configChangeListenerThread = new ConfigChangeListenerThread(propsSrc, this, interval);
182 			configChangeListenerThread.start();
183 		}
184 
185 		// initialize the jmx Bean
186 		if(isJmxEnabled()){
187 				JmxUtils.initJMXBean(this, servletContext, resourceType, jawrConfig.getConfigProperties());
188 		}
189 		
190 		if (log.isInfoEnabled()) {
191 			long totaltime = System.currentTimeMillis() - initialTime;
192 			log.info("Init method succesful. jawr started in " + (totaltime / 1000) + " seconds....");
193 		}
194 
195 	}

216 		this.initParameters = initParams;
217 
218 		if (log.isInfoEnabled())
219 			log.info("Initializing jawr config for request handler named " + (String) initParams.get("handlerName"));
220 
221 		long initialTime = System.currentTimeMillis();
222 		this.servletContext = context;
223 
224 		resourceType = (String) initParameters.get("type");
225 		resourceType = null == resourceType ? "js" : resourceType;
226 
227 		// init registry
228 		generatorRegistry = new GeneratorRegistry(resourceType);
229 
230 		// Initialize config
231 		initializeJawrConfig(configProps);
232 
233 		if (log.isInfoEnabled()) {
234 			long totaltime = System.currentTimeMillis() - initialTime;
235 			log.info("Init method succesful. jawr started in " + (totaltime / 1000) + " seconds....");
236 		}

239 	/***
240 	 * Initialize the Jawr config
241 	 * 
242 	 * @param props the properties
243 	 * @throws ServletException if an exception occurs
244 	 */
245 	protected void initializeJawrConfig(Properties props) throws ServletException {
246 		// Initialize config
247 		if (null != jawrConfig)
248 			jawrConfig.invalidate();
249 
250 		createJawrConfig(props);
251 		
252 		jawrConfig.setContext(servletContext);
253 		jawrConfig.setGeneratorRegistry(generatorRegistry);
254 
255 		// Set the content type to be used for every request.
256 		contentType = "text/";
257 		contentType += "js".equals(resourceType) ? "javascript" : "css";
258 		contentType += "; charset=" + jawrConfig.getResourceCharset().name();
259 
260 		// Set mapping, to be used by the tag lib to define URLs that point to this servlet.
261 		String mapping = (String) initParameters.get("mapping");
262 		if (null != mapping)
263 			jawrConfig.setServletMapping(mapping);
264 
265 		if (jawrConfig.isUsingClasspathCssImageServlet() && resourceType.equals("css")) {
266 			ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) servletContext.getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
267 			if (imgRsHandler == null) {
268 				log.error("You are using the CSS classpath image feature, but the JAWR Image servlet is yet initialized.\n"
269 						+ "The JAWR Image servlet must be initialized before the JAWR CSS servlet.\n"
270 						+ "Please check you web application configuration.");
271 			}
272 		}
273 
274 		if (log.isDebugEnabled()) {
275 			log.debug("Configuration read. Current config:");
276 			log.debug(jawrConfig);
277 		}
278 
279 		// Create a resource handler to read files from the WAR archive or exploded dir.
280 		ResourceHandler rsHandler = initResourceHandler();
281 		PropertiesBasedBundlesHandlerFactory factory = new PropertiesBasedBundlesHandlerFactory(props, resourceType, rsHandler, jawrConfig
282 				.getGeneratorRegistry());
283 		try {
284 			bundlesHandler = factory.buildResourceBundlesHandler(jawrConfig);
285 		} catch (DuplicateBundlePathException e) {
286 			throw new ServletException(e);
287 		}
288 
289 		if (resourceType.equals(JawrConstant.JS_TYPE))
290 			servletContext.setAttribute(JawrConstant.JS_CONTEXT_ATTRIBUTE, bundlesHandler);
291 		else
292 			servletContext.setAttribute(JawrConstant.CSS_CONTEXT_ATTRIBUTE, bundlesHandler);
293 
294 		this.clientSideScriptRequestHandler = new ClientSideHandlerScriptRequestHandler(bundlesHandler, jawrConfig);
295 
296 		if (log.isDebugEnabled()) {
297 			log.debug("content type set to: " + contentType);
298 		}
299 
300 		// Warn when in debug mode
301 		if (jawrConfig.isDebugModeOn()) {
302 			log.warn("Jawr initialized in DEVELOPMENT MODE. Do NOT use this mode in production or integration servers. ");
303 		}
304 	}

307 	 * Initialize the resource handler
308 	 * @return the resource handler
309 	 */
310 	protected ResourceHandler initResourceHandler() {
311 		ResourceHandler rsHandler = null;
312 		if(jawrConfig.getUseBundleMapping() && StringUtils.isNotEmpty(jawrConfig.getJawrWorkingDirectory())){
313 			rsHandler = new ServletContextResourceHandler(servletContext, jawrConfig.getJawrWorkingDirectory(), jawrConfig.getResourceCharset(), jawrConfig.getGeneratorRegistry(), resourceType);
314 		}else{
315 			rsHandler = new ServletContextResourceHandler(servletContext, jawrConfig.getResourceCharset(), jawrConfig.getGeneratorRegistry(), resourceType);
316 		}
317 		return rsHandler;
318 	}

338 	/***
339 	 * Handles a resource request by getting the requested path from the request object and invoking processRequest.
340 	 * 
341 	 * @param request the request
342 	 * @param response the response
343 	 * @throws ServletException if a servlet exception occurs
344 	 * @throws IOException if an IO exception occurs.
345 	 */
346 	public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
347 		
348 		try{
349 			// Initialize the Thread local for the Jawr context
350 			if(isJmxEnabled()){
351 				ThreadLocalJawrContext.setJawrConfigMgrObjectName(JmxUtils.getMBeanObjectName(request.getContextPath(), resourceType));
352 			}
353 			
354 			RendererRequestUtils.setRequestDebuggable(request, jawrConfig);
355 			
356 			String requestedPath = "".equals(jawrConfig.getServletMapping()) ? request.getServletPath() : request.getPathInfo();
357 			processRequest(requestedPath, request, response);
358 			
359 		} catch (Exception e) {
360 			throw new ServletException(e);
361 		}finally{
362 			
363 			// Reset the Thread local for the Jawr context
364 			ThreadLocalJawrContext.reset();
365 		}
366 		
367 	}

369 	/***
370 	 * Handles a resource request.
371 	 * <ul>
372 	 * <li>If the request contains an If-Modified-Since header, the 304 status is set and no data is written to the response</li>
373 	 * <li>If the requested path begins with the gzip prefix, a gzipped version of the resource is served, with the corresponding content-encoding
374 	 * header.</li>
375 	 * <li>Otherwise, the resource is written as text to the response.</li>
376 	 * <li>If the resource is not found, the response satus is set to 404 and no response is written.</li>
377 	 * </ul>
378 	 * 
379 	 * @param requestedPath the requested path
380 	 * @param request the request
381 	 * @param response the response
382 	 * @throws ServletException if a servlet exception occurs
383 	 * @throws IOException if an IO exception occurs
384 	 */
385 	public void processRequest(String requestedPath, HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
386 
387 		// manual reload request
388 		if (this.jawrConfig.getRefreshKey().length() > 0 && null != request.getParameter("refreshKey")
389 				&& this.jawrConfig.getRefreshKey().equals(request.getParameter("refreshKey"))) {
390 			this.configChanged(propertiesSource.getConfigProperties());
391 		}
392 
393 		if (log.isDebugEnabled())
394 			log.debug("Request received for path:" + requestedPath);
395 
396 		if (CLIENTSIDE_HANDLER_REQ_PATH.equals(requestedPath)) {
397 			this.clientSideScriptRequestHandler.handleClientSideHandlerRequest(request, response);
398 			return;
399 		}
400 
401 		// CSS images would be requested through this handler in case servletMapping is used
402 		// if( this.jawrConfig.isDebugModeOn() && !("".equals(this.jawrConfig.getServletMapping())) && null == request.getParameter(GENERATION_PARAM)) {
403 		if (JawrConstant.CSS_TYPE.equals(resourceType) && 
404 				!JawrConstant.CSS_TYPE.equals(getExtension(requestedPath)) &&
405 				this.imgMimeMap.containsKey(getExtension(requestedPath))) {
406 
407 			if (null == bundlesHandler.resolveBundleForPath(requestedPath)) {
408 				if (log.isDebugEnabled())
409 					log.debug("Path '" + requestedPath + "' does not belong to a bundle. Forwarding request to the server. ");
410 				request.getRequestDispatcher(requestedPath).forward(request, response);
411 				return;
412 			}
413 		}
414 
415 		// If debug mode is off, check for If-Modified-Since and If-none-match headers and set response caching headers.
416 		if (!this.jawrConfig.isDebugModeOn()) {
417 			// If a browser checks for changes, always respond 'no changes'.
418 			if (null != request.getHeader(IF_MODIFIED_SINCE_HEADER) || null != request.getHeader(IF_NONE_MATCH_HEADER)) {
419 				response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
420 				if (log.isDebugEnabled())
421 					log.debug("Returning 'not modified' header. ");
422 				return;
423 			}
424 
425 			// Add caching headers
426 			setResponseHeaders(response);
427 		} else if (null != request.getParameter(GENERATION_PARAM))
428 			requestedPath = request.getParameter(GENERATION_PARAM);
429 
430 		// By setting content type, the response writer will use appropiate encoding
431 		response.setContentType(contentType);
432 
433 		try {
434 			// Send gzipped resource if user agent supports it.
435 			if (requestedPath.startsWith(BundleRenderer.GZIP_PATH_PREFIX)) {
436 				requestedPath = "/" + requestedPath.substring(BundleRenderer.GZIP_PATH_PREFIX.length(), requestedPath.length());
437 				response.setHeader("Content-Encoding", "gzip");
438 				bundlesHandler.streamBundleTo(requestedPath, response.getOutputStream());
439 			} else {
440 
441 				// In debug mode, we take in account the image defined in the classpath
442 				// The following code will rewrite the URL path for the classpath images,
443 				// because in debug mode, we are retrieving the CSS ressources directly from the webapp
444 				// and if the CSS contains image classpath, we should rewrite the URL.
445 				//
446 				// TODO Create a temporary file which will store the result,
447 				// and use a map which will allow us to associate a path to a hashcode.
448 				// We will process the file only if the hashcode of the content change
449 				ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) servletContext.getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
450 				if (imgRsHandler != null && this.jawrConfig.isDebugModeOn() && resourceType.equals(JawrConstant.CSS_TYPE)) {
451 
452 					// Write the content of the CSS in the Stringwriter
453 					Writer writer = new StringWriter();
454 					bundlesHandler.writeBundleTo(requestedPath, writer);
455 					String content = writer.toString();
456 
457 					String imageServletMapping = imgRsHandler.getJawrConfig().getServletMapping();
458 					
459 					if (imageServletMapping == null) {
460 						imageServletMapping = "";
461 					}
462 
463 					// Define the replacement pattern for the image define in the classpath (like jar:img/myImg.png)
464 					String relativeRootUrlPath = getRootRelativeCssUrlPath(request, requestedPath);
465 					String replacementPattern = PathNormalizer.normalizePath("$1" + relativeRootUrlPath + imageServletMapping + "/cpCbDebug/" + "$4$5");
466 					
467 					Matcher matcher = CSS_CLASSPATH_IMG_PATTERN.matcher(content);
468 
469 					// Rewrite the images define in the classpath, to point to the image servlet
470 					StringBuffer result = new StringBuffer();
471 					while (matcher.find()) {
472 						matcher.appendReplacement(result, replacementPattern);
473 					}
474 					matcher.appendTail(result);
475 					Writer out = response.getWriter();
476 					out.write(result.toString());
477 				} else {
478 
479 					Writer out = response.getWriter();
480 					bundlesHandler.writeBundleTo(requestedPath, out);
481 				}
482 
483 			}
484 		} catch (ResourceNotFoundException e) {
485 			response.setStatus(HttpServletResponse.SC_NOT_FOUND);
486 			if (log.isInfoEnabled())
487 				log.info("Received a request for a non existing bundle: " + requestedPath);
488 			return;
489 		}
490 
491 		if (log.isDebugEnabled())
492 			log.debug("request succesfully attended");
493 	}

505 	/***
506 	 * Returns the relative path of an url to go back to the root.
507 	 * For example : if the url path is defined as "/cssServletPath/css/myStyle.css" -> "../../"
508 	 * 
509 	 * @param request the request
510 	 * @param url the requested url
511 	 * @return the relative path of an url to go back to the root.
512 	 */
513 	private String getRootRelativeCssUrlPath(HttpServletRequest request, String url) {
514 
515 		String servletPath = "".equals(jawrConfig.getServletMapping()) ? "" : request.getServletPath();
516 		String originalRequestPath = "".equals(jawrConfig.getServletMapping()) ? request.getServletPath() : request.getPathInfo();
517 		// Deals with Jawr generated resource path containing /jawr_generator.css
518 		if(originalRequestPath.startsWith(ResourceGenerator.CSS_DEBUGPATH)){
519 			url = ResourceGenerator.CSS_DEBUGPATH;
520 		}
521 
522 		url = PathNormalizer.asPath(servletPath + url);
523 		
524 		Matcher matcher = URL_SEPARATOR_PATTERN.matcher(url);
525 		StringBuffer result = new StringBuffer();
526 		int i = 0;
527 		while (matcher.find()) {
528 			if (i == 0) {
529 				matcher.appendReplacement(result, "");
530 				i++;
531 			} else {
532 				matcher.appendReplacement(result, ROOT_REPLACE_PATTERN);
533 			}
534 
535 		}
536 
537 		return result.toString();
538 	}

540 	/***
541 	 * Adds aggresive caching headers to the response in order to prevent browsers requesting the same file twice.
542 	 * 
543 	 * @param resp
544 	 */
545 	protected void setResponseHeaders(HttpServletResponse resp) {
546 		// Force resource caching as best as possible
547 		resp.setHeader(CACHE_CONTROL_HEADER, CACHE_CONTROL_VALUE);
548 		resp.setHeader(LAST_MODIFIED_HEADER, LAST_MODIFIED_VALUE);
549 		resp.setHeader(ETAG_HEADER, ETAG_VALUE);
550 		Calendar cal = Calendar.getInstance();
551 		cal.roll(Calendar.YEAR, 10);
552 		resp.setDateHeader(EXPIRES_HEADER, cal.getTimeInMillis());
553 	}
