59  public class JSMin {
60  	private static final int EOF = -1;
61  
62  	private PushbackInputStream in;
63  	private OutputStream out;
64  
65  	private int theA;
66  	private int theB;
67  	
68  	private int currentByteIndex;
69  	
70  	public JSMin(InputStream in, OutputStream out) {
71  		this.in = new PushbackInputStream(in);
72  		this.out = out;
73  	}
74  
75  	/**
76  	 * isAlphanum -- return true if the character is a letter, digit,
77  	 * underscore, dollar sign, or non-ASCII character.
78  	 */
79  	static boolean isAlphanum(int c) {
80  		return ( (c >= 'a' && c <= 'z') || (c >= '0' && c <= '9') || 
81  				 (c >= 'A' && c <= 'Z') || c == '_' || c == '$' || c == '\\' || 
82  				 c > 126);
83  	}
84  
85  	/**
86  	 * get -- return the next character from stdin. Watch out for lookahead. If
87  	 * the character is a control character, translate it to a space or
88  	 * linefeed.
89  	 */
90  	int get() throws IOException {
91  		int c = in.read();
92  		currentByteIndex++;
93  
94  		if (c >= ' ' || c == '\n' || c == EOF) {
95  			return c;
96  		}
97  
98  		if (c == '\r') {
99  			return '\n';
100 		}
101 		
102 		return ' ';
103 	}
104

107 	/**
108 	 * Get the next character without getting it.
109 	 */
110 	int peek() throws IOException {
111 		int lookaheadChar = in.read();
112 		in.unread(lookaheadChar);
113 		return lookaheadChar;
114 	}
115 
116 	/**
117 	 * next -- get the next character, excluding comments. peek() is used to see
118 	 * if a '/' is followed by a '/' or '*'.
119 	 */
120 	int next() throws IOException, UnterminatedCommentException {
121 		int c = get();
122 		if (c == '/') {
123 			switch (peek()) {
124 			case '/':
125 				for (;;) {
126 					c = get();
127 					if (c <= '\n') {
128 						return c;
129 					}
130 				}
131 
132 			case '*':
133 				get();
134 				for (;;) {
135 					switch (get()) {
136 					case '*':
137 						if (peek() == '/') {
138 							get();
139 							return ' ';
140 						}
141 						break;
142 					case EOF:
143 						throw new UnterminatedCommentException(currentByteIndex);
144 					}
145 				}
146 
147 			default:
148 				return c;
149 			}
150 
151 		}
152 		return c;
153 	}
154 
155 	/**
156 	 * action -- do something! What you do is determined by the argument: 1
157 	 * Output A. Copy B to A. Get the next B. 2 Copy B to A. Get the next B.
158 	 * (Delete A). 3 Get the next B. (Delete B). action treats a string as a
159 	 * single character. Wow! action recognizes a regular expression if it is
160 	 * preceded by ( or , or =.
161 	 */
162 
163 	void action(int d) throws IOException, UnterminatedRegExpLiteralException,
164 			UnterminatedCommentException, UnterminatedStringLiteralException {
165 		switch (d) {
166 		case 1:
167 			out.write(theA);
168 		case 2:
169 			theA = theB;
170 
171 			if (theA == '\'' || theA == '"') {
172 				for (;;) {
173 					out.write(theA);
174 					theA = get();
175 					if (theA == theB) {
176 						break;
177 					}
178 					if (theA <= '\n') {
179 						throw new UnterminatedStringLiteralException(currentByteIndex);
180 					}
181 					if (theA == '\\') {
182 						out.write(theA);
183 						theA = get();
184 					}
185 				}
186 			}
187 			
188 		case 3:
189 			theB = next();
190 			if (theB == '/' && (theA == '(' || theA == ',' || theA == '=' ||
191                     			theA == ':' || theA == '[' || theA == '!' || 
192                     			theA == '&' || theA == '|' || theA == '?' || 
193                     			theA == '{' || theA == '}' || theA == ';' || 
194                     			theA == '\n')) {
195 				out.write(theA);
196 				out.write(theB);
197 				for (;;) {
198 					theA = get();
199 					if (theA == '/') {
200 						break;
201 					} else if (theA == '\\') {
202 						out.write(theA);
203 						theA = get();
204 					} else if (theA <= '\n') {
205 						throw new UnterminatedRegExpLiteralException(currentByteIndex);
206 					}
207 					out.write(theA);
208 				}
209 				theB = next();
210 			}
211 		}
212 	}

214 	/**
215 	 * jsmin -- Copy the input to the output, deleting the characters which are
216 	 * insignificant to JavaScript. Comments will be removed. Tabs will be
217 	 * replaced with spaces. Carriage returns will be replaced with linefeeds.
218 	 * Most spaces and linefeeds will be removed.
219 	 */
220 	public void jsmin() throws IOException, JSMinException {
221 		theA = '\n';
222 		action(3);
223 		while (theA != EOF) {
224 			switch (theA) {
225 			case ' ':
226 				if (isAlphanum(theB)) {
227 					action(1);
228 				} else {
229 					action(2);
230 				}
231 				break;
232 			case '\n':
233 				switch (theB) {
234 				case '{':
235 				case '[':
236 				case '(':
237 				case '+':
238 				case '-':
239 					action(1);
240 					break;
241 				case ' ':
242 					action(3);
243 					break;
244 				default:
245 					if (isAlphanum(theB)) {
246 						action(1);
247 					} else {
248 						action(2);
249 					}
250 				}
251 				break;
252 			default:
253 				switch (theB) {
254 				case ' ':
255 					if (isAlphanum(theA)) {
256 						action(1);
257 						break;
258 					}
259 					action(3);
260 					break;
261 				case '\n':
262 					switch (theA) {
263 					case '}':
264 					case ']':
265 					case ')':
266 					case '+':
267 					case '-':
268 					case '"':
269 					case '\'':
270 						action(1);
271 						break;
272 					default:
273 						if (isAlphanum(theA)) {
274 							action(1);
275 						} else {
276 							action(3);
277 						}
278 					}
279 					break;
280 				default:
281 					action(1);
282 					break;
283 				}
284 			}
285 		}
286 		out.flush();
287 	}
