266 	/***
267 	 * Initialize configuration using params contained in the initialization properties file.
268 	 * 
269 	 * @param props the properties
270 	 */
271 	public JawrConfig(Properties props) {
272 		this.configProperties = props;
273 		if (null != props.getProperty(JAWR_DEBUG_ON)) {
274 			this.debugModeOn = Boolean.valueOf(props.getProperty(JAWR_DEBUG_ON)).booleanValue();
275 		}
276 		// If system flag is available, override debug mode from properties
277 		if (null != System.getProperty(DEBUG_MODE_SYSTEM_FLAG)) {
278 			this.debugModeOn = Boolean.valueOf(System.getProperty(DEBUG_MODE_SYSTEM_FLAG)).booleanValue();
279 		}
280 		if (null != props.getProperty(JAWR_DEBUG_OVERRIDE_KEY)) {
281 			this.debugOverrideKey = props.getProperty(JAWR_DEBUG_OVERRIDE_KEY);
282 		}
283 		
284 		if (null != props.getProperty(JAWR_USE_BUNDLE_MAPPING)) {
285 			this.useBundleMapping = Boolean.valueOf(props.getProperty(JAWR_USE_BUNDLE_MAPPING)).booleanValue();
286 		}
287 		
288 		if (null != props.getProperty(JAWR_WORKING_DIRECTORY)) {
289 			this.jawrWorkingDirectory = props.getProperty(JAWR_WORKING_DIRECTORY);
290 		}
291 		
292 		if (null != props.getProperty(JAWR_GZIP_ON)) {
293 			this.gzipResourcesModeOn = Boolean.valueOf(props.getProperty(JAWR_GZIP_ON)).booleanValue();
294 		}
295 		if (null != props.getProperty(JAWR_CHARSET_NAME)) {
296 			setCharsetName(props.getProperty(JAWR_CHARSET_NAME));
297 		}
298 		if (null != props.getProperty(JAWR_GZIP_IE6_ON)) {
299 			this.gzipResourcesForIESixOn = Boolean.valueOf(props.getProperty(JAWR_GZIP_IE6_ON)).booleanValue();
300 		}
301 		if (null != props.getProperty(JAWR_DEBUG_IE_FORCE_CSS_BUNDLE)) {
302 			this.forceCssBundleInDebugForIEOn = Boolean.valueOf(props.getProperty(JAWR_DEBUG_IE_FORCE_CSS_BUNDLE)).booleanValue();
303 		}
304 		if (null != props.getProperty(JAWR_URL_CONTEXTPATH_OVERRIDE)) {
305 			this.contextPathOverride = props.getProperty(JAWR_URL_CONTEXTPATH_OVERRIDE);
306 		}
307 		if (null != props.getProperty(JAWR_URL_CONTEXTPATH_SSL_OVERRIDE)) {
308 			this.contextPathSslOverride = props.getProperty(JAWR_URL_CONTEXTPATH_SSL_OVERRIDE);
309 		}
310 		
311 		if (null != props.getProperty(JAWR_USE_URL_CONTEXTPATH_OVERRIDE_IN_DEBUG_MODE)) {
312 			this.useContextPathOverrideInDebugMode = Boolean.valueOf(props.getProperty(JAWR_USE_URL_CONTEXTPATH_OVERRIDE_IN_DEBUG_MODE)).booleanValue();
313 		}
314 		
315 		if (null != props.getProperty(JAWR_CONFIG_RELOAD_REFRESH_KEY)) {
316 			this.refreshKey = props.getProperty(JAWR_CONFIG_RELOAD_REFRESH_KEY);
317 		}
318 		
319 		if (null != props.getProperty(JAWR_DWR_MAPPING)) {
320 			this.dwrMapping = props.getProperty(JAWR_DWR_MAPPING);
321 		}
322 
323 		if (null != props.getProperty(JAWR_LOCALE_RESOLVER)) {
324 			localeResolver = (LocaleResolver) ClassLoaderResourceUtils.buildObjectInstance(props.getProperty(JAWR_LOCALE_RESOLVER));
325 		} else
326 			localeResolver = new DefaultLocaleResolver();
327 
328 		if (null != props.getProperty(JAWR_CSSLINKS_FLAVOR)) {
329 			setCssLinkFlavor(props.getProperty(JAWR_CSSLINKS_FLAVOR).trim());
330 		}
331 
332 		if (null != props.getProperty(JAWR_CSS_IMG_USE_CLASSPATH_SERVLET)
333 				 && null != props.getProperty(JAWR_CSS_CLASSPATH_HANDLE_IMAGE)) {
334 			throw new IllegalStateException("You are using in the same configuration file '"+JAWR_CSS_IMG_USE_CLASSPATH_SERVLET+"' and '"+JAWR_CSS_CLASSPATH_HANDLE_IMAGE+"'. " +
335 					"The property '"+JAWR_CSS_IMG_USE_CLASSPATH_SERVLET+"' is deprecated. Please use only the property '"+JAWR_CSS_CLASSPATH_HANDLE_IMAGE+"'");
336 		}
337 		if (null != props.getProperty(JAWR_CSS_IMG_USE_CLASSPATH_SERVLET)) {
338 			this.classpathCssHandleImage = Boolean.valueOf(props.getProperty(JAWR_CSS_IMG_USE_CLASSPATH_SERVLET)).booleanValue();
339 		}
340 		if (null != props.getProperty(JAWR_CSS_CLASSPATH_HANDLE_IMAGE)) {
341 			this.classpathCssHandleImage = Boolean.valueOf(props.getProperty(JAWR_CSS_CLASSPATH_HANDLE_IMAGE)).booleanValue();
342 		}
343 		
344 		if (null != props.getProperty(JAWR_IMAGE_HASH_ALGORITHM)) {
345 			this.imageHashAlgorithm = props.getProperty(JAWR_IMAGE_HASH_ALGORITHM).trim();
346 		}
347 		
348 		if (null != props.getProperty(JAWR_IMAGE_RESOURCES)) {
349 			this.imageResourcesDefinition = props.getProperty(JAWR_IMAGE_RESOURCES).trim();
350 		}
351 		
352 	}
