66  	/***
67  	 * Initialize the JMX Bean 
68  	 */
69  	public static void initJMXBean(JawrRequestHandler requestHandler, ServletContext servletContext, String resourceType, Properties configProperties) {
70  		
71  		// Skip the initialisation if no JMX jar is find.
72  		try {
73  			JmxUtils.class.getClassLoader().loadClass("javax.management.MBeanServer");
74  		} catch (ClassNotFoundException e1) {
75  			log.info("JMX API is not define in the classpath.");
76  			return;
77  		}
78  		
79  		try {
80  
81  			MBeanServer mbs = JmxUtils.getMBeanServer();
82  			if(mbs != null){
83  				
84  				ObjectName jawrConfigMgrObjName = JmxUtils.getMBeanObjectName(servletContext, resourceType);
85  				JawrApplicationConfigManager appConfigMgr = (JawrApplicationConfigManager) servletContext.getAttribute(JawrConstant.JAWR_APPLICATION_CONFIG_MANAGER);
86  				if(appConfigMgr == null){
87  					appConfigMgr = new JawrApplicationConfigManager();
88  					servletContext.setAttribute(JawrConstant.JAWR_APPLICATION_CONFIG_MANAGER, appConfigMgr);
89  				}
90  				
91  				// register the jawrApplicationConfigManager if it's not already done
92  				ObjectName appJawrMgrObjectName = JmxUtils.getAppJawrConfigMBeanObjectName(servletContext);
93  				if(!mbs.isRegistered(appJawrMgrObjectName)){
94  					mbs.registerMBean(appConfigMgr, appJawrMgrObjectName);
95  				}
96  				
97  				// Create the MBean for the current Request Handler
98  				JawrConfigManager mbean = new JawrConfigManager(requestHandler, configProperties);
99  				if(mbs.isRegistered(jawrConfigMgrObjName)){
100 					log.warn("The MBean '"+jawrConfigMgrObjName.getCanonicalName()+"' already exists. It will be unregisterd and registered with the new JawrConfigManagerMBean.");
101 					mbs.unregisterMBean(jawrConfigMgrObjName);
102 				}
103 				
104 				// Initialize the jawrApplicationConfigManager
105 				if(resourceType.equals(JawrConstant.JS_TYPE)){
106 					appConfigMgr.setJsMBean(mbean);
107 				}else if(resourceType.equals(JawrConstant.CSS_TYPE)){
108 					appConfigMgr.setCssMBean(mbean);
109 				}else{
110 					appConfigMgr.setImgMBean(mbean);
111 				}
112 				
113 				mbs.registerMBean(mbean, jawrConfigMgrObjName);
114 			}
115 			
116 		} catch (Exception e) {
117 			log.error("Unable to instanciate the Jawr MBean for resource type '"+resourceType+"'", e);
118 		}
119 
120 	}

122 	/***
123 	 * Returns the current MBean server or create a new one if not exist.
124 	 * 
125 	 * @return the current MBean server or create a new one if not exist.
126 	 */
127 	public static MBeanServer getMBeanServer() {
128 
129 		MBeanServer mbs = null;
130 		
131 		// Check if JMX is enable
132 		if(System.getProperty(JMX_ENABLE_FLAG_SYSTEL_PROPERTY) == null){
133 			return null;
134 		}
135 		
136 		if (System.getProperty(JAVA_VERSION_SYSTEM_PROPERTY).startsWith(JAVA_VERSION_1_4_PREFIX)) {
137 
138 			List servers = MBeanServerFactory.findMBeanServer(null);
139 			if (servers.size() > 0) {
140 				if (log.isDebugEnabled()){
141 					log.debug("Retrieving the JMX MBeanServer.");
142 				}
143 				
144 				mbs = (MBeanServer) servers.get(0);
145 			} else {
146 				if (log.isDebugEnabled()){
147 					log.debug("Creating the JMX MBeanServer.");
148 				}
149 				
150 				mbs = MBeanServerFactory.createMBeanServer();
151 				
152 			}
153 		} else {
154 
155 			try {
156 				Class managementFactoryClass = JmxUtils.class.getClassLoader().loadClass(JAVA_LANG_MANAGEMENT_MANAGEMENT_FACTORY_CLASSNAME);
157 				Method getPlatformMBeanServerMethod = managementFactoryClass.getMethod(GET_PLATFORM_M_BEAN_SERVER_METHOD, new Class[] {});
158 				mbs = (MBeanServer) getPlatformMBeanServerMethod.invoke(null, null);
159 			} catch (Exception e) {
160 				log.error("Enable to get the JMX MBeanServer.");
161 			}
162 		}
163 
164 		return mbs;
165 	}

180 	/***
181 	 * Returns the context path associated to the servlet context
182 	 * @param servletContext the servlet context
183 	 * @return the context path associated to the servlet context
184 	 * @throws Exception if an exception occurs
185 	 */
186 	public static String getContextPath(ServletContext servletContext) throws Exception {
187 		String contextPath = null;
188 		
189 		// Get the context path
190 		
191 		// If the servlet API version is greater or equals to 2.5, use the getContextPath method
192 		if(servletContext.getMajorVersion() > 2 || servletContext.getMajorVersion() == 2 && servletContext.getMinorVersion() >= 5){
193 			
194 			Method getServletContextPathMethod = servletContext.getClass().getMethod(GET_CONTEXT_PATH_METHOD, new Class[] {});
195 			contextPath = (String) getServletContextPathMethod.invoke(servletContext, null);
196 		}else{ // Retrieve the context path from the init parameter or the servlet context
197 			contextPath = servletContext.getInitParameter(CONTEXT_PATH_PARAM_NAME);
198 		}
199 		
200 		if(contextPath == null){
201 			log.warn("No context path defined for this web application. You will face issues, if you are deploying mutiple web app, without defining the context.\n" +
202 					"If you are using a server with Servlet API less than 2.5, please use the context parameter 'contextPath' in your web.xml to define the context path of the application.");
203 			
204 			contextPath = DEFAULT_CONTEXT_PATH;
205 		}
206 		
207 		return contextPath;
208 	}

210 	/***
211 	 * Returns the object name for the Jawr configuration Manager MBean
212 	 * @param contextPath the context path
213 	 * @param resourceType  the resource type
214 	 * @return the object name for the Jawr configuration Manager MBean
215 	 * @throws Exception if an exception occurs
216 	 */
217 	public static ObjectName getMBeanObjectName(String contextPath, String resourceType) throws Exception {
218 		
219 		if(contextPath == null){
220 			log.warn("No context path defined for this web application. You will face issues, if you are deploying mutiple web app, without defining the context.\n" +
221 				"If you are using a server with Servlet API less than 2.5, please use the context parameter 'contextPath' in your web.xml to define your context path of the application.");
222 	
223 			contextPath = DEFAULT_CONTEXT_PATH_NAME;
224 		}
225 		
226 		if(contextPath.startsWith("/")){
227 			contextPath = contextPath.substring(1);
228 		}
229 		String objectNameStr = "net.jawr.web.jmx:type=JawrConfigManager,webappContext="+contextPath+",name="+resourceType+"MBean";
230 		
231 		return new ObjectName(objectNameStr);
232 	}
