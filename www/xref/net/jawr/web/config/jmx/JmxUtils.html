77  	/***
78  	 * Initialize the JMX Bean 
79  	 */
80  	public static void initJMXBean(JawrRequestHandler requestHandler, ServletContext servletContext, String resourceType, Properties configProperties) {
81  		
82  		// Skip the initialisation if no JMX jar is find.
83  		try {
84  			JmxUtils.class.getClassLoader().loadClass("javax.management.MBeanServer");
85  		} catch (ClassNotFoundException e1) {
86  			LOGGER.info("JMX API is not define in the classpath.");
87  			return;
88  		}
89  		
90  		try {
91  
92  			MBeanServer mbs = JmxUtils.getMBeanServer();
93  			if(mbs != null){
94  				
95  				ObjectName jawrConfigMgrObjName = JmxUtils.getMBeanObjectName(servletContext, resourceType);
96  				JawrApplicationConfigManager appConfigMgr = (JawrApplicationConfigManager) servletContext.getAttribute(JawrConstant.JAWR_APPLICATION_CONFIG_MANAGER);
97  				if(appConfigMgr == null){
98  					appConfigMgr = new JawrApplicationConfigManager();
99  					servletContext.setAttribute(JawrConstant.JAWR_APPLICATION_CONFIG_MANAGER, appConfigMgr);
100 				}
101 				
102 				// register the jawrApplicationConfigManager if it's not already done
103 				ObjectName appJawrMgrObjectName = JmxUtils.getAppJawrConfigMBeanObjectName(servletContext);
104 				if(!mbs.isRegistered(appJawrMgrObjectName)){
105 					mbs.registerMBean(appConfigMgr, appJawrMgrObjectName);
106 				}
107 				
108 				// Create the MBean for the current Request Handler
109 				JawrConfigManager mbean = new JawrConfigManager(requestHandler, configProperties);
110 				if(mbs.isRegistered(jawrConfigMgrObjName)){
111 					LOGGER.warn("The MBean '"+jawrConfigMgrObjName.getCanonicalName()+"' already exists. It will be unregisterd and registered with the new JawrConfigManagerMBean.");
112 					mbs.unregisterMBean(jawrConfigMgrObjName);
113 				}
114 				
115 				// Initialize the jawrApplicationConfigManager
116 				if(resourceType.equals(JawrConstant.JS_TYPE)){
117 					appConfigMgr.setJsMBean(mbean);
118 				}else if(resourceType.equals(JawrConstant.CSS_TYPE)){
119 					appConfigMgr.setCssMBean(mbean);
120 				}else{
121 					appConfigMgr.setImgMBean(mbean);
122 				}
123 				
124 				mbs.registerMBean(mbean, jawrConfigMgrObjName);
125 			}
126 			
127 		} catch (Exception e) {
128 			LOGGER.error("Unable to instanciate the Jawr MBean for resource type '"+resourceType+"'", e);
129 		}
130 
131 	}

133 	/***
134 	 * Returns the current MBean server or create a new one if not exist.
135 	 * 
136 	 * @return the current MBean server or create a new one if not exist.
137 	 */
138 	public static MBeanServer getMBeanServer() {
139 
140 		MBeanServer mbs = null;
141 		
142 		// Check if JMX is enable
143 		if(System.getProperty(JMX_ENABLE_FLAG_SYSTEL_PROPERTY) == null){
144 			return null;
145 		}
146 		
147 		if (System.getProperty(JAVA_VERSION_SYSTEM_PROPERTY).startsWith(JAVA_VERSION_1_4_PREFIX)) {
148 
149 			List servers = MBeanServerFactory.findMBeanServer(null);
150 			if (servers.size() > 0) {
151 				if (LOGGER.isDebugEnabled()){
152 					LOGGER.debug("Retrieving the JMX MBeanServer.");
153 				}
154 				
155 				mbs = (MBeanServer) servers.get(0);
156 			} else {
157 				if (LOGGER.isDebugEnabled()){
158 					LOGGER.debug("Creating the JMX MBeanServer.");
159 				}
160 				
161 				mbs = MBeanServerFactory.createMBeanServer();
162 				
163 			}
164 		} else {
165 
166 			try {
167 				Class managementFactoryClass = JmxUtils.class.getClassLoader().loadClass(JAVA_LANG_MANAGEMENT_MANAGEMENT_FACTORY_CLASSNAME);
168 				Method getPlatformMBeanServerMethod = managementFactoryClass.getMethod(GET_PLATFORM_M_BEAN_SERVER_METHOD, new Class[] {});
169 				mbs = (MBeanServer) getPlatformMBeanServerMethod.invoke(null, null);
170 			} catch (Exception e) {
171 				LOGGER.error("Enable to get the JMX MBeanServer.");
172 			}
173 		}
174 
175 		return mbs;
176 	}

192 	 * Returns the context path associated to the servlet context
193 	 * @param servletContext the servlet context
194 	 * @return the context path associated to the servlet context
195 	 * @throws Exception if an exception occurs
196 	 */
197 	public static String getContextPath(ServletContext servletContext) {
198 		String contextPath = null;
199 		
200 		// Get the context path
201 		
202 		// If the servlet API version is greater or equals to 2.5, use the getContextPath method
203 		if(servletContext.getMajorVersion() > 2 || servletContext.getMajorVersion() == 2 && servletContext.getMinorVersion() >= 5){
204 			
205 			contextPath = getContextPathForNewServletApi(servletContext);
206 			
207 		}else{ // Retrieve the context path from the init parameter or the servlet context
208 			contextPath = servletContext.getInitParameter(CONTEXT_PATH_PARAM_NAME);
209 		}
210 		
211 		if(contextPath == null){
212 			LOGGER.warn("No context path defined for this web application. You will face issues, if you are deploying mutiple web app, without defining the context.\n" +
213 					"If you are using a server with Servlet API less than 2.5, please use the context parameter 'contextPath' in your web.xml to define the context path of the application.");
214 			
215 			contextPath = DEFAULT_CONTEXT_PATH;
216 		}
217 		
218 		return contextPath;
219 	}

221 	/***
222 	 * Returns the context path using the servlet API version 2.5 method
223 	 * @param servletContext the servlet context
224 	 * @return the context path
225 	 */
226 	private static String getContextPathForNewServletApi(
227 			ServletContext servletContext) {
228 		String contextPath = null;
229 		try {
230 			Method getServletContextPathMethod = servletContext.getClass().getMethod(GET_CONTEXT_PATH_METHOD, new Class[] {});
231 			contextPath = (String) getServletContextPathMethod.invoke(servletContext, null);
232 		} catch (SecurityException e) {
233 			throw new JmxConfigException(e);
234 		} catch (NoSuchMethodException e) {
235 			throw new JmxConfigException(e);
236 		} catch (IllegalArgumentException e) {
237 			throw new JmxConfigException(e);
238 		} catch (IllegalAccessException e) {
239 			throw new JmxConfigException(e);
240 		} catch (InvocationTargetException e) {
241 			throw new JmxConfigException(e);
242 		}
243 		return contextPath;
244 	}

246 	/***
247 	 * Returns the object name for the Jawr configuration Manager MBean
248 	 * @param contextPath the context path
249 	 * @param resourceType  the resource type
250 	 * @return the object name for the Jawr configuration Manager MBean
251 	 * @throws Exception if an exception occurs
252 	 */
253 	public static ObjectName getMBeanObjectName(final String contextPath, final String resourceType) {
254 		
255 		String curCtxPath = contextPath;
256 		if(curCtxPath == null){
257 			LOGGER.warn("No context path defined for this web application. You will face issues, if you are deploying mutiple web app, without defining the context.\n" +
258 				"If you are using a server with Servlet API less than 2.5, please use the context parameter 'contextPath' in your web.xml to define your context path of the application.");
259 	
260 			curCtxPath = DEFAULT_CONTEXT_PATH_NAME;
261 		}
262 		
263 		if(curCtxPath.charAt(0) == ('/')){
264 			curCtxPath = curCtxPath.substring(1);
265 		}
266 		String objectNameStr = "net.jawr.web.jmx:type=JawrConfigManager,webappContext="+curCtxPath+",name="+resourceType+"MBean";
267 		
268 		return getObjectName(objectNameStr);
269 	}

290 	 * @param name the name
291 	 * @return the objectName
292 	 */
293 	private static ObjectName getObjectName(String name){
294 		
295 		ObjectName mBeanName = null;
296 		try {
297 			mBeanName = new ObjectName(name);
298 		} catch (MalformedObjectNameException e) {
299 			throw new JmxConfigException(e);
300 		} catch (NullPointerException e) {
301 			throw new JmxConfigException(e);
302 		}
303 		
304 		return mBeanName;
