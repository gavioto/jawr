372 	/***
373 	 * Returns the string value of the configuration managers
374 	 * 
375 	 * @param property the property to retrieve
376 	 * @return the string value of the configuration managers
377 	 */
378 	public String getStringValue(String property) {
379 
380 		List mBeans = getInitializedConfigurationManagers();
381 		try {
382 
383 			if (mBeans.size() == 3) {
384 
385 				if (areEquals(PropertyUtils.getProperty(jsMBean, property),
386 						PropertyUtils.getProperty(cssMBean, property), PropertyUtils.getProperty(imgMBean, property))) {
387 
388 					return PropertyUtils.getProperty(jsMBean, property);
389 				} else {
390 					return NOT_IDENTICAL_VALUES;
391 				}
392 			}
393 
394 			if (mBeans.size() == 2) {
395 				JawrConfigManagerMBean mBean1 = (JawrConfigManagerMBean) mBeans
396 						.get(0);
397 				JawrConfigManagerMBean mBean2 = (JawrConfigManagerMBean) mBeans
398 						.get(1);
399 
400 				if (areEquals(PropertyUtils.getProperty(mBean1, property),
401 						PropertyUtils.getProperty(mBean2, property))) {
402 					return PropertyUtils.getProperty(mBean1, property);
403 				} else {
404 					return NOT_IDENTICAL_VALUES;
405 				}
406 			}
407 
408 			JawrConfigManagerMBean mBean1 = (JawrConfigManagerMBean) mBeans
409 					.get(0);
410 
411 			return PropertyUtils.getProperty(mBean1, property);
412 
413 		} catch (Exception e) {
414 			return ERROR_VALUE;
415 		}
416 
417 	}
