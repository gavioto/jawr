14  package net.jawr.web.context;

16  import javax.management.ObjectName;
17  
18  /***
19   * This class defines the context for Jawr, it holds the context in a ThreadLocal object.
20   * 
21   * @author Matt Ruby
22   * @author Ibrahim Chaehoi
23   */
24  public final class ThreadLocalJawrContext {
25  
26  	/*** The java version system property name */
27  	private static final String JAVA_VERSION_SYSTEM_PROPERTY = "java.version";
28  
29  	/*** The java 1.4 version prefix */
30  	private static final String JAVA_VERSION_1_4_PREFIX = "1.4";
31  
32  	/***
33  	 * debugOverride will allow us to override production mode on a request by request basis.
34  	 * ThreadLocal is used to hold the overridden status throughout a given request.
35  	 */
36  	private static ThreadLocal jawrContext = new ThreadLocal(){
37  
38  		/* (non-Javadoc)
39  		 * @see java.lang.ThreadLocal#initialValue()
40  		 */
41  		protected Object initialValue() {
42  			return new JawrContext();
43  		}
44  	    
45  	};
46  	
47  	/***
48  	 * The debugOverride will be automatially set to false
49  	 */
50  	private ThreadLocalJawrContext() {
51  		
52  	}
53  	
54  	/***
55  	 * Returns the mbean object name of the Jawr config manager
56  	 * @return the mbean object name of the Jawr config manager
57  	 */
58  	public static ObjectName getJawrConfigMgrObjectName() {
59  		
60  		return ((JawrContext) jawrContext.get()).getJawrConfigMgrObjectName();
61  	}
62  
63  	/***
64  	 * Sets the mbean object name of the Jawr config manager
65  	 * @param mbeanObjectName the mbean object name of the Jawr config manager
66  	 */
67  	public static void setJawrConfigMgrObjectName(ObjectName mbeanObjectName) {
68  
69  		((JawrContext) jawrContext.get()).setJawrConfigMgrObjectName(mbeanObjectName);
70  	}
71  	
72  	/***
73  	 * Get the flag stating that production mode should be overridden
74  	 * @return the flag stating that production mode should be overridden
75  	 */
76  	public static boolean isDebugOverriden() {
77  		
78  		return ((JawrContext) jawrContext.get()).isDebugOverriden();
79  	}
80  
81  	/***
82  	 * Set the override flag that will live only for this request
83  	 * @param override the flag to set
84  	 */
85  	public static void setDebugOverriden(boolean override) {
86  
87  		((JawrContext) jawrContext.get()).setDebugOverriden(override);
88  	}
89  	
90  	/***
91  	 * Returns the flag indicating that we are using making a bundle processing at build time
92  	 * @return the flag indicating that we are using making a bundle processing at build time
93  	 */
94  	public static boolean isBundleProcessingAtBuildTime() {
95  		return ((JawrContext) jawrContext.get()).isBundleProcessingAtBuildTime();
96  	}
97  
98  	/***
99  	 * Sets the flag indicating that we are using making a bundle processing at build time
100 	 * @param bundleProcessingAtBuildTime the flag to set
101 	 */
102 	public static void setBundleProcessingAtBuildTime(boolean bundleProcessingAtBuildTime) {
103 		((JawrContext) jawrContext.get()).setBundleProcessingAtBuildTime(bundleProcessingAtBuildTime);
104 	}
105 	
106 	/***
107 	 * Returns the current request
108 	 * @return the request
109 	 */
110 	public static String getRequestURL() {
111 		return ((JawrContext) jawrContext.get()).getRequestURL();
112 	}
113 
114 	/***
115 	 * Sets the request
116 	 * @param request the request to set
117 	 */
118 	public static void setRequest(String requestURL) {
119 		((JawrContext) jawrContext.get()).setRequestURL(requestURL);
120 	}
121 	
122 	/***
123 	 * Sets the mbean object name
124 	 * @param mbeanObjectName the mbean object name
125 	 */
126 	public static void reset() {
127 
128 		if (System.getProperty(JAVA_VERSION_SYSTEM_PROPERTY).startsWith(JAVA_VERSION_1_4_PREFIX)) {
129 			((JawrContext) jawrContext.get()).reset();
130 		}else{
131 			jawrContext.remove();
132 		}
133 		
134 	}
