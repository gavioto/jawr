14  package net.jawr.web.taglib.el;
15  
16  import java.beans.IntrospectionException;
17  import java.beans.PropertyDescriptor;
18  import java.beans.SimpleBeanInfo;
19  import java.util.ArrayList;
20  import java.util.List;
21  
22  /**
23   * This is the <code>BeanInfo</code> descriptor for the
24   * <code>net.jawr.web.taglib.el.ELImagePathTag</code> class. It is needed to
25   * override the default mapping of custom tag attribute names to class attribute
26   * names.
27   * <p>
28   * This is because the value of the unevaluated EL expression has to be kept
29   * separately from the evaluated value, which is stored in the base class. This
30   * is related to the fact that the JSP compiler can choose to reuse different
31   * tag instances if they received the same original attribute values, and the
32   * JSP compiler can choose to not re-call the setter methods, because it can
33   * assume the same values are already set.
34   */
35  public class ELImagePathTagBeanInfo extends SimpleBeanInfo {
36  	/*
37  	 * (non-Javadoc)
38  	 * 
39  	 * @see java.beans.SimpleBeanInfo#getPropertyDescriptors()
40  	 */
41  	public PropertyDescriptor[] getPropertyDescriptors() {
42  		List<PropertyDescriptor> proplist = new ArrayList<PropertyDescriptor>();
43  
44  		try {
45  			proplist.add(new PropertyDescriptor("base64",
46  					ELImageInputTag.class, null, "setBase64Expr"));
47  		} catch (IntrospectionException ex) {
48  		}
49  
50  		try {
51  			proplist.add(new PropertyDescriptor("src", ELImageInputTag.class,
52  					null, "setSrcExpr"));
53  		} catch (IntrospectionException ex) {
54  		}
55  
56  		PropertyDescriptor[] result = new PropertyDescriptor[proplist.size()];
57  
58  		return ((PropertyDescriptor[]) proplist.toArray(result));
59  	}
60  }
