14  package net.jawr.web.taglib.jsf;
15  
16  import java.io.IOException;
17  
18  import javax.faces.context.FacesContext;
19  import javax.faces.context.ResponseWriter;
20  
21  /***
22   * This class defines the JSF image input tag.
23   * 
24   * @author Ibrahim Chaehoi
25   *
26   */
27  public class ImageInputTag extends AbstractHtmlImageTag {
28  
29  	/* (non-Javadoc)
30  	 * @see net.jawr.web.taglib.jsf.AbstractImageTag#render(javax.faces.context.FacesContext)
31  	 */
32  	protected void render(FacesContext context) throws IOException {
33  
34  		// Generate the name definition or image element
35  		StringBuffer results = new StringBuffer(getElementOpen());
36  
37  		prepareImageUrl(context, results);
38  
39  		prepareAttribute(results, "name", getAttribute("name"));
40  		prepareAttribute(results, "align", getAttribute("align"));
41  		prepareAttribute(results, "border", getAttribute("border"));
42  		prepareAttribute(results, "value", getAttribute("value"));
43  		prepareAttribute(results, "accesskey", getAttribute("accesskey"));
44  		prepareAttribute(results, "tabindex", getAttribute("tabindex"));
45  		prepareAttribute(results, "disabled", getAttribute("disabled"));
46  		results.append(prepareStyles());
47  		results.append(prepareEventHandlers());
48  		results.append(" />");
49  
50  		ResponseWriter writer = context.getResponseWriter();
51  		writer.write(results.toString());
52  	}
53  
54  	/***
55       * Prepares the keyboard event handlers, appending them to the the given
56       * StringBuffer.
57       *
58       * @param handlers The StringBuffer that output will be appended to.
59       */
60      protected void prepareKeyEvents(StringBuffer handlers) {
61          
62      	super.prepareKeyEvents(handlers);
63          prepareAttribute(handlers, "onchange", getAttribute("onchange"));
64          prepareAttribute(handlers, "onfocus",  getAttribute("onfocus"));
65          prepareAttribute(handlers, "onblur", getAttribute("onblur"));
66      }
67  
68  	/***
69  	 * Render the opening element.
70  	 * 
71  	 * @return The opening part of the element.
72  	 */
73  	protected String getElementOpen() {
74  		return "<input type=\"image\"";
75  	}
76  }
