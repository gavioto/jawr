190 	/***
191 	 * Render the IMG tag.

193 	 * @throws JspException if a JSP exception has occurred
194 	 */
195 	public int doEndTag() throws JspException {
196 		// Generate the name definition or image element
197 		HttpServletResponse response = (HttpServletResponse) pageContext.getResponse();
198 		StringBuffer results = new StringBuffer(getElementOpen());
199 
200 		prepareImageUrl(response, results);
201 
202 		prepareAttribute(results, "name", getName());
203 		prepareAttribute(results, "align", getAlign());
204 		prepareAttribute(results, "border", getBorder());
205 		prepareAttribute(results, "value", getValue());
206 		prepareAttribute(results, "accesskey", getAccesskey());
207 		prepareAttribute(results, "tabindex", getTabindex());
208 		prepareAttribute(results, "disabled", getDisabled());
209 		results.append(prepareStyles());
210 		results.append(prepareEventHandlers());
211 		results.append(" />");
212 
213 		try {
214 			pageContext.getOut().write(results.toString());
215 		} catch (IOException e) {
216 			throw new JspException(e);
217 		}
218 
219 		return (EVAL_PAGE);
220 	}
