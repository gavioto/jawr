14  package net.jawr.web.taglib;
15  
16  import java.io.IOException;
17  
18  import javax.servlet.http.HttpServletRequest;
19  import javax.servlet.jsp.JspException;
20  import javax.servlet.jsp.tagext.TagSupport;
21  
22  import net.jawr.web.resource.bundle.renderer.BundleRenderer;
23  import net.jawr.web.resource.bundle.renderer.BundleRendererContext;
24  import net.jawr.web.servlet.RendererRequestUtils;

26  /***
27   * Abstract implementation of a tag lib component which will retrieve a Jawr
28   * config object from the servlet context and use it to render bundles of
29   * resources according to its src attribute.
30   * 
31   * @author Jordi Hernández Sellés
32   * @author Ibrahim Chaehoi
33   */
34  public abstract class AbstractResourceBundleTag extends TagSupport {
35  
36  	/*** The serial version UID */
37  	private static final long serialVersionUID = -9114179136913388470L;
38  
39  	/*** The source path */
40  	protected String src;
41  
42  	/*** The bundle renderer */
43  	protected BundleRenderer renderer;
44  
45  	/*** The flag indicating if we should use the random parameter */
46  	protected boolean useRandomParam = true;
47  
48  	/*
49  	 * (non-Javadoc)
50  	 * 
51  	 * @see javax.servlet.jsp.tagext.TagSupport#doStartTag()
52  	 */
53  	public int doStartTag() throws JspException {
54  
55  		// Renderer istance which takes care of generating the response
56  		this.renderer = createRenderer();
57  
58  		HttpServletRequest request = (HttpServletRequest) pageContext
59  				.getRequest();
60  
61  		// set the debug override
62  		RendererRequestUtils.setRequestDebuggable(request, renderer
63  				.getBundler().getConfig());
64  
65  		try {
66  			BundleRendererContext ctx = RendererRequestUtils
67  					.getBundleRendererContext(request, renderer);
68  			renderer.renderBundleLinks(src, ctx, pageContext.getOut());
69  		} catch (IOException ex) {
70  			throw new JspException(
71  					"Unexpected IOException when writing script tags for path "
72  							+ src, ex);
73  		}
74  
75  		return super.doStartTag();
76  	}
77  
78  	/***
79  	 * Set the source of the resource or bundle to retrieve.
80  	 * 
81  	 * @param src
82  	 */
83  	public void setSrc(String src) {
84  		this.src = src;
85  	}
86  
87  	/***
88  	 * Set wether random param will be added in development mode to generated
89  	 * urls.
90  	 * 
91  	 * @param useRandomParam
92  	 */
93  	public void setUseRandomParam(boolean useRandomParam) {
94  		this.useRandomParam = useRandomParam;
95  	}
96  
97  	/***
98  	 * Retrieve the ResourceCollector from context. Each implementation will use
99  	 * a different key to retrieve it.
100 	 * 
101 	 * @return
102 	 */
103 	protected abstract BundleRenderer createRenderer();
104 
105 	/*
106 	 * (non-Javadoc)
107 	 * 
108 	 * @see javax.servlet.jsp.tagext.TagSupport#release()
109 	 */
110 	public void release() {
111 
112 		src = null;
113 	}

115 }
