56  	/***
57  	 * Returns the image URL generated by Jawr from a source image path
58  	 * @param imgSrc the source image path
59  	 * @param base64 the flag indicating if we must encode in base64 
60       * @param imgRsHandler the image resource handler
61  	 * @param request the request
62  	 * @param response the response
63  	 * @return the image URL generated by Jawr from a source image path
64  	 */
65  	public static String getImageUrl(String imgSrc, boolean base64, ImageResourcesHandler imgRsHandler, HttpServletRequest request, HttpServletResponse response) {
66  		
67  		String imgUrl = null;
68  		boolean isIE6orIE7 = RendererRequestUtils.isIE7orLess(request);
69  		
70  		if(!isIE6orIE7 && base64){
71  			imgUrl = getBase64EncodedImage(imgSrc, imgRsHandler, request);
72  		}else{
73  			imgUrl = getImageUrl(imgSrc, imgRsHandler, request, response);
74  		}
75  		return imgUrl;
76  	}

78  	/***
79  	 * Returns the image URL generated by Jawr from a source image path
80  	 * @param imgSrc the source image path
81       * @param imgRsHandler the image resource handler
82  	 * @param request the request
83  	 * @param response the response
84  	 * @return the image URL generated by Jawr from a source image path
85  	 */
86  	public static String getImageUrl(String imgSrc, ImageResourcesHandler imgRsHandler, HttpServletRequest request, HttpServletResponse response) {
87  		
88  		if(imgRsHandler == null){
89  			throw new IllegalStateException("You are using a Jawr image tag while the Jawr Image servlet has not been initialized. Initialization of Jawr Image servlet either failed or never occurred.");
90  		}
91  	
92  		String contextPath = request.getContextPath();
93  		
94  		imgSrc = getFullImagePath(imgSrc, imgRsHandler, request);
95  		
96  		String newUrl = (String) imgRsHandler.getCacheUrl(imgSrc);
97  		
98          JawrConfig jawrConfig = imgRsHandler.getJawrConfig();
99  		if(newUrl == null){
100         	try {
101 				newUrl = CheckSumUtils.getCacheBustedUrl(imgSrc, imgRsHandler.getRsReaderHandler(), jawrConfig);
102 				imgRsHandler.addMapping(imgSrc, newUrl);
103         	} catch (IOException e) {
104 	    		LOGGER.info("Unable to create the checksum for the image '"+imgSrc+"' while generating image tag.");
105 			} catch (ResourceNotFoundException e) {
106 				LOGGER.info("Unable to find the image '"+imgSrc+"' while generating image tag.");
107 			}
108     	}
109         
110         if(newUrl == null){
111         	newUrl = imgSrc;
112         }
113         
114         String imageServletMapping = jawrConfig.getServletMapping();
115 		if("".equals(imageServletMapping)){
116 			if(newUrl.startsWith("/")){
117 				newUrl = newUrl.substring(1);
118 			}
119 		}else{
120 			newUrl = PathNormalizer.joinDomainToPath(imageServletMapping, newUrl);
121 		}
122 		
123 		boolean sslRequest = RendererRequestUtils.isSslRequest(request);
124 		
125 		newUrl = RendererRequestUtils.getRenderedUrl(newUrl, jawrConfig, contextPath, sslRequest);
126 		
127 		return response.encodeURL(newUrl);
128 	}

130 	/***
131 	 * Returns the full image path to handle the relative path
132 	 * @param imgSrc the image source path
133 	 * @param imgRsHandler the image resource handler
134 	 * @param request the request
135 	 * @return the full image path
136 	 */
137 	private static String getFullImagePath(String imgSrc,
138 			ImageResourcesHandler imgRsHandler, HttpServletRequest request) {
139 		
140 		String contextPath = request.getContextPath();
141 		// relative path
142 		if(!imgRsHandler.getJawrConfig().getGeneratorRegistry().isGeneratedImage(imgSrc) && !imgSrc.startsWith("/")){ 
143 			imgSrc = PathNormalizer.concatWebPath(request.getRequestURI(), imgSrc);
144 			int idx = imgSrc.indexOf(contextPath);
145 			if(idx > -1){
146 				imgSrc = imgSrc.substring(idx+contextPath.length());
147 			}
148 		}
149 		return imgSrc;
150 	}

176 	/***
177 	 * Returns the base64 image of the image path given in parameter
178 	 * @param imgSrc the image path
179 	 * @param pageContext the page context
180 	 * @return the encoded base64 image
181 	 * @throws JawrLinkRenderingException if an exception occurs
182 	 */
183 	public static String getBase64EncodedImage(String imgSrc, ImageResourcesHandler imgRsHandler, HttpServletRequest request) {
184 		
185 		String encodedResult = null;
186 		if (null == imgRsHandler){
187 			throw new JawrLinkRenderingException(
188 					"You are using a Jawr image tag while the Jawr Image servlet has not been initialized. Initialization of Jawr Image servlet either failed or never occurred.");
189 		}
190 	
191 		imgSrc = getFullImagePath(imgSrc, imgRsHandler, request);
192 		
193 		encodedResult = imgRsHandler.getCacheUrl(BASE64_KEY_PREFIX+imgSrc);
194 		if(encodedResult == null){
195 			try{
196 				String fileExtension = FileNameUtils.getExtension(imgSrc);
197 				String fileMimeType = (String) MIMETypesSupport
198 						.getSupportedProperties(ImageTagUtils.class).get(fileExtension);
199 				
200 				InputStream is = imgRsHandler.getRsReaderHandler()
201 					.getResourceAsStream(imgSrc);
202 	
203 				ByteArrayOutputStream out = new ByteArrayOutputStream();
204 				IOUtils.copy(is, out, true);
205 				byte[] data = out.toByteArray();
206 				encodedResult = new String(Base64Encoder.encode(data));
207 				encodedResult = DATA_PREFIX + fileMimeType + ";base64," + encodedResult;
208 				imgRsHandler.addMapping(BASE64_KEY_PREFIX+imgSrc, encodedResult);
209 			}catch(ResourceNotFoundException e){
210 				LOGGER.warn("Unable to find the image '"+imgSrc+"' while generating image tag.");
211 			} catch (IOException e) {
212 				LOGGER.warn("Unable to copy the image '"+imgSrc+"' while generating image tag.");
213 			}
214 		}
215 		
216 		return encodedResult;
217 	}
