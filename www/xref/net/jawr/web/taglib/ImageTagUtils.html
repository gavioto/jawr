33  /***
34   * Utility class for image tags.
35   * 
36   * @author ibrahim Chaehoi
37   */
38  public final class ImageTagUtils {
39  
40  	/*** The logger */
41  	private static final Logger LOGGER = Logger.getLogger(ImageTagUtils.class);
42  	
43  	/***
44  	 * Returns the image URL generated by Jawr from a source image path
45  	 * @param imgSrc the source image path
46       * @param imgRsHandler the image resource handler
47  	 * @param request the request
48  	 * @param response the response
49  	 * @return the image URL generated by Jawr from a source image path
50  	 */
51  	public static String getImageUrl(String imgSrc, ImageResourcesHandler imgRsHandler, HttpServletRequest request, HttpServletResponse response) {
52  		
53  		if(imgRsHandler == null){
54  			throw new IllegalStateException("You are using a Jawr image tag while the Jawr Image servlet has not been initialized. Initialization of Jawr Image servlet either failed or never occurred.");
55  		}
56  	
57  		String contextPath = request.getContextPath();
58  		
59  		// relative path
60  		if(!imgRsHandler.getJawrConfig().getGeneratorRegistry().isGeneratedImage(imgSrc) && !imgSrc.startsWith("/")){ 
61  			imgSrc = PathNormalizer.concatWebPath(request.getRequestURI(), imgSrc);
62  			int idx = imgSrc.indexOf(contextPath);
63  			if(idx > -1){
64  				imgSrc = imgSrc.substring(idx+contextPath.length());
65  			}
66  		}
67  		
68  		String newUrl = (String) imgRsHandler.getCacheUrl(imgSrc);
69  		
70          JawrConfig jawrConfig = imgRsHandler.getJawrConfig();
71  		if(newUrl == null){
72          	try {
73  				newUrl = CheckSumUtils.getCacheBustedUrl(imgSrc, imgRsHandler.getRsReaderHandler(), jawrConfig);
74  				imgRsHandler.addMapping(imgSrc, newUrl);
75          	} catch (IOException e) {
76  	    		LOGGER.info("Unable to create the checksum for the image '"+imgSrc+"' while generating image tag.");
77  			} catch (ResourceNotFoundException e) {
78  				LOGGER.info("Unable to find the image '"+imgSrc+"' while generating image tag.");
79  			}
80      	}
81          
82          if(newUrl == null){
83          	newUrl = imgSrc;
84          }
85          
86          String imageServletMapping = jawrConfig.getServletMapping();
87  		if("".equals(imageServletMapping)){
88  			if(newUrl.startsWith("/")){
89  				newUrl = newUrl.substring(1);
90  			}
91  		}else{
92  			newUrl = PathNormalizer.joinDomainToPath(imageServletMapping, newUrl);
93  		}
94  		
95  		boolean sslRequest = RendererRequestUtils.isSslRequest(request);
96  		
97  		newUrl = RendererRequestUtils.getRenderedUrl(newUrl, jawrConfig, contextPath, sslRequest);
98  		
99  		return response.encodeURL(newUrl);
100 	}
101 
102 	/***
103 	 * Sames as its counterpart, only meant to be used as a JSP EL function. 
104 	 * @param imgSrc
105 	 * @param pageContext
106 	 * @return
107 	 * @throws JspException
108 	 */
109 	public static String getImageUrl(String imgSrc, PageContext pageContext) throws JspException {
110 		ImageResourcesHandler imgRsHandler = 
111 			(ImageResourcesHandler) pageContext.getServletContext().getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
112 		if (null == imgRsHandler)
113 			throw new JspException(
114 					"You are using a Jawr image tag while the Jawr Image servlet has not been initialized. Initialization of Jawr Image servlet either failed or never occurred.");
115 
116 		HttpServletResponse response = (HttpServletResponse) pageContext
117 		.getResponse();
118 
119 		HttpServletRequest request = (HttpServletRequest) pageContext
120 				.getRequest();
121 		
122 		return getImageUrl(imgSrc, imgRsHandler,request,response);
123 		
124 	}
