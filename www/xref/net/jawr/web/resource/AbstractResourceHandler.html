187 	 * @param tempDirRoot the temporary directory root

189 	 */
190 	private void initTempDirectory(String tempDirRoot, boolean createTempSubDir) {
191 
192 		tempDirPath = tempDirRoot;
193 
194 		if (createTempSubDir) {
195 			tempDirPath = tempDirPath + File.separator + TEMP_SUBDIR;
196 		}
197 
198 		// In windows, pathnames with spaces are returned as %20
199 		if (tempDirPath.indexOf("%20") != -1)
200 			tempDirPath = tempDirPath.replaceAll("%20", " ");
201 
202 		this.textDirPath = tempDirPath + File.separator + TEMP_TEXT_SUBDIR;
203 		this.gzipDirPath = tempDirPath + File.separator + TEMP_GZIP_SUBDIR;
204 		this.cssClasspathDirPath = tempDirPath + File.separator
205 				+ TEMP_CSS_CLASSPATH_SUBDIR;
206 
207 		if (createTempSubDir) {
208 			try {
209 				createDir(tempDirPath);
210 				createDir(textDirPath);
211 				createDir(gzipDirPath);
212 				createDir(cssClasspathDirPath);
213 			} catch (IOException e) {
214 				throw new RuntimeException(
215 						"Unexpected IOException creating temporary jawr directory",
216 						e);
217 			}

499 		storeBundle(bundleName, bundleContent, false, textDirPath);

508 		while (entryIterator.hasNext()) {

516 	/***
517 	 * Stores a resource bundle either in text or binary gzipped format.
518 	 * 
519 	 * @param bundleName the bundle name
520 	 * @param bundledResources the bundledRessources
521 	 * @param gzipFile a fag defining if the file is gzipped or not
522 	 * @param rootDir the root directory
523 	 */
524 	private void storeBundle(String bundleName, String bundledResources,
525 			boolean gzipFile, String rootdir) {
526 		if (log.isDebugEnabled()) {
527 			String msg = "Storing a generated "
528 					+ (gzipFile ? "and gzipped" : "")
529 					+ " bundle with an id of:" + bundleName;
530 			log.debug(msg);
531 		}
532 
533 		try {
534 			// Create subdirs if needed
535 			if (bundleName.indexOf('/') != -1) {
536 				StringTokenizer tk = new StringTokenizer(bundleName, "/");
537 				String pathName = rootdir;
538 				while (tk.hasMoreTokens()) {
539 					String name = tk.nextToken();
540 					if (tk.hasMoreTokens()) {
541 						pathName += File.separator;
542 						pathName += name;
543 						createDir(pathName);
544 					}
545 				}
546 				bundleName = bundleName.replace('/', File.separatorChar);
547 			}
548 
549 			File store = createNewFile(rootdir + File.separator + bundleName);
550 
551 			if (gzipFile) {
552 				FileOutputStream fos = new FileOutputStream(store);
553 				GZIPOutputStream gzOut = new GZIPOutputStream(fos);
554 				byte[] data = bundledResources.toString().getBytes(
555 						charset.name());
556 				gzOut.write(data, 0, data.length);
557 				gzOut.close();
558 			} else {
559 				FileOutputStream fos = new FileOutputStream(store);
560 				FileChannel channel = fos.getChannel();
561 				Writer wr = Channels.newWriter(channel, charset.newEncoder(),
562 						-1);
563 				wr.write(bundledResources.toString());
564 				wr.close();
565 			}
566 		} catch (IOException e) {
567 			e.printStackTrace();
568 			throw new RuntimeException(
569 					"Unexpected IOException creating temporary jawr file", e);
570 		}
571 	}

573 	/***
574 	 * Creates a directory. If dir is note created for some reson a runtimeexception is thrown.
575 	 * 
576 	 * @param dir
577 	 * @throws IOException
578 	 */
579 	private File createDir(String path) throws IOException {
580 		// In windows, pathnames with spaces are returned as %20
581 		if (path.indexOf("%20") != -1)
582 			path = path.replaceAll("%20", " ");
583 		File dir = new File(path);
584 		if (!dir.exists() && !dir.mkdir())
585 			throw new RuntimeException(
586 					"Error creating temporary jawr directory with path:"
587 							+ dir.getPath());
588 
589 		if (log.isDebugEnabled())
590 			log.debug("Created dir: " + dir.getCanonicalPath());
591 		return dir;
592 	}

601 	private File createNewFile(String path) throws IOException {
602
