25  import org.apache.commons.lang.StringUtils;

27  /***
28   * This class rewrites is used to rewrite CSS URLs according to the new
29   * relative locations of the references, from the original CSS path to a new one. 
30   * Since the path changes, the URLs must be rewritten accordingly. URLs in css files are
31   * expected to be according to the css spec (see
32   * http://www.w3.org/TR/REC-CSS2/syndata.html#value-def-uri). Thus, single
33   * double, or no quotes enclosing the url are allowed (and remain as they are
34   * after rewriting). Escaped parens and quotes are allowed within the url.
35   * 
36   * @author Ibrahim Chaehoi
37   */
38  public class CssImageUrlRewriter {
39  
40  	/*** The URL separator */
41  	private static final String URL_SEPARATOR = "/";
42  
43  	/*** The URL regexp pattern */
44  	public static String URL_REGEXP = "url//(//s*" // 'url('
45  		// and any number of whitespaces
46  		+ "(?!(\"|')?(data|mhtml|cid):)(((//////))|[^)])*)" // any sequence of
47  		// characters not
48  		// starting with
49  		// 'data:',
50  		// 'mhtml:', or
51  		// 'cid:', except an
52  		// unescaped ')'
53  		+ "//s*//)"; // Any number of whitespaces, then ')'
54  
55  	/*** The url pattern */
56  	public static final Pattern URL_PATTERN = Pattern.compile(URL_REGEXP, // Any number of whitespaces, then ')'
57  			Pattern.CASE_INSENSITIVE); // works with 'URL('
58  
59  	/*** The context path */
60  	protected String contextPath;
61  	
62  	/***
63  	 * Constructor
64  	 */
65  	public CssImageUrlRewriter() {
66  		
67  	}
68  	
69  	/***
70  	 * Constructor
71  	 * @param config the jawr config
72  	 */
73  	public CssImageUrlRewriter(JawrConfig config) {
74  		setContextPath(config.getProperty(JawrConstant.JAWR_CSS_URL_REWRITER_CONTEXT_PATH));
75  	}
76  
77  	/***
78  	 * Sets the context path
79  	 * @param contextPath the contextPath to set
80  	 */
81  	public void setContextPath(String contextPath) {
82  		if(StringUtils.isNotEmpty(contextPath)){
83  			if(contextPath.charAt(0) != '/'){
84  				contextPath = '/'+contextPath;
85  			}
86  			if(contextPath.charAt(contextPath.length()-1) != '/'){
87  				contextPath = contextPath+'/';
88  			}
89  			this.contextPath = contextPath;
90  		}else{
91  			this.contextPath = null;
92  		}
93  	}
94  
95  	/***
96  	 * Rewrites the image URL
97  	 * @param originalCssPath the original CSS path
98  	 * @param newCssPath the new CSS path
99  	 * @param originalCssContent the original CSS content
100 	 * @return the new CSS content with image path rewritten
101 	 * @throws IOException
102 	 */
103 	public StringBuffer rewriteUrl(String originalCssPath, String newCssPath, String originalCssContent) throws IOException {
104 		
105 		// Rewrite each css image url path
106 		Matcher matcher = URL_PATTERN.matcher(originalCssContent);
107 		StringBuffer sb = new StringBuffer();
108 		while (matcher.find()) {
109 
110 			String url = getUrlPath(matcher.group(), originalCssPath, newCssPath);
111 			matcher.appendReplacement(sb, RegexUtil
112 					.adaptReplacementToMatcher(url));
113 		}
114 		matcher.appendTail(sb);
115 		return sb;
116 	}
117 
118 	/***
119 	 * Transform a matched url so it points to the proper relative path with
120 	 * respect to the given path.
121 	 * 
122 	 * @param match
123 	 *            the matched URL
124 	 * @param newCssPath
125 	 *            the full bundle path
126 	 * @param status
127 	 *            the bundle processing status
128 	 * @return the image URL path
129 	 * @throws IOException
130 	 *             if an IO exception occurs
131 	 */
132 	protected String getUrlPath(String match, String originalPath, String newCssPath) throws IOException {
133 
134 		String url = match.substring(match.indexOf('(') + 1,
135 				match.lastIndexOf(')')).trim();
136 
137 		// To keep quotes as they are, first they are checked and removed.
138 		String quoteStr = "";
139 		if (url.startsWith("'") || url.startsWith("\"")) {
140 			quoteStr = url.charAt(0) + "";
141 			url = url.substring(1, url.length() - 1);
142 		}
143 
144 		// Check if the URL is absolute, but in the application itself
145 		if(StringUtils.isNotEmpty(contextPath) && url.startsWith(contextPath)){
146 			String rootRelativePath = PathNormalizer.getRootRelativePath(originalPath);
147 			url = rootRelativePath + url.substring(contextPath.length());
148 		}
149 		
150 		// Check if the URL is absolute, if it is return it as is.
151 		int firstSlash = url.indexOf('/');
152 		if (0 == firstSlash
153 				|| (firstSlash != -1 && url.charAt(++firstSlash) == '/')) {
154 			StringBuffer sb = new StringBuffer("url(");
155 			sb.append(quoteStr).append(url).append(quoteStr).append(")");
156 			return sb.toString();
157 		}
158 
159 		if (url.startsWith(URL_SEPARATOR))
160 			url = url.substring(1, url.length());
161 		else if (url.startsWith("./"))
162 			url = url.substring(2, url.length());
163 
164 		String imgUrl = getRewrittenImagePath(originalPath, newCssPath, url);
165 
166 		// Start rendering the result, starting by the initial quote, if any.
167 		String finalUrl = "url("+quoteStr+imgUrl+quoteStr+")";
168 		Matcher urlMatcher = URL_PATTERN.matcher(finalUrl);
169 		if(urlMatcher.find()){ // Normalize only if a real URL
170 			finalUrl = PathNormalizer.normalizePath(finalUrl);
171 		}
172 		return finalUrl;
173 	}
174 
175 	/***
176 	 * Returns the rewritten image path
177 	 * @param originalCssPath the original Css path
178 	 * @param newCssPath the new Css path
179 	 * @param url the image URL
180 	 * @return the rewritten image path
181 	 * @throws IOException if an IOException occurs
182 	 */
183 	protected String getRewrittenImagePath(String originalCssPath, String newCssPath, String url)
184 			throws IOException {
185 
186 		// Here we generate the full path of the CSS image
187 		// to be able to define the relative path from the full bundle path
188 		String fullImgPath = PathNormalizer.concatWebPath(originalCssPath, url);
189 		
190 		String imgUrl = PathNormalizer.getRelativeWebPath(PathNormalizer
191 				.getParentPath(newCssPath), fullImgPath);
192 		
193 		return imgUrl;
194 	}
