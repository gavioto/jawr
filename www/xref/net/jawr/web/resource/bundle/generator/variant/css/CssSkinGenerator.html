144 	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#createResource(net.jawr.web.resource.bundle.generator.GeneratorContext)
145 	 */
146 	public Reader createResource(GeneratorContext context) {
147 		
148 		Reader reader = null;
149 		ResourceReaderHandler readerHandler = context.getResourceReaderHandler();
150 		String path = context.getPath();
151 		
152 		String skinRootDir = getSkinRootDir(path, skinMapping.keySet());
153 		Map ctxVariantSetMap = (Map) skinMapping.get(skinRootDir);
154 		
155 		String skinVariantPath = path.substring(skinRootDir.length());
156 		String[] paths = skinVariantPath.split(JawrConstant.URL_SEPARATOR);
157 		
158 		VariantResourceReaderStrategy strategy = getVariantStrategy(context, ctxVariantSetMap);
159 		Map variantMap = null;
160 		do {
161 			variantMap = strategy.nextVariantMapConbination();
162 			if(variantMap != null){
163 				reader = getResourceReader(readerHandler, skinRootDir, paths, variantMap);
164 			}
165 		}while(variantMap != null && reader == null);
166 		
167 		if(!context.isProcessingBundle() && reader != null){
168 			
169 			// Rewrite the image URL
170 			StringWriter writer = new StringWriter();
171 			try {
172 				IOUtils.copy(reader, writer);
173 				CssImageUrlRewriter rewriter = new CssImageUrlRewriter(context.getConfig());
174 				String bundlePath = PathNormalizer.concatWebPath(context.getConfig().getServletMapping(), ResourceGenerator.CSS_DEBUGPATH);
175 				StringBuffer result = rewriter.rewriteUrl(path, bundlePath, writer.toString());
176 				reader = new StringReader(result.toString());
177 			} catch (IOException e) {
178 				throw new BundlingProcessException(e);
179 			}
180 		}
181 		
182 		return reader;
183 	}

226 	 * Returns the skin mapping from the Jawr config, here the resource mapping type is skin_locale 
227 	 * @param rsBrowser the resource browser
228 	 * @param config the jawr config
229 	 * @param skinMapping the skin mapping
230 	 * @param skinRootDirectories the skin root directories
231 	 */
232 	private void updateSkinMappingUsingTypeSkinLocale(ResourceBrowser rsBrowser,
233 			JawrConfig config, Map skinMapping, Set skinRootDirectories) {
234 		
235 		String defaultSkinName = null;
236 		String defaultLocaleName = null;
237 		
238 		// Check if there are no directory which is contained in another root directory
239 		for (Iterator itRootDir = skinRootDirectories.iterator(); itRootDir.hasNext();) {
240 			String defaultSkinDir = PathNormalizer.asDirPath((String) itRootDir.next());
241 			String defaultSkinDirName = PathNormalizer.getPathName(defaultSkinDir);
242 			String skinRootDir = PathNormalizer.getParentPath(defaultSkinDir);
243 			
244 			String skinName = null;
245 			String localeName = null;
246 			
247 			if(LocaleUtils.LOCALE_SUFFIXES.contains(defaultSkinDirName)){ // The rootDir point to a locale directory
248 				localeName = defaultSkinDirName;
249 				skinName = PathNormalizer.getPathName(skinRootDir);
250 				skinRootDir = PathNormalizer.getParentPath(skinRootDir);
251 				
252 			}else{
253 				skinName = defaultSkinDirName;
254 			}
255 			
256 			if(defaultSkinName == null){
257 				defaultSkinName = skinName;
258 			}else if(!defaultSkinName.equals(skinName)){
259 				throw new BundlingProcessException("The default skin for the skin root directories are not the same. Please check your configuration.");
260 			}
261 			
262 			if(defaultLocaleName == null){
263 				defaultLocaleName = localeName;
264 			}else if(defaultLocaleName != null && !defaultLocaleName.equals(localeName)){
265 				throw new BundlingProcessException("The default locale for the skin root directories are not the same. Please check your configuration.");
266 			}
267 			
268 			checkRootDirectoryNotOverlap(defaultSkinDir, skinRootDirectories);
269 			
270 			Map variantsMap = getVariants(rsBrowser, skinRootDir, skinName,
271 					localeName, true);
272 			
273 			skinMapping.put(skinRootDir, variantsMap);
274 		}
275 		
276 		CssSkinVariantResolver resolver = new CssSkinVariantResolver(defaultSkinName, config.getSkinCookieName());
277 		config.getGeneratorRegistry().registerVariantResolver(resolver);
278 	}

352 	/***
353 	 * Initialize the skinMapping from the parent path
354 	 * @param rsBrowser the resource browser
355 	 * @param rootDir the skin root dir path
356 	 * @param defaultSkinName the default skin name
357 	 * @param defaultLocaleName the default locale name
358 	 */
359 	private Map getVariants(ResourceBrowser rsBrowser, String rootDir,
360 			String defaultSkinName, String defaultLocaleName, boolean mappingSkinLocale) {
361 		Set paths = rsBrowser.getResourceNames(rootDir);
362 		Set skinNames = new HashSet();
363 		Set localeVariants = new HashSet();
364 		for (Iterator itPath = paths.iterator(); itPath.hasNext();) {
365 			String path = rootDir+ (String) itPath.next();
366 			if(rsBrowser.isDirectory(path)){
367 				String dirName = PathNormalizer.getPathName(path);
368 				if(mappingSkinLocale){
369 					skinNames.add(dirName);
370 					// check if there are locale variants for this skin,
371 					// and update the localeVariants if needed
372 					updateLocaleVariants(rsBrowser, path, localeVariants);
373 				}else{
374 					if(LocaleUtils.LOCALE_SUFFIXES.contains(dirName)){
375 						localeVariants.add(dirName);
376 						// check if there are skin variants for this locales,
377 						// and update the skinVariants if needed
378 						updateSkinVariants(rsBrowser, path, skinNames);
379 					}
380 				}
381 			}
382 		}
383 		
384 		// Initialize the variant mapping for the skin root directory
385 		return getVariants(defaultSkinName, skinNames, defaultLocaleName, localeVariants);
386 	}

454 	 * Check if there are no directory which is contained in another root directory,
455 	 * @param dir the root directory to check
456 	 * @param skinRootDirectories the root directories
457 	 */
458 	private void checkRootDirectoryNotOverlap(String dir,
459 			Set skinRootDirectories) {
460 		
461 		String rootDir = removeLocaleSuffixIfExist(dir);
462 		for (Iterator itSkinDir = skinRootDirectories.iterator(); itSkinDir.hasNext();) {
463 			String skinDir = PathNormalizer.asDirPath((String) itSkinDir.next());
464 			if(!skinDir.equals(dir)){
465 				skinDir = removeLocaleSuffixIfExist(skinDir);
466 				if(skinDir.startsWith(rootDir)){
467 					throw new BundlingProcessException("There is a misconfiguration. It is not allowed to have a skin root directory containing another one.");
468 				}
469 			}
470 		}
471 	}

474 	 * Removes from the directory path the locale suffix if it exists
475 	 * @param dir the directory path
476 	 * 
477 	 * <pre>
478      * removeLocaleSuffixIfExist( "/css/panel/default/" )         = "/css/panel/default/"
479      * removeLocaleSuffixIfExist( "/css/panel/default/en_US" )   = "/css/panel/default/"
480      * </pre>
481      * 
482 	 * @return the directory path, with the locale suffix removed if it exists
483 	 */
484 	private String removeLocaleSuffixIfExist(String dir){
485 		
486 		String result = dir;
487 		String dirName = PathNormalizer.getPathName(dir);
488 		if(LocaleUtils.LOCALE_SUFFIXES.contains(dirName)){
489 			result = dir.substring(0, dir.indexOf("/"+dirName)+1);
490 		}
491 		return result;
492 	}

515 	/***
516 	 * Returns the reader for the resource path define in parameter
517 	 * @param skinRootDir the root directory
518 	 * @param paths the array of path
519 	 * @param variantMap the variant map
520 	 * @return the reader
521 	 */
522 	private Reader getResourceReader(ResourceReaderHandler readerHandler, String skinRootDir, String[] paths,
523 			Map variantMap) {
524 		
525 		Reader reader = null;
526 		StringBuffer path = new StringBuffer(skinRootDir);
527 		String skinVariant = (String) variantMap.get(JawrConstant.SKIN_VARIANT_TYPE);
528 		String localeVariant = (String) variantMap.get(JawrConstant.LOCALE_VARIANT_TYPE);
529 		
530 		if(skinMappingType.equals(JawrConstant.SKIN_TYPE_MAPPING_SKIN_LOCALE)){
531 			paths[0] = skinVariant;
532 			if(localeVariant != null){
533 				paths[1] = localeVariant;
534 			}
535 		}else{
536 			paths[0] = localeVariant;
537 			if(skinVariant != null){
538 				paths[1] = skinVariant;
539 			}
540 		}
541 		
542 		for (int i = 0; i < paths.length; i++) {
543 			path.append(paths[i]);
544 			if(i+1 < paths.length){
545 				path.append(JawrConstant.URL_SEPARATOR);
546 			}
547 		}
548 		
549 		try {
550 			reader = readerHandler.getResource(path.toString());
551 		} catch (ResourceNotFoundException e) {
552 			// Nothing to do
553 		}
554 		
555 		return reader;
556 	}
