33  /***
34   * Registry for resource generators, which create scripts or CSS data dynamically, 
35   * as opposed to the usual behavior of reading a resource from the war file. 
36   * It provides methods to determine if a path mapping should be handled by a generator, 
37   * and to actually render the resource using the appropiate generator. 
38   * Path mappings which require generation will use a prefix (preferrably one which ends with 
39   * a colon, such as 'messages:'). 
40   * Generators provided with Jawr will be automatically mapped. 
41   * 
42   * @author Jordi Hernández Sellés
43   * @author Ibrahim Chaehoi
44   */
45  public class GeneratorRegistry {

47  	/*** The message bundle prefix */
48  	public static final String MESSAGE_BUNDLE_PREFIX = "messages";
49  	
50  	/*** The classpath resource bundle prefix */
51  	public static final String CLASSPATH_RESOURCE_BUNDLE_PREFIX = "jar";
52  	
53  	/*** The old classpath CSS bundle prefix (usage of this prefix is deprecated since Jawr 2.8) */
54  	public static final String OLD_CLASSPATH_CSS_BUNDLE_PREFIX = "jar_css";
55  	
56  	/*** The DWR bundle prefix */
57  	public static final String DWR_BUNDLE_PREFIX = "dwr";
58  	
59  	/*** The commons validator bundle prefix */
60  	public static final String COMMONS_VALIDATOR_PREFIX = "acv";
61  	
62  	/*** The generator prefix separator */
63  	public static final String PREFIX_SEPARATOR = ":";

65  	/*** The generator registry */
66  	private static final Map registry = ConcurrentCollectionsFactory.buildConcurrentHashMap();
67  	
68  	/*** The generator prefix registry */
69  	private static final List prefixRegistry = ConcurrentCollectionsFactory.buildCopyOnWriteArrayList();
70  	
71  	/*** The resource type */
72  	private String resourceType;
73  	
74  	/*** The Jawr config */
75  	private JawrConfig config;

80  		prefixRegistry.add(CLASSPATH_RESOURCE_BUNDLE_PREFIX + PREFIX_SEPARATOR);
81  		prefixRegistry.add(OLD_CLASSPATH_CSS_BUNDLE_PREFIX+PREFIX_SEPARATOR);
82  		prefixRegistry.add(DWR_BUNDLE_PREFIX + PREFIX_SEPARATOR);
83  		prefixRegistry.add(COMMONS_VALIDATOR_PREFIX + PREFIX_SEPARATOR);

111 	/***
112 	 * Lazy loads generators, to avoid the need for undesired dependencies. 
113 	 * 
114 	 * @param generatorKey the generator key
115 	 */
116 	private void loadGenerator(String generatorKey) {
117 		if((MESSAGE_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
118 			registry.put(generatorKey, new ResourceBundleMessagesGenerator());
119 		}
120 		else if((CLASSPATH_RESOURCE_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
121 			if(resourceType.equals(JawrConstant.JS_TYPE)){
122 				registry.put(generatorKey, new ClasspathJSGenerator());
123 			}else{
124 				registry.put(generatorKey, new ClassPathCSSGenerator());
125 			}
126 		}
127 		else if((OLD_CLASSPATH_CSS_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
128 			registry.put(generatorKey, new ClassPathCSSGenerator());
129 		}else if((DWR_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
130 			registry.put(generatorKey, DWRGeneratorFactory.createDWRGenerator());
131 		}
132 		else if((COMMONS_VALIDATOR_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
133 			registry.put(generatorKey, new CommonsValidatorGenerator());
134 		}
135 	}

137 	/***
138 	 * Register a generator mapping it to the specified prefix. 
139 	 * 
140 	 * @param clazz the classname of the generator
141 	 */
142 	public void registerGenerator(String clazz){
143 		ResourceGenerator generator = (ResourceGenerator) ClassLoaderResourceUtils.buildObjectInstance(clazz);
144 		
145 		if(null == generator.getMappingPrefix() || "".equals(generator.getMappingPrefix()) )
146 			throw new IllegalStateException("The getMappingPrefix() method must return something at " + clazz);
147 		
148 		String fullPrefix = generator.getMappingPrefix() + PREFIX_SEPARATOR;
149 		
150 		// Verify this prefix is unused
151 		if(prefixRegistry.contains(fullPrefix)) {
152 			String generatorName = registry.get(fullPrefix).getClass().getName();
153 			if(!clazz.equals(generatorName)) {
154 				String errorMsg = "Cannot register the generator of class " 
155 								+ generator.getClass().getName()
156 								+ " using the prefix " + generator.getMappingPrefix() + " since such prefix is being used by "
157 								+ generatorName + ". Please pecify a different return value at the getMappingPrefix() method.";
158 				throw new IllegalStateException(errorMsg);
159 			}
160 		}
161 		
162 		prefixRegistry.add(generator.getMappingPrefix() + PREFIX_SEPARATOR);
163 		registry.put(generator.getMappingPrefix() + PREFIX_SEPARATOR, generator);
164 	}

168 	 * @param path the resource path

175 	/***
176 	 * Creates the contents corresponding to a path, by using the appropriate generator. 
177 	 * @param path the resource path
178 	 * @param resourceHandler the current resource handler
179 	 * @param processingBundle the flag indicating if if we are currently processing the bundles
180 	 * @return the reader for the contents
181 	 */
182 	public Reader createResource(String path,
183 			ResourceHandler resourceHandler, boolean processingBundle) {
184 		String key = matchPath(path);
185 		Locale locale = null;
186 		
187 		if(path.indexOf('@') != -1){
188 			String localeKey = path.substring(path.indexOf('@')+1);
189 			path = path.substring(0,path.indexOf('@'));
190 			
191 			// Resourcebundle should be doing this for me...
192 			String[] params = localeKey.split("_");			
193 			switch(params.length) {
194 				case 3:
195 					locale = new Locale(params[0],params[1],params[2]);
196 					break;
197 				case 2: 
198 					locale = new Locale(params[0],params[1]);
199 					break;
200 				default:
201 					locale = new Locale(localeKey);
202 			}
203 		}
204 		GeneratorContext context = new GeneratorContext(config, path.substring(key.length()));
205 		context.setLocale(locale);
206 		context.setResourceHandler(resourceHandler);
207 		context.setProcessingBundle(processingBundle);
208 		
209 		return ((ResourceGenerator)registry.get(key)).createResource(context);
210 	}

223 	/***
224 	 * Get the key from the mappings that corresponds to the specified path. 
225 	 * @param path the resource path
226 	 * @return the registry key corresponding to the path
227 	 */
228 	private String matchPath(String path) {
229 		String rets = null;
230 		for(Iterator it = prefixRegistry.iterator();it.hasNext() && rets == null;) {
231 			String prefix = (String) it.next();
232 			if(path.startsWith(prefix))
233 				rets = prefix;			
234 		}	
235 		// Lazy load generator
236 		if(null != rets && !registry.containsKey(rets))
237 			loadGenerator(rets);
238 		
239 		return rets;
240 	}
