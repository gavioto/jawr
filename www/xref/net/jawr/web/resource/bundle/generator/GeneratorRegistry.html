48  /***
49   * Registry for resource generators, which create scripts or CSS data dynamically, 
50   * as opposed to the usual behavior of reading a resource from the war file. 
51   * It provides methods to determine if a path mapping should be handled by a generator, 
52   * and to actually render the resource using the appropriate generator. 
53   * Path mappings which require generation will use a prefix (preferably one which ends with 
54   * a colon, such as 'messages:'). 
55   * Generators provided with Jawr will be automatically mapped. 
56   * 
57   * @author Jordi Hernández Sellés
58   * @author Ibrahim Chaehoi

154 	/***
155 	 * Lazy loads generators, to avoid the need for undesired dependencies. 
156 	 * 
157 	 * @param generatorKey the generator key
158 	 */
159 	private void loadGenerator(String generatorKey) {
160 		PrefixedResourceGenerator generator = null;
161 		if((MESSAGE_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
162 			generator = new ResourceBundleMessagesGenerator();
163 		}
164 		else if((CLASSPATH_RESOURCE_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
165 			if(resourceType.equals(JawrConstant.JS_TYPE)){
166 				generator = new ClasspathJSGenerator();
167 			}else if(resourceType.equals(JawrConstant.CSS_TYPE)){
168 				generator = new ClassPathCSSGenerator(config.isCssClasspathImageHandledByClasspathCss());
169 			}else{
170 				generator = new ClassPathImgResourceGenerator();
171 			}
172 		}
173 		else if((DWR_BUNDLE_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
174 			generator = DWRGeneratorFactory.createDWRGenerator();
175 		}
176 		else if((COMMONS_VALIDATOR_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
177 			generator = new CommonsValidatorGenerator();
178 		}else if(resourceType.equals(JawrConstant.CSS_TYPE) && (IE_CSS_GENERATOR_PREFIX + PREFIX_SEPARATOR).equals(generatorKey)){
179 			generator = new IECssBundleGenerator();
180 		}else if((resourceType.equals(JawrConstant.CSS_TYPE) ||
181 				resourceType.equals(JawrConstant.IMG_TYPE)) && (SPRITE_GENERATOR_PREFIX+PREFIX_SEPARATOR).equals(generatorKey)){
182 			generator = new SpriteGenerator(rsHandler, config);
183 		}else if(resourceType.equals(JawrConstant.CSS_TYPE) && (SKIN_GENERATOR_PREFIX+PREFIX_SEPARATOR).equals(generatorKey)){
184 			generator = new CssSkinGenerator(rsHandler, config);
185 		}else if(resourceType.equals(JawrConstant.JS_TYPE) && (SKIN_SWTICHER_GENERATOR_PREFIX+PREFIX_SEPARATOR).equals(generatorKey)){
186 			generator = new SkinSwitcherJsGenerator();
187 		}
188 		
189 		if(generator != null){
190 			initializeGeneratorProperties(generator);
191 			updateRegistries(generator, generatorKey);
192 			ResourceReader proxy = ResourceGeneratorReaderProxyFactory.getResourceReaderProxy(generator, rsHandler, config);
193 			rsHandler.addResourceReaderToEnd(proxy);
194 		}
195 	}

202 	private void updateRegistries(PrefixedResourceGenerator generator, String generatorKey) {
203 		
204 		registry.put(generatorKey, generator);
205 
206 		if(generator instanceof StreamResourceGenerator){
207 			imageResourceGeneratorPrefixRegistry.add(generatorKey);
208 		}
209 		if(generator instanceof CssResourceGenerator){

211 				cssImageResourceGeneratorPrefixRegistry.add(generatorKey);

213 		}

226 	/***
227 	 * Register a variant resolver
228 	 * @param resolver the variant resolver
229 	 */
230 	public void registerVariantResolver(VariantResolver resolver){
231 		
232 		for (Iterator itResolver = variantResolvers.values().iterator(); itResolver.hasNext();) {
233 			VariantResolver variantResolver = (VariantResolver) itResolver.next();
234 			if(StringUtils.isEmpty(resolver.getVariantType())){
235 				throw new IllegalStateException("The getVariantType() method must return something at " + resolver.getClass());
236 			}
237 			
238 			if(resolver.getVariantType().equals(variantResolver.getVariantType())){
239 				throw new IllegalStateException("There are 2 resolvers defined for the variant type '"+resolver.getVariantType()+"' : "+variantResolver.getClass()+";" + resolver.getClass());
240 			}
241 		}
242 		variantResolvers.put(resolver.getVariantType(), resolver);
243 	}

245 	/***
246 	 * Register a generator mapping it to the specified prefix. 
247 	 * 
248 	 * @param clazz the classname of the generator
249 	 */
250 	public void registerGenerator(String clazz){
251 		PrefixedResourceGenerator generator = (PrefixedResourceGenerator) ClassLoaderResourceUtils.buildObjectInstance(clazz);
252 		
253 		if(null == generator.getMappingPrefix() || "".equals(generator.getMappingPrefix()) )
254 			throw new IllegalStateException("The getMappingPrefix() method must return something at " + clazz);
255 		
256 		String fullPrefix = generator.getMappingPrefix() + PREFIX_SEPARATOR;
257 		
258 		// Verify this prefix is unused
259 		if(prefixRegistry.contains(fullPrefix)) {
260 			String generatorName = registry.get(fullPrefix).getClass().getName();
261 			if(!clazz.equals(generatorName)) {
262 				String errorMsg = "Cannot register the generator of class " 
263 								+ generator.getClass().getName()
264 								+ " using the prefix " + generator.getMappingPrefix() + " since such prefix is being used by "
265 								+ generatorName + ". Please pecify a different return value at the getMappingPrefix() method.";
266 				throw new IllegalStateException(errorMsg);
267 			}
268 		}
269 		
270 		initializeGeneratorProperties(generator);
271 		
272 		prefixRegistry.add(generator.getMappingPrefix() + PREFIX_SEPARATOR);
273 		updateRegistries(generator, generator.getMappingPrefix() + PREFIX_SEPARATOR);
274 		rsHandler.addResourceReaderToEnd(ResourceGeneratorReaderProxyFactory.getResourceReaderProxy(generator, rsHandler, config));
275 	}

278 	 * Initializes the generator properties.
279 	 * 
280 	 * @param generator the generator
281 	 */
282 	private void initializeGeneratorProperties(
283 			PrefixedResourceGenerator generator) {
284 		// Initialize the generator
285 		if(generator instanceof InitializingResourceGenerator){
286 			if(generator instanceof ConfigurationAwareResourceGenerator){
287 				((ConfigurationAwareResourceGenerator) generator).setConfig(config);
288 			}
289 			if(generator instanceof TypeAwareResourceGenerator){
290 				((TypeAwareResourceGenerator) generator).setResourceType(resourceType);
291 			}
292 			if(generator instanceof PostInitializationAwareResourceGenerator){
293 				((PostInitializationAwareResourceGenerator) generator).afterPropertiesSet();
294 			}
295 		}
296 	}

318 	 * Returns the path to use in the "build time process" to generate the resource path for debug mode. 
319 	 * @param path the resource path
320 	 * @return the path to use in the "build time process" to generate the resource path for debug mode. 
321 	 */
322 	public String getDebugModeBuildTimeGenerationPath(String path){
323 		
324 		int idx = path.indexOf("?");
325 		String debugModeGeneratorPath = path.substring(0, idx);
326 		debugModeGeneratorPath = debugModeGeneratorPath.replaceAll("//.", "/");
327 		
328 		int jawrGenerationParamIdx = path.indexOf(JawrRequestHandler.GENERATION_PARAM);
329 		String parameter = path.substring(jawrGenerationParamIdx+JawrRequestHandler.GENERATION_PARAM.length()+1); // Add 1 for the '=' character 
330 		String key = matchPath(parameter);
331 		ResourceGenerator resourceGenerator = (ResourceGenerator)registry.get(key);
332 		String suffixPath = null;
333 		if(resourceGenerator instanceof SpecificCDNDebugPathResourceGenerator){
334 			suffixPath = ((SpecificCDNDebugPathResourceGenerator)resourceGenerator).getDebugModeBuildTimeGenerationPath(parameter);
335 		}else{
336 			suffixPath = parameter.replaceFirst(GeneratorRegistry.PREFIX_SEPARATOR, JawrConstant.URL_SEPARATOR);
337 		}
338 		return debugModeGeneratorPath+"/"+suffixPath;
339 	}

341 	/***
342 	 * Get the key from the mappings that corresponds to the specified path. 
343 	 * @param path the resource path
344 	 * @return the registry key corresponding to the path
345 	 */
346 	private String matchPath(String path) {
347 		String generatorKey = null;
348 		for(Iterator it = prefixRegistry.iterator();it.hasNext() && generatorKey == null;) {
349 			String prefix = (String) it.next();
350 			if(path.startsWith(prefix))
351 				generatorKey = prefix;			
352 		}	
353 		// Lazy load generator
354 		if(null != generatorKey && !registry.containsKey(generatorKey))
355 			loadGenerator(generatorKey);
356 		
357 		return generatorKey;
358 	}

369 	/***
370 	 * Returns the available variant for a bundle
371 	 * @param bundle the bundle
372 	 * @return the available variant for a bundle
373 	 */
374 	public Map getAvailableVariants(String bundle) {
375 		
376 		Map availableVariants = new TreeMap();
377 		String generatorKey = matchPath(bundle);
378 		if(generatorKey != null){
379 			ResourceGenerator generator = (ResourceGenerator) registry.get(generatorKey);
380 			if(generator instanceof VariantResourceGenerator){
381 				
382 				Map tempResult = ((VariantResourceGenerator)generator).getAvailableVariants(bundle.substring(generatorKey.length()));
383 				if(tempResult != null){
384 					availableVariants = tempResult;
385 				}
386 			}else if(generator instanceof LocaleAwareResourceGenerator){
387 				List availableLocales = ((LocaleAwareResourceGenerator)generator).getAvailableLocales(bundle.substring(generatorKey.length()));
388 				if(availableLocales != null){
389 					availableVariants = new HashMap();
390 					VariantSet variantSet = new VariantSet(JawrConstant.LOCALE_VARIANT_TYPE, "", availableLocales);
391 					availableVariants.put(JawrConstant.LOCALE_VARIANT_TYPE, variantSet);
392 				}
393 			}
394 		}
395 		
396 		return availableVariants;
397 	}

459 	/***
460 	 * Resolve the variants for the request passed in parameter
461 	 * @param request the request
462 	 * @return the map of variants defined in the request
463 	 */
464 	public Map resolveVariants(HttpServletRequest request){
465 		
466 		Map variants = new TreeMap();
467 		for (Iterator itVariantResolver = variantResolvers.values().iterator(); itVariantResolver.hasNext();) {
468 			VariantResolver resolver = (VariantResolver) itVariantResolver.next();
469 			String value = resolver.resolveVariant(request);
470 			if(value != null){
471 				variants.put(resolver.getVariantType(), value);
472 			}
473 		}
474 		
475 		return variants;
476 	}

479 	 * Returns the available variants.
480 	 * @param variants the current bundle variants
481 	 * @param curVariants the current variant
482 	 * @return the available variants
483 	 */
484 	public Map getAvailableVariantMap(Map variants, Map curVariants){
485 		
486 		Map availableVariantMap = new HashMap();
487 		for (Iterator iterator = curVariants.entrySet().iterator(); iterator.hasNext();) {
488 			Entry entry = (Entry) iterator.next();
489 			String variantType = (String) entry.getKey();
490 			String variant = (String) entry.getValue();
491 			
492 			if(variants.containsKey(variantType)){
493 				VariantResolver resolver = (VariantResolver) variantResolvers.get(variantType);
494 				if(resolver != null){
495 					variant = resolver.getAvailableVariant(variant, (VariantSet) variants.get(variantType));
496 				}
497 				
498 				availableVariantMap.put(variantType, variant);
499 			}
500 		}
501 		return availableVariantMap;
502 	}
