28  /***
29   * This class defines the Factory which creates a proxy to ResourceReader objects from a Resource generator.
30   *  
31   * @author Ibrahim Chaehoi
32   *
33   */
34  public class ResourceGeneratorReaderProxyFactory {
35  
36  	/***
37  	 * Creates a Resource reader from a resource generator.
38  	 * 
39  	 * @param generator the resource generator, which can be a text or an image resource generator.
40  	 * @param rsReaderHandler the resourceReaderHandler
41  	 * @param config the jawr config
42  	 * @return
43  	 */
44  	public static ResourceReader getResourceReaderProxy(PrefixedResourceGenerator generator, ResourceReaderHandler rsReaderHandler, JawrConfig config){
45  		
46  		ResourceReader proxy = null;
47  		
48  		// Defines the interfaces to be implemented by the ResourceReader 
49  		int nbExtraInterface = 0;
50  		Class[] extraInterfaces = new Class[2];
51  		boolean isResourceGenerator = generator instanceof ResourceGenerator;
52  		boolean isStreamResourceGenerator= generator instanceof StreamResourceGenerator;
53  		
54  		if(isResourceGenerator){
55  			extraInterfaces[nbExtraInterface++] = TextResourceReader.class;
56  		}
57  		if(isStreamResourceGenerator){
58  			extraInterfaces[nbExtraInterface++] = StreamResourceReader.class;
59  		}
60  		
61  		Class[] generatorInterfaces = getGeneratorInterfaces(generator);
62  		Class[] implementedInterfaces = new Class[generatorInterfaces.length+nbExtraInterface];
63  		for (int i = 0; i < generatorInterfaces.length; i++) {
64  			implementedInterfaces[i] = generatorInterfaces[i];
65  		}
66  		for (int i = 0; i < nbExtraInterface; i++) {
67  			implementedInterfaces[generatorInterfaces.length+i] = extraInterfaces[i];
68  		}
69  		
70  		// Creates the proxy
71  		InvocationHandler handler = new ResourceGeneratorReaderWrapperInvocationHandler(generator, rsReaderHandler, config);	
72  		proxy = (ResourceReader) Proxy.newProxyInstance(ResourceGeneratorReaderProxyFactory.class.getClassLoader(),
73  					implementedInterfaces,
74  	                handler);
75  		
76  		
77  		return proxy;
78  	}
79  
80  	/***
81  	 * Returns the array of interfaces implemented by the PrefixedResourceGenerator
82  	 * @param generator the generator
83  	 * @return the array of interfaces implemented by the PrefixedResourceGenerator
84  	 */
85  	private static Class[] getGeneratorInterfaces(
86  			PrefixedResourceGenerator generator) {
87  		
88  		Set interfaces = new HashSet();
89  		addInterfaces(generator, interfaces);
90  		return (Class[]) interfaces.toArray(new Class[]{});
91  	}
92  
93  	/***
94  	 * Adds all the interfaces of the object passed in parameter in the set of interfaces.
95  	 * @param obj the object
96  	 * @param interfaces the set of interfaces to update
97  	 */
98  	private static void addInterfaces(Object obj,
99  			Set interfaces) {
100 		
101 		Class[] generatorInterfaces = null;
102 		Class superClass = null;
103 		if(obj instanceof Class){
104 			generatorInterfaces = ((Class)obj).getInterfaces();
105 			superClass = ((Class)obj).getSuperclass();
106 		}else{
107 			generatorInterfaces = obj.getClass().getInterfaces();
108 			superClass = obj.getClass().getSuperclass();
109 		}
110 		for (int i = 0; i < generatorInterfaces.length; i++) {
111 			interfaces.add(generatorInterfaces[i]);
112 			addInterfaces(generatorInterfaces[i], interfaces);
113 		}
114 
115 		if(superClass != null && superClass != Object.class){
116 			addInterfaces(superClass, interfaces);
117 		}
118 	}

120 	/***
121 	 * This class defines the InvocationHandler which will handle the call to the ResourceReader proxy
122 	 * 
123 	 * @author Ibrahim Chaehoi
124 	 */
125 	private static class ResourceGeneratorReaderWrapperInvocationHandler implements InvocationHandler {
126 		
127 		/*** The resource generator wrapped */
128 		private PrefixedResourceGenerator generator;
129 		
130 		/*** The resource readre wrapper */
131 		private ResourceGeneratorReaderWrapper rsReaderWrapper;
132 		
133 		/*** The resource readre wrapper */
134 		private StreamResourceGeneratorReaderWrapper streamRsReaderWrapper;
135 		
136 		/*** The resource reader methods */
137 		private static Method[] resourceReaderMethods = TextResourceReader.class.getMethods();
138 		
139 		/*** The stream resource reader methods */
140 		private static Method[] streamResourceReaderMethods = StreamResourceReader.class.getMethods();
141 		
142 		/***
143 		 * Constructor
144 		 * @param generator the generator
145 		 * @param rsReaderHandler the resourceReaderHandler
146 		 * @param config the jawr config
147 		 */
148 		public ResourceGeneratorReaderWrapperInvocationHandler(
149 				PrefixedResourceGenerator generator, ResourceReaderHandler rsReaderHandler, JawrConfig config) {
150 			this.generator = generator;
151 			if(generator instanceof ResourceGenerator){
152 				this.rsReaderWrapper = new ResourceGeneratorReaderWrapper((ResourceGenerator)generator, rsReaderHandler, config);
153 			}
154 			if(generator instanceof StreamResourceGenerator){
155 				this.streamRsReaderWrapper = new StreamResourceGeneratorReaderWrapper((StreamResourceGenerator)generator, rsReaderHandler, config);
156 			}
157 		}
158 
159 		/* (non-Javadoc)
160 		 * @see java.lang.reflect.InvocationHandler#invoke(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
161 		 */
162 		public Object invoke(Object proxy, Method method, Object[] args)
163 				throws Throwable {
164 			
165 			Object result = null;
166 			if(rsReaderWrapper != null && methodBelongsTo(method, resourceReaderMethods)){
167 				result = method.invoke(rsReaderWrapper, args);
168 			
169 			}else if(streamRsReaderWrapper != null && methodBelongsTo(method, streamResourceReaderMethods)){
170 				result = method.invoke(streamRsReaderWrapper, args);
171 			
172 			}else{
173 				result = method.invoke(generator, args);
174 			}
175 			return result;
176 		}
177 		
178 		/***
179 		 * Returns true if the method is in the method array.
180 		 * @param m the method
181 		 * @param methods the method array
182 		 * @return  true if the method is in the method array.
183 		 */
184 		private boolean methodBelongsTo(Method m, Method[] methods){
185 			boolean result = false;
186 			
187 			for (int i = 0; i < methods.length && !result; i++) {
188 				if(methods[i].equals(m)){
189 					result = true;
190 				}
191 			}
192 			
193 			return result;
194 		}
195 	}
