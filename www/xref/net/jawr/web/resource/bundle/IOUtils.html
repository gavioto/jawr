28  	// the buffer size for reading data
29  	private static final int BUFFER_SIZE = 16384;

31  	/***
32  	 * Writes all the contents of a Reader to a Writer.
33  	 * 
34  	 * @param reader the reader to read from
35  	 * @param writer the writer to write to
36  	 */
37  	public static void copy(Reader reader, Writer writer) throws IOException {
38  		char[] buf = new char[BUFFER_SIZE];
39  		int num = 0;
40  
41  		while ((num = reader.read(buf, 0, buf.length)) != -1) {
42  			writer.write(buf, 0, num);
43  		}
44  	}

52  	public static void copy(InputStream input, OutputStream output)
53  			throws IOException {
54  		byte[] buf = new byte[BUFFER_SIZE];

73  	/***
74  	 * Writes all the contents of an InputStream to an OutStream.
75  	 * 
76  	 * @param input the input stream to read from
77  	 * @param output the output stream to write to
78  	 */
79  	public static void copy(InputStream input, OutputStream output,
80  			boolean closeStreams) throws IOException {
81  		try {
82  			copy(input, output);
83  		} finally {
84  			if (closeStreams) {
85  				close(input);
86  				close(output);
87  			}
88  		}
89  	}

91  	/***
92  	 * Copy the readable byte channel to the writable byte channel
93  	 * 
94  	 * @param inChannel the readable byte channel
95  	 * @param outChannel the writable byte channel
96  	 * @throws IOException if an IOException occurs.
97  	 */
98  	public static void copy(ReadableByteChannel inChannel,
99  			WritableByteChannel outChannel) throws IOException {
100 
101 		if (inChannel instanceof FileChannel) {
102 			((FileChannel) inChannel).transferTo(0, ((FileChannel) inChannel)
103 					.size(), outChannel);
104 		} else {
105 
106 			final ByteBuffer buffer = ByteBuffer.allocateDirect(16 * 1024);
107 			try {
108 
109 				while (inChannel.read(buffer) != -1) {
110 					// prepare the buffer to be drained
111 					buffer.flip();
112 					// write to the channel, may block
113 					outChannel.write(buffer);
114 					// If partial transfer, shift remainder down
115 					// If buffer is empty, same as doing clear()
116 					buffer.compact();
117 				}
118 				// EOF will leave buffer in fill state
119 				buffer.flip();
120 				// make sure the buffer is fully drained.
121 				while (buffer.hasRemaining()) {
122 					outChannel.write(buffer);
123 				}
124 			} finally {
125 				IOUtils.close(inChannel);
126 				IOUtils.close(outChannel);
127 			}
128 		}
129 	}
