36  /***
37   * This class defines the Post processor which handle the inclusion of the CSS define with @import statement
38   * 
39   * @author Ibrahim Chaehoi
40   * 
41   */
42  public class CSSImportPostProcessor extends
43  		AbstractChainedResourceBundlePostProcessor {
44  
45  	/*** The url pattern */
46  	private static final Pattern IMPORT_PATTERN = Pattern.compile(	"@import//s*url//(//s*" // 'url(' and any number of whitespaces 
47  																+ "[\"']?([^\"']*)[\"']?" // any sequence of characters, except an unescaped ')'
48  																+ "//s*//)//s*(//w+)?//s*;?",  // Any number of whitespaces, then ')'
49  																Pattern.CASE_INSENSITIVE); // works with 'URL('
50  	
51  	/***
52  	 * Constructor
53  	 * @param id the Id of the post processor
54  	 */
55  	public CSSImportPostProcessor() {
56  		super(PostProcessFactoryConstant.CSS_IMPORT);
57  	}
58  
59  	/*
60  	 * (non-Javadoc)
61  	 * 
62  	 * @see
63  	 * net.jawr.web.resource.bundle.postprocess.AbstractChainedResourceBundlePostProcessor#doPostProcessBundle(net.jawr.web.resource.bundle.postprocess
64  	 * .BundleProcessingStatus, java.lang.StringBuffer)
65  	 */
66  	protected StringBuffer doPostProcessBundle(BundleProcessingStatus status,
67  			StringBuffer bundleData) throws IOException {
68  
69  		String data = bundleData.toString();
70  		
71  		// Rewrite each css url path
72  		Matcher matcher = IMPORT_PATTERN.matcher(data);
73  		StringBuffer sb = new StringBuffer();
74  		while(matcher.find()) {
75  		
76  			String content = getCssPathContent(matcher.group(1), matcher.group(2), status);
77  			matcher.appendReplacement(sb, RegexUtil.adaptReplacementToMatcher(content));
78  		}
79  		matcher.appendTail(sb);
80  		return sb;
81  		
82  	}
83  
84  	/***
85  	 * Retrieve the content of the css to import
86  	 * @param cssPathToImport the path of the css to import
87  	 * @param media the media
88  	 * @param status the bundle processing status
89  	 * @return the content of the css to import
90  	 * @throws IOException if an IOException occurs
91  	 */
92  	private String getCssPathContent(String cssPathToImport, String media, BundleProcessingStatus status) throws IOException {
93  		
94  		String currentCssPath = status.getLastPathAdded();
95  		
96  		String path = cssPathToImport;
97  
98  		JawrConfig jawrConfig = status.getJawrConfig();
99  		if(!cssPathToImport.startsWith("/") && !jawrConfig.getGeneratorRegistry().isPathGenerated(path)){ // relative URL
100 			path = PathNormalizer.concatWebPath(currentCssPath, cssPathToImport);
101 		}
102 		
103 		Reader reader = null;
104 		
105 		try {
106 			reader = status.getRsReader().getResource(path, true);
107 		} catch (ResourceNotFoundException e) {
108 			throw new IOException("Css to import '"+path+"' was not found");
109 		}
110 		
111 		StringWriter content = new StringWriter();
112 		IOUtils.copy(reader, content, true);
113 		
114 		// Retrieve the image servlet mapping
115 		ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) jawrConfig.getContext().getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
116 		if(imgRsHandler != null){
117 			jawrConfig = imgRsHandler.getJawrConfig();
118 		}
119 		// Rewrite image URL
120 		CssImportedUrlRewriter urlRewriter = new CssImportedUrlRewriter(jawrConfig);
121 		StringBuffer result = new StringBuffer();
122 		boolean isMediaAttributeSet = StringUtils.isNotEmpty(media);
123 		if(isMediaAttributeSet){
124 			result.append("@media "+media+" {\n");
125 		}
126 		result.append(urlRewriter.rewriteUrl(path, currentCssPath, content.getBuffer().toString()));
127 		if(isMediaAttributeSet){
128 			result.append("\n}\n");
129 		}
130 		return result.toString();
131 	}

133 	/***
134 	 * This class rewrite the image URL for the imported CSS.
135 	 * 
136 	 * @author Ibrahim Chaehoi
137 	 */
138 	private static class CssImportedUrlRewriter extends CssImageUrlRewriter {
139 		
140 		/*** The generator registry */
141 		private GeneratorRegistry generatorRegistry;
142 		
143 		/***
144 		 * Constructor
145 		 * @param generatorRegistry the generator registry
146 		 */
147 		public CssImportedUrlRewriter(JawrConfig jawrConfig) {
148 
149 			super(jawrConfig);
150 			this.generatorRegistry = jawrConfig.getGeneratorRegistry();
151 		}
152 
153 		
154 		/* (non-Javadoc)
155 		 * @see net.jawr.web.resource.bundle.css.CssImageUrlRewriter#getRewrittenImagePath(java.lang.String, java.lang.String, java.lang.String)
156 		 */
157 		protected String getRewrittenImagePath(String originalCssPath,
158 				String newCssPath, String url) throws IOException {
159 			
160 			String currentPath = originalCssPath;
161 			
162 			String imgPath = PathNormalizer.concatWebPath(currentPath, url);
163 			if(!generatorRegistry.isGeneratedImage(imgPath) && !generatorRegistry.isHandlingCssImage(originalCssPath)){
164 				imgPath = PathNormalizer.getRelativeWebPath(PathNormalizer
165 						.getParentPath(newCssPath), imgPath);
166 			}
167 			
168 			return imgPath;
169 		}
170 	}
