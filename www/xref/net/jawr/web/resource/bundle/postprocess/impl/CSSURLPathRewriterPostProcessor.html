34  /***
35   * Single file postprocessor used to rewrite CSS URLs according to the new relative locations of the references when
36   * added to a bundle. Since the path changes, the URLs must be rewritten accordingly.  
37   * URLs in css files are expected to be according to the css spec (see http://www.w3.org/TR/REC-CSS2/syndata.html#value-def-uri). 
38   * Thus, single double, or no quotes enclosing the url are allowed (and remain as they are after rewriting). Escaped parens and quotes 
39   * are allowed within the url.   
40   *  
41   * @author Jordi Hernández Sellés
42   * @author Ibrahim Chaehoi
43   */
44  public class CSSURLPathRewriterPostProcessor extends
45  		AbstractChainedResourceBundlePostProcessor {
46

48  	/*** The URL separator */
49  	private static final String URL_SEPARATOR = "/";
50  
51  	/*** The cache buster separator */
52  	private static final String OLD_CLASSPATH_CSS_PREFIX = GeneratorRegistry.OLD_CLASSPATH_CSS_BUNDLE_PREFIX+GeneratorRegistry.PREFIX_SEPARATOR;
53  
54  	/*** The url pattern */
55  	private static final Pattern urlPattern = Pattern.compile(	"url//(//s*" // 'url(' and any number of whitespaces 
56  																+ "((//////))|[^)])*" // any sequence of characters, except an unescaped ')'
57  																+ "//s*//)",  // Any number of whitespaces, then ')'
58  																Pattern.CASE_INSENSITIVE); // works with 'URL('
59  	
60  	/*** The back reference regexp */
61  	private static final String backRefRegex = "(//.//./)";
62  	
63  	/*** The back reference regexp pattern */
64  	private static final Pattern backRefRegexPattern = Pattern.compile(backRefRegex);
65  	
66  	/*** The URL separator pattern */
67  	private static final Pattern URLSeparatorPattern = Pattern.compile(URL_SEPARATOR);
68  	
69  	/* (non-Javadoc)
70  	 * @see net.jawr.web.resource.bundle.postprocess.impl.AbstractChainedResourceBundlePostProcessor#doPostProcessBundle(net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus, java.lang.StringBuffer)
71  	 */
72  	protected StringBuffer doPostProcessBundle(BundleProcessingStatus status,
73  			StringBuffer bundleData) throws IOException {
74  		
75  		String data = bundleData.toString();
76  		
77  		// Initial backrefs (number of backtracking paths needed, i.e. '../').
78  		int initialBackRefs = 1;
79  		
80  		JawrConfig jawrConfig = status.getJawrConfig();
81  		if(! "".equals(jawrConfig.getServletMapping())){
82  			StringTokenizer tk = new StringTokenizer(jawrConfig.getServletMapping(),URL_SEPARATOR);
83  			initialBackRefs += tk.countTokens();
84  		}
85  		
86  		String bundleName = status.getCurrentBundle().getName();
87  		initialBackRefs += numberOfForwardReferences(bundleName);
88  		List resourceBackRefs = getPathNamesInResource(status.getLastPathAdded());
89  		Matcher matcher = urlPattern.matcher(data);
90  		StringBuffer sb = new StringBuffer();
91  		while(matcher.find()) {
92  			List backRefs = new ArrayList();
93  			backRefs.addAll(resourceBackRefs);
94  			
95  			String url = getUrlPath(matcher.group(), initialBackRefs, backRefs, status);
96  			
97  			matcher.appendReplacement(sb, RegexUtil.adaptReplacementToMatcher(url));
98  		}
99  		matcher.appendTail(sb);
100 		return sb;
101 	}

104 	/***
105 	 * Transform a matched url so it points to the proper relative path with respect to the given path.  
106 	 * @param match the matched URL
107 	 * @param initialBackRefs the initial backward references
108 	 * @param resourceBackRefs the list of resources backward references
109 	 * @param status the bundle processing status
110 	 * @return the image URL path
111 	 * @throws IOException if an IO exception occurs
112 	 */
113 	private String getUrlPath(String match, int initialBackRefs, List resourceBackRefs, BundleProcessingStatus status) throws IOException {
114 
115 		JawrConfig jawrConfig = status.getJawrConfig();
116 		String imagePathOverride = jawrConfig.getCssImagePathOverride();
117 		boolean useClassPathCssImgServlet = jawrConfig.isUsingClasspathCssImageServlet();
118 		ImageResourcesHandler imgRsHandler = (ImageResourcesHandler) jawrConfig.getContext().getAttribute(JawrConstant.IMG_CONTEXT_ATTRIBUTE);
119 		String classPathImgServletPath = "";
120 		
121 		if(imgRsHandler != null){
122 			classPathImgServletPath = imgRsHandler.getJawrConfig().getServletMapping();
123 		}
124 		
125 		String url = match.substring(match.indexOf('(')+1,match.lastIndexOf(')'))
126 					.trim();
127 
128 		// To keep quotes as they are, first they are checked and removed. 
129 		String quoteStr = "";
130 		if( url.startsWith("'") || url.startsWith("\"")) {
131 			quoteStr = url.charAt(0)+"";
132 			url = url.substring(1,url.length()-1);
133 		}
134 		
135 		// Check if the URL is absolute, if it is return it as is. 
136 		int firstSlash = url.indexOf('/');
137 		if(0 == firstSlash || (firstSlash != -1 && url.charAt(++firstSlash) == '/')){
138 			StringBuffer sb = new StringBuffer("url(");
139 			sb.append(quoteStr).append(url).append(quoteStr).append(")");
140 			return sb.toString();
141 		}
142 		
143 		// Check if the URL is embedded data (RFC2397), if it is return it as is
144 		if(url.trim().toLowerCase().startsWith("data:")) {
145 			StringBuffer sb = new StringBuffer("url(");
146 			sb.append(quoteStr).append(url).append(quoteStr).append(")");
147 			return sb.toString();
148 		}
149 		
150 		int backRefsInURL = 0;
151 		// Remove leading slash
152 		if(url.startsWith("../")) {
153 			Matcher backUrls = backRefRegexPattern.matcher(url);
154 			while(backUrls.find())
155 				backRefsInURL++;
156 			url = url.replaceAll(backRefRegex, "");
157 		}
158 		else if(url.startsWith(URL_SEPARATOR))
159 			url = url.substring(1,url.length());
160 		else if(url.startsWith("./"))
161 			url = url.substring(2,url.length());
162 		
163 		// append the override url here if need be
164 		if(imagePathOverride != null){
165 			StringBuffer sb = new StringBuffer("url(");
166 			sb.append(quoteStr).append(imagePathOverride).append(url).append(quoteStr).append(")");
167 			return sb.toString();
168 		}
169 		
170 		// Adjust the forward pathnames according to the backrefs in the url
171 		if(backRefsInURL > 0 && resourceBackRefs.size() > 0) {
172 			int size = resourceBackRefs.size();
173 			int counter = 0;
174 			while(size > 0) {
175 				counter++;
176 				if(counter >= 50)
177 					throw new RuntimeException("Infinite loop on url: " + url + " :size:" + size);
178 				size--;
179 				backRefsInURL--;
180 				resourceBackRefs.remove(size);
181 				if(backRefsInURL == 0)
182 					break;
183 			}
184 		}		
185 		
186 		String root = "";
187 		if(!resourceBackRefs.isEmpty()){
188 			root = (String) resourceBackRefs.get(0);
189 		}
190 		boolean classpathCss = !resourceBackRefs.isEmpty()
191 		&& (root.startsWith(JawrConstant.CLASSPATH_RESOURCE_PREFIX) || 
192 			root.startsWith(OLD_CLASSPATH_CSS_PREFIX)) && useClassPathCssImgServlet;
193 		
194 		if (classpathCss) {
195 			
196 			int idx = root.indexOf(GeneratorRegistry.PREFIX_SEPARATOR);
197 			resourceBackRefs.set(0, root.substring(idx+1));
198 			
199 			// If we are in Debug mode, the Jawr CSS generator will be used.
200 			// As the path of this generator is define as root level,
201 			// we remove the initial backreference.
202 			if(jawrConfig.isDebugModeOn()){
203 				initialBackRefs = 0;
204 			}
205 		}
206 		
207 		// Start rendering the result, starting by the initial quote, if any. 
208 		StringBuffer urlPrefix = new StringBuffer("url(").append(quoteStr);
209 		
210 		// Add the back references as needed
211 		for (int i = 0; i < initialBackRefs; i++) {
212 			urlPrefix.append("../");
213 		}
214 		
215 		if (classpathCss) {
216 			// If path starts with "/", remove it
217 			String servletPath = classPathImgServletPath;
218 			if(!"".equals(servletPath)){
219 				if(servletPath.startsWith(URL_SEPARATOR) && servletPath.length() >= 1){
220 					servletPath = servletPath.substring(1);
221 				}
222 				// Add image servlet path in the URL
223 				urlPrefix.append(servletPath+URL_SEPARATOR);
224 			}
225 			
226 		}
227 		
228 		StringBuffer urlBuffer = new StringBuffer();
229 		for(Iterator it = resourceBackRefs.iterator();it.hasNext(); ) {
230 			urlBuffer.append(it.next()).append(URL_SEPARATOR);			
231 		}
232 		urlBuffer.append(url);
233 		if (classpathCss) {
234 			url = addCacheBuster(status, urlBuffer.toString());
235 			if(url.startsWith("/")){
236 				url = url.substring(1);
237 			}
238 		}else{
239 			url = urlBuffer.toString();
240 		}
241 		
242 		return PathNormalizer.normalizePath(urlPrefix.append(url).append(quoteStr).append(")").toString());
243 	}

246 	/***
247 	 * Adds the cache buster to the CSS image
248 	 * @param status the bundle processing status
249 	 * @param url the URL
250 	 * @return the url of the CSS image with a cache buster
251 	 * @throws IOException if an IO exception occurs
252 	 */
253 	private String addCacheBuster(BundleProcessingStatus status, String url) throws IOException {
254 		
255 		String newUrl = status.getImageMapping(url);
256 		 
257 		if(newUrl != null){
258 			return newUrl;
259 		}
260 		
261 		newUrl = CheckSumUtils.getCacheBustedUrl(url, status.getRsHandler(), status.getJawrConfig(), true);
262 		
263 		// Set the result in a map, so we will not search it the next time
264 		status.setImageMapping(url, newUrl);
265 		
266 		return newUrl;
267 	}

272 	/***
273 	 * Gets the path names within a resource URL, excluding the resource file name. 
274 	 * @param resourceURL the resource URL
275 	 * @return the list of path names within a resource URL, excluding the resource file name. 
276 	 */
277 	private List getPathNamesInResource(String resourceURL) {
278 		List names = new ArrayList();
279 		String[] namesArray = resourceURL.split(URL_SEPARATOR);
280 		
281 		// Add all but the last pathname
282 		for (int i = 0; i < namesArray.length -1 ; i++) {
283 			if(!"".equals(namesArray[i]))
284 				names.add(namesArray[i]);
285 		}
286 		
287 		return names;
288 	}

290 	/***
291 	 * Find the number of occurrences of / within a url. 
292 	 * @param url the url
293 	 * @return the number of occurrences of / within a url. 
294 	 */
295 	private int numberOfForwardReferences(String url) {
296 		Matcher backUrls = URLSeparatorPattern.matcher(url);
297 		int numMatches = 0;
298 		while(backUrls.find())
299 			numMatches++;
300 		
301 		if(url.startsWith(URL_SEPARATOR))
302 			numMatches--;
303 		
304 		return numMatches;
305 	}
