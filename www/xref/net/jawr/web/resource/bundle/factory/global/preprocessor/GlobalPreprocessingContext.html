14  package net.jawr.web.resource.bundle.factory.global.preprocessor;
15  
16  import net.jawr.web.config.JawrConfig;
17  import net.jawr.web.resource.bundle.global.processor.AbstractGlobalProcessingContext;
18  import net.jawr.web.resource.handler.reader.ResourceReaderHandler;
19  
20  /**
21   * This class defines the context for global postprocessing.
22   *  
23   * @author Ibrahim Chaehoi
24   */
25  public class GlobalPreprocessingContext extends AbstractGlobalProcessingContext {
26  
27  	/** The resource handler */
28  	private ResourceReaderHandler rsHandler;
29  
30  	/**
31  	 * Constructor
32  	 * @param config the jawr config
33  	 * @param bundleHandler the bundle handler
34  	 * @param resourceHandler the resource handler
35  	 * @param processBundle the process bundle flag
36  	 */
37  	public GlobalPreprocessingContext(JawrConfig config,
38  			ResourceReaderHandler resourceHandler, boolean processBundle) {
39  		super(config, processBundle);
40  		this.rsHandler = resourceHandler;
41  	}
42  
43  	/**
44  	 * Returns the resource reader handler.
45  	 * @return the resource reader Handler
46  	 */
47  	public ResourceReaderHandler getRsReaderHandler() {
48  		return rsHandler;
49  	}
50  	
51  	/**
52  	 * Sets the resource handler 
53  	 * @param rsHandler the rsHandler to set
54  	 */
55  	public void setRsReaderHandler(ResourceReaderHandler rsHandler) {
56  		this.rsHandler = rsHandler;
57  	}
58  	
59  }
