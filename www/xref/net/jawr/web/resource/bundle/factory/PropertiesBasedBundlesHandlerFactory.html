60  	/***
61  	 * Create a PropertiesBasedBundlesHandlerFactory using the specified
62  	 * properties.
63  	 * 
64  	 * @param properties the properties
65  	 * @param resourceType
66  	 *            js or css
67  	 * @param rsHandler
68  	 *            ResourceHandler to access files.
69  	 */
70  	public PropertiesBasedBundlesHandlerFactory(Properties properties,
71  			String resourceType, ResourceHandler rsHandler,
72  			GeneratorRegistry generatorRegistry) {
73  		this.props = new PropertiesConfigHelper(properties, resourceType);
74  
75  		// Create the BundlesHandlerFactory
76  		factory = new BundlesHandlerFactory();
77  		factory.setResourceHandler(rsHandler);
78  		factory.setBundlesType(resourceType);
79  
80  		// Root resources dir
81  		factory.setBaseDir(props.getProperty(PropertiesBundleConstant.RESOURCES_BASEDIR, "/"));
82  
83  		// Use cache by default
84  		factory.setUseInMemoryCache(Boolean.valueOf(
85  				props.getProperty(PropertiesBundleConstant.RESOURCES_USE_CACHE, "true")).booleanValue());
86  
87  		// Postprocessor definitions
88  		factory.setGlobalPostProcessorKeys(props
89  				.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_POSTPROCESSOR));
90  		factory.setUnitPostProcessorKeys(props
91  				.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_FILE_POSTPROCESSOR));
92  
93  		// Single or multiple bundle for orphans settings.
94  		factory.setUseSingleResourceFactory(Boolean.valueOf(
95  				props.getProperty(PropertiesBundleConstant.FACTORY_USE_SINGLE_BUNDLE, "false"))
96  				.booleanValue());
97  		factory.setSingleFileBundleName(props
98  				.getProperty(PropertiesBundleConstant.FACTORY_SINGLE_FILE_NAME));
99  		
100 		// Use orphans resolution at all, on by default. FACTORY_PROCESS_ORPHANS
101 		factory.setScanForOrphans(Boolean.valueOf(
102 				props.getProperty(PropertiesBundleConstant.FACTORY_PROCESS_ORPHANS, "true")).booleanValue());
103 
104 		// Use the automatic directory-as-bundle mapper.
105 		factory.setUseDirMapperFactory(Boolean.valueOf(
106 				props.getProperty(PropertiesBundleConstant.FACTORY_USE_DIR_MAPPER, "false"))
107 				.booleanValue());
108 		factory.setExludedDirMapperDirs(props
109 				.getPropertyAsSet(PropertiesBundleConstant.FACTORY_DIR_MAPPER_EXCLUSION));
110 
111 		// Initialize custom generators
112 		Iterator generators = props.getPropertyAsSet(PropertiesBundleConstant.CUSTOM_GENERATORS)
113 				.iterator();
114 		while (generators.hasNext()) {
115 			String generatorClass = (String) generators.next();
116 			generatorRegistry.registerGenerator(generatorClass);
117 		}
118 
119 		// Initialize custom bundles
120 		Set customBundles = new HashSet();
121 		// Check if we should use the bundle names property or
122 		// find the bundle name using the bundle id declaration :
123 		// jawr.<type>.bundle.<name>.id
124 		if(null != props.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_NAMES)) {
125 			StringTokenizer tk = new StringTokenizer(props
126 					.getProperty(PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_NAMES), ",");
127 			while (tk.hasMoreTokens()) {
128 				customBundles.add(buildCustomBundleDefinition(tk.nextToken()
129 						.trim(), false, generatorRegistry));
130 			}
131 		} else {
132 			Iterator bundleNames = props.getPropertyBundleNameSet().iterator();
133 			while (bundleNames.hasNext()) {
134 				customBundles.add(buildCustomBundleDefinition(
135 						(String) bundleNames.next(), false, generatorRegistry));
136 			}
137 		}
138 		
139 		factory.setBundleDefinitions(customBundles);
140 
141 		// Check if we should use the custom postprocessor names property or
142 		// find the postprocessor name using the postprocessor class declaration :
143 		// jawr.custom.postprocessors.<name>.class
144 		Map customPostprocessors = new HashMap();
145 		if (null != properties.getProperty(PropertiesBundleConstant.CUSTOM_POSTPROCESSORS
146 				+ PropertiesBundleConstant.CUSTOM_POSTPROCESSORS_NAMES)) {
147 			StringTokenizer tk = new StringTokenizer(properties
148 					.getProperty(PropertiesBundleConstant.CUSTOM_POSTPROCESSORS
149 							+ PropertiesBundleConstant.CUSTOM_POSTPROCESSORS_NAMES), ",");
150 
151 			while (tk.hasMoreTokens()) {
152 				String processorKey = tk.nextToken();
153 				String processorClass = properties
154 						.getProperty(PropertiesBundleConstant.CUSTOM_POSTPROCESSORS + "." + processorKey
155 								+ PropertiesBundleConstant.CUSTOM_POSTPROCESSORS_CLASS);
156 				if (null != processorClass)
157 					customPostprocessors.put(processorKey, processorClass);
158 			}
159 		}else{
160 			
161 			customPostprocessors = props.getCustomPostProcessorMap();
162 		}
163 		
164 		factory.setCustomPostprocessors(customPostprocessors);
165 		
166 	}

183 	/***
184 	 * Create a BundleDefinition based on the properties file.
185 	 * 
186 	 * @param bundleName the bundle name
187 	 * @param generatorRegistry the generator registry
188 	 * @return the bundleDefinition
189 	 */
190 	private ResourceBundleDefinition buildCustomBundleDefinition(
191 			String bundleName, boolean isChildBundle, GeneratorRegistry generatorRegistry) {
192 
193 		// Id for the bundle
194 		String bundleId = props.getCustomBundleProperty(bundleName,
195 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_ID);
196 		if (null == bundleId && !isChildBundle)
197 			throw new IllegalArgumentException(
198 					"No id defined for the bundle with name:" + bundleName
199 							+ ". Please specify one in configuration. ");
200 
201 		// Wether it's a composite or not
202 		boolean isComposite = Boolean.valueOf(
203 				props.getCustomBundleProperty(bundleName,
204 						PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_COMPOSITE_FLAG, "false"))
205 				.booleanValue();
206 
207 		// Create definition and set its id
208 		ResourceBundleDefinition bundle = new ResourceBundleDefinition();
209 		bundle.setBundleId(bundleId);
210 		bundle.setBundleName(bundleName);
211 		
212 		// Wether it's global or not
213 		Boolean isGlobal = Boolean.valueOf(props.getCustomBundleProperty(
214 				bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_GLOBAL_FLAG, "false"));
215 		bundle.setGlobal(isGlobal.booleanValue());
216 
217 		// Set order if its a global bundle
218 		if (isGlobal.booleanValue()) {
219 			Integer order = Integer.valueOf(props.getCustomBundleProperty(
220 					bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_ORDER, "0"));
221 			bundle.setInclusionOrder(order.intValue());
222 		}
223 
224 		// Override bundle postprocessor
225 		if (null != props.getCustomBundleProperty(bundleName,
226 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_POSTPROCESSOR))
227 			bundle.setBundlePostProcessorKeys(props.getCustomBundleProperty(
228 					bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_POSTPROCESSOR));
229 
230 		// Override unitary postprocessor
231 		if (null != props.getCustomBundleProperty(bundleName,
232 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_FILE_POSTPROCESSOR))
233 			bundle.setUnitaryPostProcessorKeys(props.getCustomBundleProperty(
234 					bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_FILE_POSTPROCESSOR));
235 
236 		// Use only with debug mode on
237 		Boolean isDebugOnly = Boolean.valueOf(props.getCustomBundleProperty(
238 				bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEBUGONLY, "false"));
239 		bundle.setDebugOnly(isDebugOnly.booleanValue());
240 
241 		// Use only with debug mode off
242 		Boolean isDebugNever = Boolean.valueOf(props.getCustomBundleProperty(
243 				bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_DEBUGNEVER, "false"));
244 		bundle.setDebugNever(isDebugNever.booleanValue());
245 
246 		// Set conditional comment for IE, in case one is specified
247 		if (null != props.getCustomBundleProperty(bundleName,
248 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_IE_CONDITIONAL_EXPRESSION))
249 			bundle.setIeConditionalExpression(props
250 					.getCustomBundleProperty(bundleName,
251 							PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_IE_CONDITIONAL_EXPRESSION));
252 
253 		// Sets the alternate URL for production mode. 
254 		if (null != props.getCustomBundleProperty(bundleName,
255 				PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_PRODUCTION_ALT_URL))
256 			bundle.setAlternateProductionURL(props.getCustomBundleProperty(bundleName,
257 					PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_PRODUCTION_ALT_URL));
258 			
259 		
260 		if (isComposite) {
261 			String childBundlesProperty = props.getCustomBundleProperty(
262 					bundleName, PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_COMPOSITE_NAMES);
263 			if (null == childBundlesProperty)
264 				throw new IllegalArgumentException(
265 						"No child bundle names were defined for the composite bundle with name:"
266 								+ bundleName
267 								+ ". Please specify at least one in configuration. ");
268 
269 			bundle.setComposite(true);
270 
271 			// add children
272 			List children = new ArrayList();
273 			StringTokenizer tk = new StringTokenizer(childBundlesProperty, ",");
274 			while (tk.hasMoreTokens()) {
275 				ResourceBundleDefinition childDef = buildCustomBundleDefinition(
276 						tk.nextToken().trim(), true, generatorRegistry);
277 				childDef.setBundleId(bundleId);
278 				children.add(childDef);
279 			}
280 			bundle.setChildren(children);
281 		} else {
282 			String mappingsProperty = props.getCustomBundleProperty(bundleName,
283 					PropertiesBundleConstant.BUNDLE_FACTORY_CUSTOM_MAPPINGS);
284 			if (null == mappingsProperty)
285 				throw new IllegalArgumentException(
286 						"No mappings were defined for the bundle with name:"
287 								+ bundleName
288 								+ ". Please specify at least one in configuration. ");
289 
290 			// Add the mappings
291 			List mappings = new ArrayList();
292 			Set localeKeys = new HashSet();
293 			StringTokenizer tk = new StringTokenizer(mappingsProperty, ",");
294 			while (tk.hasMoreTokens()){
295 				String mapping = tk.nextToken().trim();
296 				mappings.add(mapping);
297 				// Add local variants
298 				if(generatorRegistry.isMessageResourceGenerator(mapping)){
299 					int idx = mapping.indexOf(GeneratorRegistry.PREFIX_SEPARATOR);
300 					String msgBundle = mapping.substring(idx+1);
301 					localeKeys.addAll(LocaleUtils.getAvailableLocaleSuffixesForBundle(msgBundle));
302 				}
303 			}
304 			bundle.setMappings(mappings);
305 			bundle.setLocaleVariantKeys(Collections.list(Collections.enumeration(localeKeys)));
306 		}
307 
308 		return bundle;
309 	}
