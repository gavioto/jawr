108 	/***
109 	 * Build a ResourceBundlesHandler. Must be invoked after setting at least the ResourceHandler.
110 	 * 
111 	 * @param jawrConfig the jawr config
112 	 * @return the resource bundles handler
113 	 * @throws DuplicateBundlePathException if two bundles are defined with the same path
114 	 */
115 	public ResourceBundlesHandler buildResourceBundlesHandler()
116 			throws DuplicateBundlePathException {
117 		if (log.isInfoEnabled())
118 			log.info("Building resources handler... ");
119 
120 		// Ensure state is correct
121 		if (null == jawrConfig)
122 			throw new IllegalStateException(
123 					"Must set the JawrConfig for this factory before invoking buildResourceBundlesHandler(). ");
124 
125 		if (null == resourceHandler)
126 			throw new IllegalStateException(
127 					"Must set the resourceHandler for this factory before invoking buildResourceBundlesHandler(). ");
128 		if (useSingleResourceFactory && null == singleFileBundleName)
129 			throw new IllegalStateException(
130 					"Must set the singleFileBundleName when useSingleResourceFactory is set to true. Please check the documentation. ");
131 
132 		// Initialize custom postprocessors before using the factory to build the postprocessing chains
133 		if (null != customPostprocessors)
134 			chainFactory.setCustomPostprocessors(customPostprocessors);
135 
136 		// List of bundles
137 		List resourceBundles = new ArrayList();
138 
139 		boolean processBundle = !jawrConfig.getUseBundleMapping()
140 				|| !resourceHandler.isExistingMappingFile();
141 		if (processBundle) {
142 			initResourceBundles(resourceBundles);
143 		} else {
144 			initResourceBundlesFromFullMapping(resourceBundles);
145 		}
146 
147 		// Build the postprocessor for bundles
148 		ResourceBundlePostProcessor processor = null;
149 		if (null == this.globalPostProcessorKeys)
150 			processor = this.chainFactory.buildDefaultProcessorChain();
151 		else
152 			processor = this.chainFactory
153 					.buildPostProcessorChain(globalPostProcessorKeys);
154 
155 		// Build the postprocessor to use on resources before adding them to the bundle.
156 		ResourceBundlePostProcessor unitProcessor = null;
157 		if (null == this.unitPostProcessorKeys)
158 			unitProcessor = this.chainFactory.buildDefaultUnitProcessorChain();
159 		else
160 			unitProcessor = this.chainFactory
161 					.buildPostProcessorChain(unitPostProcessorKeys);
162 
163 		// Build the handler
164 		ResourceBundlesHandler collector = new ResourceBundlesHandlerImpl(
165 				resourceBundles, resourceHandler, jawrConfig, processor,
166 				unitProcessor);
167 
168 		// Use the cached proxy if specified when debug mode is off.
169 		if (useInMemoryCache && !jawrConfig.isDebugModeOn())
170 			collector = new CachedResourceBundlesHandler(collector);
171 
172 		collector.initAllBundles();
173 
174 		return collector;
175 	}

178 	 * Initialize the resource bundles from the mapping file
179 	 */
180 	private void initResourceBundlesFromFullMapping(List resourceBundles) {
181 
182 		if (log.isInfoEnabled()){
183 			log.info("Building bundles from the full bundle mapping. The bundles will not be processed.");
184 		}
185 		Properties mappingProperties = resourceHandler.getJawrBundleMapping();
186 		FullMappingPropertiesBasedBundlesHandlerFactory factory = new FullMappingPropertiesBasedBundlesHandlerFactory(resourceType, 
187 				resourceHandler, chainFactory);
188 		
189 		resourceBundles.addAll(factory.getResourceBundles(mappingProperties));
190 	}

192 	/***
193 	 * Initialize the resource bundles
194 	 * 
195 	 * @param resourceBundles the resource bundles
196 	 * @throws DuplicateBundlePathException if two bundles are defined with the same path 
197 	 */
198 	private void initResourceBundles(List resourceBundles)
199 			throws DuplicateBundlePathException {
200 
201 		// Create custom defined bundles
202 		if (null != bundleDefinitions) {
203 			if (log.isInfoEnabled())
204 				log.info("Adding custom bundle definitions. ");
205 			for (Iterator it = bundleDefinitions.iterator(); it.hasNext();) {
206 				ResourceBundleDefinition def = (ResourceBundleDefinition) it
207 						.next();
208 
209 				// If this is a composite bundle
210 				if (def.isComposite()) {
211 					List childBundles = new ArrayList();
212 					for (Iterator childIterator = def.getChildren().iterator(); childIterator
213 							.hasNext();) {
214 						ResourceBundleDefinition child = (ResourceBundleDefinition) childIterator
215 								.next();
216 						childBundles.add(buildResourcebundle(child));
217 					}
218 					resourceBundles.add(buildCompositeResourcebundle(def,
219 							childBundles));
220 				} else
221 					resourceBundles.add(buildResourcebundle(def));
222 			}
223 		}
224 
225 		// Use the dirmapper if specified
226 		if (useDirMapperFactory) {
227 			if (log.isInfoEnabled())
228 				log.info("Using ResourceBundleDirMapper. ");
229 
230 			ResourceBundleDirMapper dirFactory = new ResourceBundleDirMapper(
231 					baseDir, resourceHandler, resourceBundles, fileExtension,
232 					excludedDirMapperDirs);
233 			Map mappings = dirFactory.getBundleMapping();
234 			for (Iterator it = mappings.entrySet().iterator(); it.hasNext();) {
235 				Entry entry = (Entry) it.next();
236 				resourceBundles.add(buildDirMappedResourceBundle((String) entry.getKey(),
237 						(String) entry.getValue()));
238 			}
239 		}
240 
241 		if (this.scanForOrphans) {
242 			// Add all orphan bundles
243 			OrphanResourceBundlesMapper orphanFactory = new OrphanResourceBundlesMapper(
244 					baseDir, resourceHandler, resourceBundles, fileExtension);
245 			List orphans = orphanFactory.getOrphansList();
246 
247 			// Orphans may be added separately or as one single resource bundle.
248 			if (useSingleResourceFactory) {
249 				// Add extension to the filename
250 				if (!singleFileBundleName.endsWith(fileExtension))
251 					singleFileBundleName += fileExtension;
252 
253 				if (log.isInfoEnabled())
254 					log
255 							.info("Building bundle of orphan resources with the name: "
256 									+ singleFileBundleName);
257 
258 				resourceBundles.add(buildOrphansResourceBundle(
259 						singleFileBundleName, orphans));
260 
261 			} else {
262 				if (log.isInfoEnabled())
263 					log.info("Creating mappings for orphan resources. ");
264 				for (Iterator it = orphans.iterator(); it.hasNext();) {
265 					resourceBundles.add(buildOrphanResourceBundle((String) it
266 							.next()));
267 				}
268 			}
269 		} else if (log.isDebugEnabled()) {
270 			log.debug("Skipping orphan file auto processing. ");
271 			if ("".equals(jawrConfig.getServletMapping()))
272 				log
273 						.debug("Note that there is no specified mapping for Jawr "
274 								+ "(it has been seet to serve *.js or *.css requests). "
275 								+ "The orphan files will become unreachable through the server.");
276 		}
277 	}

279 	/***
280 	 * Build a Composite resource bundle using a ResourceBundleDefinition
281 	 * 
282 	 * @param definition the bundle definition
283 	 * @param childBundles the list of child bundles
284 	 * @return a Composite resource bundle
285 	 */
286 	private JoinableResourceBundle buildCompositeResourcebundle(
287 			ResourceBundleDefinition definition, List childBundles) {
288 
289 		if (log.isDebugEnabled())
290 			log.debug("Init composite bundle with id:"
291 					+ definition.getBundleId());
292 
293 		InclusionPattern include = new InclusionPattern(definition.isGlobal(),
294 				definition.getInclusionOrder(), definition.isDebugOnly(),
295 				definition.isDebugNever());
296 
297 		CompositeResourceBundle composite = new CompositeResourceBundle(
298 				definition.getBundleId(), definition.getBundleName(), 
299 				childBundles, include, resourceHandler, fileExtension,
300 				jawrConfig);
301 		if (null != definition.getBundlePostProcessorKeys())
302 			composite.setBundlePostProcessor(chainFactory
303 					.buildPostProcessorChain(definition
304 							.getBundlePostProcessorKeys()));
305 
306 		if (null != definition.getUnitaryPostProcessorKeys())
307 			composite.setUnitaryPostProcessor(chainFactory
308 					.buildPostProcessorChain(definition
309 							.getUnitaryPostProcessorKeys()));
310 
311 		if (null != definition.getIeConditionalExpression())
312 			composite.setExplorerConditionalExpression(definition
313 					.getIeConditionalExpression());
314 
315 		if (null != definition.getAlternateProductionURL())
316 			composite.setAlternateProductionURL(definition
317 					.getAlternateProductionURL());
318 
319 		return composite;
320 	}

323 	 * Build a JoinableResourceBundle using a ResourceBundleDefinition
324 	 * 
325 	 * @param definition the resource bundle definition
326 	 * @return a JoinableResourceBundle
327 	 */
328 	private JoinableResourceBundle buildResourcebundle(
329 			ResourceBundleDefinition definition) {
330 		if (log.isDebugEnabled())
331 			log.debug("Init bundle with id:" + definition.getBundleId());
332 
333 		InclusionPattern include = new InclusionPattern(definition.isGlobal(),
334 				definition.getInclusionOrder(), definition.isDebugOnly(),
335 				definition.isDebugNever());
336 
337 		JoinableResourceBundleImpl newBundle = new JoinableResourceBundleImpl(
338 				definition.getBundleId(), definition.getBundleName(), 
339 				fileExtension, include, definition.getMappings(),
340 				resourceHandler);
341 		if (null != definition.getBundlePostProcessorKeys())
342 			newBundle.setBundlePostProcessor(chainFactory
343 					.buildPostProcessorChain(definition
344 							.getBundlePostProcessorKeys()));
345 
346 		if (null != definition.getUnitaryPostProcessorKeys())
347 			newBundle.setUnitaryPostProcessor(chainFactory
348 					.buildPostProcessorChain(definition
349 							.getUnitaryPostProcessorKeys()));
350 
351 		if (null != definition.getIeConditionalExpression())
352 			newBundle.setExplorerConditionalExpression(definition
353 					.getIeConditionalExpression());
354 
355 		if (null != definition.getLocaleVariantKeys())
356 			newBundle.setLocaleVariantKeys(definition.getLocaleVariantKeys());
357 
358 		if (null != definition.getAlternateProductionURL())
359 			newBundle.setAlternateProductionURL(definition
360 					.getAlternateProductionURL());
361 
362 		return newBundle;
363 	}

381 	/***
382 	 * Generates the bundle ID from the bundle name
383 	 * 
384 	 * @param bundleId the bundle name
385 	 * @return the generated bundle ID
386 	 */
387 	private String generateBundleNameFromBundleId(String bundleId) {
388 		String bundleName = bundleId;
389 		if(bundleName.startsWith("/")){
390 			bundleName = bundleName.substring(1);
391 		}
392 		int idxExtension = FileNameUtils.indexOfExtension(bundleName);
393 		if(idxExtension != -1){
394 			bundleName = bundleName.substring(0, idxExtension);
395 		}
396 		return bundleName.replaceAll("/", "_");
397 	}
