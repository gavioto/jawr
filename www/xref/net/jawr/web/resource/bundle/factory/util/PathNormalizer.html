65  	/**
66  	 * Removes the URL prefix defined in the configuration from a path. If the prefix contains a variant information, it adds it to the name.
67  	 * 
68  	 * @param path the path
69  	 * @return the path without the prefix
70  	 */
71  	public static String removeVariantPrefixFromPath(String path) {
72  		
73  		String resultPath = path;
74  		
75  		// Remove first slash
76  		if(path.charAt(0) == '/'){
77  			resultPath = path.substring(1);
78  		}
79  		
80  		// eval the existence of a suffix
81  		String prefix = resultPath.substring(0, resultPath.indexOf("/"));
82  
83  		// The prefix also contains variant information after a '.'
84  		if (prefix.indexOf('.') != -1) {
85  			String variantPrefix = prefix.substring(prefix.indexOf('.') + 1);
86  			String suffix = '@' + variantPrefix + resultPath.substring(resultPath.lastIndexOf('.'));
87  			resultPath = resultPath.substring(resultPath.indexOf("/"), resultPath.lastIndexOf('.')) + suffix;
88  		} else
89  			resultPath = resultPath.substring(resultPath.indexOf("/"));
90  		
91  		return resultPath;
92  	}

94  	/**
95  	 * Removes the URL prefix defined in the configuration from a path. If the prefix contains a variant information, it adds it to the name.
96  	 * 
97  	 * @param path the path
98  	 * @return the bundle info from the  
99  	 */
100 	public static String[] extractBundleInfoFromPath(String path) {
101 		
102 		String[] result = new String[3];
103 		
104 		String resultPath = null;
105 		if(path.startsWith(BundleRenderer.GZIP_PATH_PREFIX)){
106 			// Remove the gzip prefix
107 			resultPath = path.substring(BundleRenderer.GZIP_PATH_PREFIX.length());
108 		}else{
109 			// Remove first slash
110 			resultPath = path.substring(1);
111 		}
112 		
113 		// eval the existence of a suffix
114 		int realPathStartIdx = resultPath.indexOf("/");
115 		String prefix = "";
116 		if(realPathStartIdx != -1){
117 			prefix = resultPath.substring(0, realPathStartIdx);
118 		}
119 		
120 		String hashcode = null;
121 		String variantPrefix = null;
122 		// The prefix also contains variant information after a '.'
123 		int hashCodeVariantSeparatorIdx = prefix.indexOf('.');
124 		if (hashCodeVariantSeparatorIdx != -1) {
125 			hashcode = prefix.substring(0, hashCodeVariantSeparatorIdx);
126 			variantPrefix = prefix.substring(hashCodeVariantSeparatorIdx + 1);
127 			resultPath = resultPath.substring(realPathStartIdx);
128 		} else{
129 			hashcode = prefix.substring(0, realPathStartIdx);
130 			resultPath = resultPath.substring(realPathStartIdx);
131 		}
132 		
133 		// Remove leading slash if it's a bundle containing a generator prefix
134 		if(resultPath.startsWith("/") && resultPath.contains(":")){
135 			resultPath = resultPath.substring(1);
136 		}
137 		
138 		result[0] = resultPath;
139 		result[1] = variantPrefix;
140 		result[2] = hashcode;
141 		
142 		return result;
143 	}

209 	 * Normalizes two paths and joins them as a single path. 
210 	 * @param prefix
211 	 * @param path
212 	 * @return the joined path
213 	 */
214 	public static String joinPaths(String prefix,String path) {
215 		
216 		String joinedPath = null;
217 		if(prefix.startsWith(JawrConstant.HTTP_URL_PREFIX) || prefix.startsWith(JawrConstant.HTTPS_URL_PREFIX) || prefix.startsWith("//")){
218 			joinedPath = joinDomainToPath(prefix, path);
219 		}else{
220 			
221 			String normalizedPrefix = PathNormalizer.normalizePath(prefix);
222 			
223 			StringBuffer sb = new StringBuffer(JawrConstant.URL_SEPARATOR);
224 			if(!"".equals(normalizedPrefix))
225 				sb.append(normalizedPrefix).append(JawrConstant.URL_SEPARATOR);
226 			sb.append(PathNormalizer.normalizePath(path));
227 			joinedPath = sb.toString();
228 		}
229 		
230 		return joinedPath; 
231 	}

253 	/**
254 	 * Removes leading and trailing separators from a path, and removes 
255 	 * double separators (// is replaced by /). 
256 	 * @param path the path to normalize
257 	 * @return the normalized path
258 	 */
259 	public static final String normalizePath(String path) {
260 		String normalizedPath = path.replaceAll("//", JawrConstant.URL_SEPARATOR);
261 		StringTokenizer tk = new StringTokenizer(normalizedPath,JawrConstant.URL_SEPARATOR);
262 		StringBuffer sb = new StringBuffer();
263 		while(tk.hasMoreTokens()) {
264 			sb.append(tk.nextToken());
265 			if(tk.hasMoreTokens())
266 				sb.append(JawrConstant.URL_SEPARATOR);
267 		}
268 		return sb.toString();
269 		
270 	}

273 	 * Normalizes all the paths in a Set. 
274 	 * @param paths
275 	 * @return
276 	 */
277 	public static final Set<String> normalizePaths(Set<String> paths) {
278 		Set<String> ret = new HashSet<String>();
279 		for(Iterator<String> it = paths.iterator();it.hasNext();) {
280 			String path = normalizePath((String)it.next());
281 			ret.add(path);
282 		}
283 		return ret;

293

297 			requestPath = registry.getDebugModeGenerationPath(path);

308 			throw new JawrLinkRenderingException("Something went unexpectedly wrong while encoding a URL for a generator. ",

312 	}

345     /**
346      * Determines the parent path of a filename or a directory.
347      * 
348      * <pre>
349      * PathUtils.getParentPath( null )                           = ""
350      * PathUtils.getParentPath( "" )                             = ""
351      * PathUtils.getParentPath( "/" )                            = "/"
352      * PathUtils.getParentPath( "/usr/local/" )                  = "/usr/local/"
353      * PathUtils.getRelativePath( "/usr/local/bin/java.sh" )     = ""/usr/local/bin/"
354      * </pre>
355      * 
356      * @param path the path
357      * @return the parent path.
358      */
359     public static String getParentPath(String path){
360     
361     	String parentPath = null;
362     	if(StringUtils.isEmpty(path)){
363     		parentPath = "";
364     	}else{
365     		
366     		parentPath = path;
367 	    	if(parentPath.length() > 1 && parentPath.endsWith(JawrConstant.URL_SEPARATOR)){
368 	    		parentPath = parentPath.substring(0, parentPath.length()-2);
369 	    	}
370 	    	int index = parentPath.lastIndexOf(JawrConstant.URL_SEPARATOR);
371 	    	if(index > 0){
372 	    		return parentPath.substring(0, index+1);
373 	    	}else{
374 	    		parentPath = JawrConstant.URL_SEPARATOR;
375 	    	}
376     	}
377     	
378     	return parentPath; 
379     }

417     /**
418      * Determines the relative path of a filename from a base directory.
419      * This method is useful in building relative links within pages of
420      * a web site.  It provides similar functionality to Anakia's
421      * <code>$relativePath</code> context variable.  The arguments to
422      * this method may contain either forward or backward slashes as
423      * file separators.  The relative path returned is formed using
424      * forward slashes as it is expected this path is to be used as a
425      * link in a web page (again mimicking Anakia's behavior).
426      * <p/>
427      * This method is thread-safe.
428      * <br/>
429      * <pre>
430      * PathUtils.getRelativePath( null, null )                                   = ""
431      * PathUtils.getRelativePath( null, "/usr/local/java/bin" )                  = ""
432      * PathUtils.getRelativePath( "/usr/local/", null )                          = ""
433      * PathUtils.getRelativePath( "/usr/local/", "/usr/local/java/bin" )         = ".."
434      * PathUtils.getRelativePath( "/usr/local/", "/usr/local/java/bin/java.sh" ) = "../.."
435      * PathUtils.getRelativePath( "/usr/local/java/bin/java.sh", "/usr/local/" ) = ""
436      * </pre>
437      *
438      * @param basedir The base directory.
439      * @param filename The filename that is relative to the base
440      * directory.
441      * @return The relative path of the filename from the base
442      * directory.  This value is not terminated with a forward slash.
443      * A zero-length string is returned if: the filename is not relative to
444      * the base directory, <code>basedir</code> is null or zero-length,
445      * or <code>filename</code> is null or zero-length.
446      */
447     public static final String getRelativePath( String basedir, String filename )
448     {
449         String basedirPath = uppercaseDrive(basedir);
450         String filenamePath = uppercaseDrive(filename);
451 
452         /*
453          * Verify the arguments and make sure the filename is relative
454          * to the base directory.
455          */
456         if ( basedirPath == null || basedirPath.length() == 0 || filenamePath == null
457             || filenamePath.length() == 0 || !filenamePath.startsWith( basedirPath ) )
458         {
459             return "";
460         }
461 
462         /*
463          * Normalize the arguments.  First, determine the file separator
464          * that is being used, then strip that off the end of both the
465          * base directory and filename.
466          */
467         String separator = determineSeparator( filenamePath );
468         basedirPath = StringUtils.chompLast( basedirPath, separator );
469         filenamePath = StringUtils.chompLast( filenamePath, separator );
470 
471         /*
472          * Remove the base directory from the filename to end up with a
473          * relative filename (relative to the base directory).  This
474          * filename is then used to determine the relative path.
475          */
476         String relativeFilename = filenamePath.substring( basedirPath.length() );
477 
478         return determineRelativePath( relativeFilename, separator );
479     }

481     /**
482      * Concatenates a filename to a base web path.
483      * If the base path doesn't end with "/", it will consider as base path the parent folder of the base path passed as parameter.
484      * 
485      * <pre>
486      * PathUtils.concatWebPath("", null));								   			= null
487 	 * PathUtils.concatWebPath(null, null));								   		= null
488 	 * PathUtils.concatWebPath(null, ""));								   			= null
489 	 * PathUtils.concatWebPath(null, "a"));								   			= null			
490 	 * PathUtils.concatWebPath(null, "/a"));								   		= "/a" 
491  	 * PathUtils.concatWebPath( "/css/folder/subfolder/", "icons/img.png" )      	= "/css/folder/subfolder/icons/img.png"
492      * PathUtils.concatWebPath( "/css/folder/subfolder/style.css", "icons/img.png") = "/css/folder/subfolder/icons/img.png"
493      * PathUtils.concatWebPath( "/css/folder/", "../icons/img.png" )      			= "/css/icons/img.png"
494      * PathUtils.concatWebPath( "/css/folder/style.css", "../icons/img.png" ) 		= "/css/icons/img.png"
495      * </pre>
496      * 
497      * @param basePath the base path
498      * @param fullFilenameToAdd the file name to add
499      * @return the concatenated path, or null if invalid
500      */
501     public static String concatWebPath(String basePath, String fullFilenameToAdd){
502      	
503     	if(fullFilenameToAdd == null || basePath == null && (fullFilenameToAdd.length() == 0
504     			|| fullFilenameToAdd.charAt(0) != JawrConstant.URL_SEPARATOR_CHAR)){
505     		return null;
506     	}
507     	
508     	if(basePath == null){
509     		basePath = "";
510     	}
511     	// If the basePath is pointing to a file, set the base path to the parent directory
512     	if(basePath.length() > 1 && basePath.charAt(basePath.length()-1) != '/'){
513     		basePath = getParentPath(basePath);
514     	}
515     	
516     	int len = basePath.length();
517     	String fullPath = null;
518         if (len == 0) {
519         	return doNormalizeIgnoreOtherSeparator(fullFilenameToAdd, true);
520         }
521         
522     	char ch = basePath.charAt(len - 1);
523         if (ch == JawrConstant.URL_SEPARATOR_CHAR) {
524         	fullPath = basePath + fullFilenameToAdd;
525         } else {
526         	fullPath = basePath + '/' + fullFilenameToAdd;
527         }
528         
529         return doNormalizeIgnoreOtherSeparator(fullPath, true); 
530         
531     }

533     /**
534      * Internal method to perform the normalization.
535      *
536      * @param filename  the filename
537      * @param keepSeparator  true to keep the final separator
538      * @return the normalized filename
539      */
540     private static String doNormalizeIgnoreOtherSeparator(String filename, boolean keepSeparator) {
541         if (filename == null) {
542             return null;
543         }
544         int size = filename.length();
545         if (size == 0) {
546             return filename;
547         }
548         int prefix = 0;
549 //        int prefix = getPrefixLength(filename);
550 //        if (prefix < 0) {
551 //            return null;
552 //        }
553         
554         char[] array = new char[size + 2];  // +1 for possible extra slash, +2 for arraycopy
555         filename.getChars(0, filename.length(), array, 0);
556         
557         // add extra separator on the end to simplify code below
558         boolean lastIsDirectory = true;
559         if (array[size - 1] != JawrConstant.URL_SEPARATOR_CHAR) {
560             array[size++] = JawrConstant.URL_SEPARATOR_CHAR;
561             lastIsDirectory = false;
562         }
563         
564         // adjoining slashes
565         for (int i = prefix + 1; i < size; i++) {
566             if (array[i] == JawrConstant.URL_SEPARATOR_CHAR && array[i - 1] == JawrConstant.URL_SEPARATOR_CHAR) {
567                 System.arraycopy(array, i, array, i - 1, size - i);
568                 size--;
569                 i--;
570             }
571         }
572         
573         // dot slash
574         for (int i = prefix + 1; i < size; i++) {
575             if (array[i] == JawrConstant.URL_SEPARATOR_CHAR && array[i - 1] == '.' &&
576                     (i == prefix + 1 || array[i - 2] == JawrConstant.URL_SEPARATOR_CHAR)) {
577                 if (i == size - 1) {
578                     lastIsDirectory = true;
579                 }
580                 System.arraycopy(array, i + 1, array, i - 1, size - i);
581                 size -=2;
582                 i--;
583             }
584         }
585         
586         // double dot slash
587         outer:
588         for (int i = prefix + 2; i < size; i++) {
589             if (array[i] == JawrConstant.URL_SEPARATOR_CHAR && array[i - 1] == '.' && array[i - 2] == '.' &&
590                     (i == prefix + 2 || array[i - 3] == JawrConstant.URL_SEPARATOR_CHAR)) {
591                 if (i == prefix + 2) {
592                     return null;
593                 }
594                 if (i == size - 1) {
595                     lastIsDirectory = true;
596                 }
597                 int j;
598                 for (j = i - 4 ; j >= prefix; j--) {
599                     if (array[j] == JawrConstant.URL_SEPARATOR_CHAR) {
600                         // remove b/../ from a/b/../c
601                         System.arraycopy(array, i + 1, array, j + 1, size - i);
602                         size -= (i - j);
603                         i = j + 1;
604                         continue outer;
605                     }
606                 }
607                 // remove a/../ from a/../c
608                 System.arraycopy(array, i + 1, array, prefix, size - i);
609                 size -= (i + 1 - prefix);
610                 i = prefix + 1;
611             }
612         }
613         
614         if (size <= 0) {  // should never be less than 0
615             return "";
616         }
617         if (size <= prefix) {  // should never be less than prefix
618             return new String(array, 0, size);
619         }
620         if (lastIsDirectory && keepSeparator) {
621             return new String(array, 0, size);  // keep trailing separator
622         }
623         return new String(array, 0, size - 1);  // lose trailing separator
624     }

626     /**
627      * This method can calculate the relative path between two pathes on a web site.
628      * <br/>
629      * <pre>
630      * PathUtils.getRelativeWebPath( null, null )                                          = ""
631      * PathUtils.getRelativeWebPath( null, "http://plexus.codehaus.org/" )                 = ""
632      * PathUtils.getRelativeWebPath( "http://plexus.codehaus.org/", null )                 = ""
633      * PathUtils.getRelativeWebPath( "http://plexus.codehaus.org/",
634      *                      "http://plexus.codehaus.org/plexus-utils/index.html" )        = "plexus-utils/index.html"
635      * PathUtils.getRelativeWebPath( "http://plexus.codehaus.org/plexus-utils/index.html",
636      *                      "http://plexus.codehaus.org/"                                 = "../../"
637      * </pre>
638      *
639      * @param oldPath
640      * @param newPath
641      * @return a relative web path from <code>oldPath</code>.
642      */
643     public static final String getRelativeWebPath( final String oldPath, final String newPath )
644     {
645         if ( StringUtils.isEmpty( oldPath ) || StringUtils.isEmpty( newPath ) )
646         {
647             return "";
648         }
649 
650         String resultPath = buildRelativePath( newPath, oldPath, '/' );
651 
652         if ( newPath.endsWith( "/" ) && !resultPath.endsWith( "/" ) )
653         {
654             return resultPath + "/";
655         }
656 
657         return resultPath;
658     }

660     /**
661      * This method can calculate the relative path between two pathes on a file system.
662      * <br/>
663      * <pre>
664      * PathUtils.getRelativeFilePath( null, null )                                   = ""
665      * PathUtils.getRelativeFilePath( null, "/usr/local/java/bin" )                  = ""
666      * PathUtils.getRelativeFilePath( "/usr/local", null )                           = ""
667      * PathUtils.getRelativeFilePath( "/usr/local", "/usr/local/java/bin" )          = "java/bin"
668      * PathUtils.getRelativeFilePath( "/usr/local", "/usr/local/java/bin/" )         = "java/bin"
669      * PathUtils.getRelativeFilePath( "/usr/local/java/bin", "/usr/local/" )         = "../.."
670      * PathUtils.getRelativeFilePath( "/usr/local/", "/usr/local/java/bin/java.sh" ) = "java/bin/java.sh"
671      * PathUtils.getRelativeFilePath( "/usr/local/java/bin/java.sh", "/usr/local/" ) = "../../.."
672      * PathUtils.getRelativeFilePath( "/usr/local/", "/bin" )                        = "../../bin"
673      * PathUtils.getRelativeFilePath( "/bin", "/usr/local/" )                        = "../usr/local"
674      * </pre>
675      * Note: On Windows based system, the <code>/</code> character should be replaced by <code>\</code> character.
676      *
677      * @param oldPath
678      * @param newPath
679      * @return a relative file path from <code>oldPath</code>.
680      */
681     public static final String getRelativeFilePath( final String oldPath, final String newPath )
682     {
683         if ( StringUtils.isEmpty( oldPath ) || StringUtils.isEmpty( newPath ) )
684         {
685             return "";
686         }
687 
688         // normalise the path delimiters
689         String fromPath = new File( oldPath ).getPath();
690         String toPath = new File( newPath ).getPath();
691 
692         // strip any leading slashes if its a windows path
693         if ( toPath.matches( "^\\[a-zA-Z]:" ) )
694         {
695             toPath = toPath.substring( 1 );
696         }
697         if ( fromPath.matches( "^\\[a-zA-Z]:" ) )
698         {
699             fromPath = fromPath.substring( 1 );
700         }
701 
702         // lowercase windows drive letters.
703         if ( fromPath.startsWith( ":", 1 ) )
704         {
705             fromPath = Character.toLowerCase( fromPath.charAt( 0 ) ) + fromPath.substring( 1 );
706         }
707         if ( toPath.startsWith( ":", 1 ) )
708         {
709             toPath = Character.toLowerCase( toPath.charAt( 0 ) ) + toPath.substring( 1 );
710         }
711 
712         // check for the presence of windows drives. No relative way of
713         // traversing from one to the other.
714         if ( ( toPath.startsWith( ":", 1 ) && fromPath.startsWith( ":", 1 ) )
715                         && ( !toPath.substring( 0, 1 ).equals( fromPath.substring( 0, 1 ) ) ) )
716         {
717             // they both have drive path element but they dont match, no
718             // relative path
719             return null;
720         }
721 
722         if ( ( toPath.startsWith( ":", 1 ) && !fromPath.startsWith( ":", 1 ) )
723                         || ( !toPath.startsWith( ":", 1 ) && fromPath.startsWith( ":", 1 ) ) )
724         {
725             // one has a drive path element and the other doesnt, no relative
726             // path.
727             return null;
728         }
729 
730         String resultPath = buildRelativePath( toPath, fromPath, File.separatorChar );
731 
732         if ( newPath.endsWith( File.separator ) && !resultPath.endsWith( File.separator ) )
733         {
734             return resultPath + File.separator;
735         }
736 
737         return resultPath;
738     }

744     /**
745      * Determines the relative path of a filename.  For each separator
746      * within the filename (except the leading if present), append the
747      * "../" string to the return value.
748      *
749      * @param filename The filename to parse.
750      * @param separator The separator used within the filename.
751      * @return The relative path of the filename.  This value is not
752      * terminated with a forward slash.  A zero-length string is
753      * returned if: the filename is zero-length.
754      */
755     private static final String determineRelativePath( String filename,
756                                                        String separator )
757     {
758         if ( filename.length() == 0 )
759         {
760             return "";
761         }
762 
763 
764         /*
765          * Count the slashes in the relative filename, but exclude the
766          * leading slash.  If the path has no slashes, then the filename
767          * is relative to the current directory.
768          */
769         int slashCount = StringUtils.countMatches( filename, separator ) - 1;
770         if ( slashCount <= 0 )
771         {
772             return ".";
773         }
774 
775         /*
776          * The relative filename contains one or more slashes indicating
777          * that the file is within one or more directories.  Thus, each
778          * slash represents a "../" in the relative path.
779          */
780         StringBuffer sb = new StringBuffer();
781         for ( int i = 0; i < slashCount; i++ )
782         {
783             sb.append( "../" );
784         }
785 
786         /*
787          * Finally, return the relative path but strip the trailing
788          * slash to mimic Anakia's behavior.
789          */
790         return StringUtils.chop( sb.toString() );
791     }

793     /**
794      * Helper method to determine the file separator (forward or
795      * backward slash) used in a filename.  The slash that occurs more
796      * often is returned as the separator.
797      *
798      * @param filename The filename parsed to determine the file
799      * separator.
800      * @return The file separator used within <code>filename</code>.
801      * This value is either a forward or backward slash.
802      */
803     private static final String determineSeparator( String filename )
804     {
805         int forwardCount = StringUtils.countMatches( filename, "/" );
806         int backwardCount = StringUtils.countMatches( filename, "\\" );
807 
808         return forwardCount >= backwardCount ? "/" : "\\";
809     }

831     private static final String buildRelativePath( String toPath,  String fromPath, final char separatorChar )
832     {
833         // use tokeniser to traverse paths and for lazy checking
834         StringTokenizer toTokeniser = new StringTokenizer( toPath, String.valueOf( separatorChar ) );
835         StringTokenizer fromTokeniser = new StringTokenizer( fromPath, String.valueOf( separatorChar ) );
836 
837         int count = 0;
838 
839         // walk along the to path looking for divergence from the from path
840         while ( toTokeniser.hasMoreTokens() && fromTokeniser.hasMoreTokens() )
841         {
842             if ( separatorChar == '\\' )
843             {
844                 if ( !fromTokeniser.nextToken().equalsIgnoreCase( toTokeniser.nextToken() ) )
845                 {
846                     break;
847                 }
848             }
849             else
850             {
851                 if ( !fromTokeniser.nextToken().equals( toTokeniser.nextToken() ) )
852                 {
853                     break;
854                 }
855             }
856 
857             count++;
858         }
859 
860         // reinitialise the tokenisers to count positions to retrieve the
861         // gobbled token
862 
863         toTokeniser = new StringTokenizer( toPath, String.valueOf( separatorChar ) );
864         fromTokeniser = new StringTokenizer( fromPath, String.valueOf( separatorChar ) );
865 
866         while ( count-- > 0 )
867         {
868             fromTokeniser.nextToken();
869             toTokeniser.nextToken();
870         }
871 
872         StringBuffer relativePath = new StringBuffer();
873 
874         // add back refs for the rest of from location.
875         while ( fromTokeniser.hasMoreTokens() )
876         {
877             fromTokeniser.nextToken();
878 
879             relativePath.append("..");
880 
881             if ( fromTokeniser.hasMoreTokens() )
882             {
883             	relativePath.append(separatorChar);
884             }
885         }
886 
887         if ( relativePath.length() != 0 && toTokeniser.hasMoreTokens() )
888         {
889         	relativePath.append(separatorChar);
890         }
891 
892         // add fwd fills for whatevers left of newPath.
893         while ( toTokeniser.hasMoreTokens() )
894         {
895         	relativePath.append(toTokeniser.nextToken());
896 
897             if ( toTokeniser.hasMoreTokens() )
898             {
899             	relativePath.append(separatorChar);
900             }
901         }
902         return relativePath.toString();
903     }

905     /**
906 	 * Returns the relative path of an url to go back to the root.
907 	 * For example : if the url path is defined as "/cssServletPath/css/myStyle.css" -> "../../"
908 	 * 
909 	 * @param url the requested url
910 	 * @return the relative path of an url to go back to the root.
911 	 */
912 	public static String getRootRelativePath(String url) {
913 
914 		Matcher matcher = URL_SEPARATOR_PATTERN.matcher(url);
915 		StringBuffer result = new StringBuffer();
916 		boolean first = true;
917 		while (matcher.find()) {
918 			if (first) {
919 				matcher.appendReplacement(result, "");
920 				first = false;
921 			} else {
922 				matcher.appendReplacement(result, ROOT_REPLACE_PATTERN);
923 			}
924 		}
925 
926 		return result.toString();
927 	}
