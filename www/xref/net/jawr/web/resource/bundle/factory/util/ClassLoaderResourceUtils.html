43  	/***
44  	 * Attempots to load a resource from the classpath, either usinf the caller's class loader or the current thread's 
45  	 * context classloader. 
46  	 * 
47  	 * @param resourcePath
48  	 * @param source
49  	 * @return
50  	 * @throws FileNotFoundException
51  	 */
52  	public static InputStream getResourceAsStream(String resourcePath, Object source) throws FileNotFoundException {
53  		
54  		// Try the current classloader
55  		InputStream is = source.getClass().getResourceAsStream(resourcePath);
56  		
57  		// Weblogic 10 likes this one better..
58  		if(null == is) {
59  			ClassLoader cl = source.getClass().getClassLoader();
60  			if(null != cl)
61  				is = cl.getResourceAsStream(resourcePath);
62  		}
63  		
64  		// If current classloader failed, try with the Threads context classloader. If that fails ott, the resource is either not on the 
65  		// classpath or inaccessible from the current context. 
66  		
67  		if(null == is) {
68  			is = Thread.currentThread().getContextClassLoader().getResourceAsStream(resourcePath);
69  		}
70  		
71  		if(null == is){
72  			
73  			// Check if we are using JMX
74  			if(System.getProperty(JawrConstant.JMX_ENABLE_FLAG_SYSTEL_PROPERTY) != null){
75  			
76  				// Try to use the classloader of the current Jawr Config Manager MBean
77  				// This will be used when a refresh is done in the configuration using the JMX MBean
78  				MBeanServer mbs = JmxUtils.getMBeanServer();
79  				if(mbs != null){
80  					
81  					ObjectName name = ThreadLocalJawrContext.getJawrConfigMgrObjectName();
82  					if(name != null){
83  						try {
84  							
85  							ClassLoader cl = mbs.getClassLoaderFor(name);
86  							is = cl.getResourceAsStream(resourcePath);
87  						} catch (Exception e) {
88  							log.error("Unable to instanciate the Jawr MBean '"+name.getCanonicalName()+"'", e);
89  						}
90  					}
91  				}
92  			}
93  		}
94  		
95  		// Try to retrieve by URL
96  		if(null == is) {
97  			try {
98  				URL url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
99  				
100 				
101 				// Last chance, hack in the classloader
102 				if(null == url) {
103 					ClassLoader threadClassLoader = Thread.currentThread().getContextClassLoader();
104 					try {
105 						 Thread.currentThread().setContextClassLoader(source.getClass().getClassLoader());
106 						 if(Thread.currentThread().getContextClassLoader() != null){
107 							 url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
108 						 }
109 					}
110 					finally {
111 						 Thread.currentThread().setContextClassLoader(threadClassLoader);
112 					}
113 					
114 				}
115 				if(null == url){
116 					throw new FileNotFoundException( resourcePath + " could not be found. ");
117 				}
118 				
119 				is = new FileInputStream(new File(url.getFile()));			
120 			} 
121 			catch (IOException e) {
122 				throw new FileNotFoundException( resourcePath + " could not be found. ");
123 			}
124 		}
125 		
126 		// Everything failed... exception is trown. 
127 		if(null == is)
128 			throw new FileNotFoundException( resourcePath + " could not be found. ");
129 		return is;
130 	}

132 	/***
133 	 * Attempts to find the URL of a resource from the classpath, either usinf the caller's class loader or the current thread's 
134 	 * context classloader. 
135 	 * 
136 	 * @param resourcePath the resource path
137 	 * @param source the object
138 	 * @return the URL.
139 	 * @throws ResourceNotFoundException if the resource is not found
140 	 */
141 	public static URL getResourceURL(String resourcePath, Object source) throws ResourceNotFoundException {
142 		
143 		// Try the current classloader
144 		URL url = source.getClass().getResource(resourcePath);
145 		
146 		// Weblogic 10 likes this one better..
147 		if(null == url) {
148 			ClassLoader cl = source.getClass().getClassLoader();
149 			if(null != cl)
150 				url = cl.getResource(resourcePath);
151 		}
152 		
153 		// If current classloader failed, try with the Threads context classloader. If that fails ott, the resource is either not on the 
154 		// classpath or inaccessible from the current context. 
155 		
156 		if(null == url) {
157 			url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
158 		}
159 		
160 		// Try to retrieve by URL
161 		if(null == url) {
162 			url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
163 			
164 			// Last chance, hack in the classloader
165 			if(null == url) {
166 				ClassLoader threadClassLoader = Thread.currentThread().getContextClassLoader();
167 				try {
168 					 Thread.currentThread().setContextClassLoader(source.getClass().getClassLoader());
169 					 if(Thread.currentThread().getContextClassLoader() != null){
170 						 url = Thread.currentThread().getContextClassLoader().getResource(resourcePath);
171 					 }
172 				}
173 				finally {
174 					 Thread.currentThread().setContextClassLoader(threadClassLoader);
175 				}
176 				
177 			}
178 			if(null == url){
179 				throw new ResourceNotFoundException( resourcePath + " could not be found. ");
180 			}
181 		}
182 		
183 		return url;
184 	}

188 	 * @param classname the class to build an instance of.

191 	public static Object buildObjectInstance(String classname) {
192 		Object rets = null;
193 		Class clazz = locateClass(classname);
194 		try {
195 			rets = clazz.newInstance();
196 		}catch(Exception e) {
197 				throw new RuntimeException(e.getMessage() 
198 											+ " [The custom class " 
199 											+ classname 
200 											+ " could not be instantiated, check wether it is available on the classpath and" 
201 											+ " verify that it has a zero-arg constructor].\n" 
202 											+ " The specific error message is: " + e.getClass().getName() + ":" + e.getMessage(),e);			
203 		}
204 		return rets;
205 	}

207 	private static Class locateClass(String classname) {
208 		Class clazz = null;
209 		try {
210 			clazz = Class.forName(classname);
211 			
212 		} catch (Exception e) {
213 			// Try the second approach 
214 		}
215 		if(null == clazz) {
216 			Exception classNotFoundEx = null;
217 			try {
218 				clazz = Class.forName(classname, true, new ClassLoaderResourceUtils().getClass().getClassLoader());
219 			} catch (Exception e) {
220 				// Try the third approach
221 				classNotFoundEx = e;
222 			}
223 			if(null == clazz){
224 				ClassLoader threadClassLoader = Thread.currentThread().getContextClassLoader();
225 				if(null != threadClassLoader){
226 					try{
227 						clazz = Class.forName(classname, true, threadClassLoader);
228 					}catch(Exception e){
229 						throw new RuntimeException(e.getMessage() 
230 								+ " [The custom class " 
231 								+ classname 
232 								+ " could not be instantiated, check wether it is available on the classpath and" 
233 								+ " verify that it has a zero-arg constructor].\n" 
234 								+ " The specific error message is: " + e.getClass().getName() + ":" + e.getMessage(),e);
235 					}
236 				}else{
237 					throw new RuntimeException(classNotFoundEx.getMessage() 
238 							+ " [The custom class " 
239 							+ classname 
240 							+ " could not be instantiated, check wether it is available on the classpath and" 
241 							+ " verify that it has a zero-arg constructor].\n" 
242 							+ " The specific error message is: " + classNotFoundEx.getClass().getName() + ":" + classNotFoundEx.getMessage(),classNotFoundEx);
243 				}
244 				
245 			}
246 		}
247 		return clazz;
248 	}

250 	/***
251 	 * Builds a class instance using reflection, by using its classname. The class must have a zero-arg constructor. 
252 	 * @param classname the class to build an instance of. 
253 	 * @return
254 	 */
255 	public static Object buildObjectInstance(String classname, Object[] params) {
256 		Object rets = null;
257 		
258 		Class[] paramTypes = new Class[params.length];
259 		for(int x = 0; x < params.length; x++){
260 			paramTypes[x] = params[x].getClass();
261 		}
262 		
263 		try {
264 			Class clazz = locateClass(classname);
265 			rets = clazz.getConstructor(paramTypes).newInstance(params);
266 			
267 		} catch (Exception e) {
268 			throw new RuntimeException(e.getMessage() 
269 										+ " [The custom class " 
270 										+ classname 
271 										+ " could not be instantiated, check wether it is available on the classpath and" 
272 										+ " verify that it has a zero-arg constructor].\n" 
273 										+ " The specific error message is: " + e.getClass().getName() + ":" + e.getMessage(),e);
274 		}
275 		return rets;
276 	}
