30  /***
31   * A generator that creates a script from message bundles.
32   * The generated script can be used to reference the message literals easily from javascript.  
33   * 
34   * @author Jordi Hernández Sellés
35   * @author Ibrahim Chaehoi
36   *
37   */
38  public class ResourceBundleMessagesGenerator extends AbstractJavascriptGenerator implements ResourceGenerator, LocaleAwareResourceGenerator {
39  	
40  	/*** The logger */
41  	private static final Logger log = Logger.getLogger(ResourceBundleMessagesGenerator.class);
42  	
43  	public static final String GRAILS_WAR_DEPLOYED = "jawr.grails.war.deployed";
44  	
45  	private static final String GRAILS_MESSAGE_CREATOR = "net.jawr.web.resource.bundle.locale.message.GrailsMessageBundleScriptCreator";
46  	
47  	/* (non-Javadoc)
48  	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#createResource(java.lang.String, java.nio.charset.Charset)
49  	 */
50  	public Reader createResource(GeneratorContext context) {
51  		MessageBundleScriptCreator creator = null;
52  		// In grails apps, the generator uses a special implementation
53  		if(null == context.getServletContext().getAttribute(ResourceBundleMessagesGenerator.GRAILS_WAR_DEPLOYED)){
54  			if(log.isDebugEnabled())
55  				log.debug("Using standard messages generator. ");
56  			creator = new MessageBundleScriptCreator(context);
57  		}
58  		else {
59  			if(log.isDebugEnabled())
60  				log.debug("Using grails messages generator. ");
61  			// Loading this way prevents unwanted dependencies in non grails applications. 
62  			Object[] param = {context};
63  			creator = (MessageBundleScriptCreator) ClassLoaderResourceUtils.buildObjectInstance(GRAILS_MESSAGE_CREATOR,param);
64  		}
65  		return creator.createScript(context.getCharset());
66  	}
67  
68  	/* (non-Javadoc)
69  	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#getMappingPrefix()
70  	 */
71  	public String getMappingPrefix() {
72  		return GeneratorRegistry.MESSAGE_BUNDLE_PREFIX;
73  	}
74  	
75  	/* (non-Javadoc)
76  	 * @see net.jawr.web.resource.bundle.generator.ResourceGenerator#getDebugModeBuildTimeGenerationPath(java.lang.String)
77  	 */
78  	public String getDebugModeBuildTimeGenerationPath(String path) {
79  		
80  		path = path.replaceFirst(GeneratorRegistry.PREFIX_SEPARATOR, JawrConstant.URL_SEPARATOR);
81  		if(path.endsWith("@")){
82  			path = path.replaceAll("@", "");
83  		}else{
84  			path = path.replaceAll("@", "_");
85  			path = path.replaceAll("//|", "_");
86  		}
87  		return path+"."+JawrConstant.JS_TYPE;
88  	}
89  
90  	/* (non-Javadoc)
91  	 * @see net.jawr.web.resource.handler.LocaleAwareResourceReader#getAvailableLocales(java.lang.String)
92  	 */
93  	public List getAvailableLocales(String resource) {
94  		
95  		return LocaleUtils.getAvailableLocaleSuffixesForBundle(resource);
96  	}
97  }
