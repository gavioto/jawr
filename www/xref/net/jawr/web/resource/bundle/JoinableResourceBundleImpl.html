166 	/***
167 	 * Detects all files that belong to this bundle and adds them to the items path list.
168 	 */
169 	private void initPathList() {
170 		if (LOGGER.isDebugEnabled()){
171 			LOGGER.debug("Creating bundle path List for " + this.id);
172 		}
173 		
174 		for (Iterator it = pathMappings.iterator(); it.hasNext();) {
175 			String pathMapping = (String) it.next();
176 			boolean isGeneratedPath = generatorRegistry.isPathGenerated(pathMapping);
177 			// Handle generated resources
178 			// path ends in /, the folder is included without subfolders
179 			if (pathMapping.endsWith("/")) {
180 				addItemsFromDir(pathMapping, false);
181 			}
182 			// path ends in /, the folder is included with all subfolders
183 			else if (pathMapping.endsWith("/**")) {
184 				addItemsFromDir(pathMapping.substring(0, pathMapping
185 						.lastIndexOf("**")), true);
186 			} else if (pathMapping.endsWith(fileExtension)) {
187 				itemPathList.add(asPath(pathMapping, isGeneratedPath));
188 			} else if (generatorRegistry.isPathGenerated(pathMapping)) {
189 				itemPathList.add(pathMapping);
190 			}else if (pathMapping.endsWith(LICENSES_FILENAME)) {
191 				licensesPathList.add(asPath(pathMapping, isGeneratedPath));
192 			} else
193 				LOGGER.warn("Wrong mapping [" + pathMapping + "] for bundle ["
194 						+ this.name + "]. Please check configuration. ");
195 		}
196 		if (LOGGER.isDebugEnabled()){
197 			LOGGER.debug("Finished creating bundle path List for " + this.id);
198 		}
199 	}

201 	/***
202 	 * Adds all the resources within a path to the item path list.
203 	 * 
204 	 * @param dirName
205 	 * @param addSubDirs boolean If subfolders will be included. In such case, every folder below the path is included.
206 	 */
207 	private void addItemsFromDir(String dirName, boolean addSubDirs) {
208 		Collection resources = resourceReaderHandler.getResourceNames(dirName);
209 		boolean isGeneratedPath = generatorRegistry.isPathGenerated(dirName);
210 		if (LOGGER.isDebugEnabled()) {
211 			LOGGER.debug("Adding " + resources.size() + " resources from path ["
212 					+ dirName + "] to bundle " + getId());
213 		}
214 
215 		boolean isSortingResourcesAlphabetically = false;
216 		if(isSortingResourcesAlphabetically){
217 			Collections.sort(new ArrayList(resources));
218 		}
219 		
220 		// If the directory contains a sorting file, it is used to order the resources.
221 		if (resources.contains(SORT_FILE_NAME)
222 				|| resources.contains("/" + SORT_FILE_NAME)) {
223 			
224 			String sortFilePath = joinPaths(dirName,
225 					SORT_FILE_NAME, isGeneratedPath);
226 			
227 			Reader reader;
228 			try {
229 				reader = resourceReaderHandler.getResource(sortFilePath);
230 			} catch (ResourceNotFoundException e) {
231 				throw new BundlingProcessException(
232 						"Unexpected ResourceNotFoundException when reading a sorting file["
233 								+ sortFilePath + "]", e);
234 			}
235 			
236 			SortFileParser parser = new SortFileParser(reader, resources, dirName);
237 
238 			List sortedResources = parser.getSortedResources();
239 			for (Iterator it = sortedResources.iterator(); it.hasNext();) {
240 				String resourceName = (String) it.next();
241 
242 				// Add subfolders or files
243 				if (resourceName.endsWith(fileExtension)) {
244 					itemPathList.add(asPath(resourceName, isGeneratedPath));
245 					if (LOGGER.isDebugEnabled())
246 						LOGGER
247 								.debug("Added to item path list from the sorting file:"
248 										+ resourceName);
249 				} else if (addSubDirs
250 						&& resourceReaderHandler.isDirectory(resourceName))
251 					addItemsFromDir(resourceName, true);
252 			}
253 		}
254 
255 		// Add licenses file
256 		if (resources.contains(LICENSES_FILENAME)
257 				|| resources.contains("/" + LICENSES_FILENAME)) {
258 			licensesPathList.add(joinPaths(dirName,
259 					LICENSES_FILENAME, isGeneratedPath));
260 		}
261 
262 		// Add remaining resources (remaining after sorting, or all if no sort file present)
263 		List folders = new ArrayList();
264 		for (Iterator it = resources.iterator(); it.hasNext();) {
265 			String resourceName = (String) it.next();
266 			String resourcePath = joinPaths(dirName,
267 					resourceName, isGeneratedPath);
268 			if (resourceName.endsWith(fileExtension)) {
269 				itemPathList.add(asPath(resourcePath, isGeneratedPath));
270 
271 				if (LOGGER.isDebugEnabled())
272 					LOGGER.debug("Added to item path list:"
273 							+ asPath(resourcePath, isGeneratedPath));
274 			} else if (addSubDirs
275 					&& resourceReaderHandler.isDirectory(resourcePath))
276 				folders.add(resourceName);
277 		}
278 
279 		// Add subfolders if requested. Subfolders are added last unless specified in sorting file.
280 		if (addSubDirs) {
281 			for (Iterator it = folders.iterator(); it.hasNext();) {
282 				String folderName = (String) it.next();
283 				addItemsFromDir(joinPaths(dirName, folderName, isGeneratedPath),
284 						true);
285 			}
286 		}
287 	}

289 	/***
290 	 * Normalizes a path and adds a separator at its start, if it's not a generated resource. 
291 	 * @param path the path
292 	 * @param generatedResource the flag indicating if the resource has been generated
293 	 * @return the normalized path
294 	 */
295 	private String asPath(String path, boolean generatedResource){
296 		
297 		String result = path;
298 		if(!generatedResource){
299 			result = PathNormalizer.asPath(path);
300 		}
301 		return result;
302 	}

304 	/***
305 	 * Normalizes two paths and joins them as a single path. 
306 	 * @param prefix the path prefix
307 	 * @param path the path
308 	 * @param generatedResource the flag indicating if the resource has been generated
309 	 * @return the normalized path
310 	 */
311 	private String joinPaths(String dirName, String folderName, boolean generatedResource){
312 		
313 		String result = null;
314 		if(generatedResource){
315 			result = PathNormalizer.joinDomainToPath(dirName, folderName);
316 		}else{
317 			result = PathNormalizer.joinPaths(dirName, folderName);
318 		}
319 		return result;
320 	}

498 	public List getItemPathList(Map variants) {
499 		if (variants == null || variants.isEmpty())
500 			return itemPathList;
501 
502 		List rets = new ArrayList();
503 		
504 		for (Iterator it = itemPathList.iterator(); it.hasNext();) {
505 			String path = (String) it.next();
506 			if (generatorRegistry.isPathGenerated(path)) {
507 				Set variantTypes = generatorRegistry.getGeneratedResourceVariantTypes(path);
508 				String variantKey = VariantUtils.getVariantKey(variants, variantTypes);
509 				if(StringUtils.isNotEmpty(variantKey)){
510 					rets.add(VariantUtils.getVariantBundleName(path, variantKey));
511 				}else{
512 					rets.add(path);
513 				}
514 			} else{
515 				rets.add(path);
516 			}
517 		}
518 		return rets;
519 	}

554 	/* (non-Javadoc)
555 	 * @see net.jawr.web.resource.bundle.JoinableResourceBundle#getURLPrefix(java.util.Map)
556 	 */
557 	public String getURLPrefix(Map variants) {
558 		
559 		if (null == this.urlPrefix)
560 			throw new IllegalStateException(
561 					"The bundleDataHashCode must be set before accessing the url prefix.");
562 
563 		if (variants != null && !variants.isEmpty()) {
564 			String key = getAvailableVariant(variants);
565 			if (StringUtils.isNotEmpty(key)){
566 				return prefixMap.get(key) + "." + key + "/";
567 			}
568 		}
569 		return this.urlPrefix + "/";
570 	}
