14  package net.jawr.web.resource;
15  
16  import java.io.File;
17  import java.io.InputStream;
18  import java.io.InputStreamReader;
19  import java.io.Reader;
20  import java.nio.charset.Charset;
21  import java.util.HashSet;
22  import java.util.Iterator;
23  import java.util.Set;
24  
25  import javax.servlet.ServletContext;
26  
27  import net.jawr.web.exception.ResourceNotFoundException;
28  import net.jawr.web.resource.bundle.generator.GeneratorRegistry;
29  
30  /***
31   * Implementation of resourcehandler that gets its resources
32   * from a ServletContext instance, as resources contained in 
33   * a deployed war archive (or exploded directory). The generated
34   * files are stored in the web server temporary folder. 
35   * 
36   * @author Jordi Hernández Sellés
37   * @author Ibrahim Chaehoi
38   *
39   */
40  public class ServletContextResourceHandler extends AbstractResourceHandler implements ResourceHandler {
41  	
42  	/*** The servlet temp directory property name */
43  	private static final String SERVLET_CONTEXT_TEMPDIR = "javax.servlet.context.tempdir";
44  	
45  	/*** The servlet context */
46  	private ServletContext context;
47  	
48  	/***
49  	 * Constructor
50  	 * @param context the servlet context
51  	 * @param charset the charset
52  	 * @param generatorRegistry the generator registry
53  	 * @param resourceType the resource type
54  	 */
55  	public ServletContextResourceHandler(ServletContext context, Charset charset,GeneratorRegistry generatorRegistry, String resourceType) {
56  		super((File) context.getAttribute(SERVLET_CONTEXT_TEMPDIR),charset,generatorRegistry, resourceType);		
57  		this.context  = context;
58  		this.charset = charset;
59  	}
60  
61  	/***
62  	 * Constructor
63  	 * @param context the servlet context
64  	 * @param workingDirectory the working directory
65  	 * @param charset the charset
66  	 * @param generatorRegistry the generator registry
67  	 * @param resourceType the resource type
68  	 */
69  	public ServletContextResourceHandler(ServletContext context, String workingDirectory, Charset charset,GeneratorRegistry generatorRegistry, String resourceType) {
70  		super(workingDirectory,charset,generatorRegistry, resourceType, false);		
71  		this.context  = context;
72  		this.charset = charset;
73  	}
74  	
75  	/* (non-Javadoc)
76  	 * @see net.jawr.web.resource.AbstractResourceHandler#doGetResourceAsStream(java.lang.String)
77  	 */
78  	protected InputStream doGetResourceAsStream(String resourceName) {
79  		
80  		return context.getResourceAsStream(resourceName);	
81  	}
82  	
83  	/* (non-Javadoc)
84  	 * @see net.jawr.web.resource.bundle.ResourceHandler#getResource(java.lang.String)
85  	 */
86  	public Reader doGetResource(String resourceName)  throws ResourceNotFoundException{
87  		
88  		InputStream is = getResourceAsStream(resourceName);		
89  		return new InputStreamReader(is, charset);
90  	}
91  
92  	/* (non-Javadoc)
93  	 * @see net.jawr.web.resource.bundle.ResourceHandler#getResourceNames(java.lang.String)
94  	 */
95  	public Set getResourceNames(String path) {
96  		Set paths = context.getResourcePaths(path);
97  		Set names = new HashSet();
98  		int length = path.length();
99  		if(null != paths) {
100 			for(Iterator it = paths.iterator();it.hasNext();) {
101 				String resourcePath = (String) it.next();
102 				names.add(resourcePath.substring(length, resourcePath.length()));
103 			}
104 		}
105 		return names;
106 	}
107 
108 	/* (non-Javadoc)
109 	 * @see net.jawr.web.resource.bundle.ResourceHandler#isDirectory(java.lang.String)
110 	 */
111 	public boolean isDirectory(String path) {
112 		Set paths = context.getResourcePaths(path);
113 		return (null != paths && paths.size() > 0);
114 	}
115 
116 }
